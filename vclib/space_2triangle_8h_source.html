326 static PointT
327 circumcenter(const PointT& p0, const PointT& p1, const PointT& p2)
328 {
329 ScalarType a2 = p1.squaredDist(p2);
330 ScalarType b2 = p2.squaredDist(p0);
331 ScalarType c2 = p0.squaredDist(p1);
332
333 PointType c = p0 * a2 * (-a2 + b2 + c2) + p1 * b2 * (a2 - b2 + c2) +
334 p2 * c2 * (a2 + b2 - c2);
335
336 c /= 2 * (a2 * b2 + a2 * c2 + b2 * c2) - a2 * a2 - b2 * b2 - c2 * c2;
337
338 return c;
339 }

368 static ScalarType area(const PointT& p0, const PointT& p1, const PointT& p2)
369 {
370 if constexpr(DIM == 2) {
371 return ((p1[0] - p0[0]) * (p2[1] - p0[1]) -
372 (p2[0] - p0[0]) * (p1[1] - p0[1])) /
373 2;
374 }
375 if constexpr(DIM == 3) {
376 return normal(p0, p1, p2).norm() / 2;
377 }
378 else {
379 // heron's formula
380 ScalarType s = perimeter() / 2;
381 return std::sqrt(
382 s * (s - p0.dist(p1)) * (s - p1.dist(p2)) * (s - p2.dist(p0)));
383 }
384 }

403 static ScalarType
404 quality(const PointT& p0, const PointT& p1, const PointT& p2)
405 {
406 ScalarType a = area();
407 if (a == 0)
408 return 0; // Area zero triangles have surely quality==0;
409
410 PointType d10 = p1 - p0;
411 PointType d20 = p2 - p0;
412 PointType d12 = p1 - p2;
413 double b =
414 std::min({d10.squaredNorm(), d20.squaredNorm(), d12.squaredNorm()});
415 if (b == 0)
416 return 0; // Again: area zero triangles have surely quality==0;
417 return (2 * a) / b;
418 }

437 static ScalarType
438 qualityRadii(const PointT& p0, const PointT& p1, const PointT& p2)
439 {
440 double a = p0.dist(p1);
441 double b = p2.dist(p0);
442 double c = p1.dist(p2);
443
444 double sum = (a + b + c) * 0.5;
445 double area2 = sum * (a + b - sum) * (a + c - sum) * (b + c - sum);
446
447 if (area2 <= 0)
448 return 0;
449
450 return (8 * area2) / (a * b * c * sum);
451 }
452
473 static ScalarType
474 qualityMeanRatio(const PointT& p0, const PointT& p1, const PointT& p2)
475 {
476 double a = p0.dist(p1);
477 double b = p2.dist(p0);
478 double c = p1.dist(p2);
479
480 double sum = (a + b + c) * 0.5; // semiperimeter
481 double area2 = sum * (a + b - sum) * (a + c - sum) * (b + c - sum);
482 if (area2 <= 0)
483 return 0;
484 return (4.0 * std::sqrt(3.0) * std::sqrt(area2)) /
485 (a * a + b * b + c * c);
486 }
