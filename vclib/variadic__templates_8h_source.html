54/*
55 * Possibility to get the index of a Type T in a pack of types (variadic templates).
56 * The pack is composed of U and Us...
57 *
58 * https://stackoverflow.com/a/71477756/5851101
59 */
60template <typename T, typename U, typename... Us>
61constexpr uint indexInTypePack() {
62 if constexpr (std::is_same_v<T, U>) {
63 return 0;
64 }
65 else {
66 if constexpr (sizeof...(Us)) { // there is at least another type to check
67 return 1 + indexInTypePack<T, Us...>();
68 }
69 else { // not found
70 return UINT_NULL;
71 }
72 }
73}
74
75template<typename T, typename... Us>
76struct IndexInTypes

78 static constexpr uint value = indexInTypePack<T, Us...>();
79};

81template<typename T, typename... Us>
