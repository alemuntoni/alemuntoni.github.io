38template<typename GridType, typename ValueType>
39class StaticGrid :
40 public AbstractGrid<
41 GridType,
42 ValueType,
43 StaticGrid<GridType, ValueType>>
44{
45 using AbsGrid =
46 AbstractGrid<GridType, ValueType, StaticGrid<GridType, ValueType>>;
47
48 using PairType = std::pair<uint, ValueType>;
49 using PairComparator = FirstElementPairComparator<PairType>;
50
51 friend AbsGrid;
52
53 PairComparator comparator = PairComparator();
54
55 // each value is stored as a pair: [cell index of the grid - value]
56 // when the grid is built, this vector is sorted by the cell indices
57 std::vector<PairType> values;
58
59 // for each cell of the grid, we store the index (in the values vector ) of
60 // the first ValueType object contained in the cell, or values.size() if the
61 // cell is empty
62 std::vector<uint> grid;
63
64public:
65 using KeyType = AbsGrid::KeyType;
66 using IsInCellFunction = AbsGrid::IsInCellFunction;
67
68 using Iterator = StaticGridIterator<KeyType, ValueType, GridType>;
69 using ConstIterator = ConstStaticGridIterator<KeyType, ValueType, GridType>;
70
71 StaticGrid() {}
72
73 StaticGrid(const GridType& g) : AbsGrid(g) {}

92 void build()
93 {
94 uint totCellNumber = 1;
95 for (uint i = 0; i < GridType::DIM; ++i) {
96 totCellNumber *= GridType::cellNumber(i);
97 }
98
99 grid.resize(totCellNumber);
100
101 std::sort(values.begin(), values.end(), comparator);
102
103 // values[vi].first points to the next non empty cell in the grid
104 uint vi = 0;
105
106 // for each cell
107 for (uint ci = 0; ci < grid.size(); ++ci) {
108 // values[vi] is in the cell ci
109 if (vi < values.size() && values[vi].first == ci)
110 grid[ci] = vi;
111 else { // there are no values in this grid cell
112 grid[ci] = values.size(); // set the sentinel value
113 }
114
115 // move vi to the next non-empty cell
116 // skip all the values that are in the current cell ci
117 // won't increment vi if the values[vi].first is not equal to ci
118 while (vi < values.size() && values[vi].first == ci) {
119 vi++;
120 }
121 }
122 }
123
124 bool empty() const { return values.empty(); }
125
126 bool cellEmpty(const KeyType& k) const
127 {
128 uint ind = GridType::indexOfCell(k);
129 return grid[ind] == values.size();
130 }
131
132 std::set<KeyType> nonEmptyCells() const
133 {
134 std::set<KeyType> keys;
135 uint actualInd = values.size();
136 for (uint i = 0; i < values.size(); ++i) {
137 if (values[i].first != actualInd) {
138 actualInd = values[i].first;
139 keys.insert(GridType::cellOfIndex(actualInd));
140 }
141 }
142 return keys;
143 }
144
145 std::size_t countInCell(const KeyType& k) const
146 {
147 uint ind = GridType::indexOfCell(k);
148 uint i = grid[ind];
149 uint cnt = 0;
150
151 while(i < values.size() && values[i].first == ind) {
152 i++;
153 cnt++;
154 }
155 return cnt;
156 }
157
158 std::pair<Iterator, Iterator> valuesInCell(const KeyType& k)
159 {
160 uint ind = GridType::indexOfCell(k);
161 uint i = grid[ind];
162 if (i < values.size()) {
163 std::pair<Iterator, Iterator> p;
164 p.first = Iterator(values.begin() + i, (const GridType&)*this);
165 while(i < values.size() && values[i].first == ind) {
166 i++;
167 }
168 auto it = i < values.size() ? values.begin() + i : values.end();
169 p.second = Iterator(it, (const GridType&)*this);
170 return p;
171 }
172 else {
173 return std::make_pair(end(), end());
174 }
175 }
176
177 std::pair<ConstIterator, ConstIterator> valuesInCell(const KeyType& k) const
178 {
179 uint ind = GridType::indexOfCell(k);
180 uint i = grid[ind];
181 if (i < values.size()) {
182 std::pair<ConstIterator, ConstIterator> p;
183 p.first = ConstIterator(values.begin() + i, (const GridType&)*this);
184 while(i < values.size() && values[i].first == ind) {
185 i++;
186 }
187 auto it = i < values.size() ? values.begin() + i : values.end();
188 p.second = ConstIterator(it, (const GridType&)*this);
189 return p;
190 }
191 else {
192 return std::make_pair(end(), end());
193 }
194 }
195
196 Iterator begin()
197 {
198 return Iterator(values.begin(), (const GridType&)*this);
199 }
200
201 ConstIterator begin() const
202 {
203 return ConstIterator(values.begin(), (const GridType&) *this);
204 }
205
206 Iterator end() { return Iterator(values.end(), (const GridType&) *this); }
207
208 ConstIterator end() const
209 {
210 return ConstIterator(values.end(), (const GridType&)*this);
211 }
212
213private:
214 // not available member functions
215 using AbsGrid::erase;
216 using AbsGrid::eraseAllInCell;
217 using AbsGrid::eraseInSphere;
218
219 bool insertInCell(const KeyType& cell, const ValueType& v)
220 {
221 uint cellIndex = GridType::indexOfCell(cell);
222 values.emplace_back(cellIndex, v);
223 return true;
224 }
225
226 // not allowing to erase
227 bool eraseInCell(const KeyType&, const ValueType&) { return false; };
228};
