46template<MatrixConcept MatrixType, Point3Concept PointType, typename ScalarType>
47void setTransformMatrixRotation(
48 MatrixType& matrix,
49 PointType axis,
50 const ScalarType& angleRad)
51{
52 ScalarType c = std::cos(angleRad);
53 ScalarType s = std::sin(angleRad);
54 ScalarType q = 1 - c;
55 axis.normalize();
56 matrix(0, 0) = axis[0] * axis[0] * q + c;
57 matrix(0, 1) = axis[0] * axis[1] * q - axis[2] * s;
58 matrix(0, 2) = axis[0] * axis[2] * q + axis[1] * s;
59 matrix(1, 0) = axis[1] * axis[0] * q + axis[2] * s;
60 matrix(1, 1) = axis[1] * axis[1] * q + c;
61 matrix(1, 2) = axis[1] * axis[2] * q - axis[0] * s;
62 matrix(2, 0) = axis[2] * axis[0] * q - axis[1] * s;
63 matrix(2, 1) = axis[2] * axis[1] * q + axis[0] * s;
64 matrix(2, 2) = axis[2] * axis[2] * q + c;
65}

102template<MatrixConcept MatrixType, Point3Concept PointType>
103void setTransformMatrixRotation(
104 MatrixType& matrix,
105 const PointType& fromVector,
106 const PointType& toVector)
107{
108 if (fromVector == toVector) {
109 matrix.block(0, 0, 3, 3).setIdentity();
110 return;
111 }
112 PointType axis = fromVector.cross(toVector);
113 auto angle = std::acos(
114 fromVector.dot(toVector) / (fromVector.norm() * toVector.norm()));
115 setTransformMatrixRotation(matrix, axis, angle);
116}
