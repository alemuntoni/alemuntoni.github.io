34/*
35 * Utility to get clean type from an input type that could have a reference or a pointer.
36 */
37template<typename T>
38using RemoveCVRefAndPointer =
39 typename std::remove_cvref_t<std::remove_pointer_t<T>>;

41/*
42 * Utility to get a pointer type without the constness.
43 * e.g. If T is const int*, the resulting type is int*
44 * If the type is not a pointer, it is left as it was
45 */
46template<typename T>
47using RemoveConstFromPointer =
48 std::conditional_t<std::is_pointer_v<T>,
49 std::add_pointer_t<typename std::remove_cv_t<typename std::remove_pointer_t<T>>>,
50 T>;
