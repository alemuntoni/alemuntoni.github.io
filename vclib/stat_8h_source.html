50template <MeshConcept MeshType>
51typename MeshType::VertexType::CoordType barycenter(const MeshType& m)
52{
53 using VertexType = MeshType::VertexType;
54 using CoordType = VertexType::CoordType;
55
56 CoordType bar;
57
58 for (const VertexType& v : m.vertices()) {
59 bar += v.coord();
60 }
61
62 return bar / m.vertexNumber();
63}

80template<MeshConcept MeshType>
81typename MeshType::VertexType::CoordType

116template<FaceMeshConcept MeshType>
117typename MeshType::VertexType::CoordType shellBarycenter(const MeshType& m)
118{
119 using VertexType = MeshType::VertexType;
120 using FaceType = MeshType::FaceType;
121 using CoordType = VertexType::CoordType;
122 using ScalarType = CoordType::ScalarType;
123
124 CoordType bar;
125 bar.setZero();
126 ScalarType areaSum = 0;
127
128 for (const FaceType& f : m.faces()) {
129 ScalarType area = faceArea(f);
130 bar += faceBarycenter(f) * area;
131 areaSum += area;
132 }
133
134 return bar / areaSum;
135}

198template<PointConcept PointType>
199auto covarianceMatrixOfPointCloud(const std::vector<PointType>& pointVec)
200{
201 Matrix33<typename PointType::ScalarType> m;
202 m.setZero();
203 PointType barycenter =
204 Polygon<PointType>::barycenter(pointVec.begin(), pointVec.end());
205
206 // compute covariance matrix
207 for (const PointType& p : pointVec){
208 auto e = (p-barycenter).eigenVector();
209 m += e.transpose()*e; // outer product
210 }
211 return m;
212}

248template<PointConcept PointType>
249auto weightedCovarianceMatrixOfPointCloud(
250 const std::vector<PointType>& pointVec,
251 const std::vector<typename PointType::ScalarType>& weights)
252{
253 Matrix33<typename PointType::ScalarType> m;
254 m.setZero();
255 PointType barycenter = polygonWeighedBarycenter(
256 pointVec.begin(), pointVec.end(), weights.begin());
257
258 // compute covariance matrix
259 typename PointType::ScalarType wsum = 0;
260 for (uint i = 0; i < pointVec.size(); ++i){
261 auto e = ((pointVec[i]-barycenter)*weights[i]).eigenVector();
262 m += e.transpose()*e; // outer product
263 wsum += weights[i];
264 }
265 return m / wsum;
266}

276template<FaceMeshConcept MeshType>
277auto covarianceMatrixOfMesh(const MeshType& m)
278{
279 using VertexType = MeshType::VertexType;
280 using FaceType = MeshType::FaceType;
281 using CoordType = VertexType::CoordType;
282 using ScalarType = CoordType::ScalarType;
283
284 CoordType bar = shellBarycenter(m);
285 Matrix33<ScalarType> C;
286 C.setZero();
287 Matrix33<ScalarType> C0;
288 C0.setZero();
289 C0(0,0) = C0(1,1) = 2.0;
290 C0(0,1) = C0(1,0) = 1.0;
291 C0*=1/24.0;
292 // integral of (x,y,0) in the same triangle
293 Eigen::Vector3<ScalarType> x;
294 x << 1/6.0,1/6.0,0;
295 Matrix33<ScalarType> A; // matrix that bring the vertices to (v1-v0,v2-v0,n)
296 Matrix33<ScalarType> DC;
297
298 for (const FaceType& f : m.faces()) {
299 const CoordType& p0 = f.vertex(0)->coord();
300 const CoordType& p1 = f.vertex(1)->coord();
301 const CoordType& p2 = f.vertex(2)->coord();
302 CoordType n = (p1-p0).cross(p2-p0);
303 double da = n.norm();
304 n /= da*da;
305
306 CoordType tmpp = p1 - p0;
307 for (uint j = 0; j < 3; j++)
308 A(j, 0) = tmpp(j);
309 tmpp = p2 - p0;
310 for (uint j = 0; j < 3; j++)
311 A(j, 1) = tmpp(j);
312 for (uint j = 0; j < 3; j++)
313 A(j, 2) = n(j);
314 Eigen::Vector3<ScalarType> delta;
315 tmpp = p0 - bar;
316 for (uint j = 0; j < 3; j++)
317 delta(j) = tmpp(j);
318
319 /* DC is calculated as integral of (A*x+delta) * (A*x+delta)^T over the
320 * triangle, where delta = v0-bary */
321 DC.setZero();
322 DC += A* C0 * A.transpose();
323 Matrix33<ScalarType> tmp = (A*x) * delta.transpose();
324 DC += tmp + tmp.transpose();
325 DC += tmp;
326 tmp = delta * delta.transpose();
327 DC+=tmp*0.5;
328 DC*=da; // the determinant of A is also the double area of f
329 C+=DC;
330 }
331 return C;
332}

348template<MeshConcept MeshType, typename ScalarType>
349std::vector<ScalarType> vertexRadiusFromWeights(
350 const MeshType& m,
351 const std::vector<ScalarType>& weights,
352 double diskRadius,
353 double radiusVariance,
354 bool invert = false)
355{
356 using VertexType = MeshType::VertexType;
357
358 std::vector<ScalarType> radius(m.vertexContainerSize());
359 auto minmax = std::minmax_element(weights.begin(), weights.end());
360
361 float minRad = diskRadius;
362 float maxRad = diskRadius * radiusVariance;
363 float deltaQ = *minmax.second - *minmax.first;
364 float deltaRad = maxRad - minRad;
365 for (const VertexType& v : m.vertices()) {
366 ScalarType w = weights[m.index(v)];
367 radius[m.index(v)] =
368 minRad +
369 deltaRad *
370 ((invert ? *minmax.second - w : w - *minmax.first) / deltaQ);
371 }
372
373 return radius;
374}
