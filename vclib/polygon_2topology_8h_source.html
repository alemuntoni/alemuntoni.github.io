55template<FaceConcept FaceType>
56bool isFaceManifoldOnEdge(const FaceType& f, uint edge)
57 requires comp::HasAdjacentFaces<FaceType>
58{
59 // Check if the AdjacentFaces component is available for the given face.
60 if (! comp::isAdjacentFacesAvailableOn(f)) {
61 throw vcl::MissingComponentException(
62 "Face has no Adjacent Faces component.");
63 }
64
65 // Check if the edge is a boundary edge.
66 if (f.adjFace(edge) == nullptr) {
67 return true;
68 }
69 else { // Check if the edge is shared by exactly two faces.
70 return f.adjFace(edge)->indexOfAdjFace(&f) != UINT_NULL;
71 }
72}

136template<FaceConcept FaceType>
137bool checkFlipEdge(const FaceType& f, uint edge)
138 requires comp::HasAdjacentFaces<FaceType>
139{
140 if (! comp::isAdjacentFacesAvailableOn(f)) {
141 throw vcl::MissingComponentException(
142 "Face has no Adjacent Faces component.");
143 }
144
145 using VertexType = FaceType::VertexType;
146
147 if (f.vertexNumber() > 3)
148 return false;
149
150 if (vcl::isFaceEdgeOnBorder(f, edge))
151 return false;
152
153 const VertexType* v0 = f.vertex(edge);
154 const VertexType* v1 = f.vertexMod(edge+1);
155
156 const FaceType* of = f.adjFace(edge);
157 uint oe = of->indexOfAdjFace(&f);
158 assert(oe != UINT_NULL);
159
160 // check if the vertices of the edge are the same
161 // e.g. the mesh has to be well oriented
162 if (of->vertex(oe) != v1 || of->vertexMod(oe+1) != v0)
163 return false;
164
165 // check if the flipped edge is already present in the mesh
166 // f_v2 and of_v2 are the vertices of the new edge
167 const VertexType* f_v2 = f.vertexMod(edge+2);
168 const VertexType* of_v2 = of->vertexMod(oe+2);
169
170 MeshPos<FaceType> pos(&f, f_v2);
171 MeshPos<FaceType> startPos = pos;
172 // loop in the one ring of f_v2
173 do {
174 pos.nextEdgeAdjacentToV();
175 if (pos.adjVertex() == of_v2)
176 return false;
177 } while(pos != startPos);
178
179 return true;
180}

204 requires comp::HasAdjacentFaces<FaceType>

212 uint cnt = 0;

277template<FaceConcept FaceType>
278void detachAdjacentFacesOnEdge(FaceType& f, uint edge)
279 requires comp::HasAdjacentFaces<FaceType>
280{
281 if (! comp::isAdjacentFacesAvailableOn(f)) {
282 throw vcl::MissingComponentException(
283 "Face has no Adjacent Faces component.");
284 }
285
286 FaceType* nextFace = f.adjFace(edge);
287
288 // if nextFace == nullptr there is nothing to do
289 // the face is already detached on the edge
290 if (nextFace != nullptr) {
291
292 FaceType* prevFace;
293 ConstEdgeAdjFaceIterator<FaceType> begin(f, edge), end;
294 for (auto it = begin ; it != end; ++it) {
295 prevFace = *it;
296 }
297
298 if (nextFace == prevFace) { // manifold case
299 uint en = nextFace->indexOfAdjFace(&f);
300 assert(en != UINT_NULL);
301 nextFace->adjFace(en) = nullptr;
302 }
303 else { // non manifold case
304 // the prev face does not have to point to f anymore, but to
305 // nextFace
306 uint pn = prevFace->indexOfAdjFace(&f);
307 assert(pn != UINT_NULL);
308 prevFace->adjFace(pn) = nextFace;
309 }
310 f.adjFace(edge) = nullptr;
311 }
312}

335template <FaceConcept FaceType>
336void detachFace(FaceType& f) requires comp::HasAdjacentFaces<FaceType>
337{
338 if (! comp::isAdjacentFacesAvailableOn(f)) {
339 throw vcl::MissingComponentException(
340 "Face has no Adjacent Faces component.");
341 }
342
343 using VertexType = FaceType::VertexType;
344
345 for (uint e = 0; e < f.vertexNumber(); ++e) {
346 detachAdjacentFacesOnEdge(f, e);
347
348 // if the vertices have adjacent faces
349 if constexpr (comp::HasAdjacentFaces<VertexType>) {
350 if (comp::isAdjacentFacesAvailableOn(f.vertex(e))) {
351 VertexType* v = f.vertex(e);
352 uint vpos = v->indexOfAdjFace(&f);
353 if (vpos != UINT_NULL) { // may happen if vertex adj faces are
354 // not initialized / updated
355 v->eraseAdjFace(vpos); // the vertex v has not anymore the
356 // adjacent face f
357 }
358 }
359 }
360 }

414template<FaceMeshConcept MeshType, FaceConcept FaceType>
415void addTriangleFacesFromPolygon(
416 MeshType& m,
417 FaceType& f,
418 const std::vector<uint>& polygon)
419{
420 using VertexType = MeshType::VertexType;
421 using CoordType = VertexType::CoordType;
422
423 // from the ids, create a polygon of coordinates
424 std::vector<CoordType> polCoords(polygon.size());
425 for (uint i = 0; i < polygon.size(); ++i) {
426 if (polygon[i] >= m.vertexContainerSize()) {
427 throw BadVertexIndexException(
428 "Index " + std::to_string(polygon[i]) + " is out of range in "
429 "Vertex Container.");
430 }
431 if (m.vertex(polygon[i]).deleted()) {
432 throw BadVertexIndexException(
433 "Vertex " + std::to_string(polygon[i]) + " is deleted.");
434 }
435 polCoords[i] = m.vertex(polygon[i]).coord();
436 }
437
438 // compute earcut of the polygons
439 std::vector<uint> tris =
440 Polygon<CoordType>::earCut(polCoords.begin(), polCoords.end());
441
442 // faux edges management: create a set of unordered edges of the polygon
443 // note: we use indices from 0 to polygon.size() because that are the output
444 // indices given by the earcut algorithm
445 std::set<std::pair<uint, uint>, UnorderedPairComparator<uint>>
446 unorderedEdges;
447 for (uint i = 0; i < polygon.size(); ++i)
448 unorderedEdges.emplace(i, (i+1) % (uint)polygon.size());
449
450 if constexpr (FaceType::VERTEX_NUMBER < 0) {
451 f.resizeVertices(3);
452 }
453
454 // set the first triangle of the loaded polygon
455 uint i = 0; // from 0 to 2, vertices indices of the triangle
456 for (auto& v : f.vertices()) {
457 v = &m.vertex(polygon[tris[i]]);
458 ++i;
459 }
460
461 if constexpr(face::HasFaceBitFlags<FaceType>) {
462 if (unorderedEdges.find(std::make_pair(tris[0], tris[1])) ==
463 unorderedEdges.end())
464 f.edgeFaux(0) = true;
465 if (unorderedEdges.find(std::make_pair(tris[1], tris[2])) ==
466 unorderedEdges.end())
467 f.edgeFaux(1) = true;
468 if (unorderedEdges.find(std::make_pair(tris[2], tris[0])) ==
469 unorderedEdges.end())
470 f.edgeFaux(2) = true;
471 }
472
473 // remaining triangles, need to create more faces in the mesh
474 for (uint i = 3; i < tris.size(); i += 3) {
475 uint ff = m.addFace();
476
477 if constexpr (FaceType::VERTEX_NUMBER < 0) {
478 m.face(ff).resizeVertices(3);
479 }
480
481 uint j = 0; // from 0 to 2, vertices indices of the triangle
482 for (auto& v : m.face(ff).vertices()) {
483 v = &m.vertex(polygon[tris[i + j]]);
484 ++j;
485 }
486
487 if constexpr(face::HasFaceBitFlags<FaceType>) {
488 if (unorderedEdges.find(std::make_pair(tris[i], tris[i + 1])) ==
489 unorderedEdges.end())
490 m.face(ff).edgeFaux(0) = true;
491 if (unorderedEdges.find(std::make_pair(tris[i + 1], tris[i + 2])) ==
492 unorderedEdges.end())
493 m.face(ff).edgeFaux(1) = true;
494 if (unorderedEdges.find(std::make_pair(tris[i + 2], tris[i + 0])) ==
495 unorderedEdges.end())
496 m.face(ff).edgeFaux(2) = true;
497 }
498 }
499}
