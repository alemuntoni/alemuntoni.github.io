40template<typename Scalar>
41class Distribution
42{
43 std::set<Scalar> set;
44
45 Scalar minValue = std::numeric_limits<Scalar>::max();
46 Scalar maxValue = std::numeric_limits<Scalar>::lowest();
47
48 Scalar valSum = 0;
49 Scalar sqrdValSum = 0;
50 Scalar valAvg = 0;
51 Scalar sqrdValAvg = 0;
52 Scalar valRMS = 0;
53
54public:
58 Distribution() = default;
59
63 void clear()
64 {
65 set.clear();
66 minValue = std::numeric_limits<Scalar>::max();
67 maxValue = std::numeric_limits<Scalar>::lowest();
68 valSum = 0;
69 sqrdValSum = 0;
70 valAvg = 0;
71 sqrdValAvg = 0;
72 valRMS = 0;
73 }
74
79 void add(Scalar v)
80 {
81 set.insert(v);
82 if (v < minValue)
83 minValue = v;
84 if (v > maxValue)
85 maxValue = v;
86 valSum += v;
87 sqrdValSum += v * v;
88 valAvg = valSum / set.size();
89 sqrdValAvg = sqrdValSum / set.size();
90 valRMS = std::sqrt(sqrdValAvg);
91 }
92
99 Scalar min() const { return minValue; }
100
107 Scalar max() const { return maxValue; }
108
113 uint size() const { return set.size(); }
114
119 Scalar sum() const { return valSum; }
120
125 Scalar average() const { return valAvg; }
126
131 Scalar rootMeanSquare() const { return valRMS; }
132
137 Scalar variance() const { return sqrdValAvg - valAvg * valAvg; }
138
143 Scalar standardDeviation() const { return std::sqrt(variance()); }
144
151 Scalar percentile(Scalar perc) const
152 {
153 assert(!set.empty());
154 assert(perc >= 0 && perc <= 1);
155
156 int index = set.size() * perc - 1;
157 if (index < 0)
158 index = 0;
159 auto it = set.begin();
160 std::advance(it, index);
161 return *it;
162 }
163};
