31template<
32 template<typename, typename...>

34 typename T,

36class ElementContainerIterator
37{
38 using ContIt = std::conditional_t<
39 CNST,
40 typename Container<T>::const_iterator,

42
43public:
44 using difference_type = ptrdiff_t;
45 using value_type = T;
46 using reference = std::conditional_t<CNST, const T&, T&>;
47 using pointer = std::conditional_t<CNST, const T*, T*>;
48 using iterator_category = std::random_access_iterator_tag;
49
50private:
51 ContIt mIt; // the actual iterator
52
53 const Container<T>* mVec = nullptr; // need to check end when jump elements
54
55 /* Pointers to functions */
56
57 // pointer to mIncrementFun function

75 difference_type) = &ElementContainerIterator::assignSumJump;
76
77 // pointer to diff between iterators function
78 difference_type (ElementContainerIterator::*mDiffFun)(
79 const ElementContainerIterator& oi) const =

82public:
83 ElementContainerIterator() = default;

85 ElementContainerIterator(
86 ContIt it,

88 bool jumpDeleted = true) :

179private:
184 ElementContainerIterator& incrementJump()
185 {
186 do {
187 ++mIt;
188 } while (mIt != mVec->end() && mIt->deleted());
189 return *this;
190 }

196 ElementContainerIterator postIncrementJump()
197 {
198 ElementContainerIterator old = *this;
199 do {
200 ++mIt;

209 ElementContainerIterator& incrementFast()

211 ++mIt;

230 ElementContainerIterator& decrementJump()

232 do {

242 ElementContainerIterator postDecrementJump()

255 ElementContainerIterator& decrementFast()

257 --mIt;

260
265 ElementContainerIterator postDecrementFast()
266 {
267 ElementContainerIterator old = *this;
268 --mIt;
269 return old;
270 }
271
272 ElementContainerIterator& assignSumJump(difference_type n)
273 {
274 difference_type m = n;
275
276 if (m >= 0)
277 while (m--)
278 this->operator++();
279 else
280 while (m++)
281 this->operator--();
282 return *this;
283 }
284
285 ElementContainerIterator& assignSumFast(difference_type n)
286 {
287 mIt += n;
288 return *this;
289 }
290
291 difference_type diffJump(const ElementContainerIterator& oi) const
292 {
293 ElementContainerIterator i = oi;
294 difference_type diff = 0;
295 while (i != *this) {
296 ++diff;
297 ++i;
298 }
299 return diff;
300 }
301
302 difference_type diffFast(const ElementContainerIterator& oi) const
303 {
304 return this->mIt - oi.mIt;
305 }
