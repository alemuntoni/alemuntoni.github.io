42template<FaceMeshConcept MeshType, FaceConcept FaceType>
43void saveFaceIndices(
44 std::ofstream& file,
45 Property p,
46 const MeshType& m,
47 const std::vector<uint>& vIndices,
48 const FaceType& f,
49 bool bin)
50{
51 using VertexType = MeshType::VertexType;
52
53 uint fsize = f.vertexNumber();
54 io::internal::writeProperty(file, fsize, p.listSizeType, bin);
55 for (const VertexType* v : f.vertices()) {
56 io::internal::writeProperty(file, vIndices[m.index(v)], p.type, bin);
57 }

60template<FaceMeshConcept MeshType, FaceConcept FaceType>
61void setFaceIndices(FaceType& f, MeshType& m, const std::vector<uint>& vids)
62{
63 bool splitFace = false;
64 // we have a polygonal mesh
65 if constexpr (FaceType::VERTEX_NUMBER < 0) {
66 // need to resize the face to the right number of verts
67 f.resizeVertices(vids.size());
68 }
69 else if (FaceType::VERTEX_NUMBER != vids.size()) {
70 // we have faces with static sizes (triangles), but we are loading faces
71 // with number of verts > 3. Need to split the face we are loading in n
72 // faces!
73 splitFace = true;
74 }
75
76 if (!splitFace) { // classic load, no split needed
77 uint i = 0;
78 for (auto& v : f.vertices()) {
79 if (vids[i] >= m.vertexNumber()) {
80 throw vcl::MalformedFileException(
81 "Bad vertex index for face " + std::to_string(i));
82 }
83 v = &m.vertex(vids[i]);
84 i++;
85 }
86 }
87 else { // split needed
88 addTriangleFacesFromPolygon(m, f, vids);
89 }
90}

92template<FaceMeshConcept MeshType, FaceConcept FaceType, typename Scalar>
93void setFaceWedgeTexCoords(
94 FaceType& f,
95 MeshType& m,
96 const std::vector<uint>& vids,
97 const std::vector<std::pair<Scalar, Scalar>>& wedges)
98{
99 bool splitFace = false;
100 if (FaceType::VERTEX_NUMBER > 0 && FaceType::VERTEX_NUMBER != wedges.size())
101 splitFace = true;
102 if (!splitFace) {
103 for (uint i = 0; i < wedges.size(); ++i) {
104 f.wedgeTexCoord(i).u() = wedges[i].first;
105 f.wedgeTexCoord(i).v() = wedges[i].second;
106 }
107 }
108 else { // the face has been splitted
109 // for each triangle generated by the polygon
110 for (uint ff = m.index(f); ff < m.faceNumber(); ++ff) {
111 // for each vertex/wedge of the triangle
112 for (uint i = 0; i < m.face(ff).vertexNumber(); ++i) {
113 // get the position of this vertex in the original polygon
114 uint vid = m.index(m.face(ff).vertex(i));
115 auto it = std::find(vids.begin(), vids.end(), vid);
116 if (it == vids.end()) { // should never happen
117 // if this happens, it means that this triangle was not
118 // generated by the polygon
119 throw vcl::MalformedFileException(
120 "Bad vertex index for face " + std::to_string(ff));
121 }
122 // p is the position of the vertex in the polygon
123 uint p = it - vids.begin();
124 m.face(ff).wedgeTexCoord(i).u() = wedges[p].first;
125 m.face(ff).wedgeTexCoord(i).v() = wedges[p].second;
126 }
127 }
128 }

131template<FaceMeshConcept MeshType, FaceConcept FaceType, typename Stream>
132void loadFaceProperty(Stream& file, MeshType& mesh, FaceType& f, ply::Property p)
133{
134 bool hasBeenRead = false;
135 std::vector<uint> vids; // contains the vertex ids of the actual face
136 if (p.name == ply::vertex_indices) { // loading vertex indices
137 uint fSize = io::internal::readProperty<uint>(file, p.listSizeType);
138 vids.resize(fSize);
139 for (uint i = 0; i < fSize; ++i) {
140 vids[i] = io::internal::readProperty<size_t>(file, p.type);
141 }
142 hasBeenRead = true;
143 // will manage the case of loading a polygon in a triangle mesh
144 setFaceIndices(f, mesh, vids);
145 }
146 if (p.name == ply::texcoord) { // loading wedge texcoords
147 if constexpr (vcl::HasPerFaceWedgeTexCoords<MeshType>) {
148 if (vcl::isPerFaceWedgeTexCoordsAvailable(mesh)) {
149 using Scalar = FaceType::WedgeTexCoordType::ScalarType;
150 uint uvSize =
151 io::internal::readProperty<uint>(file, p.listSizeType);
152 uint fSize = uvSize / 2;
153 std::vector<std::pair<Scalar, Scalar>> wedges(fSize);
154 for (uint i = 0; i < fSize; ++i) {
155 Scalar u = io::internal::readProperty<Scalar>(file, p.type);
156 Scalar v = io::internal::readProperty<Scalar>(file, p.type);
157 wedges[i].first = u;
158 wedges[i].second = v;
159 }
160 hasBeenRead = true;
161 setFaceWedgeTexCoords(f, mesh, vids, wedges);
162 }
163 }
164 }
165 // loading texture id associated to ALL the wedges
166 if (p.name == ply::texnumber) {
167 if constexpr (vcl::HasPerFaceWedgeTexCoords<MeshType>) {
168 if (vcl::isPerFaceWedgeTexCoordsAvailable(mesh)) {
169 uint n = io::internal::readProperty<uint>(file, p.type);
170 hasBeenRead = true;
171 // in case the loaded polygon has been triangulated in the last
172 // n triangles of mesh
173 for (uint ff = mesh.index(f); ff < mesh.faceNumber(); ++ff) {
174 mesh.face(ff).textureIndex() = n;
175 }
176 }
177 }
178 }
179 // loading one of the normal components
180 if (p.name >= ply::nx && p.name <= ply::nz) {
181 if constexpr (vcl::HasPerFaceNormal<MeshType>) {
182 if (vcl::isPerFaceNormalAvailable(mesh)) {
183 using Scalar = FaceType::NormalType::ScalarType;
184 int a = p.name - ply::nx;
185 Scalar n = io::internal::readProperty<Scalar>(file, p.type);
186 hasBeenRead = true;
187 // in case the loaded polygon has been triangulated in the last
188 // n triangles of mesh
189 for (uint ff = mesh.index(f); ff < mesh.faceNumber(); ++ff) {
190 mesh.face(ff).normal()[a] = n;
191 }
192 }
193 }
194 }
195 // loading one of the color components
196 if (p.name >= ply::red && p.name <= ply::alpha) {
197 if constexpr (vcl::HasPerFaceColor<MeshType>) {
198 if (vcl::isPerFaceColorAvailable(mesh)) {
199 int a = p.name - ply::red;
200 unsigned char c =
201 io::internal::readProperty<unsigned char>(file, p.type);
202 hasBeenRead = true;
203 // in case the loaded polygon has been triangulated in the last
204 // n triangles of mesh
205 for (uint ff = mesh.index(f); ff < mesh.faceNumber(); ++ff) {
206 mesh.face(ff).color()[a] = c;
207 }
208 }
209 }
210 }
211 if (p.name == ply::quality) { // loading the quality component
212 if constexpr (vcl::HasPerFaceQuality<MeshType>) {
213 using QualityType = FaceType::QualityType;
214 if (vcl::isPerFaceQualityAvailable(mesh)) {
215 QualityType s =
216 io::internal::readProperty<QualityType>(file, p.type);
217 hasBeenRead = true;
218 // in case the loaded polygon has been triangulated in the last
219 // n triangles of mesh
220 for (uint ff = mesh.index(f); ff < mesh.faceNumber(); ++ff) {
221 mesh.face(ff).quality() = s;
222 }
223 }
224 }
225 }
226 if (p.name == ply::unknown) {
227 if constexpr (vcl::HasPerFaceCustomComponents<MeshType>) {
228 if (mesh.hasPerFaceCustomComponent(p.unknownPropertyName)){
229 io::internal::readCustomComponent(
230 file, f, p.unknownPropertyName, p.type);
231 hasBeenRead = true;
232 }
233 }
234 }
235 // if nothing has been read, it means that there is some data we don't know
236 // we still need to read and discard what we read
237 if (!hasBeenRead) {
238 if (p.list) {
239 uint s = io::internal::readProperty<int>(file, p.listSizeType);
240 for (uint i = 0; i < s; ++i)
241 io::internal::readProperty<int>(file, p.type);
242 }
243 else {
244 io::internal::readProperty<int>(file, p.type);
245 }
246 }
247}

249template<FaceConcept FaceType, MeshConcept MeshType>
250void loadFaceTxt(
251 std::ifstream& file,
252 FaceType& f,
253 MeshType& mesh,
254 const std::list<ply::Property>& faceProperties)
255{
256 vcl::Tokenizer spaceTokenizer =
257 vcl::io::internal::nextNonEmptyTokenizedLine(file);
258 vcl::Tokenizer::iterator token = spaceTokenizer.begin();
259 for (const ply::Property& p : faceProperties) {
260 if (token == spaceTokenizer.end()) {
261 throw vcl::MalformedFileException("Unexpected end of line.");
262 }
263 loadFaceProperty(token, mesh, f, p);
264 }
265}

281template<FaceMeshConcept MeshType>
282void saveFaces(
283 std::ofstream& file,
284 const PlyHeader& header,
285 const MeshType& mesh)
286{
287 using FaceType = MeshType::FaceType;
288
289 bool bin = header.format() == ply::BINARY;
290
291 // indices of vertices that do not consider deleted vertices
292 std::vector<uint> vIndices = mesh.vertexCompactIndices();
293
294 for (const FaceType& f : mesh.faces()) {
295 for (const ply::Property& p : header.faceProperties()) {
296 bool hasBeenWritten = false;
297 if (p.name == ply::vertex_indices) {
298 internal::saveFaceIndices(file, p, mesh, vIndices, f, bin);
299 hasBeenWritten = true;
300 }
301 if (p.name >= ply::nx && p.name <= ply::nz) {
302 if constexpr (vcl::HasPerFaceNormal<MeshType>) {
303 io::internal::writeProperty(
304 file, f.normal()[p.name - ply::nx], p.type, bin);
305 hasBeenWritten = true;
306 }
307 }
308 if (p.name >= ply::red && p.name <= ply::alpha) {
309 if constexpr (vcl::HasPerFaceColor<MeshType>) {
310 io::internal::writeProperty(
311 file, f.color()[p.name - ply::red], p.type, bin);
312 hasBeenWritten = true;
313 }
314 }
315 if (p.name == ply::quality) {
316 if constexpr (vcl::HasPerFaceQuality<MeshType>) {
317 io::internal::writeProperty(file, f.quality(), p.type, bin);
318 hasBeenWritten = true;
319 }
320 }
321 if (p.name == ply::texcoord) {
322 if constexpr (vcl::HasPerFaceWedgeTexCoords<MeshType>) {
323 io::internal::writeProperty(
324 file, f.vertexNumber() * 2, p.listSizeType, bin);
325 for (const auto& tc : f.wedgeTexCoords()) {
326 io::internal::writeProperty(file, tc.u(), p.type, bin);
327 io::internal::writeProperty(file, tc.v(), p.type, bin);
328 }
329 hasBeenWritten = true;
330 }
331 }
332 if (p.name == ply::texnumber) {
333 if constexpr (vcl::HasPerFaceWedgeTexCoords<MeshType>) {
334 io::internal::writeProperty(
335 file, f.textureIndex(), p.type, bin);
336 hasBeenWritten = true;
337 }
338 }
339 if (p.name == ply::unknown) {
340 if constexpr (vcl::HasPerFaceCustomComponents<MeshType>) {
341 if (mesh.hasPerFaceCustomComponent(p.unknownPropertyName)) {
342 io::internal::writeCustomComponent(
343 file, f, p.unknownPropertyName, p.type, bin);
344 hasBeenWritten = true;
345 }
346 }
347 }
348 if (!hasBeenWritten) {
349 // be sure to write something if the header declares some
350 // property that is not in the mesh
351 io::internal::writeProperty(file, 0, p.type, bin);
352 }
353 }
354 if (!bin)
355 file << std::endl;
356 }
357}

359template<FaceMeshConcept MeshType>
360void loadFaces(std::ifstream& file, const PlyHeader& header, MeshType& mesh)
361{
362 using FaceType = MeshType::FaceType;
363 mesh.reserveFaces(header.numberFaces());
364 for (uint fid = 0; fid < header.numberFaces(); ++fid) {
365 uint ffid = mesh.addFace();
366 FaceType& f = mesh.face(ffid);
367 if (header.format() == ply::ASCII) {
368 internal::loadFaceTxt(file, f, mesh, header.faceProperties());
369 }
370 else {
371 internal::loadFaceBin(file, f, mesh, header.faceProperties());
372 }
373 }
