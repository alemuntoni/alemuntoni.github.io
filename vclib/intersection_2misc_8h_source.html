39inline void findMinMax(
40 ScalarType x0,
41 ScalarType x1,
42 ScalarType x2,
43 ScalarType &min,
44 ScalarType &max) {
45 min = max = x0;
46 if (x1 < min)
47 min = x1;
48 if (x1 > max)
49 max = x1;
50 if (x2 < min)
51 min = x2;
52 if (x2 > max)
53 max = x2;
54}

57template<typename ScalarType, PointConcept PointType>
58inline bool axisTestX01(
59 ScalarType a,
60 ScalarType b,
61 ScalarType fa,
62 ScalarType fb,
63 const PointType& v0,
64 const PointType& v2,
65 const PointType& bHalfSixe)
66{
67 ScalarType p0 = a * v0.y() - b * v0.z();
68 ScalarType p2 = a * v2.y() - b * v2.z();
69 ScalarType min, max;
70 if (p0 < p2) {
71 min = p0;
72 max = p2;
73 } else {
74 min = p2;
75 max = p0;
76 }
77 ScalarType rad = fa * bHalfSixe.y() + fb * bHalfSixe.z();
78 if (min > rad || max < -rad)
79 return false;
80 return true;
81}

231template<PlaneConcept PlaneType, Box3Concept BoxType>
232bool planeBoxIntersect(const PlaneType& p, const BoxType& box)
233{
234 using PointType = BoxType::PointType;
235 using ScalarType = PointType::ScalarType;
236
237 // Convert AABB to center-extents representation
238 PointType c = (box.max() + box.min()) * 0.5f; // Compute AABB center
239 PointType e = box.max() - c; // Compute positive extents
240
241 PointType n = p.direction();
242 // Compute the projection interval radius of b onto L(t) = b.c + t * p.n
243 ScalarType r =
244 e[0] * std::abs(n[0]) + e[1] * std::abs(n[1]) + e[2] * std::abs(n[2]);
245
246 // Compute distance of box center from plane
247 ScalarType s = n.dot(c) - p.offset();
248
249 // Intersection occurs when distance s falls within [-r,+r] interval
250 return std::abs(s) <= r;
251}

266template<PlaneConcept PlaneType, Segment3Concept SegmentType>
267bool planeSegmentIntersect(
268 const PlaneType& p,
269 const SegmentType& s,
270 typename SegmentType::PointType& intersection)
271{
272 using ScalarType = SegmentType::ScalarType;
273
274 // Compute the projection of the segment endpoints onto the plane.
275 ScalarType p1_proj = s.p1() * p.direction() - p.offset();
276 ScalarType p0_proj = s.p0() * p.direction() - p.offset();
277
278 // If both endpoints are on the same side of the plane, there is no
279 // intersection.
280 if ( (p1_proj>0)-(p0_proj<0))
281 return false;
282
283 // If both endpoints have the same projection onto the plane, there is no
284 // intersection.
285 if(p0_proj == p1_proj)
286 return false;
287
288 // check that we perform the computation in a way that is independent with
289 // v0 v1 swaps
290 if(p0_proj < p1_proj)
291 intersection = s.p0() + (s.p1() - s.p0()) *
292 std::abs(p0_proj / (p1_proj - p0_proj));
293 if(p0_proj > p1_proj)
294 intersection = s.p1() + (s.p0() - s.p1()) *
295 std::abs(p1_proj / (p0_proj - p1_proj));
296
297 return true;
298}

354template<ConstTriangle2Concept TriangleType, Point2Concept PointType>
355bool trianglePointIntersect(const TriangleType& tr, const PointType& p)
356{
357 using TP = TriangleType::PointType;
358 using ScalarType = TP::ScalarType;
359
360 const TP& p0 = tr.point(0);
361 const TP& p1 = tr.point(1);
362 const TP& p2 = tr.point(2);
363
364 ScalarType A = tr.area();
365 ScalarType sign = A < 0 ? -1 : 1;
366
367 ScalarType s = (p0.y() * p2.x() - p0.x() * p2.y() +
368 (p2.y() - p0.y()) * p.x() + (p0.x() - p2.x()) * p.y()) *
369 sign;
370 ScalarType t = (p0.x() * p1.y() - p0.y() * p1.x() +
371 (p0.y() - p1.y()) * p.x() + (p1.x() - p0.x()) * p.y()) *
372 sign;
373
374 return s > 0 && t > 0 && (s + t) < 2 * A * sign;
375}

390template<ConstTriangle3Concept TriangleType, Point3Concept PointType>
391bool trianglePointIntersect(const TriangleType& t, const PointType& p)
392{
393 PointType v1 = t.point(1) - t.point(0);
394 PointType v2 = t.point(2) - t.point(0);
395 PointType v3 = p - t.point(0);
396
397 return v1.dot(v2.cross(v3)) > 0;
398}

404template<ConstTriangle3Concept TriangleType, Box3Concept BoxType>
405bool triangleBoxIntersect(
406 const TriangleType& t,
407 const BoxType& box)
408{
409 using PointType = TriangleType::PointType;
410 using ScalarType = PointType::ScalarType;
411
412 PointType boxcenter = box.center();
413 PointType bHalfSixe = box.size() / 2;
414
415 /* use separating axis theorem to test overlap between triangle and box
416 * need to test for overlap in these directions:
417 * 1) the {x,y,z}-directions (actually, since we use the AABB of the
418 * triangle we do not even need to test these)
419 * 2) normal of the triangle
420 * 3) crossproduct(edge from tri, {x,y,z}-directin)
421 * this gives 3x3=9 more tests
422 */
423 ScalarType min, max;
424 PointType normal;
425
426 /* This is the fastest branch on Sun */
427 /* move everything so that the boxcenter is in (0,0,0) */
428 PointType v0 = t.point(0) - boxcenter;
429 PointType v1 = t.point(1) - boxcenter;
430 PointType v2 = t.point(2) - boxcenter;
431
432 /* compute triangle edges */
433 PointType e0 = v1 - v0;
434 PointType e1 = v2 - v1;
435 PointType e2 = v0 - v2;
436
437 /* Bullet 3: */
438 /* test the 9 tests first (this was faster) */
439 ScalarType fex = std::abs(e0.x());
440 ScalarType fey = std::abs(e0.y());
441 ScalarType fez = std::abs(e0.z());
442
443 if (!internal::axisTestX01(e0.z(), e0.y(), fez, fey, v0, v2, bHalfSixe))
444 return false;
445 if (!internal::axisTestY02(e0.z(), e0.x(), fez, fex, v0, v2, bHalfSixe))
446 return false;
447 if (!internal::axisTestZ12(e0.y(), e0.x(), fey, fex, v1, v2, bHalfSixe))
448 return false;
449
450 fex = std::abs(e1.x());
451 fey = std::abs(e1.y());
452 fez = std::abs(e1.z());
453
454 if (!internal::axisTestX01(e1.z(), e1.y(), fez, fey, v0, v2, bHalfSixe))
455 return false;
456 if (!internal::axisTestY02(e1.z(), e1.x(), fez, fex, v0, v2, bHalfSixe))
457 return false;
458 if (!internal::axisTestZ0(e1.y(), e1.x(), fey, fex, v0, v1, bHalfSixe))
459 return false;
460
461 fex = std::abs(e2.x());
462 fey = std::abs(e2.y());
463 fez = std::abs(e2.z());
464 if (!internal::axisTestX2(e2.z(), e2.y(), fez, fey, v0, v1, bHalfSixe))
465 return false;
466 if (!internal::axisTestY1(e2.z(), e2.x(), fez, fex, v0, v1, bHalfSixe))
467 return false;
468 if (!internal::axisTestZ12(e2.y(), e2.x(), fey, fex, v1, v2, bHalfSixe))
469 return false;
470
471 /* Bullet 1:
472 * first test overlap in the {x,y,z}-directions
473 * find min, max of the triangle each direction, and test for overlap in
474 * that direction -- this is equivalent to testing a minimal AABB around
475 * the triangle against the AABB
476 */
477
478 /* test in X-direction */
479 internal::findMinMax(v0.x(), v1.x(), v2.x(), min, max);
480 if (min > bHalfSixe.x() || max < -bHalfSixe.x())
481 return false;
482
483 /* test in Y-direction */
484 internal::findMinMax(v0.y(), v1.y(), v2.y(), min, max);
485 if (min > bHalfSixe.y() || max < -bHalfSixe.y())
486 return false;
487
488 /* test in Z-direction */
489 internal::findMinMax(v0.z(), v1.z(), v2.z(), min, max);
490 if (min > bHalfSixe.z() || max < -bHalfSixe.z())
491 return false;
492
493 /* Bullet 2:
494 * test if the box intersects the plane of the triangle
495 * compute plane equation of triangle: normal*x+d=0
496 */
497 normal = e0.cross(e1);
498 vcl::Plane<ScalarType> plane(t.point(0), t.point(1), t.point(2));
499 if (!planeBoxIntersect(plane, box))
500 return false;
501
502 return true; /* box and triangle overlaps */
503}

516template<
517 ConstTriangle3Concept TriangleType,
518 SphereConcept SphereType,
519 Point3Concept PointType,
520 typename ScalarType>
521bool triangleSphereItersect(
522 const TriangleType& t,
523 const SphereType& sphere,
524 PointType& witness,
525 std::pair<ScalarType, ScalarType>& res)
526{
527 bool penetrationDetected = false;
528
529 ScalarType radius = sphere.radius();
530 PointType center = sphere.center();
531 PointType p0 = t.point0() - center;
532 PointType p1 = t.point1() - center;
533 PointType p2 = t.point2() - center;
534
535 PointType p10 = p1 - p0;
536 PointType p21 = p2 - p1;
537 PointType p20 = p2 - p0;
538
539 ScalarType delta0_p01 = p10.dot(p1);
540 ScalarType delta1_p01 = -p10.dot(p0);
541 ScalarType delta0_p02 = p20.dot(p2);
542 ScalarType delta2_p02 = -p20.dot(p0);
543 ScalarType delta1_p12 = p21.dot(p2);
544 ScalarType delta2_p12 = -p21.dot(p1);
545
546 // the closest point can be one of the vertices of the triangle
547 if (delta1_p01 <= ScalarType(0.0) && delta2_p02 <= ScalarType(0.0))
548 witness = p0;
549 else if (delta0_p01 <= ScalarType(0.0) && delta2_p12 <= ScalarType(0.0))
550 witness = p1;
551 else if (delta0_p02 <= ScalarType(0.0) && delta1_p12 <= ScalarType(0.0))
552 witness = p2;
553 else {
554 ScalarType temp = p10.dot(p2);
555 ScalarType delta0_p012 = delta0_p01 * delta1_p12 + delta2_p12 * temp;
556 ScalarType delta1_p012 = delta1_p01 * delta0_p02 - delta2_p02 * temp;
557 ScalarType delta2_p012 =
558 delta2_p02 * delta0_p01 - delta1_p01 * (p20.dot(p1));
559
560 // otherwise, can be a point lying on same edge of the triangle
561 if (delta0_p012 <= ScalarType(0.0)) {
562 ScalarType denominator = delta1_p12 + delta2_p12;
563 ScalarType mu1 = delta1_p12 / denominator;
564 ScalarType mu2 = delta2_p12 / denominator;
565
566 witness = (p1 * mu1 + p2 * mu2);
567 }
568 else if (delta1_p012 <= ScalarType(0.0)) {
569 ScalarType denominator = delta0_p02 + delta2_p02;
570 ScalarType mu0 = delta0_p02 / denominator;
571 ScalarType mu2 = delta2_p02 / denominator;
572
573 witness = (p0 * mu0 + p2 * mu2);
574 }
575 else if (delta2_p012 <= ScalarType(0.0)) {
576 ScalarType denominator = delta0_p01 + delta1_p01;
577 ScalarType mu0 = delta0_p01 / denominator;
578 ScalarType mu1 = delta1_p01 / denominator;
579
580 witness = (p0 * mu0 + p1 * mu1);
581 }
582 else {
583 // or else can be an point internal to the triangle
584 ScalarType denominator = delta0_p012 + delta1_p012 + delta2_p012;
585 ScalarType lambda0 = delta0_p012 / denominator;
586 ScalarType lambda1 = delta1_p012 / denominator;
587 ScalarType lambda2 = delta2_p012 / denominator;
588
589 witness = p0 * lambda0 + p1 * lambda1 + p2 * lambda2;
590 }
591 }
592
593 ScalarType witness_norm = witness.norm();
594
595 res.first = std::max<ScalarType>(witness_norm - radius, ScalarType(0.0));
596 res.second = std::max<ScalarType>(radius - witness_norm, ScalarType(0.0));
597
598 penetrationDetected = (witness.squaredNorm() <= (radius * radius));
599 witness += center;
600 return penetrationDetected;
601}
