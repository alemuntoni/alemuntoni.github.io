54template<FaceConcept FaceType>
55bool isFaceManifoldOnEdge(const FaceType& f, uint edge)
56 requires comp::HasAdjacentFaces<FaceType>
57{
58 // Check if the AdjacentFaces component is available for the given face.
59 if (!comp::isAdjacentFacesAvailableOn(f)) {
60 throw vcl::MissingComponentException(
61 "Face has no Adjacent Faces component.");
62 }
63
64 // Check if the edge is a boundary edge.
65 if (f.adjFace(edge) == nullptr) {
66 return true;
67 }
68 else { // Check if the edge is shared by exactly two faces.
69 return f.adjFace(edge)->indexOfAdjFace(&f) != UINT_NULL;
70 }
71}

135template<FaceConcept FaceType>
136bool checkFlipEdge(const FaceType& f, uint edge)
137 requires comp::HasAdjacentFaces<FaceType>
138{
139 if (!comp::isAdjacentFacesAvailableOn(f)) {
140 throw vcl::MissingComponentException(
141 "Face has no Adjacent Faces component.");
142 }
143
144 using VertexType = FaceType::VertexType;
145
146 if (f.vertexNumber() > 3)
147 return false;
148
149 if (vcl::isFaceEdgeOnBorder(f, edge))
150 return false;
151
152 const VertexType* v0 = f.vertex(edge);
153 const VertexType* v1 = f.vertexMod(edge + 1);
154
155 const FaceType* of = f.adjFace(edge);
156 uint oe = of->indexOfAdjFace(&f);
157 assert(oe != UINT_NULL);
158
159 // check if the vertices of the edge are the same
160 // e.g. the mesh has to be well oriented
161 if (of->vertex(oe) != v1 || of->vertexMod(oe + 1) != v0)
162 return false;
163
164 // check if the flipped edge is already present in the mesh
165 // f_v2 and of_v2 are the vertices of the new edge
166 const VertexType* f_v2 = f.vertexMod(edge + 2);
167 const VertexType* of_v2 = of->vertexMod(oe + 2);
168
169 MeshPos<FaceType> pos(&f, f_v2);
170 MeshPos<FaceType> startPos = pos;
171 // loop in the one ring of f_v2
172 do {
173 pos.nextEdgeAdjacentToV();
174 if (pos.adjVertex() == of_v2)
175 return false;
176 } while (pos != startPos);
177
178 return true;
179}

270template<FaceConcept FaceType>
271auto faceDihedralAngleOnEdge(const FaceType& f, uint e)
272 requires comp::HasAdjacentFaces<FaceType>
273{
274 if (!comp::isAdjacentFacesAvailableOn(f)) {
275 throw vcl::MissingComponentException(
276 "Face has no Adjacent Faces component.");
277 }
278
279 /*
280 * v0 ___________ vf1
281 * |\ |
282 * | e\ f1 |
283 * | \e1 |
284 * |f \ |
285 * | \ |
286 * |__________\|
287 * vf0 v1
288 */
289
290 assert(f.adjFace(e) != nullptr);
291 const FaceType& f1 = *(f.adjFace(e));
292
293 uint e1 = f1.indexOfAdjFace(&f);
294 assert(e1 != UINT_NULL);
295
296 const auto& vf0 = *(f.vertexMod((int) e - 1));
297 const auto& vf1 = *(f1.vertexMod(e1 - 1));
298
299 auto n0 = faceNormal(f);
300 auto n1 = faceNormal(f1);
301
302 auto off0 = n0 * vf0.coord();
303 auto off1 = n1 * vf1.coord();
304
305 auto dist01 = off0 - n0 * vf1.coord();
306 auto dist10 = off1 - n1 * vf0.coord();
307
308 auto sign = std::abs(dist01) > std::abs(dist10) ? dist01 : dist10;
309
310 auto angleRad = n0.angle(n1);
311 return sign > 0 ? angleRad : -angleRad;
312}

339template<FaceConcept FaceType>
340void detachAdjacentFacesOnEdge(FaceType& f, uint edge)
341 requires comp::HasAdjacentFaces<FaceType>
342{
343 if (!comp::isAdjacentFacesAvailableOn(f)) {
344 throw vcl::MissingComponentException(
345 "Face has no Adjacent Faces component.");
346 }
347
348 FaceType* nextFace = f.adjFace(edge);
349
350 // if nextFace == nullptr there is nothing to do
351 // the face is already detached on the edge
352 if (nextFace != nullptr) {
353 FaceType* prevFace;
354 ConstEdgeAdjFaceIterator<FaceType> begin(f, edge), end;
355 for (auto it = begin; it != end; ++it) {
356 prevFace = *it;
357 }
358
359 if (nextFace == prevFace) { // manifold case
360 uint en = nextFace->indexOfAdjFace(&f);
361 assert(en != UINT_NULL);
362 nextFace->setAdjFace(en, nullptr);
363 }
364 else { // non manifold case
365 // the prev face does not have to point to f anymore, but to
366 // nextFace
367 uint pn = prevFace->indexOfAdjFace(&f);
368 assert(pn != UINT_NULL);
369 prevFace->setAdjFace(pn, nextFace);
370 }
371 f.setAdjFace(edge, nullptr);
372 }
373}

396template<FaceConcept FaceType>
397void detachFace(FaceType& f) requires comp::HasAdjacentFaces<FaceType>
398{
399 if (!comp::isAdjacentFacesAvailableOn(f)) {
400 throw vcl::MissingComponentException(
401 "Face has no Adjacent Faces component.");
402 }
403
404 using VertexType = FaceType::VertexType;
405
406 for (uint e = 0; e < f.vertexNumber(); ++e) {
407 detachAdjacentFacesOnEdge(f, e);
408
409 // if the vertices have adjacent faces
410 if constexpr (comp::HasAdjacentFaces<VertexType>) {
411 if (comp::isAdjacentFacesAvailableOn(f.vertex(e))) {
412 VertexType* v = f.vertex(e);
413 uint vpos = v->indexOfAdjFace(&f);
414 if (vpos != UINT_NULL) { // may happen if vertex adj faces are
415 // not initialized / updated
416 v->eraseAdjFace(vpos); // the vertex v has not anymore the
417 // adjacent face f
418 }
419 }
420 }
421 }
