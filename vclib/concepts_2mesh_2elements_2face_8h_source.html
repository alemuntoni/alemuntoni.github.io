131template<typename T>
132concept FaceConcept =
133 T::ELEMENT_TYPE == FACE &&
134 (face::IsDerivedFromFace<T>::value || face::IsAFace<T>::value) &&
135 face::HasBitFlags<T> &&
136 face::HasVertexReferences<T> &&
137 (T::VERTEX_NUMBER < 0 || T::VERTEX_NUMBER >= 3) &&
138 (!face::HasTriangleBitFlags<T> || T::VERTEX_NUMBER == 3) &&
139 comp::SanityCheckAdjacentEdges<T> &&
140 comp::SanityCheckAdjacentFaces<T> &&
141 comp::SanityCheckWedgeColors<T> &&
142 comp::SanityCheckWedgeTexCoords<T>;
143
144template<typename T>
145concept TriangleFaceConcept =
146 T::VERTEX_NUMBER == 3 && FaceConcept<T>;
147
158template<typename T>
159concept PolygonFaceConcept =
160 T::VERTEX_NUMBER < 0 && FaceConcept<T>;
161
162template<typename T>
163concept NonDcelPolygonFaceConcept =
164 T::VERTEX_NUMBER < 0 && FaceConcept<T> && !face::HasHalfEdgeReference<T>;
