43std::string::const_iterator
44findCaseInsensitive(const std::string& input, const std::string& substr)
45{
46 // link: https://stackoverflow.com/a/19839371/5851101
47
48 std::string::const_iterator it = std::search(
49 input.begin(),
50 input.end(),
51 substr.begin(),
52 substr.end(),
53 [](char ch1, char ch2) {
54 return std::toupper(ch1) == std::toupper(ch2);
55 });
56 return it;
57}
