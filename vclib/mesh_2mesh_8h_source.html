397 template<typename OtherMeshType>
398 void importFrom(const OtherMeshType& m)
399 {
400 // This function will first:
401 // Call, for each Container and Component of the mesh, its importFrom
402 // function. In case of containers, it first creates the same number of
403 // elements in the container, and then calls the importFrom function for
404 // each new element. Pointers are not managed here, since they need
405 // additional parameters to be imported
406
407 (Args::importFrom(m), ...);
408
409 // Set to all element containers their parent mesh (this)
410 updateAllParentMeshPointers();
411
412 // after importing ordinary components, we need to convert the pointers
413 // between containers. each container can import more than one pointer
414 // type, e.g.:
415 // - VertexContainer could import vertex pointers (adjacent vertices),
416 // face pointers (adjacent faces), and so on;
417 // - FaceContainer will always import vertex pointers, but could also
418 // import face pointers (adjacent faces), edge pointers (adjacent
419 // edges)...
420 // for each container of this Mesh, we'll call the importPointers
421 // passing the container (Args) as template parameter. This parameter
422 // will be used to call all the possible import functions available
423 // (vertices, faces, edges, half edges)
424
425 (importPointers<Args>(m), ...);
426
427 if constexpr (mesh::HasFaceContainer<Mesh<Args...>>) {
428 // Now I need to manage imports between different types of meshes
429 // (same type of meshes are already managed from importFrom and
430 // importPointersFrom member functions).
431 //
432 // Generally speaking, Polygon meshes can import from any other type
433 // of mesh. We need to take care when this mesh has static vertex
434 // pointers number in the face container (VERTEX_NUMBER >= 3).
435 //
436 // The follwing case don't need to be managed:
437 // - import polygon mesh from triangle mesh
438 //
439 // I can manage the following cases:
440 // - import triangle mesh from polygon mesh: need triangulation
441 //
442 // I cannot manage the follwing cases:
443 // - import static non-triangle mesh from polygon mesh or from a
444 // mesh with different VERTEX_NUMBER
445
446 // in case of import from poly to triangle mesh, I need to manage
447 // triangulation of polygons and create additional triangle faces
448 // for each of the imported polygons. This function statically
449 // asserts that the import can be done.
450 using FaceContainer = typename Mesh<Args...>::FaceContainer;
451 FaceContainer::manageImportTriFromPoly(m);
452 }
453 }

459 void swap(Mesh& m2)
460 {
461 Mesh<Args...>& m1 = *this;
462
463 constexpr uint N_CONTAINERS =
464 NumberOfTypes<typename Mesh<Args...>::Containers>::value;
465 static_assert(N_CONTAINERS != 0);
466
467 // container bases of each container for m1 and m2
468 // we save the bases of the containers before swap
469 std::array<const void*, N_CONTAINERS> m1Bases =
470 Mesh<Args...>::getContainerBases(m1);
471 std::array<const void*, N_CONTAINERS> m2Bases =
472 Mesh<Args...>::getContainerBases(m2);
473
474 // actual swap of all the containers and the components of the mesh
475 // using pack expansion: swap will be called for each of the containers
476 // (or components!) that compose the Mesh
477 using std::swap;
478 (swap((Args&) m1, (Args&) m2), ...);
479
480 // Set to all elements their parent mesh
481 m1.updateAllParentMeshPointers();
482 m2.updateAllParentMeshPointers();
483
484 // update all the pointers to m1 and m2: old base of m1 is now "old
485 // base" of m2, and viceversa
486 (updatePointersOfContainerType<Args>(m1, m2Bases), ...);
487 (updatePointersOfContainerType<Args>(m2, m1Bases), ...);
488 }

1329 template<typename... A>
1330 static auto getContainerBases(const Mesh<A...>& m)
1331 {
1332 using Containers = Mesh<A...>::Containers;
1333
1334 // the number of containers in Mesh<A...>
1335 constexpr uint N_CONTAINERS = NumberOfTypes<Containers>::value;
1336 // each element of this array will contain the base pointer of the
1337 // vector of elements contained in each container of Mesh<A...>
1338 std::array<const void*, N_CONTAINERS> bases;
1339
1340 // for each container/component of Mesh<A...>, we set call the function
1341 // that sets the base of the container in its index
1342 (setContainerBase<IndexInTypes<A, Containers>::value, A>(m, bases),
1343 ...);
1344
1345 return bases;
1346 }

1364 template<typename Cont, typename Array, typename... A>
1365 static void updatePointersOfContainerType(
1366 Mesh<A...>& m,
1367 const Array& bases)
1368 {
1369 // since this function is called using pack expansion, it means that
1370 // Cont could be a mesh component and not a cointainer. We check if Cont
1371 // is a container
1372 if constexpr (mesh::ElementContainerConcept<Cont>) {
1373 // The element type contained in the container
1374 // We need it to get back the actual type of the element from the
1375 // old bases
1376 using ElType = Cont::ElementType;
1377
1378 using Containers = Mesh<A...>::Containers;
1379 constexpr uint I = IndexInTypes<Cont, Containers>::value;
1380 static_assert(I >= 0 && I != UINT_NULL);
1381
1382 // for each Container A in m, we update the pointers of ElType.
1383 // old base is contained in the array bases, the new base is the
1384 // base of the container
1385 (m.template updatePointers<A>(
1386 (const ElType*) bases[I], m.Cont::vec.data()),
1387 ...);
1388 }
