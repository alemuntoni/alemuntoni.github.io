125 void startNewTask(double fromPerc, double toPerc, const std::string& action)
126 override final
127 {
128 printLine(action, START);
129
130 assert(fromPerc >= 0);
131 assert(toPerc <= 100);
132 std::pair<double, double> newP;
133 newP.first = mIntervals.top().first +
134 (mIntervals.top().second - mIntervals.top().first) *
135 (fromPerc / 100);
136 newP.second =
137 (mIntervals.top().second - mIntervals.top().first) * (toPerc / 100);
138 mGlobalPercProgress = newP.first;
139 mIntervals.push(newP);
140 updateStep();
141 }
142
143 void endTask(const std::string& action) override final
144 {
145 mGlobalPercProgress = mIntervals.top().second;
146 if (mIntervals.size() > 1) {
147 mIntervals.pop();
148 updateStep();
149
150 printLine(action, END);
151 }
152 }
153
154 double percentage() const override final
155 {
156 double k = std::pow(10, mPercPrecision);
157 uint c = mGlobalPercProgress * k;
158 return c / k;
159 }

188 void startProgress(
189 const std::string& msg,
190 uint progressSize,
191 uint percPrintProgress = 10,
192 uint startPerc = 0,
193 uint endPerc = 100) override final
194 {
195 assert(percPrintProgress > 0);
196 assert((endPerc - startPerc) > 0);
197 mIsProgressActive = true;
198 mProgressMessage = msg;
199 mProgressSize = progressSize;
200 mProgressPerc = startPerc;
201 mProgressPercStep = percPrintProgress;
202 mProgressStep =
203 (progressSize + 1) / ((endPerc - startPerc) / percPrintProgress);
204 if (mProgressStep == 0)
205 mProgressStep = progressSize;
206 mLastProgress = 0;
207 }
208
209 void endProgress() override final
210 {
211 progress(mProgressSize);
212 mIsProgressActive = false;
213 }
214
215 void progress(uint n) override final
216 {
217 mMutex.lock();
218 assert(mIsProgressActive);
219 uint progress = n / mProgressStep;
220 if (mLastProgress < progress) {
221 mProgressPerc = progress * mProgressPercStep;
222 if (mPrintMsgDuringProgress)
223 log(mProgressPerc, PROGRESS, mProgressMessage);
224 else
225 setPercentage(mProgressPerc);
226 mLastProgress = progress;
227 }
228 mMutex.unlock();
229 }

250private:
251 void updateStep()
252 {
253 mStep = (mIntervals.top().second - mIntervals.top().first) / 100;
254 }
255
256 void printLine(const std::string& msg, uint lvl)
257 {
258 if (!mPrintPerc && msg.empty())
259 return;
260
261 // lvl could be also InternalLogLevel BEGIN or END
262 LogLevel l = PROGRESS;
263 if (lvl <= DEBUG) {
264 l = (LogLevel) lvl;
265 }
266 if (l > mPrintLevel)
267 return;
268 Stream* stream = levelStream(l);
269
270 if (stream) {
271 uint s = 0;
272
273 if (mPrintPerc) {
274 s = printPercentage(*stream);
275 }
276 s += printIndentation(*stream);
277 printMessage(*stream, msg, lvl, s);
278 printElapsedTime(*stream);
279 *stream << "\n";
280 }
281 flush(*stream);
282 }
283
284 uint printPercentage(Stream& o)
285 {
286 uint size = 3;
287 if (mPercPrecision > 0)
288 size += 1 + mPercPrecision;
289
290 o << "[";
291 alignRight(o);
292 setWidth(o, size);
293 o << percentage() << "%]";
294 return size + 3;
295 }
296
297 uint printIndentation(Stream& o)
298 {
299 uint s = 0;
300 if (mIndent) {
301 uint n = mIntervals.size() - 1;
302 for (uint i = 0; i < n; i++) {
303 o << " ";
304 s += 2;
305 }
306 }
307 return s;
308 }
309
310 void printMessage(Stream& o, const std::string& msg, uint lvl, uint n)
311 {
312 uint maxMsgSize = mLineWidth - n;
313 if (mPrintTimer)
314 maxMsgSize -= TIMER_MAX_CHAR_NUMBER;
315 switch (lvl) {
316 case ERROR:
317 maxMsgSize -= 8;
318 o << " ERROR: ";
319 break;
320 case WARNING:
321 maxMsgSize -= 10;
322 o << " WARNING: ";
323 break;
324 case PROGRESS:
325 maxMsgSize -= 1;
326 o << " ";
327 break;
328 case DEBUG:
329 maxMsgSize -= 9;
330 o << " (debug) ";
331 break;
332 case START:
333 maxMsgSize -= 7;
334 o << " Start ";
335 break;
336 case END:
337 maxMsgSize -= 5;
338 o << " End ";
339 break;
340 }
341 setWidth(o, maxMsgSize);
342 alignLeft(o);
343 o << msg.c_str();
344 }
345
346 void printElapsedTime(Stream& o)
347 {
348 if (mPrintTimer) {
349 o << "[";
350 setWidth(o, TIMER_MAX_CHAR_NUMBER - 3);
351 alignRight(o);
352 o << mTimer.delay() << "s]";
353 }
354 }
355};
