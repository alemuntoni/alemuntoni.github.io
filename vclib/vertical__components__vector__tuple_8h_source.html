38template<typename ...Comp>
39class VerticalComponentsVectorTuple
40{
41 using ComponentTypes = std::tuple<Comp...>;
42
43 static constexpr uint COMP_NUMBER =
44 std::tuple_size_v<std::tuple<std::vector<Comp>...>>;
45
46 std::tuple<std::vector<typename Comp::DataValueType>...> tuple;
47
48 std::array<bool, COMP_NUMBER> vecEnabled;
49 std::size_t siz = 0;
50
51public:

54 (setComponentEnabled<Comp, !vcl::comp::IsOptionalComponent<Comp>>(), ...);

163private:
164 template<typename C>
165 static constexpr uint indexOfType()
166 {
167 return vcl::IndexInTypes<C, Comp...>::value;
168 }
169
170 template<std::size_t N>
171 void vectorResize(std::size_t size)
172 {
173 if (vecEnabled[N]) {
174 std::get<N>(tuple).resize(size);
175 }
176 if constexpr (N != 0)
177 vectorResize<N-1>(size);
178 }
179
180 template<std::size_t N>
181 void vectorReserve(std::size_t size)
182 {
183 if (vecEnabled[N]) {
184 std::get<N>(tuple).reserve(size);
185 }
186 if constexpr (N != 0)
187 vectorReserve<N-1>(size);
188 }
189
190 template<std::size_t N>
191 void vectorCompact(const std::vector<uint>& newIndices)
192 {
193 if (vecEnabled[N]) {
194 vcl::compactVector(std::get<N>(tuple), newIndices);
195 }
196 if constexpr (N != 0)
197 vectorCompact<N-1>(newIndices);
198 }
199
200 template<typename C, bool E>
201 void setComponentEnabled()
202 {
203 if constexpr (E) {
204 enableComponent<C>();
205 }
206 else {
207 disableComponent<C>();
208 }
209 }
210
211 template<typename C, bool E>
212 void setComponentEnabledIfOptional()
213 {
214 if constexpr (vcl::comp::IsOptionalComponent<C>) {
215 setComponentEnabled<C, E>();
216 }
217 }
218};
