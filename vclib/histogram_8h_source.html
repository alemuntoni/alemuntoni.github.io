35template<typename ScalarType>
36class Histogram
37{
38 std::vector<ScalarType> H; // Counters for bins
39 std::vector<ScalarType> R; // Range for bins
40
41 ScalarType minRangeVal = 0; // Minimum range value
42 ScalarType maxRangeVal = 1; // Maximum range value
43
44 ScalarType minVal = std::numeric_limits<ScalarType>::max();
45 ScalarType maxVal = std::numeric_limits<ScalarType>::lowest();
46
47 // Number of valid bins stored between minRangeValue and maxRangeValue
48 uint nBins = 0;
49
50 ScalarType cnt = 0; // Number of accumulated values
51 ScalarType sum = 0; // Sum of values
52 ScalarType rms = 0; // Root mean square
53
54public:
55 Histogram() = default;
56
67 Histogram(
68 ScalarType minRangeValue,
69 ScalarType maxRangeValue,
70 uint nBins,
71 ScalarType gamma = 1.0) :
72 minRangeVal(minRangeValue),
73 maxRangeVal(maxRangeValue), nBins(nBins)
74 {
75 H.resize(nBins + 2, 0);
76
77 R.resize(nBins + 3);
78
79 R[0] = -std::numeric_limits<ScalarType>::max();
80 R[nBins + 2] = std::numeric_limits<ScalarType>::max();
81
82 double delta = (maxRangeVal - minRangeVal);
83 if (gamma == 1) {
84 for (uint i = 0; i <= nBins; ++i)
85 R[i + 1] = minRangeVal + delta * ScalarType(i) / nBins;
86 }
87 else {
88 for (uint i = 0; i <= nBins; ++i)
89 R[i + 1] = minRangeVal +
90 delta * std::pow(ScalarType(i) / nBins, gamma);
91 }
92 }
93
97 void clear()
98 {
99 H.clear();
100 R.clear();
101 cnt = 0;
102 sum = 0;
103 rms = 0;
104 nBins = 0;
105 minRangeVal = 0;
106 maxRangeVal = 1;
107 minVal = std::numeric_limits<ScalarType>::max();
108 maxVal = std::numeric_limits<ScalarType>::lowest();
109 }
110
117 void addValue(ScalarType value, ScalarType increment = 1.0)
118 {
119 uint pos = binIndex(value);
120 if (value < minVal)
121 minVal = value;
122 if (value > maxVal)
123 maxVal = value;
124 assert((pos >= 0) && (pos <= nBins + 1));
125 H[pos] += increment;
126 cnt += increment;
127 sum += value * increment;
128 rms += (value * value) * increment;
129 }

214 ScalarType rangeCount(ScalarType rangeMin, ScalarType rangeMax) const
215 {
216 uint firstBin = binIndex(rangeMin);
217 uint lastBin = binIndex(rangeMax);
218
219 ScalarType sum = 0;
220 for (uint i = firstBin; i <= lastBin; ++i)
221 sum += H[i];
222 return sum;
223 }
224
232 ScalarType percentile(ScalarType frac) const
233 {
234 if (H.size() == 0)
235 return 0;
236
237 // check percentile range
238 assert(frac >= 0 && frac <= 1);
239
240 ScalarType sum = cnt * frac;
241 ScalarType partsum = 0;
242
243 uint i = 0;
244 while (partsum < sum) {
245 partsum += H[i];
246 ++i;
247 }
248
249 return R[i + 1];
250 }
251
256 ScalarType average() const { return sum / cnt; }
257
262 ScalarType rootMeanSquare() const { return std::sqrt(rms / cnt); }
263
268 ScalarType variance() const
269 {
270 return std::abs(rms / cnt - average() * average());
271 }
272
277 ScalarType standardDeviation() const { return std::sqrt(variance()); }
278
279private:
283 uint binIndex(ScalarType elem) const
284 {
285 // lower_bound returns the furthermost iterator i in [first, last) such
286 // that, for every iterator j in [first, i), *j < value. E.g. An
287 // iterator pointing to the first element "not less than" val, or end()
288 // if every element is less than val.
289 auto it = std::lower_bound(R.begin(),R.end(), elem);
290
291 assert(it != R.begin());
292 assert(it != R.end());
293
294 uint pos = it - R.begin();
295 pos -= 1;
296
297 assert (R[pos] < elem);
298 assert (elem <= R[pos+1] );
299 return pos;
300 }

302};
