415 uint indexOfEdge(const Vertex* v1, const Vertex* v2) const
416 {
417 if constexpr (STORE_INDICES) {
418 return indexOfEdge(
419 Base::indexFromPointer(v1), Base::indexFromPointer(v2));
420 }
421 else {
422 uint vid = indexOfVertex(v1);
423 if (vid == UINT_NULL) {
424 return UINT_NULL;
425 }
426 else if (vertexMod(vid + 1) == v2) {
427 return vid;
428 }
429 else if (vertexMod((int) vid - 1) == v2) {
430 int n = vertexNumber(); // n must be int to avoid unwanted casts
431 return (((int) vid - 1) % n + n) % n;
432 }
433 else {
434 return UINT_NULL;
435 }
436 }
437 }

455 uint indexOfEdge(uint vi1, uint vi2) const

469 else {

471 }
472 }

657 // Component interface functions
658 template<typename Element>
659 void importFrom(const Element& e, bool importRefs = true)
660 {
661 if (importRefs) {
662 if constexpr (HasVertexReferences<Element>) {
663 if constexpr (N > 0) {
664 // same size non-polygonal faces
665 if constexpr (N == Element::VERTEX_NUMBER) {
666 importIndicesFrom(e);
667 }
668 // from polygonal to fixed size, but the polygon size == the
669 // fixed face size
670 else if constexpr (Element::VERTEX_NUMBER < 0) {
671 if (e.vertexNumber() == N) {
672 importIndicesFrom(e);
673 }
674 }
675 else {
676 // do not import in this case: cannot import from a face
677 // of different size
678 }
679 }
680 else {
681 // from fixed to polygonal size: need to resize first, then
682 // import
683 resizeVertices(e.vertexNumber());
684 importIndicesFrom(e);
685 }
686 }
687 }

690 void serialize(std::ostream& os) const
691 {
692 // regardless of the type of the container (indices or pointers), the
693 // serialization is always done using the indices
694 if constexpr (N < 0) {
695 vcl::serialize(os, vertexNumber());

698 vcl::serialize(os, vertexIndex(i));
699 }

702 void deserialize(std::istream& is)
703 {

708 }
709 for (uint i = 0; i < vertexNumber(); ++i) {
710 uint vi;
711 vcl::deserialize(is, vi);

713 }
