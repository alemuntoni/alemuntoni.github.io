223template<uint COMPONENT_ID, ElementOrMeshConcept T>
224bool isComponentAvailableOn(const T& obj)
225{
226 if constexpr (HasOptionalComponentOfType<T, COMPONENT_ID>) {
227 using ComponentType =
228 ComponentOfType<COMPONENT_ID, typename T::Components>;
229 const ComponentType& c = static_cast<const ComponentType&>(obj);
230 return c.isAvailable();
231 }
232 else
233 return HasComponentOfType<T, COMPONENT_ID>;
234}
