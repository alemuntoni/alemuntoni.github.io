147template<uint COMPONENT_TYPE, typename T>
148bool isComponentEnabledOn(const T& element)
149{
150 if constexpr (HasOptionalComponentOfType<T, COMPONENT_TYPE>) {
151 using ComponentType = ComponentOfType<COMPONENT_TYPE, typename T::Components>;
152 const ComponentType& c = static_cast<const ComponentType&>(element);

154 }
