46// From:
47// https://blog.nobel-joergensen.com/2013/01/29/debugging-opengl-using-glgeterror/
48inline void _check_gl_error(const char* file, int line)
49{
50 GLenum err(glGetError());
51
52 while (err != GL_NO_ERROR) {
53 std::string error;
54
55 switch (err) {
56 case GL_INVALID_OPERATION: error = "INVALID_OPERATION"; break;
57 case GL_INVALID_ENUM: error = "INVALID_ENUM"; break;
58 case GL_INVALID_VALUE: error = "INVALID_VALUE"; break;
59 case GL_OUT_OF_MEMORY: error = "OUT_OF_MEMORY"; break;
60#ifdef unix
61 case GL_INVALID_FRAMEBUFFER_OPERATION:
62 error = "INVALID_FRAMEBUFFER_OPERATION";
63 break;
64#endif
65 }
66
67 std::cerr << "GL_" << error.c_str() << " - " << file << ":" << line
68 << std::endl;
69 err = glGetError();
70 }
71}

114 void draw() const
115 {
116 if (mrs.isVisible()) {
117 if (mrs.isWireframeVisible()) {
118 if (mrs.isPointCloudVisible()) {
119 glDisable(GL_LIGHTING);
120 glShadeModel(GL_FLAT);
121 glPolygonMode(GL_FRONT_AND_BACK, GL_LINE);
122 glDepthRange(0.0, 1.0);
123 renderPass();
124 glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
125 }
126 if (mrs.isSurfaceVisible()) {
127 if (mrs.isSurfaceShadingFlat()) {
128 glEnable(GL_LIGHTING);
129 glShadeModel(GL_FLAT);
130 glDepthRange(0.01, 1.0);
131 renderPass();
132
133 glDisable(GL_LIGHTING);
134 glPolygonMode(GL_FRONT_AND_BACK, GL_LINE);
135 glDepthRange(0.0, 1.0);
136 glDepthFunc(GL_LEQUAL);
137 renderPass();
138 glDepthFunc(GL_LESS);
139 glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
140 }
141 else if (mrs.isSurfaceShadingSmooth()) {
142 glEnable(GL_LIGHTING);
143 glShadeModel(GL_SMOOTH);
144 glDepthRange(0.01, 1.0);
145 renderPass();
146
147 glDisable(GL_LIGHTING);
148 glPolygonMode(GL_FRONT_AND_BACK, GL_LINE);
149 glDepthRange(0.0, 1.0);
150 glDepthFunc(GL_LEQUAL);
151 renderPass();
152 glDepthFunc(GL_LESS);
153 glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
154 }
155 }
156 else {
157 glDisable(GL_LIGHTING);
158 glShadeModel(GL_FLAT);
159 glPolygonMode(GL_FRONT_AND_BACK, GL_LINE);
160 glDepthRange(0.0, 1.0);
161 renderPass();
162 glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
163 }
164 }
165 else { // no wireframe
166 if (mrs.isPointCloudVisible()) {
167 glDisable(GL_LIGHTING);
168 renderPass();
169 }
170 if (mrs.isSurfaceVisible()) {
171 if (mrs.isSurfaceShadingFlat()) {
172 glEnable(GL_LIGHTING);
173 glShadeModel(GL_FLAT);
174 renderPass();
175 }
176 else if (mrs.isSurfaceShadingSmooth()) {
177 glEnable(GL_LIGHTING);
178 glShadeModel(GL_SMOOTH);
179 renderPass();
180 }
181 }
182 }
183 if (mrs.isBboxEnabled()) {
184 vcl::drawBox3(mrb.bbMin(), mrb.bbMax(), vcl::Color(0, 0, 0));
185 }
186 }
187 }

196 void renderPass() const
197 {
198 uint nv = mrb.vertexNumber();
199 uint nt = mrb.triangleNumber();
200
201 const float* coords = mrb.vertexBufferData();
202 const int* triangles = mrb.triangleBufferData();
203 const float* vertexNormals = mrb.vertexNormalBufferData();
204 const float* vertexColors = mrb.vertexColorBufferData();
205 const float* triangleNormals = mrb.triangleNormalBufferData();
206 const float* triangleColors = mrb.triangleColorBufferData();
207 const float* vertTexCoords = mrb.vertexTexCoordsBufferData();
208 const float* wedgTexCoords = mrb.wedgeTexCoordsBufferData();
209
210 if (mrs.isPointCloudVisible()) {
211 glEnableClientState(GL_VERTEX_ARRAY);
212 glVertexPointer(3, GL_FLOAT, 0, coords);
213
214 if (mrs.isPointCloudColorPerVertex()) {
215 glEnableClientState(GL_COLOR_ARRAY);
216 glColorPointer(3, GL_FLOAT, 0, vertexColors);
217 }
218 else if (mrs.isPointCloudColorPerMesh()) {
219 glColor4fv(mrb.meshColorBufferData());
220 }
221 else if (mrs.isPointCloudColorUserDefined()) {
222 glColor4fv(mrs.pointCloudUserColorData());
223 }
224
225 glPointSize(mrs.pointWidth());
226
227 glDrawArrays(GL_POINTS, 0, nv);
228
229 glDisableClientState(GL_COLOR_ARRAY);
230 glDisableClientState(GL_VERTEX_ARRAY);
231 }
232
233 if (mrs.isSurfaceVisible()) {
234 // Old fashioned, verbose and slow rendering.
235 if (mrs.isSurfaceColorPerFace()) {
236 int n_tris = nt;
237 for (int tid = 0; tid < n_tris; ++tid) {
238 int tid_ptr = 3 * tid;
239 int vid0 = triangles[tid_ptr + 0];
240 int vid1 = triangles[tid_ptr + 1];
241 int vid2 = triangles[tid_ptr + 2];
242 int vid0_ptr = 3 * vid0;
243 int vid1_ptr = 3 * vid1;
244 int vid2_ptr = 3 * vid2;
245
246 if (mrs.isSurfaceShadingSmooth()) {
247 glBegin(GL_TRIANGLES);
248 glColor3fv(&(triangleColors[tid_ptr]));
249 glNormal3fv(&(vertexNormals[vid0_ptr]));
250 glVertex3fv(&(coords[vid0_ptr]));
251 glNormal3fv(&(vertexNormals[vid1_ptr]));
252 glVertex3fv(&(coords[vid1_ptr]));
253 glNormal3fv(&(vertexNormals[vid2_ptr]));
254 glVertex3fv(&(coords[vid2_ptr]));
255 glEnd();
256 }
257 else {
258 glBegin(GL_TRIANGLES);
259 glColor3fv(&(triangleColors[tid_ptr]));
260 glNormal3fv(&(triangleNormals[tid_ptr]));
261 glVertex3fv(&(coords[vid0_ptr]));
262 glNormal3fv(&(triangleNormals[tid_ptr]));
263 glVertex3fv(&(coords[vid1_ptr]));
264 glNormal3fv(&(triangleNormals[tid_ptr]));
265 glVertex3fv(&(coords[vid2_ptr]));
266 glEnd();
267 }
268 }
269 }
270 else if (mrs.isSurfaceColorPerVertex()) {
271 if (mrs.isSurfaceShadingSmooth()) {
272 glEnableClientState(GL_VERTEX_ARRAY);
273 glVertexPointer(3, GL_FLOAT, 0, coords);
274
275 glEnableClientState(GL_NORMAL_ARRAY);
276 glNormalPointer(GL_FLOAT, 0, vertexNormals);
277
278 glEnableClientState(GL_COLOR_ARRAY);
279 glColorPointer(3, GL_FLOAT, 0, vertexColors);
280
281 glDrawElements(
282 GL_TRIANGLES, nt * 3, GL_UNSIGNED_INT, triangles);
283
284 glDisableClientState(GL_COLOR_ARRAY);
285 glDisableClientState(GL_NORMAL_ARRAY);
286 glDisableClientState(GL_VERTEX_ARRAY);
287 }
288 else {
289 glShadeModel(GL_SMOOTH);
290 int n_tris = nt;
291 for (int tid = 0; tid < n_tris; ++tid) {
292 int tid_ptr = 3 * tid;
293 int vid0 = triangles[tid_ptr + 0];
294 int vid1 = triangles[tid_ptr + 1];
295 int vid2 = triangles[tid_ptr + 2];
296 int vid0_ptr = 3 * vid0;
297 int vid1_ptr = 3 * vid1;
298 int vid2_ptr = 3 * vid2;
299
300 glBegin(GL_TRIANGLES);
301 glColor3fv(&(vertexColors[vid0_ptr]));
302 glNormal3fv(&(triangleNormals[tid_ptr]));
303 glVertex3fv(&(coords[vid0_ptr]));
304 glColor3fv(&(vertexColors[vid1_ptr]));
305 glNormal3fv(&(triangleNormals[tid_ptr]));
306 glVertex3fv(&(coords[vid1_ptr]));
307 glColor3fv(&(vertexColors[vid2_ptr]));
308 glNormal3fv(&(triangleNormals[tid_ptr]));
309 glVertex3fv(&(coords[vid2_ptr]));
310 glEnd();
311 }
312 }
313 }
314 else if (
315 mrs.isSurfaceColorPerMesh() || mrs.isSurfaceColorUserDefined())
316 {
317 if (mrs.isSurfaceShadingSmooth()) {
318 glEnableClientState(GL_VERTEX_ARRAY);
319 glVertexPointer(3, GL_FLOAT, 0, coords);
320
321 glEnableClientState(GL_NORMAL_ARRAY);
322 glNormalPointer(GL_FLOAT, 0, vertexNormals);
323
324 if (mrs.isSurfaceColorPerMesh()) {
325 glColor4fv(mrb.meshColorBufferData());
326 }
327 else {
328 glColor4fv(mrs.surfaceUserColorData());
329 }
330
331 glDrawElements(
332 GL_TRIANGLES, nt * 3, GL_UNSIGNED_INT, triangles);
333
334 glDisableClientState(GL_COLOR_ARRAY);
335 glDisableClientState(GL_NORMAL_ARRAY);
336 glDisableClientState(GL_VERTEX_ARRAY);
337 }
338 else {
339 if (mrs.isSurfaceColorPerMesh()) {
340 glColor4fv(mrb.meshColorBufferData());
341 }
342 else {
343 glColor4fv(mrs.surfaceUserColorData());
344 }
345 int n_tris = nt;
346 for (int tid = 0; tid < n_tris; ++tid) {
347 int tid_ptr = 3 * tid;
348 int vid0 = triangles[tid_ptr + 0];
349 int vid1 = triangles[tid_ptr + 1];
350 int vid2 = triangles[tid_ptr + 2];
351 int vid0_ptr = 3 * vid0;
352 int vid1_ptr = 3 * vid1;
353 int vid2_ptr = 3 * vid2;
354
355 glBegin(GL_TRIANGLES);
356 glNormal3fv(&(triangleNormals[tid_ptr]));
357 glVertex3fv(&(coords[vid0_ptr]));
358 glNormal3fv(&(triangleNormals[tid_ptr]));
359 glVertex3fv(&(coords[vid1_ptr]));
360 glNormal3fv(&(triangleNormals[tid_ptr]));
361 glVertex3fv(&(coords[vid2_ptr]));
362 glEnd();
363 }
364 }
365 }
366 else if (mrs.isSurfaceColorPerVertexTexcoords()) {
367 if (mrs.isSurfaceShadingSmooth()) {
368 short texture = textID[0];
369 glBindTexture(GL_TEXTURE_2D, texture);
370 glColor4f(1,1,1,1);
371 glEnableClientState(GL_VERTEX_ARRAY);
372 glVertexPointer(3, GL_FLOAT, 0, coords);
373
374 glEnableClientState(GL_NORMAL_ARRAY);
375 glNormalPointer(GL_FLOAT, 0, vertexNormals);
376
377 glEnableClientState(GL_TEXTURE_COORD_ARRAY);
378 glTexCoordPointer(2, GL_FLOAT, 0, vertTexCoords);
379
380 glDrawElements(
381 GL_TRIANGLES, nt * 3, GL_UNSIGNED_INT, triangles);
382
383 glDisableClientState(GL_TEXTURE_COORD_ARRAY);
384 glDisableClientState(GL_NORMAL_ARRAY);
385 glDisableClientState(GL_VERTEX_ARRAY);
386 glBindTexture(GL_TEXTURE_2D, 0);
387 }
388 else {
389 glShadeModel(GL_SMOOTH);
390 int n_tris = nt;
391 for (int tid = 0; tid < n_tris; ++tid) {
392 int tid_ptr = 3 * tid;
393 int vid0 = triangles[tid_ptr + 0];
394 int vid1 = triangles[tid_ptr + 1];
395 int vid2 = triangles[tid_ptr + 2];
396 int vid0_ptr = 3 * vid0;
397 int vid1_ptr = 3 * vid1;
398 int vid2_ptr = 3 * vid2;
399 short texture = textID[0];
400 glBindTexture(GL_TEXTURE_2D, texture);
401 glBegin(GL_TRIANGLES);
402 glColor4f(1,1,1,1);
403 glTexCoord2f(
404 vertTexCoords[vid0 * 2 + 0],
405 vertTexCoords[vid0 * 2 + 1]);
406 glNormal3fv(&(vertexNormals[vid0_ptr]));
407 glVertex3fv(&(coords[vid0_ptr]));
408 glTexCoord2f(
409 vertTexCoords[vid1 * 2 + 0],
410 vertTexCoords[vid1 * 2 + 1]);
411 glNormal3fv(&(vertexNormals[vid1_ptr]));
412 glVertex3fv(&(coords[vid1_ptr]));
413 glTexCoord2f(
414 vertTexCoords[vid2 * 2 + 0],
415 vertTexCoords[vid2 * 2 + 1]);
416 glNormal3fv(&(vertexNormals[vid2_ptr]));
417 glVertex3fv(&(coords[vid2_ptr]));
418 glEnd();
419 glBindTexture(GL_TEXTURE_2D, 0);
420 }
421 }
422 }
423 else if (mrs.isSurfaceColorPerWedgeTexcoords()) {
424 int n_tris = nt;
425 for (int tid = 0; tid < n_tris; ++tid) {
426 int tid_ptr = 3 * tid;
427 int vid0 = triangles[tid_ptr + 0];
428 int vid1 = triangles[tid_ptr + 1];
429 int vid2 = triangles[tid_ptr + 2];
430 int vid0_ptr = 3 * vid0;
431 int vid1_ptr = 3 * vid1;
432 int vid2_ptr = 3 * vid2;
433 short texture = textID[mrb.wedgeTextureIDsBufferData()[tid]];
434 glBindTexture(GL_TEXTURE_2D, texture);
435 glBegin(GL_TRIANGLES);
436 glColor4f(1,1,1,1);
437 glTexCoord2f(
438 wedgTexCoords[tid * 6 + 0], wedgTexCoords[tid * 6 + 1]);
439 glNormal3fv(&(vertexNormals[vid0_ptr]));
440 glVertex3fv(&(coords[vid0_ptr]));
441 glTexCoord2f(
442 wedgTexCoords[tid * 6 + 2], wedgTexCoords[tid * 6 + 3]);
443 glNormal3fv(&(vertexNormals[vid1_ptr]));
444 glVertex3fv(&(coords[vid1_ptr]));
445 glTexCoord2f(
446 wedgTexCoords[tid * 6 + 4], wedgTexCoords[tid * 6 + 5]);
447 glNormal3fv(&(vertexNormals[vid2_ptr]));
448 glVertex3fv(&(coords[vid2_ptr]));
449 glEnd();
450 glBindTexture(GL_TEXTURE_2D, 0);
451 }
452 }
453 }
454
455 if (mrs.isWireframeVisible()) {
456 glEnableClientState(GL_VERTEX_ARRAY);
457 glVertexPointer(3, GL_FLOAT, 0, coords);
458
459 glLineWidth(mrs.wireframeWidth());
460
461 if (mrs.isWireframeColorPerMesh()) {
462 glColor4fv(mrb.meshColorBufferData());
463 }
464 else {
465 glColor4fv(mrs.wireframeUserColorData());
466 }
467
468 glDrawElements(GL_TRIANGLES, nt * 3, GL_UNSIGNED_INT, triangles);
469
470 glDisableClientState(GL_VERTEX_ARRAY);
471 }
472 }
473
474 void bindTextures()
475 {
476 textID.resize(mrb.textureNumber());
477 glEnable(GL_TEXTURE_2D);
478 glGenTextures(mrb.textureNumber(), textID.data());
479
480 for (uint i = 0; i < mrb.textureNumber(); i++) {
481 glBindTexture(GL_TEXTURE_2D, textID[i]);
482 glTexImage2D(
483 GL_TEXTURE_2D,
484 0,
485 GL_RGB,
486 mrb.textureSize(i).x(),
487 mrb.textureSize(i).y(),
488 0,
489 GL_RGBA,
490 GL_UNSIGNED_BYTE,
491 mrb.textureBufferData(i));
492 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
493 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
494 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
495 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
496 }
497 }
498
499 void unbindTextures()
500 {
501 if (textID.size() > 0) {
502 glDeleteTextures(textID.size(), textID.data());
503 textID.clear();
504 }
505 }
