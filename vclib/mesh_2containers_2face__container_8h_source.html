158 template<Range Rng>
159 uint addFace(Rng&& r) requires (
160 RangeOfConvertibleTo<Rng, typename Face::VertexType*> ||
161 RangeOfConvertibleTo<Rng, uint>)
162 {
163 auto begin = std::ranges::begin(r);
164 auto end = std::ranges::end(r);
165
166 if (begin == end)
167 return UINT_NULL;
168 uint n = std::distance(begin, end);
169
170 uint fid = UINT_NULL;
171
172 assert(n >= 3);
173 if (n < 3)
174 return UINT_NULL;
175
176 if constexpr (T::VERTEX_NUMBER < 0) {
177 fid = addFace();
178 face(fid).resizeVertices(n);
179 }
180 else {
181 assert(n == T::VERTEX_NUMBER);
182 if (n == T::VERTEX_NUMBER)
183 fid = addFace();
184 }
185
186 if (fid != UINT_NULL) {
187 Face& f = face(fid);
188
189 unsigned int i = 0;
190 for (auto it = begin; it != end; ++it) {
191 f.setVertex(i, *it);
192 ++i;
193 }
194 }
195 return fid;
196 }

1120 template<typename OthMesh>
1121 void manageImportTriFromPoly(const OthMesh& m)
1122 {
1123 if constexpr (HasFaceContainer<OthMesh>) {
1124 using ParentMesh = Base::ParentMeshType;
1125 using VertexType = ParentMesh::VertexType;
1126 using MVertexType = OthMesh::VertexType;
1127 using MCoordType = MVertexType::CoordType;
1128 using MFaceType = OthMesh::FaceType;
1129
1130 using VertexContainer = ParentMesh::VertexContainer;
1131
1132 // if this is not a triangle mesh nor a polygon mesh (meaning that
1133 // we can't control the number of vertex pointers in this mesh), and
1134 // this mesh does not have the same number of vertex pointers of the
1135 // other, it means that we don't know how to convert these type of
1136 // meshes (e.g. we don't know how to convert a polygon mesh into a
1137 // quad mesh, or convert a quad mesh into a pentagonal mesh...)
1138 static_assert(
1139 !(FaceType::VERTEX_NUMBER != 3 && FaceType::VERTEX_NUMBER > 0 &&
1140 FaceType::VERTEX_NUMBER != MFaceType::VERTEX_NUMBER),
1141 "Cannot import from that type of Mesh. Don't know how to "
1142 "convert "
1143 "faces.");
1144
1145 // we need to manage conversion from poly or faces with cardinality
1146 // > 3 (e.g. quads) to triangle meshes. In this case, we triangulate
1147 // the polygon using the earcut algorithm.
1148 if constexpr (
1149 FaceType::VERTEX_NUMBER == 3 &&
1150 (MFaceType::VERTEX_NUMBER > 3 || MFaceType::VERTEX_NUMBER < 0))
1151 {
1152 VertexType* base = &Base::mParentMesh->vertex(0);
1153 const MVertexType* mvbase = &m.vertex(0);
1154
1155 for (const MFaceType& mf : m.faces()) {
1156 // if the current face has the same number of vertices of
1157 // this faces (3), then the vertex pointers have been
1158 // correctly imported from the import pointers function. The
1159 // import pointers function does nothing when importing from
1160 // a face with at least 4 vertices
1161 if (mf.vertexNumber() != FaceType::VERTEX_NUMBER) {
1162 // triangulate mf; the first triangle of the
1163 // triangulation will be this->face(m.index(mf)); the
1164 // other triangles will be added at the end of the
1165 // container
1166 std::vector<uint> tris = Polygon<MCoordType>::earCut(
1167 mf.vertices() | vcl::views::coords);
1168 FaceType& f = face(m.index(mf));
1169 importTriPointersHelper(f, mf, base, mvbase, tris, 0);
1170
1171 // number of other faces to add
1172 uint nf = tris.size() / 3 - 1;
1173 uint fid = addFaces(nf);
1174
1175 uint i = 3; // index that cycles into tris
1176 for (; fid < faceContainerSize(); ++fid) {
1177 FaceType& f = face(fid);
1178 importTriPointersHelper(
1179 f, mf, base, mvbase, tris, i);
1180 i += 3;
1181 }
1182 }
1183 }
1184 }
1185 }

1213 template<typename MFaceType, typename VertexType, typename MVertexType>
1214 static void importTriPointersHelper(
1215 FaceType& f,
1216 const MFaceType& mf,
1217 VertexType* base,
1218 const MVertexType* mvbase,
1219 const std::vector<uint>& tris,
1220 uint basetri)
1221 {
1222 f.importFrom(mf); // import all the components from mf
1223 for (uint i = basetri, j = 0; i < basetri + 3; i++, j++) {
1224 f.setVertex(j, base + (mf.vertex(tris[i]) - mvbase));
1225
1226 // wedge colors
1227 if constexpr (
1228 face::HasWedgeColors<FaceType> &&
1229 face::HasWedgeColors<MFaceType>)
1230 {
1231 if (comp::isWedgeColorsAvailableOn(f) &&
1232 comp::isWedgeColorsAvailableOn(mf))
1233 {
1234 f.wedgeColor(j) = mf.wedgeColor(tris[i]);
1235 }
1236 }
1237
1238 // wedge texcoords
1239 if constexpr (
1240 face::HasWedgeTexCoords<FaceType> &&
1241 face::HasWedgeTexCoords<MFaceType>)
1242 {
1243 if (comp::isWedgeTexCoordsAvailableOn(f) &&
1244 comp::isWedgeTexCoordsAvailableOn(mf))
1245 {
1246 using ST = typename FaceType::WedgeTexCoordType::ScalarType;
1247 f.wedgeTexCoord(j) =
1248 mf.wedgeTexCoord(tris[i]).template cast<ST>();
1249 }
1250 }
1251 }
1252 }
