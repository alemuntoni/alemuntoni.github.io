159 // TODO requires on iterated type - must be Vertex* or int
160 template<typename Iterator>
161 uint addFace(Iterator begin, Iterator end)
162 {
163 if (begin == end) return UINT_NULL;
164 uint n = std::distance(begin, end);
165
166 uint fid = UINT_NULL;
167
168 assert(n >= 3);
169 if (n < 3) return UINT_NULL;
170
171 if constexpr (T::VERTEX_NUMBER < 0) {
172 fid = addFace();
173 face(fid).resizeVertices(n);
174 }
175 else {
176 assert(n == T::VERTEX_NUMBER);
177 if (n == T::VERTEX_NUMBER)
178 fid = addFace();
179 }
180
181 if (fid != UINT_NULL) {
182 Face& f = face(fid);
183
184 unsigned int i = 0;
185 for (Iterator it = begin; it != end; ++it) {
186 if constexpr (std::integral<typename Iterator::value_type>)
187 f.vertex(i) = &Base::parentMesh->vertex(*it);
188 else
189 f.vertex(i) = *it;
190 ++i;
191 }
192 }
193 return fid;
194 }

1119 template<typename OthMesh>
1120 void manageImportTriFromPoly(const OthMesh& m)
1121 {
1122 if constexpr (HasFaceContainer<OthMesh>) {
1123 using ParentMesh = Base::ParentMeshType;
1124 using VertexType = ParentMesh::VertexType;
1125 using MVertexType = OthMesh::VertexType;
1126 using MCoordType = MVertexType::CoordType;
1127 using MFaceType = OthMesh::FaceType;
1128
1129 using VertexContainer = ParentMesh::VertexContainer;
1130
1131 // if this is not a triangle mesh nor a polygon mesh (meaning that
1132 // we can't control the number of vertex pointers in this mesh), and
1133 // this mesh does not have the same number of vertex pointers of the
1134 // other, it means that we don't know how to convert these type of
1135 // meshes (e.g. we don't know how to convert a polygon mesh into a
1136 // quad mesh, or convert a quad mesh into a pentagonal mesh...)
1137 static_assert(
1138 !(FaceType::VERTEX_NUMBER != 3 && FaceType::VERTEX_NUMBER > 0 &&
1139 FaceType::VERTEX_NUMBER != MFaceType::VERTEX_NUMBER),
1140 "Cannot import from that type of Mesh. Don't know how to "
1141 "convert "
1142 "faces.");
1143
1144 // we need to manage conversion from poly or faces with cardinality
1145 // > 3 (e.g. quads) to triangle meshes. In this case, we triangulate
1146 // the polygon using the earcut algorithm.
1147 if constexpr (
1148 FaceType::VERTEX_NUMBER == 3 &&
1149 (MFaceType::VERTEX_NUMBER > 3 || MFaceType::VERTEX_NUMBER < 0))
1150 {
1151 VertexType* base = &Base::parentMesh->vertex(0);
1152 const MVertexType* mvbase = &m.vertex(0);
1153
1154 for (const MFaceType& mf : m.faces()) {
1155 // if the current face has the same number of vertices of
1156 // this faces (3), then the vertex pointers have been
1157 // correctly imported from the import pointers function. The
1158 // import pointers function does nothing when importing from
1159 // a face with at least 4 vertices
1160 if (mf.vertexNumber() != FaceType::VERTEX_NUMBER) {
1161 // triangulate mf; the first triangle of the
1162 // triangulation will be this->face(m.index(mf)); the
1163 // other triangles will be added at the end of the
1164 // container
1165 std::vector<uint> tris = Polygon<MCoordType>::earCut(
1166 mf.vertices() | vcl::views::coords);
1167 FaceType& f = face(m.index(mf));
1168 importTriPointersHelper(f, mf, base, mvbase, tris, 0);
1169
1170 // number of other faces to add
1171 uint nf = tris.size() / 3 - 1;
1172 uint fid = addFaces(nf);
1173
1174 uint i = 3; // index that cycles into tris
1175 for (; fid < faceContainerSize(); ++fid) {
1176 FaceType& f = face(fid);
1177 importTriPointersHelper(
1178 f, mf, base, mvbase, tris, i);
1179 i+=3;
1180 }
1181 }
1182 }
1183 }
1184 }

1210 template<typename MFaceType, typename VertexType, typename MVertexType>
1211 static void importTriPointersHelper(
1212 FaceType& f,
1213 const MFaceType& mf,
1214 VertexType* base,
1215 const MVertexType* mvbase,
1216 const std::vector<uint>& tris,
1217 uint basetri)
1218 {
1219 f.importFrom(mf); // import all the components from mf
1220 for (uint i = basetri, j = 0; i < basetri+3; i++, j++) {
1221 f.vertex(j) = base + (mf.vertex(tris[i]) - mvbase);
1222
1223 // wedge colors
1224 if constexpr (
1225 face::HasWedgeColors<FaceType> &&
1226 face::HasWedgeColors<MFaceType>)
1227 {
1228 if (comp::isWedgeColorsAvailableOn(f) &&
1229 comp::isWedgeColorsAvailableOn(mf))
1230 {
1231 f.wedgeColor(j) = mf.wedgeColor(tris[i]);
1232 }
1233 }
1234
1235 // wedge texcoords
1236 if constexpr (
1237 face::HasWedgeTexCoords<FaceType> &&
1238 face::HasWedgeTexCoords<MFaceType>)
1239 {
1240 if (comp::isWedgeTexCoordsAvailableOn(f) &&
1241 comp::isWedgeTexCoordsAvailableOn(mf))
1242 {
1243 using ST = typename FaceType::WedgeTexCoordType::ScalarType;
1244 f.wedgeTexCoord(j) =
1245 mf.wedgeTexCoord(tris[i]).template cast<ST>();
1246 }
1247 }
1248 }
1249 }
