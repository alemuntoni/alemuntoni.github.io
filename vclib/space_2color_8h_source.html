195 uint8_t hsvHue() const
196 {
197 uint8_t rgbMin, rgbMax;
198 uint8_t h;
199
200 rgbMin = x() < y() ? (x() < z() ? x() : z()) : (y() < z() ? y() : z());
201 rgbMax = x() > y() ? (x() > z() ? x() : z()) : (y() > z() ? y() : z());
202
203 if (rgbMax == 0) {
204 return 0;
205 }
206
207 if (255 * long(rgbMax - rgbMin) / rgbMax == 0) {
208 return 0;
209 }
210
211 if (rgbMax == x())
212 h = 0 + 43 * (y() - z()) / (rgbMax - rgbMin);
213 else if (rgbMax == y())
214 h = 85 + 43 * (z() - x()) / (rgbMax - rgbMin);
215 else
216 h = 171 + 43 * (x() - y()) / (rgbMax - rgbMin);
217
218 return h;
219 }

331 void setHsv(uint h, uint8_t s, uint8_t v, uint8_t alpha = 255)
332 {
333 w() = alpha;
334 if (s == 0) {
335 x() = v;
336 y() = v;
337 z() = v;
338 }
339 else {
340 h = (h / 360.0) * 255;
341 uint8_t region, remainder, p, q, t;
342 region = h / 43;
343 remainder = (h - (region * 43)) * 6;
344
345 p = (v * (255 - s)) >> 8;
346 q = (v * (255 - ((s * remainder) >> 8))) >> 8;
347 t = (v * (255 - ((s * (255 - remainder)) >> 8))) >> 8;
348
349 switch (region) {
350 case 0:
351 x() = v;
352 y() = t;
353 z() = p;
354 break;
355 case 1:
356 x() = q;
357 y() = v;
358 z() = p;
359 break;
360 case 2:
361 x() = p;
362 y() = v;
363 z() = t;
364 break;
365 case 3:
366 x() = p;
367 y() = q;
368 z() = v;
369 break;
370 case 4:
371 x() = t;
372 y() = p;
373 z() = v;
374 break;
375 default:
376 x() = v;
377 y() = p;
378 z() = q;
379 break;
380 }
381 }

566inline Color colorLerp(const Color& c0, const Color& c1, float value)
567{
568 Color c;
569 if (value < 0) // out of range - left
570 value = 0;
571 else if (value > 1) // out of range - right
572 value = 1;
573 for (uint i = 0; i < 4; i++)
574 c(i) = c1(i)*value + c0(i)*(1-value);
575 return c;
576}

642inline Color colorFromIntervalParula(float value)
643{
644 if (value < 0) // out of range - left
645 value = 0;
646 else if (value > 1) // out of range - right
647 value = 1;
648
649 static uint paruVal[9] = {
650 0x271680ff,
651 0x0363e1ff,
652 0x1485d4ff,
653 0x06a7c6ff,
654 0x38b99eff,
655 0x92bf73ff,
656 0xd9ba56ff,
657 0xfcce2eff,
658 0xfffa0aff};
659
660 int ind = int(floor(value * 8.0f));
661 float div = (value * 8.0f - ind);
662
663 if (div < 0)
664 div = 0;
665 else if (div > 1)
666 div = 1;
667
668 return colorLerp(
669 (Color::ColorRGBA) paruVal[ind],
670 (Color::ColorRGBA) paruVal[ind + 1],
671 div);
672}

814inline std::vector<Color>
815colorScattering(uint n, float sat = 0.3, float val = 0.9)
816{
817 std::vector<Color> scattering;
818 scattering.reserve(n);
819
820 for (uint v = 0; v < n; ++v) {
821 uint value = v, m = n;
822 uint b = 0;
823
824 for (uint k = 1; k < n; k <<= 1) {
825 if (value << 1 >= m) {
826 b += k;
827 value -= (m + 1) >> 1;
828 m >>= 1;
829 }
830 else {
831 m = (m + 1) >> 1;
832 }
833 }
834
835 Color rc;
836 rc.setHsvF(float(b) / float(n), sat, val);
837 scattering.push_back(rc);
838 }
839 return scattering;
840}
