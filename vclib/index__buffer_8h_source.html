224private:
225 static uint64_t flagsForType(PrimitiveType type)
226 {
227 switch (type) {
228 case PrimitiveType::INT:
229 case PrimitiveType::UINT: return BGFX_BUFFER_INDEX32;
230 case PrimitiveType::FLOAT:
231 return BGFX_BUFFER_COMPUTE_FORMAT_32X1 |
232 BGFX_BUFFER_COMPUTE_TYPE_FLOAT;
233 case PrimitiveType::DOUBLE: assert(0); // not supported
234 default: return BGFX_BUFFER_NONE;
235 }
236 }
237
238 static uint64_t flagsForAccess(bgfx::Access::Enum access)
239 {
240 switch (access) {
241 case bgfx::Access::Read: return BGFX_BUFFER_COMPUTE_READ;
242 case bgfx::Access::Write: return BGFX_BUFFER_COMPUTE_WRITE;
243 case bgfx::Access::ReadWrite: return BGFX_BUFFER_COMPUTE_READ_WRITE;
244 default: return BGFX_BUFFER_NONE;
245 }
