341 template<typename... I>
342 ConstPointer data(I... indices) const requires (sizeof...(indices) < N)
343 {
344 constexpr std::size_t n = sizeof...(indices);
345 if constexpr (n == 0) {
346 return mVec.data();
347 }
348 else {
349 std::size_t args[] = {static_cast<std::size_t>(indices)...};
350 std::size_t ind = args[0];

353 for (i = 1; i < n; i++) {

355 ind *= mSizes[i];
356 ind += args[i];

358 for (; i < N; i++) {

360 }

460 template<typename... Sizes>
461 void conservativeResize(Sizes... s) requires (sizeof...(s) == N)
462 {
463 std::size_t newSizes[N] = {static_cast<std::size_t>(s)...};
464 std::size_t newTotalSize = 1;
465 for (uint i = 0; i < N; i++)
466 newTotalSize *= newSizes[i];
467 std::vector<T> newVector(newTotalSize);
468
469 for (std::size_t i = 0; i < mVec.size(); i++) {
470 std::array<std::size_t, N> indices = reverseIndex(i);
471 bool outOfBound = false;
472 for (std::size_t j = 0; j < N; j++)
473 if (indices[j] >= newSizes[j] || indices[j] >= mSizes[j])
474 outOfBound = true;
475 if (!outOfBound) {
476 newVector[getIndex(indices.data(), newSizes)] = mVec[i];
477 }
478 }
479
480 for (uint i = 0; i < mSizes.size(); i++) {
481 mSizes[i] = newSizes[i];
482 }
483
484 mVec = std::move(newVector);
485 }

520 Array<T, N - 1> subArray(uint r) const requires (N > 1)
521 {

525 for (uint i = 0; i < mSizes.size() - 1; i++) {
526 sub.mSizes[i] = mSizes[i + 1];
527 size *= sub.mSizes[i];

591 std::array<std::size_t, N> reverseIndex(uint index)
592 {
593 std::array<std::size_t, N> indices;
594 for (long int i = N - 1; i >= 0; i--) {
595 indices[i] = index % mSizes[i];
596 index /= mSizes[i];
597 }
598 return indices;
599 }

601 static std::size_t getIndex(
602 const std::size_t indices[],
603 const std::size_t sizes[])
604 {
605 std::size_t ind = indices[0];
606 assert(indices[0] < sizes[0]);
607 for (uint i = 1; i < N; i++) {
608 assert(indices[i] < sizes[i]);
609 ind *= sizes[i];
610 ind += indices[i];
611 }
