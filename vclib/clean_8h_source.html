62template<typename Cont, typename MeshType>
63void setReferencedVerticesOnVector(
64 const MeshType& m,
65 std::vector<bool>& refs,
66 uint& nRefs)
67{
68 // check if the Cont container of the Mesh has vertex pointers
69 if constexpr (comp::HasVertexPointers<typename Cont::ElementType>) {
70 // if there are still some vertices non-referenced
71 if (nRefs < m.vertexNumber()) {
72 constexpr uint ELEM_ID = Cont::ElementType::ELEMENT_ID;
73 // for eache element of the Cont container
74 for (const auto& el : m.template elements<ELEM_ID>()) {
75 // for each vertex pointer of the element
76 for (const auto* v : el.vertices()) {
77 if (! refs[m.index(v)]) {
78 // set the vertex as referenced
79 refs[m.index(v)] = true;
80 nRefs++;
81 }
82 }
83 }
84 }
85 }
86}

139template<typename IndexType, typename SentinelType, int N>
140class SortedIndexContainer
141{
142public:
143 SortedIndexContainer() {}
144
145 template<Range RangeType>
146 SortedIndexContainer(SentinelType s, RangeType rng) : s(s), v(rng)
147 {
148 std::sort(v.begin(), v.end());
149 }
150
151 bool operator<(const SortedIndexContainer& s) const
152 {
153 if constexpr (N >= 0) {
154 for (uint i = 0; i < N; ++i) {
155 if (v[i] != s.v[i])
156 return v[i] < s.v[i];
157 }
158 return false;
159 }
160 else {
161 for (uint i = 0; i < v.size() && i < s.v.size(); ++i) {
162 if (v[i] != s.v[i])
163 return v[i] < s.v[i];
164 }
165 return v.size() < s.v.size();
166 }
167 }
168
169 bool operator==(const SortedIndexContainer& s) const
170 {
171 if constexpr (N >= 0) {
172 for (uint i = 0; i < N; ++i) {
173 if (v[i] != s.v[i])
174 return false;
175 }
176 return true;
177 }
178 else {
179 if (v.size() != s.v.size())
180 return false;
181 for (uint i = 0; i < v.size(); ++i) {
182 if (v[i] != s.v[i])
183 return false;
184 }
185 return true;
186 }
187 }
188
189 SentinelType sentinel() const { return s; }
190
191private:
192 vcl::Vector<IndexType, N> v;
193 SentinelType s;
194};

196template<FaceMeshConcept MeshType>
197std::vector<bool> nonManifoldVerticesVectorBool(const MeshType& m)
198 requires vcl::HasPerFaceAdjacentFaces<MeshType>
199{
200 vcl::requirePerFaceAdjacentFaces(m);
201
202 using FaceType = MeshType::FaceType;
203
204 std::vector<bool> nonManifoldVertices(m.vertexContainerSize(), false);
205
206 std::vector<uint> TD(m.vertexContainerSize(), 0);
207 std::vector<bool> nonManifoldInc(m.vertexContainerSize(), false);
208 // First Loop, count how many faces are incident on a vertex and store it in
209 // TD, and flag how many vertices are incident on non manifold edges.
210 for (const FaceType& f : m.faces()) {
211 for (uint i = 0; i < f.vertexNumber(); ++i) {
212 TD[m.index(f.vertex(i))]++;
213 if (!isFaceManifoldOnEdge(f, i)) {
214 nonManifoldInc[m.index(f.vertex(i))] = true;
215 nonManifoldInc[m.index(f.vertexMod(i+1))] = true;
216 }
217 }
218 }
219
220 std::vector<bool> visited(m.vertexContainerSize(), false);
221 for (const FaceType& f : m.faces()) {
222 for (uint i = 0; i < f.vertexNumber(); ++i){
223 if (!visited[m.index(f.vertex(i))]) {
224 visited[m.index(f.vertex(i))] = true;
225 MeshPos pos(&f, i);
226 uint starSize = pos.numberOfAdjacentFacesToV();
227 if (starSize != TD[m.index(f.vertex(i))])
228 nonManifoldVertices[m.index(f.vertex(i))] = true;
229 }
230 }
231 }
232
233 return nonManifoldVertices;
234}

236template<FaceMeshConcept MeshType>
237uint numberEdges(
238 const MeshType& m,
239 uint& numBoundaryEdges,
240 uint& numNonManifoldEdges)
241{
242 std::vector<ConstMeshEdgeUtil<MeshType>> edgeVec =
243 fillAndSortMeshEdgeUtilVector(m);
244
245 uint numEdges = 0;
246 numBoundaryEdges = 0;
247 numNonManifoldEdges = 0;
248
249 size_t f_on_cur_edge =1;
250 for(size_t i=0;i<edgeVec.size();++i) {
251 if(( (i+1) == edgeVec.size()) || !(edgeVec[i] == edgeVec[i+1])) {
252 ++numEdges;
253 if(f_on_cur_edge==1)
254 ++numBoundaryEdges;
255 if(f_on_cur_edge>2)
256 ++numNonManifoldEdges;
257 f_on_cur_edge=1;
258 }
259 else {
260 ++f_on_cur_edge;
261 }
262 }
263 return numEdges;
264}

313template<MeshConcept MeshType>
314uint removeUnreferencedVertices(MeshType& m)
315{
316 using VertexType = MeshType::VertexType;
317
318 // Generate a vector of boolean flags indicating whether each vertex is
319 // referenced by any of the mesh's elements.
320
321 uint n = 0;
322 std::vector<bool> referredVertices =
323 internal::unreferencedVerticesVectorBool(m, n);
324
325 // need to mark as deleted vertices only if the number of unreferenced is
326 // less than vn
327 if (n < m.vertexNumber()) {
328 // will store on this vector only the indices of the referenced vertices
329 std::vector<uint> refVertIndices(m.vertexContainerSize(), UINT_NULL);
330 // Iterate over all vertices in the mesh, and mark any unreferenced
331 // vertex as deleted.
332 for (const VertexType& v : m.vertices()) {
333 if (!referredVertices[m.index(v)]) {
334 m.deleteVertex(m.index(v));
335 }
336 else {
337 refVertIndices[m.index(v)] = m.index(v);
338 }
339 }
340
341 // update the vertex indices of the mesh, setting to null the indices of
342 // the unreferenced vertices (it may happen on adjacent vertices of some
343 // container).
344 m.updateVertexIndices(refVertIndices);
345 }
346
347 return n;
348}

368template<MeshConcept MeshType>
369uint removeDuplicatedVertices(MeshType& m)
370{
371 using VertexType = MeshType::VertexType;
372 using VertexPointer = MeshType::VertexType*;
373
374 if (m.vertexNumber() == 0)
375 return 0;
376
377 // a map that will be used to keep track of deleted vertices and their
378 // corresponding pointers.
379 std::vector<uint> newVertexIndices(m.vertexNumber());
380 // assigning each vertex index to itself.
381 std::iota(newVertexIndices.begin(), newVertexIndices.end(), 0);
382
383 uint deleted = 0;
384
385 std::vector<VertexPointer> perm(m.vertexNumber());
386
387 // put all the vertices into a vector for sorting.
388 uint k = 0;
389 for (VertexType& v : m.vertices())
390 perm[k++] = &v;
391
392 // sort the vector based on the vertices' spatial positions.
393 std::sort(
394 std::execution::par_unseq,
395 perm.begin(),
396 perm.end(),
397 internal::VertPositionComparator<VertexPointer>());
398
399 uint i = 0;
400
401 // compare the i-th position with the next ones while they are equal to the
402 // i-th.
403 while (i < perm.size() - 1) {
404 uint j = i + 1;
405 while (j < perm.size() && perm[i]->coord() == perm[j]->coord()) {
406 // j will be deleted, so we map its pointer to the i-th vertex's
407 // pointer.
408 newVertexIndices[m.index(perm[j])] = m.index(perm[i]); // map j -> i
409 m.deleteVertex(m.index(perm[j]));
410 j++;
411 deleted++;
412 }
413 // here perm[i] != perm[j], so we need to check perm[j] with the next
414 // vertex.
415 i = j;
416 }
417
418 // update the vertex pointers to point to the correct vertices, in every
419 // container of the mesh
420 m.updateVertexIndices(newVertexIndices);
421
422 // todo:
423 // - add a flag that removes degenerate elements after
424 return deleted;
425}

453template<FaceMeshConcept MeshType>
454uint removeDuplicatedFaces(MeshType& m)
455{
456 using VertexType = MeshType::VertexType;
457 using FaceType = MeshType::FaceType;
458
459 // create a vector of sorted tuples of indices, where each tuple represents
460 // a face's vertices and a pointer to the face.
461 std::vector<internal::SortedIndexContainer<
462 VertexType*,
463 FaceType*,
464 FaceType::VERTEX_NUMBER>>
465 fvec;
466
467 for (FaceType& f : m.faces()) {
468 fvec.emplace_back(&f, f.vertices());
469 }
470
471 // sort the vector based on the face vertex indices.
472 std::sort(std::execution::par_unseq, fvec.begin(), fvec.end());
473 uint total = 0;
474
475 // iterate over the sorted vector, and mark any duplicate faces as deleted.
476 for (uint i = 0; i < fvec.size() - 1; ++i) {
477 if (fvec[i] == fvec[i + 1]) {
478 total++;
479 m.deleteFace(fvec[i].sentinel());
480 }
481 }
482 return total;
483}

508template<MeshConcept MeshType>
509uint removeDegeneratedVertices(MeshType& m, bool deleteAlsoFaces)
510{
511 using VertexType = MeshType::VertexType;
512
513 int count_vd = 0;
514
515 // iterate over all vertices in the mesh, and mark any with invalid floating
516 // point values as deleted.
517 for (VertexType& v : m.vertices()) {
518 if (v.coord().isDegenerate()) {
519 count_vd++;
520 m.deleteVertex(&v);
521 }
522 }
523
524 // If the mesh has faces and the `deleteAlsoFaces` flag is true, delete all
525 // faces incident on deleted vertices.
526 if constexpr (HasFaces<MeshType>) {
527 using FaceType = MeshType::FaceType;
528 if (deleteAlsoFaces) {
529 for (FaceType& f : m.faces()) {
530 bool deg = false;
531 for (VertexType* v : f.vertices()) {
532 if (v->deleted()) {
533 deg = true;
534 }
535 }
536 if (deg) {
537 m.deleteFace(&f);
538 }
539 }
540 }
541 }
542
543 return count_vd;
544}

568template<FaceMeshConcept MeshType>
569uint removeDegenerateFaces(MeshType& m)
570{
571 uint count = 0;
572 using FaceType = MeshType::FaceType;
573
574 // iterate over all faces in the mesh, and mark any that are degenerate as
575 // deleted.
576 for (FaceType& f : m.faces()){
577 bool deg = false; // flag to check if a face is degenerate
578 for (uint i = 0; i < f.vertexNumber() && !deg; ++i){
579 if (f.vertex(i) == f.vertexMod(i+1)){
580 deg = true;
581 m.deleteFace(m.index(f));
582 count++;
583 }
584 }
585 }
586 return count;

661template<FaceMeshConcept MeshType>
662uint numberHoles(const MeshType& m)
663 requires vcl::HasPerFaceAdjacentFaces<MeshType>
664{
665 vcl::requirePerFaceAdjacentFaces(m);
666
667 using VertexType = MeshType::VertexType;
668 using FaceType = MeshType::FaceType;
669
670 uint loopNum=0;
671
672 // create a vector of bools to keep track of visited faces.
673 std::vector<bool> visitedFaces(m.faceContainerSize(), false);
674
675 // Traverse the mesh using a depth-first search algorithm to find all the
676 // holes.
677 for(const FaceType& f : m.faces()) {
678 uint e = 0;
679 for(const VertexType* v : f.vertices()) {
680 if(!visitedFaces[m.index(f)] && f.adjFace(e) == nullptr) {
681 MeshPos<FaceType> startPos(&f,e);
682 MeshPos<FaceType> curPos=startPos;
683 do {
684 curPos.nextEdgeOnBorderAdjacentToV();
685 curPos.flipVertex();
686 visitedFaces[m.index(curPos.face())] = true;
687 }
688 while(curPos!=startPos);
689 ++loopNum;
690 }
691 ++e;
692 }
693 }
694 return loopNum;
695}

719template <FaceMeshConcept MeshType>
720std::vector<std::set<uint>> connectedComponents(const MeshType& m)
721 requires vcl::HasPerFaceAdjacentFaces<MeshType>
722{
723 vcl::requirePerFaceAdjacentFaces(m);
724
725 using FaceType = MeshType::FaceType;
726
727 std::vector<std::set<uint>> cc;
728
729 // create a vector of bools to keep track of visited faces.
730 std::vector<bool> visitedFaces(m.faceContainerSize(), false);
731
732 // create a stack to hold the faces that need to be visited during the
733 // depth-first search.
734 std::stack<const FaceType*> sf;
735
736 // traverse the mesh using a depth-first search algorithm to find the
737 // connected components.
738 for (const FaceType& f : m.faces()){
739 if (!visitedFaces[m.index(f)]) { // first time I see this face
740 visitedFaces[m.index(f)] = true;
741
742 // new connected component
743 cc.emplace_back();
744 std::set<uint>& ccf = cc[cc.size()-1];
745 ccf.insert(m.index(f));
746
747 // while the stack is empty, visit the adjacent faces of the top
748 // face of the stack
749 sf.push(&f);
750 while(!sf.empty()){
751 const FaceType* fpt = sf.top();
752 // remove the top face and add it to the connected component
753 sf.pop();
754 ccf.insert(m.index(fpt));
755
756 // add the adjacent faces of the current visited in the stack
757 for (uint j = 0; j < fpt->vertexNumber(); ++j){
758 const FaceType* adjf = fpt->adjFace(j);
759 // if there is an adj face and it has not been visited
760 if (adjf != nullptr && !visitedFaces[m.index(adjf)]){
761 sf.push(adjf);
762 visitedFaces[m.index(adjf)] = true;
763 }
764 }
765 }
766 }
767 }
768 return cc;
769}
