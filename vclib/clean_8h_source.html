53class VertPositionComparator
54{
55public:
56 inline bool operator()(const VertexPointer& a, const VertexPointer& b)

62template<typename Cont, typename MeshType>
63void setReferencedVerticesOnVector(
64 const MeshType& m,
65 std::vector<bool>& refs,
66 uint& nRefs)
67{
68 // check if the Cont container of the Mesh has vertex pointers
69 if constexpr (comp::HasVertexReferences<typename Cont::ElementType>) {
70 // if there are still some vertices non-referenced
71 if (nRefs < m.vertexNumber()) {
72 constexpr uint ELEM_ID = Cont::ElementType::ELEMENT_ID;
73 // for eache element of the Cont container
74 for (const auto& el : m.template elements<ELEM_ID>()) {
75 // for each vertex of the element
76 for (uint vi : el.vertexIndices()) {
77 assert(vi != UINT_NULL);
78 if (!refs[vi]) {
79 // set the vertex as referenced
80 refs[vi] = true;
81 nRefs++;
82 }
83 }
84 }
85 }
86 }
87}

141template<typename IndexType, typename SentinelType, int N>
142class SortedIndexContainer
143{
144public:
145 SortedIndexContainer() {}
146
147 template<Range RangeType>
148 SortedIndexContainer(SentinelType s, RangeType rng) : s(s), v(rng)
149 {
150 std::sort(v.begin(), v.end());
151 }
152
153 bool operator<(const SortedIndexContainer& s) const
154 {
155 if constexpr (N >= 0) {
156 for (uint i = 0; i < N; ++i) {
157 if (v[i] != s.v[i])
158 return v[i] < s.v[i];
159 }
160 return false;
161 }
162 else {
163 for (uint i = 0; i < v.size() && i < s.v.size(); ++i) {
164 if (v[i] != s.v[i])
165 return v[i] < s.v[i];
166 }
167 return v.size() < s.v.size();
168 }
169 }
170
171 bool operator==(const SortedIndexContainer& s) const
172 {
173 if constexpr (N >= 0) {
174 for (uint i = 0; i < N; ++i) {
175 if (v[i] != s.v[i])
176 return false;
177 }
178 return true;
179 }
180 else {
181 if (v.size() != s.v.size())
182 return false;
183 for (uint i = 0; i < v.size(); ++i) {
184 if (v[i] != s.v[i])
185 return false;
186 }
187 return true;
188 }
189 }
190
191 SentinelType sentinel() const { return s; }
192
193private:
194 vcl::Vector<IndexType, N> v;
195 SentinelType s;
196};

198template<FaceMeshConcept MeshType>
199std::vector<bool> nonManifoldVerticesVectorBool(const MeshType& m)
200 requires vcl::HasPerFaceAdjacentFaces<MeshType>
201{
202 vcl::requirePerFaceAdjacentFaces(m);
203
204 using FaceType = MeshType::FaceType;
205
206 std::vector<bool> nonManifoldVertices(m.vertexContainerSize(), false);
207
208 std::vector<uint> TD(m.vertexContainerSize(), 0);
209 std::vector<bool> nonManifoldInc(m.vertexContainerSize(), false);
210 // First Loop, count how many faces are incident on a vertex and store it in
211 // TD, and flag how many vertices are incident on non manifold edges.
212 for (const FaceType& f : m.faces()) {
213 for (uint i = 0; i < f.vertexNumber(); ++i) {
214 TD[m.index(f.vertex(i))]++;
215 if (!isFaceManifoldOnEdge(f, i)) {
216 nonManifoldInc[m.index(f.vertex(i))] = true;
217 nonManifoldInc[m.index(f.vertexMod(i + 1))] = true;
218 }
219 }
220 }
221
222 std::vector<bool> visited(m.vertexContainerSize(), false);
223 for (const FaceType& f : m.faces()) {
224 for (uint i = 0; i < f.vertexNumber(); ++i) {
225 if (!visited[m.index(f.vertex(i))]) {
226 visited[m.index(f.vertex(i))] = true;
227 MeshPos pos(&f, i);
228 uint starSize = pos.numberOfAdjacentFacesToV();
229 if (starSize != TD[m.index(f.vertex(i))])
230 nonManifoldVertices[m.index(f.vertex(i))] = true;
231 }
232 }
233 }
234
235 return nonManifoldVertices;
236}

238template<FaceMeshConcept MeshType>
239uint numberEdges(
240 const MeshType& m,
241 uint& numBoundaryEdges,
242 uint& numNonManifoldEdges)
243{
244 std::vector<ConstMeshEdgeUtil<MeshType>> edgeVec =
245 fillAndSortMeshEdgeUtilVector(m);
246
247 uint numEdges = 0;
248 numBoundaryEdges = 0;
249 numNonManifoldEdges = 0;
250
251 size_t f_on_cur_edge = 1;
252 for (size_t i = 0; i < edgeVec.size(); ++i) {
253 if (((i + 1) == edgeVec.size()) || !(edgeVec[i] == edgeVec[i + 1])) {
254 ++numEdges;
255 if (f_on_cur_edge == 1)
256 ++numBoundaryEdges;
257 if (f_on_cur_edge > 2)
258 ++numNonManifoldEdges;
259 f_on_cur_edge = 1;
260 }
261 else {
262 ++f_on_cur_edge;
263 }
264 }
265 return numEdges;
266}

315template<MeshConcept MeshType>
316uint removeUnreferencedVertices(MeshType& m)
317{
318 using VertexType = MeshType::VertexType;
319
320 // Generate a vector of boolean flags indicating whether each vertex is
321 // referenced by any of the mesh's elements.
322
323 uint n = 0;
324 std::vector<bool> referredVertices =
325 detail::unreferencedVerticesVectorBool(m, n);
326
327 // need to mark as deleted vertices only if the number of unreferenced is
328 // less than vn
329 if (n < m.vertexNumber()) {
330 // will store on this vector only the indices of the referenced vertices
331 std::vector<uint> refVertIndices(m.vertexContainerSize(), UINT_NULL);
332 // Iterate over all vertices in the mesh, and mark any unreferenced
333 // vertex as deleted.
334 for (const VertexType& v : m.vertices()) {
335 if (!referredVertices[m.index(v)]) {
336 m.deleteVertex(m.index(v));
337 }
338 else {
339 refVertIndices[m.index(v)] = m.index(v);
340 }
341 }
342
343 // update the vertex indices of the mesh, setting to null the indices of
344 // the unreferenced vertices (it may happen on adjacent vertices of some
345 // container).
346 m.updateVertexIndices(refVertIndices);
347 }
348
349 return n;
350}

370template<MeshConcept MeshType>
371uint removeDuplicatedVertices(MeshType& m)
372{
373 using VertexType = MeshType::VertexType;
374 using VertexPointer = MeshType::VertexType*;
375
376 if (m.vertexNumber() == 0)
377 return 0;
378
379 // a map that will be used to keep track of deleted vertices and their
380 // corresponding pointers.
381 std::vector<uint> newVertexIndices(m.vertexNumber());
382 // assigning each vertex index to itself.
383 std::iota(newVertexIndices.begin(), newVertexIndices.end(), 0);
384
385 uint deleted = 0;
386
387 std::vector<VertexPointer> perm(m.vertexNumber());
388
389 // put all the vertices into a vector for sorting.
390 uint k = 0;
391 for (VertexType& v : m.vertices())
392 perm[k++] = &v;
393
394 // sort the vector based on the vertices' spatial positions.
395 std::sort(
396 std::execution::par_unseq,
397 perm.begin(),
398 perm.end(),
399 detail::VertPositionComparator<VertexPointer>());
400
401 uint i = 0;
402
403 // compare the i-th position with the next ones while they are equal to the
404 // i-th.
405 while (i < perm.size() - 1) {
406 uint j = i + 1;
407 while (j < perm.size() && perm[i]->coord() == perm[j]->coord()) {
408 // j will be deleted, so we map its pointer to the i-th vertex's
409 // pointer.
410 newVertexIndices[m.index(perm[j])] = m.index(perm[i]); // map j -> i
411 m.deleteVertex(m.index(perm[j]));
412 j++;
413 deleted++;
414 }
415 // here perm[i] != perm[j], so we need to check perm[j] with the next
416 // vertex.
417 i = j;
418 }
419
420 // update the vertex pointers to point to the correct vertices, in every
421 // container of the mesh
422 m.updateVertexIndices(newVertexIndices);
423
424 // todo:
425 // - add a flag that removes degenerate elements after
426 return deleted;
427}

455template<FaceMeshConcept MeshType>
456uint removeDuplicatedFaces(MeshType& m)
457{
458 using VertexType = MeshType::VertexType;
459 using FaceType = MeshType::FaceType;
460
461 // create a vector of sorted tuples of indices, where each tuple represents
462 // a face's vertices and a pointer to the face.
463 std::vector<detail::SortedIndexContainer<
464 VertexType*,
465 FaceType*,
466 FaceType::VERTEX_NUMBER>>
467 fvec;
468
469 for (FaceType& f : m.faces()) {
470 fvec.emplace_back(&f, f.vertices());
471 }
472
473 // sort the vector based on the face vertex indices.
474 std::sort(std::execution::par_unseq, fvec.begin(), fvec.end());
475 uint total = 0;
476
477 // iterate over the sorted vector, and mark any duplicate faces as deleted.
478 for (uint i = 0; i < fvec.size() - 1; ++i) {
479 if (fvec[i] == fvec[i + 1]) {
480 total++;
481 m.deleteFace(fvec[i].sentinel());
482 }
483 }
484 return total;
485}

510template<MeshConcept MeshType>
511uint removeDegeneratedVertices(MeshType& m, bool deleteAlsoFaces)
512{
513 using VertexType = MeshType::VertexType;
514
515 int count_vd = 0;
516
517 // iterate over all vertices in the mesh, and mark any with invalid floating
518 // point values as deleted.
519 for (VertexType& v : m.vertices()) {
520 if (v.coord().isDegenerate()) {
521 count_vd++;
522 m.deleteVertex(&v);
523 }
524 }
525
526 // If the mesh has faces and the `deleteAlsoFaces` flag is true, delete all
527 // faces incident on deleted vertices.
528 if constexpr (HasFaces<MeshType>) {
529 using FaceType = MeshType::FaceType;
530 if (deleteAlsoFaces) {
531 for (FaceType& f : m.faces()) {
532 bool deg = false;
533 for (VertexType* v : f.vertices()) {
534 if (v->deleted()) {
535 deg = true;
536 }
537 }
538 if (deg) {
539 m.deleteFace(&f);
540 }
541 }
542 }
543 }
544
545 return count_vd;
546}

570template<FaceMeshConcept MeshType>
571uint removeDegenerateFaces(MeshType& m)
572{
573 uint count = 0;
574 using FaceType = MeshType::FaceType;
575
576 // iterate over all faces in the mesh, and mark any that are degenerate as
577 // deleted.
578 for (FaceType& f : m.faces()) {
579 bool deg = false; // flag to check if a face is degenerate
580 for (uint i = 0; i < f.vertexNumber() && !deg; ++i) {
581 if (f.vertex(i) == f.vertexMod(i + 1)) {
582 deg = true;
583 m.deleteFace(m.index(f));
584 count++;
585 }
586 }
587 }
588 return count;

663template<FaceMeshConcept MeshType>
664uint numberHoles(const MeshType& m)
665 requires vcl::HasPerFaceAdjacentFaces<MeshType>
666{
667 vcl::requirePerFaceAdjacentFaces(m);
668
669 using VertexType = MeshType::VertexType;
670 using FaceType = MeshType::FaceType;
671
672 uint loopNum = 0;
673
674 // create a vector of bools to keep track of visited faces.
675 std::vector<bool> visitedFaces(m.faceContainerSize(), false);
676
677 // Traverse the mesh using a depth-first search algorithm to find all the
678 // holes.
679 for (const FaceType& f : m.faces()) {
680 uint e = 0;
681 for (const VertexType* v : f.vertices()) {
682 if (!visitedFaces[m.index(f)] && f.adjFace(e) == nullptr) {
683 MeshPos<FaceType> startPos(&f, e);
684 MeshPos<FaceType> curPos = startPos;
685 do {
686 curPos.nextEdgeOnBorderAdjacentToV();
687 curPos.flipVertex();
688 visitedFaces[m.index(curPos.face())] = true;
689 } while (curPos != startPos);
690 ++loopNum;
691 }
692 ++e;
693 }
694 }
695 return loopNum;
696}

720template<FaceMeshConcept MeshType>
721std::vector<std::set<uint>> connectedComponents(const MeshType& m)
722 requires vcl::HasPerFaceAdjacentFaces<MeshType>
723{
724 vcl::requirePerFaceAdjacentFaces(m);
725
726 using FaceType = MeshType::FaceType;
727
728 std::vector<std::set<uint>> cc;
729
730 // create a vector of bools to keep track of visited faces.
731 std::vector<bool> visitedFaces(m.faceContainerSize(), false);
732
733 // create a stack to hold the faces that need to be visited during the
734 // depth-first search.
735 std::stack<const FaceType*> sf;
736
737 // traverse the mesh using a depth-first search algorithm to find the
738 // connected components.
739 for (const FaceType& f : m.faces()) {
740 if (!visitedFaces[m.index(f)]) { // first time I see this face
741 visitedFaces[m.index(f)] = true;
742
743 // new connected component
744 cc.emplace_back();
745 std::set<uint>& ccf = cc[cc.size() - 1];
746 ccf.insert(m.index(f));
747
748 // while the stack is empty, visit the adjacent faces of the top
749 // face of the stack
750 sf.push(&f);
751 while (!sf.empty()) {
752 const FaceType* fpt = sf.top();
753 // remove the top face and add it to the connected component
754 sf.pop();
755 ccf.insert(m.index(fpt));
756
757 // add the adjacent faces of the current visited in the stack
758 for (uint j = 0; j < fpt->vertexNumber(); ++j) {
759 const FaceType* adjf = fpt->adjFace(j);
760 // if there is an adj face and it has not been visited
761 if (adjf != nullptr && !visitedFaces[m.index(adjf)]) {
762 sf.push(adjf);
763 visitedFaces[m.index(adjf)] = true;
764 }
765 }
766 }
767 }
768 }
769 return cc;
770}
