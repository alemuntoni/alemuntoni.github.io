36/*
37 * Developer Documentation
38 * A class that derives from an AbstractGrid must store, in some way, an
39 * association between a GridCell coordinate (a Point of unsigned integers) and
40 * elements of type ValueType. Each cell may contain more than one element
41 * value, and each element value may be stored in more than one cell if it
42 * intersects with more than one cell.
43 *
44 * This class is made using CRTP static polymorphism
45 * (https: *en.wikipedia.org/wiki/Curiously_recurring_template_pattern).
46 * The last template parameter, is the Derived class that derives from this
47 * grid. This allows to avoid calling virtual member functions, which would be
48 * called very often and may cause a significant overhead during queries.
49 *
50 * What you need to do is to declared your Grid Data Structure as follows:
51 *
52 * template<typename GridType, typename ValueType>
53 * class MyDSGrid :
54 * public AbstractGrid<GridType, ValueType, MyDSGrid<GridType, ValueType>>
55 * { ... };
56 *
57 * Make sure that your class:
58 * - declares the AbstractGrid class as friend (if one of the following member
59 * functions is
60 * protected/private);
61 * - implements iterators Iterator and ConstIterator that iterate over pairs of
62 * <KeyType, ValueType> (see below for more details)
63 * - implements the following member functions:
64 * - std::pair<Iterator, Iterator> valuesInCell(const KeyType) -> returns a
65 * pair of begin-end iterators that will allow to cycle over the elements
66 * contained in a cell. this function must be overloaded both in const and
67 * non-const versions, returning proper const or non-const iterators.
68 * - bool insertInCell(const KeyType&, const ValueType&) -> inserts the
69 * element in a Grid Cell. The returned type tells if the element has been
70 * inserted (in some implementations, insertion could not happen, e.g. when
71 * duplicates are not allowed).
72 * - bool eraseInCell(const KeyType&, const ValueType&) -> erases all the
73 * element equal to the given element from the given Grid Cell. The returned
74 * type tells if the element has been erased from the given cell.
75 * - Iterator end() -> classic member function that returns the end iterator
76 *
77 * Iterators of your class must:
78 * - Iterate over pairs of <KeyType, ValueType&>:
79 * - operator*() returns a pair having first member that is the key, and
80 * second member that is a reference to the value
81 * - operator->() return a pointer to the same pair
82 * - operator+() and operator+(int), that moves the iterator in the same cell if
83 * there are more values in it, or moves in the next cell-element
84 *
85 * You can reimplement in your derived class all the member functions
86 * implemented in this class that may take advantage of a speedup depending on
87 * the internal storage. You can also hide all the member functions implemented
88 * in this class that do not apply to your derived class.
89 */

183 bool insert(const ValueType& v)
184 {
185 const VT* vv = addressOfObj(v);
186
187 // if vv is a valid pointer (ValueType, or ValueType* if ValueType is
188 // not a pointer)
189 if (vv) {
190 // first and last cell where insert (could be the same)
191 KeyType bmin, bmax;
192
193 // if ValueType is Point, Point*, Vertex, Vertex*
194 if constexpr (PointConcept<VT> || VertexConcept<VT>) {
195 typename GridType::PointType p;
196 if constexpr (PointConcept<VT>)
197 p = *vv;
198 else
199 p = vv->coord();
200 bmin = bmax = GridType::cell(p);
201 }
202 else { // else, call the boundingBox function
203 // bounding box of value
204 typename GridType::BBoxType bb = vcl::boundingBox(*vv);
205
206 bmin = GridType::cell(bb.min()); // first cell where insert
207 bmax = GridType::cell(bb.max()); // last cell where insert
208 }
209
210 bool ins = false;
211
212 // custom intersection function between cell and value
213 if (mIntersectsFun) {
214 for (const auto& cell : GridType::cells(bmin, bmax)) {
215 if (mIntersectsFun(
216 GridType::cellBox(cell), dereferencePtr(v)))
217 {
218 ins |= static_cast<DerivedGrid*>(this)->insertInCell(
219 cell, v);
220 }
221 }
222 }
223 else {
224 for (const auto& cell : GridType::cells(bmin, bmax)) {
225 ins |=
226 static_cast<DerivedGrid*>(this)->insertInCell(cell, v);
227 }
228 }
229 return ins;
230 }
231 return false;
232 }

264 bool erase(const ValueType& v)
265 {
266 const VT* vv = addressOfObj(v);
267
268 if (vv) {
269 // first and last cell where erase (could be the same)
270 KeyType bmin, bmax;
271
272 if constexpr (PointConcept<VT> || VertexConcept<VT>) {
273 typename GridType::PointType p;
274 if constexpr (PointConcept<VT>)
275 p = *vv;
276 else
277 p = vv->coord();
278 bmin = bmax = GridType::cell(p);
279 }
280 else {
281 typename GridType::BBoxType bb = vcl::boundingBox(*vv);
282
283 bmin = GridType::cell(bb.min);
284 bmax = GridType::cell(bb.max);
285 }
286
287 bool found = false;
288 for (const auto& cell : GridType::cells(bmin, bmax)) {
289 found |= static_cast<DerivedGrid*>(this)->eraseInCell(cell, v);
290 }
291 return found;
292 }
293 return false;
294 }

313 // vector of iterators - return type must be auto here (we still don't know
314 // the iterator type)
315 auto valuesInSphere(const Sphere<typename GridType::ScalarType>& s) const
316 {
317 using Iter = DerivedGrid::ConstIterator;
318 using IterSet = std::set<Iter, IterComparator<Iter>>;
319
320 // will use this set only if the value type is not a point -- that is
321 // when the value can occupy more than one single cell. We therefore
322 // push the results in a set in order to avoid possible duplicates
323 IterSet valuesSet;
324
325 std::vector<Iter> resVec;
326
327 // interval of cells containing the sphere
328 KeyType first = GridType::cell(s.center() - s.radius());
329 KeyType last = GridType::cell(s.center() + s.radius());
330
331 // for each cell in the interval
332 for (const KeyType& c : GridType::cells(first, last)) {
333 // p is a pair of iterators
334 const auto& p =
335 static_cast<const DerivedGrid*>(this)->valuesInCell(c);
336 // for each value contained in the cell
337 for (auto it = p.first; it != p.second; ++it) {
338 if (valueIsInSpehere(it, s)) {
339 if constexpr (!PointConcept<VT> && !VertexConcept<VT>) {
340 valuesSet.insert(it);
341 }
342 else {
343 resVec.push_back(it);
344 }
345 }
346 }
347 }
348
349 // if the valuetype is a point (or vertex), we already pushed in resVec
350 // - faster than using set
351 if constexpr (!PointConcept<VT> && !VertexConcept<VT>) {
352 resVec = std::vector<Iter>(valuesSet.begin(), valuesSet.end());
353 }
354 return resVec;
355 }

357 void eraseInSphere(const Sphere<typename GridType::ScalarType>& s)
358 {
359 // interval of cells containing the sphere
360 KeyType first = GridType::cell(s.center() - s.radius());
361 KeyType last = GridType::cell(s.center() + s.radius());
362
363 // for each cell in the interval
364 for (const KeyType& c : GridType::cells(first, last)) {
365 // p is a pair of iterators
366 const auto& p =
367 static_cast<const DerivedGrid*>(this)->valuesInCell(c);
368 // for each value contained in the cell
369 for (auto it = p.first; it != p.second; ++it) {

377 // closest queries
378 template<typename QueryValueType>
379 auto closestValue(
380 const QueryValueType& qv,
381 QueryBoundedDistFunction<QueryValueType> distFunction,
382 typename GridType::ScalarType& dist) const
383 {
384 using ScalarType = GridType::ScalarType;
385 using PointType = GridType::PointType;
386 using ResType = DerivedGrid::ConstIterator;
387
388 using QVT = RemoveCVRefAndPointer<QueryValueType>;
389 const QVT* qvv = addressOfObj(qv);
390 ResType result = static_cast<const DerivedGrid*>(this)->end();
391
392 if (qvv) {
393 typename GridType::ScalarType maxDist = dist;
394
395 const ScalarType cellDiag = GridType::cellDiagonal();
396
397 ScalarType centerDist = cellDiag;
398 PointType center = vcl::boundingBox(*qvv).center();
399
400 // we first look just on the cells where the query value lies
401 // here, we will store also the looking interval where we need to
402 // look
403 Boxui currentIntervalBox;
404
405 // bbox of query value
406 typename GridType::BBoxType bb = vcl::boundingBox(*qvv);
407 // first cell where look for closest
408 currentIntervalBox.add(GridType::cell(bb.min()));
409 // last cell where look for closest
410 currentIntervalBox.add(GridType::cell(bb.max()));
411
412 // looking just on cells where query lies
413 ScalarType tmp = cellDiag;
414 result = closestInCells(qv, tmp, currentIntervalBox, distFunction);
415
416 // we have found (maybe) the closest value contained in the cell(s)
417 // where the query value lies (if the cells were empty, we did not
418 // found nothing). we now have to expand the search on adjacent
419 // cells starting from the center of the query value and expanding
420 // until we find a value
421
422 // now we start with the actual search, including all the cells in
423 // the current interval
424
425 if (result != static_cast<const DerivedGrid*>(this)->end()) {
426 dist = tmp;
427 centerDist = dist;
428 }
429
430 bool end = false;
431
432 do {
433 Boxui lastIntervalBox = currentIntervalBox;
434 currentIntervalBox.add(GridType::cell(center - centerDist));
435 currentIntervalBox.add(GridType::cell(center + centerDist));
436
437 ScalarType tmp = centerDist;
438 ResType winner = closestInCells(
439 qv, tmp, currentIntervalBox, distFunction, lastIntervalBox);
440
441 if (winner != static_cast<const DerivedGrid*>(this)->end()) {
442 result = winner;
443 dist = tmp;
444 }
445
446 centerDist += cellDiag;
447 end = result != static_cast<const DerivedGrid*>(this)->end();
448 end |= (centerDist > maxDist);
449 end |=
450 (center - centerDist < GridType::min() &&
451 center + centerDist > GridType::max());
452 } while (!end);
453 }
454
455 return result;
456 }

510 template<typename QueryValueType>
511 auto kClosestValues(
512 const QueryValueType& qv,
513 uint n,
514 QueryDistFunction<QueryValueType> distFunction) const
515 {
516 using ResType = std::vector<typename DerivedGrid::ConstIterator>;
517 // KClosest Types
518 using KClosestPairType = std::pair<
519 typename GridType::ScalarType,
520 typename DerivedGrid::ConstIterator>;
521 using KClosestSet = std::
522 set<KClosestPairType, DistIterPairComparator<KClosestPairType>>;
523
524 Boxui ignore; // will contain the interval of cells already visited
525
526 KClosestSet set = valuesInCellNeighborhood(qv, n, distFunction, ignore);
527
528 auto it = set.size() >= n ? std::next(set.begin(), n) : set.end();
529 // if we didn't found n values, it means that there aren't n values in
530 // the grid - nothing to do
531 if (it != set.end()) {
532 using QVT = RemoveCVRefAndPointer<QueryValueType>;
533 const QVT* qvv = addressOfObj(qv);
534
535 // bbox of query value
536 typename GridType::BBoxType bb = vcl::boundingBox(*qvv);
537 // we need to be sure that there are no values that are closest
538 // w.r.t. the n-th that we have already found by looking in the cell
539 // neighborhood we extend the bb with the distance of the n-th
540 // closest found value
541 bb.min() -= it->first;
542 bb.max() += it->first;
543
544 // and we look in all of these cells
545 Boxui currentIntervalBox;
546 // first cell where look for closest
547 currentIntervalBox.add(GridType::cell(bb.min()));
548 // last cell where look for closest
549 currentIntervalBox.add(GridType::cell(bb.max()));
550
551 // for all the cells in the current interval box
552 for (const KeyType& c : GridType::cells(
553 currentIntervalBox.min(), currentIntervalBox.max()))
554 {
555 if (!ignore.isInsideOpenBox(c)) {
556 // get the values of the cell c
557 const auto& p =
558 static_cast<const DerivedGrid*>(this)->valuesInCell(c);
559 // for each value contained in the cell c
560 for (auto it = p.first; it != p.second; ++it) {
561 auto tmp = distFunction(qv, dereferencePtr(it->second));
562 set.insert(std::make_pair(tmp, it));
563 }
564 }
565 }
566 }
567
568 ResType vec;
569 // if there are more than n values in the set, we will return n values,
570 // otherwise set.size()
571 uint retNValues = std::min((uint) set.size(), n);
572 vec.reserve(retNValues);
573 it = set.begin();
574 for (uint i = 0; i < retNValues; i++) {
575 vec.push_back(it->second);
576 it++;
577 }
578 return vec;
579 }

762
767 template<typename Iterator>
768 bool valueIsInSpehere(
769 const Iterator& it,
770 const Sphere<typename GridType::ScalarType>& s) const
771 {
772 const VT* vv = addressOfObj(it->second);
773
774 bool test = false;
775 if constexpr (PointConcept<VT> || VertexConcept<VT>) {
776 typename GridType::PointType p;
777 if constexpr (PointConcept<VT>)
778 p = *vv;
779 else
780 p = vv->coord();
781 test = vv && s.isInside(p);
782 }
783 else { // check if the bbox of the value intersects the sphere
784 test = vv && s.intersects(vcl::boundingBox(*vv));
785 }
786 return test;
787 }

794 template<typename QueryValueType>
795 auto closestInCells(
796 const QueryValueType& qv,
797 typename GridType::ScalarType& dist,
798 const Boxui& interval,
799 QueryBoundedDistFunction<QueryValueType> distFunction,
800 const Boxui& ignore = Boxui()) const
801 {
802 using ResType = DerivedGrid::ConstIterator;
803 ResType res = static_cast<const DerivedGrid*>(this)->end();
804
805 // for each cell in the interval
806 for (const KeyType& c : GridType::cells(interval.min(), interval.max()))
807 {
808 if (!ignore.isInsideOpenBox(c)) {
809 // p is a pair of iterators
810 const auto& p =
811 static_cast<const DerivedGrid*>(this)->valuesInCell(c);
812 // for each value contained in the cell
813 for (auto it = p.first; it != p.second; ++it) {
814 auto tmp =
815 distFunction(qv, dereferencePtr(it->second), dist);
816 if (tmp < dist) {
817 dist = tmp;
818 res = it;
819 }
820 }
821 }
822 }
823 return res;
824 }

826 template<typename QueryValueType>
827 auto valuesInCellNeighborhood(
828 const QueryValueType& qv,
829 uint n,
830 QueryDistFunction<QueryValueType> distFunction,
831 Boxui& ignore) const
832 {
833 // types used for K closest neighbors queries
834 using KClosestPairType = std::pair<
835 typename GridType::ScalarType,
836 typename DerivedGrid::ConstIterator>;
837 using KClosestSet = std::
838 set<KClosestPairType, DistIterPairComparator<KClosestPairType>>;
839
840 KClosestSet res;
841
842 using QVT = RemoveCVRefAndPointer<QueryValueType>;
843 const QVT* qvv = addressOfObj(qv);
844
845 if (qvv) {
846 Boxui currentIntervalBox;
847 // bbox of query value
848 typename GridType::BBoxType bb = vcl::boundingBox(*qvv);
849 // first cell where look for closest
850 currentIntervalBox.add(GridType::cell(bb.min()));
851 // last cell where look for closest
852 currentIntervalBox.add(GridType::cell(bb.max()));
853
854 ignore.setNull();
855 while (res.size() < n && currentIntervalBox != ignore) {
856 // for each cell in the interval
857 for (const KeyType& c : GridType::cells(
858 currentIntervalBox.min(), currentIntervalBox.max()))
859 {
860 if (!ignore.isInsideOpenBox(c)) {
861 const auto& p =
862 static_cast<const DerivedGrid*>(this)->valuesInCell(
863 c);
864
865 // for each value contained in the cell
866 for (auto it = p.first; it != p.second; ++it) {
867 auto tmp =
868 distFunction(qv, dereferencePtr(it->second));
869 res.emplace(tmp, it);
870 }
871 }
872 }
873 ignore = currentIntervalBox;
874 for (uint i = 0; i < currentIntervalBox.min().DIM; ++i) {
875 if (currentIntervalBox.min()(i) != 0)
876 currentIntervalBox.min()(i)--;
877 if (currentIntervalBox.max()(i) != GridType::cellNumber(i))
878 currentIntervalBox.max()(i)++;
879 }
880 }
881 }
882
883 return res;
884 }
