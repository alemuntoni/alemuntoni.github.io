110 template<typename Iterator>
111 static PointT normal(Iterator begin, Iterator end) requires (
112 std::is_same_v<typename Iterator::value_type, PointT> &&
113 PointT::DIM == 3)
114 {
115 // compute the sum of normals for each triplet of consecutive points
116 PointT sum;
117 sum.setZero();
118 for (auto i = begin; i != end; ++i) {
119 auto i1 = i; ++i1;
120 if (i1 == end) i1 = begin;
121
122 auto i2 = i1; ++i2;
123 if (i2 == end) i2 = begin;
124
125 sum += (*i1 - *i).cross(*i2 - *i);
126 }
127 sum.normalize();
128 return sum;
129 }

149 template<typename Iterator>
150 static PointT barycenter(Iterator begin, Iterator end)
151 requires (std::is_same_v<typename Iterator::value_type, PointT>)
152 {
153 PointT bar;
154 bar.setZero();
155
156 uint cnt = 0;
157 for (; begin != end; ++begin) {
158 bar += *begin;
159 ++cnt;
160 }
161
162 assert(cnt);
163
164 return bar / cnt;
165 }

190 template<typename Iterator, typename WIterator>
191 static PointT
192 weightedBarycenter(Iterator begin, Iterator end, WIterator wbegin)
193 requires (std::is_same_v<typename Iterator::value_type, PointT>)
194 {
195 using ScalarType = WIterator::value_type;
196
197 PointT bar;
198 bar.setZero();
199 ScalarType wsum = 0;
200
201 for (; begin != end; ++begin, ++wbegin){
202 bar += *begin * *wbegin;
203 wsum += *wbegin;
204 }
205
206 return bar / wsum;
207 }

237 template<typename Iterator>
238 static ScalarType perimeter(Iterator begin, Iterator end)
239 requires (std::is_same_v<typename Iterator::value_type, PointT>)
240 {
241 using Scalar = PointType::ScalarType;
242
243 Scalar per = 0;
244 for (auto i = begin; i != end; ++i){
245 const PointT& p0 = *i;
246 auto i1 = i; ++i;
247 if (i1 == end) i1 = begin;

276 template<typename Iterator>
277 static ScalarType area(Iterator begin, Iterator end)
278 requires (std::is_same_v<typename Iterator::value_type, PointT>)
279 {
280 using Scalar = PointType::ScalarType;
281
282 PointType bar = barycenter(begin, end);
283 Scalar area = 0;
284 for (auto i = begin; i != end; ++i){
285 const PointT& p0 = *i;
286 auto i1 = i; ++i;
287 if (i1 == end) i1 = begin;
288 const PointT& p1 = *i1;
289
290 area += Triangle<PointT>::area(p0, p1, bar);
291 }
292 return area;
293 }

330 template<typename Iterator>
331 static std::vector<uint> earCut(Iterator begin, Iterator end) requires (
332 std::is_same_v<typename Iterator::value_type, PointT> &&
333 PointT::DIM == 2)
334 {
335 using Scalar = PointT::ScalarType;
336
337 // Convert the polygon represented as a sequence of vertices to a vector
338 // of vectors of points. This is necessary because the earcut library
339 // expects the polygon to be represented as a vector of contours, where
340 // each contour is a vector of points. In this case, there is only one
341 // contour, which is represented as a vector of points.
342 std::vector<std::vector<Point2<Scalar>>> poly;
343 poly.emplace_back(begin, end);
344
345 // Use the earcut library to triangulate the polygon and return the
346 // result.
347 return mapbox::earcut<uint>(poly);
348 }
349
383 template<typename Iterator>
384 static std::vector<uint> earCut(Iterator begin, Iterator end) requires (
385 std::is_same_v<typename Iterator::value_type, PointT> &&
386 PointT::DIM == 3)
387 {
388 using Scalar = PointType::ScalarType;
389
390 // Calculate the normal vector of the polygon and an orthonormal basis
391 // for the plane containing the polygon.
392 PointType n = normal(begin, end);
393 PointType u, v;
394 n.orthoBase(u, v);
395
396 // Project each vertex onto the plane defined by the orthonormal basis.
397 std::vector<Point2<Scalar>> poly2D;
398
399 if constexpr (std::is_same_v<
400 typename Iterator::iterator_category,
401 std::random_access_iterator_tag>)
402 {
403 poly2D.reserve(std::distance(begin, end));
404 }
405
406 for (auto i = begin; i != end; ++i){
407 // project i-th polygon in a 2D plane
408 poly2D.emplace_back(*i*u, *i*v);
409 }
410
411 // Use the ear-cut algorithm to triangulate the polygon in the 2D plane
412 // and return the result.
413 return Polygon<Point2<Scalar>>::earCut(poly2D.begin(), poly2D.end());
414 }
415
