418 void importPointersFrom(
419 const Element& e,
420 Vertex* base,
421 const ElVType* ebase)
422 {
423 if constexpr (HasVertexPointers<Element>) {
424 if constexpr(N > 0) {
425 // same size non-polygonal faces
426 if constexpr (N == Element::VERTEX_NUMBER) {
427 importPtrsFrom(e, base, ebase);
428 }
429 // from polygonal to fixed size, but the polygon size == the
430 // fixed face size
431 else if constexpr (Element::VERTEX_NUMBER < 0){
432 if (e.vertexNumber() == N) {
433 importPtrsFrom(e, base, ebase);
434 }
435 }
436 else {
437 // do not import in this case: cannot import from a face of
438 // different size
439 }
440 }
441 else {
442 // from fixed to polygonal size: need to resize first, then
443 // import
444 resizeVertices(e.vertexNumber());
445 importPtrsFrom(e, base, ebase);
446 }
447 }
448 }
449
450private:
451 template<typename Element, typename ElVType>
452 void importPtrsFrom(const Element& e, Vertex* base, const ElVType* ebase)
453 {
454 if (ebase != nullptr && base != nullptr) {
455 for (uint i = 0; i < e.vertexNumber(); ++i){
456 if (e.vertex(i) != nullptr){
457 vertex(i) = base + (e.vertex(i) - ebase);
458 }
459 }
460 }
