71inline void drawSphere(
72 const Point3d& center,
73 float radius,
74 const Color& color,
75 int precision = 4)
76{
77 glEnable(GL_LIGHTING);
78 glShadeModel(GL_SMOOTH);
79 glColor3f(color.redF(), color.greenF(), color.blueF());
80 glPushMatrix();
81 glTranslated(center.x(), center.y(), center.z());
82 GLUquadric* sphere = gluNewQuadric();
83 gluQuadricNormals(sphere, GLU_SMOOTH);
84 gluQuadricOrientation(sphere, GLU_OUTSIDE);
85 gluSphere(sphere, radius, precision, precision);
86 glPopMatrix();
87}

99inline void drawCylinder(
100 const Point3d& a,
101 const Point3d& b,
102 float top_radius,
103 float bottom_radius,
104 const Color& color,
105 unsigned int slices = 50,
106 unsigned int stacks = 10)
107{
108 Point3d dir = b - a;
109 dir.normalize();
110 Point3d z = Point3d(0, 0, 1);
111 Point3d normal = dir.cross(z);
112 double angle = acos(dir.dot(z)) * 180 / M_PI;
113
114 bool cull = glIsEnabled(GL_CULL_FACE);
115 glDisable(GL_CULL_FACE);
116 glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
117 // cylinder
118 glColor3f(color.redF(), color.greenF(), color.blueF());
119 glPushMatrix();
120 glTranslated(a.x(), a.y(), a.z());
121 glRotatef(-angle, normal.x(), normal.y(), normal.z());
122
123 GLUquadric* cylinder = gluNewQuadric();
124 gluQuadricNormals(cylinder, GLU_SMOOTH);
125 gluQuadricOrientation(cylinder, GLU_OUTSIDE);
126 gluCylinder(
127 cylinder, top_radius, bottom_radius, (a - b).norm(), slices, stacks);
128
129 GLUquadric* disk1 = gluNewQuadric();
130 gluQuadricNormals(disk1, GLU_SMOOTH);
131 gluQuadricOrientation(disk1, GLU_INSIDE);
132 gluDisk(disk1, 0, top_radius, slices, stacks);
133
134 /*glTranslated((b-a).x(), (b-a).y(), (b-a).z());
135 GLUquadric *disk2 = gluNewQuadric();
136 gluQuadricNormals(disk2, GLU_SMOOTH);
137 gluQuadricOrientation(disk2, GLU_OUTSIDE);
138 gluDisk(disk2, 0, bottom_radius, slices, stacks);*/
139
140 glColor3f(0, 0, 0);
141 glPopMatrix();
142 if (cull)
143 glEnable(GL_CULL_FACE);
144}

228inline void drawTriangle(
229 const Point3d& p1,
230 const Point3d& p2,
231 const Point3d& p3,
232 const Color& c,
233 int width = 3,
234 bool fill = false)
235{
236 if (width != 0) {
237 vcl::drawLine3(p1, p2, c, width);
238 vcl::drawLine3(p2, p3, c, width);
239 vcl::drawLine3(p3, p1, c, width);
240 }
241 if (fill) {
242 glBegin(GL_TRIANGLES); // Begin triangle coordinates
243 glVertex3f(p1.x(), p1.y(), p1.z());
244 glVertex3f(p2.x(), p2.y(), p2.z());
245 glVertex3f(p3.x(), p3.y(), p3.z());
246 glEnd(); // End triangle coordinates
247 }
248}
