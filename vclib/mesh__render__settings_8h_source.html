475 template<MeshConcept MeshType>
476 void setRenderCapabilityFrom(const MeshType& m)
477 {
478 mDrawModeCapability0 = 0;
479
480 if (m.vertexNumber() > 0) {
481 mDrawModeCapability0 |= VCL_MRS_DRAW_MESH;
482
483 // -- Points --
484 mDrawModeCapability0 |= VCL_MRS_DRAW_POINTS;
485 mDrawModeCapability0 |= VCL_MRS_POINTS_SHADING_NONE;
486 mDrawModeCapability0 |= VCL_MRS_POINTS_PIXEL;
487 mDrawModeCapability0 |= VCL_MRS_POINTS_CIRCLE;
488 mDrawModeCapability0 |= VCL_MRS_POINTS_COLOR_USER;
489
490 if constexpr (vcl::HasPerVertexNormal<MeshType>) {
491 if (vcl::isPerVertexNormalAvailable(m)) {
492 mDrawModeCapability0 |= VCL_MRS_POINTS_SHADING_VERT;
493 }
494 }
495
496 if constexpr (vcl::HasPerVertexColor<MeshType>) {
497 if (vcl::isPerVertexColorAvailable(m)) {
498 mDrawModeCapability0 |= VCL_MRS_POINTS_COLOR_VERTEX;
499 }
500 }
501
502 if constexpr (vcl::HasColor<MeshType>) {
503 mDrawModeCapability0 |= VCL_MRS_POINTS_COLOR_MESH;
504 }
505
506 // -- Surface and Wireframe --
507 if constexpr (vcl::HasFaces<MeshType>) {
508 if (m.faceNumber() > 0) {
509 mDrawModeCapability0 |= VCL_MRS_DRAW_SURF;
510 mDrawModeCapability0 |= VCL_MRS_SURF_SHADING_NONE;
511 mDrawModeCapability0 |= VCL_MRS_SURF_COLOR_USER;
512 mDrawModeCapability0 |= VCL_MRS_DRAW_WIREFRAME;
513 mDrawModeCapability0 |= VCL_MRS_WIREFRAME_SHADING_NONE;
514 mDrawModeCapability0 |= VCL_MRS_WIREFRAME_COLOR_USER;
515
516 if constexpr (vcl::HasColor<MeshType>) {
517 mDrawModeCapability0 |= VCL_MRS_SURF_COLOR_MESH;
518 mDrawModeCapability0 |= VCL_MRS_WIREFRAME_COLOR_MESH;
519 }
520
521 if constexpr (vcl::HasPerFaceNormal<MeshType>) {
522 if (vcl::isPerFaceNormalAvailable(m)) {
523 mDrawModeCapability0 |= VCL_MRS_SURF_SHADING_FLAT;
524 }
525 }
526
527 if constexpr (vcl::HasPerVertexNormal<MeshType>) {
528 if (vcl::isPerVertexNormalAvailable(m)) {
529 mDrawModeCapability0 |= VCL_MRS_SURF_SHADING_SMOOTH;
530 mDrawModeCapability0 |=
531 VCL_MRS_WIREFRAME_SHADING_VERT;
532 }
533 }
534
535 if constexpr (vcl::HasPerFaceColor<MeshType>) {
536 if (vcl::isPerFaceColorAvailable(m))
537 mDrawModeCapability0 |= VCL_MRS_SURF_COLOR_FACE;
538 }
539
540 if constexpr (vcl::HasPerVertexColor<MeshType>) {
541 if (vcl::isPerVertexColorAvailable(m)) {
542 mDrawModeCapability0 |= VCL_MRS_SURF_COLOR_VERTEX;
543 mDrawModeCapability0 |=
544 VCL_MRS_WIREFRAME_COLOR_VERT;
545 }
546 }
547
548 if constexpr (vcl::HasTexturePaths<MeshType>) {
549 if constexpr (vcl::HasPerVertexTexCoord<MeshType>) {
550 if (vcl::isPerVertexTexCoordAvailable(m))
551 mDrawModeCapability0 |= VCL_MRS_SURF_TEX_VERTEX;
552 }
553
554 if constexpr (vcl::HasPerFaceWedgeTexCoords<MeshType>) {
555 if (vcl::isPerFaceWedgeTexCoordsAvailable(m))
556 mDrawModeCapability0 |= VCL_MRS_SURF_TEX_WEDGE;
557 }
558 }
559 }
560 }
561
562 // -- Bounding Box --
563 if constexpr (vcl::HasBoundingBox<MeshType>) {
564 mDrawModeCapability0 |= VCL_MRS_DRAW_BOUNDINGBOX;
565 }
566
567 // -- Edges --
568 if constexpr (vcl::HasEdges<MeshType>) {
569 if (m.edgeNumber() > 0) {
570 mDrawModeCapability1 |= VCL_MRS_DRAW_EDGES;
571 mDrawModeCapability1 |= VCL_MRS_EDGES_SHADING_NONE;
572 mDrawModeCapability1 |= VCL_MRS_EDGES_COLOR_USER;
573
574 if constexpr (vcl::HasColor<MeshType>) {
575 mDrawModeCapability1 |= VCL_MRS_EDGES_COLOR_MESH;
576 }
577
578 if constexpr (vcl::HasPerVertexNormal<MeshType>) {
579 if (vcl::isPerVertexNormalAvailable(m)) {
580 mDrawModeCapability1 |=
581 VCL_MRS_EDGES_SHADING_SMOOTH;
582 }
583 }
584
585 if constexpr (vcl::HasPerEdgeNormal<MeshType>) {
586 if (vcl::isPerEdgeNormalAvailable(m)) {
587 mDrawModeCapability1 |= VCL_MRS_EDGES_SHADING_FLAT;
588 }
589 }
590
591 if constexpr (vcl::HasPerEdgeColor<MeshType>) {
592 if (vcl::isPerEdgeColorAvailable(m))
593 mDrawModeCapability1 |= VCL_MRS_EDGES_COLOR_EDGE;
594 }
595
596 if constexpr (vcl::HasPerVertexColor<MeshType>) {
597 if (vcl::isPerVertexColorAvailable(m)) {
598 mDrawModeCapability1 |= VCL_MRS_EDGES_COLOR_VERTEX;
599 }
600 }
601 }
602 }
603 }
604
605 // make sure that the previous draw mode satisfies the new capabilites
606 mDrawMode0 &= mDrawModeCapability0;
607 mDrawMode1 &= mDrawModeCapability1;
608 }
609
610 void setDefaultSettingsFromCapability();
611};
