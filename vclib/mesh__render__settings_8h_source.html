59class MeshRenderSettings
60{
61 enum {
62 DRAW_MESH = 1 << 0, // mesh visibility
63
64 // points
65 DRAW_POINTS = 1 << 1, // point visibility
66 DRAW_POINTS_PIXEL = 1 << 2, // draw points as pixels
67 DRAW_POINTS_CIRCLE = 1 << 3, // draw points as circles
68 DRAW_POINTS_COLOR_VERTEX = 1 << 4, // use vertex colors for points
69 DRAW_POINTS_COLOR_MESH = 1 << 5, // use mesh color for points
70 DRAW_POINTS_COLOR_USER = 1 << 6, // use user color for points
71
72 // surface
73 DRAW_SURF = 1 << 10, // surface visibility
74 DRAW_SURF_FLAT = 1 << 11, // flat shading
75 DRAW_SURF_SMOOTH = 1 << 12, // smooth shading
76 DRAW_SURF_COLOR_FACE = 1 << 13, // use face color for surface
77 DRAW_SURF_COLOR_VERTEX = 1 << 14, // use vertex color for surface
78 DRAW_SURF_COLOR_MESH = 1 << 15, // use mesh color for surface
79 DRAW_SURF_COLOR_USER = 1 << 16, // use user color for surface
80 DRAW_SURF_TEX_VERTEX = 1 << 17, // use per vertex texcoords
81 DRAW_SURF_TEX_WEDGE = 1 << 18, // use per wedge texcoords
82
83 // wireframe
84 DRAW_WIREFRAME = 1 << 20, // draw wireframe
85 DRAW_WIREFRAME_COLOR_USER = 1 << 21,
86 DRAW_WIREFRAME_COLOR_MESH = 1 << 22,
87
88 // bounding box
89 DRAW_BOUNDINGBOX = 1 << 25

699 template <MeshConcept MeshType>
700 void setRenderCapabilityFrom(const MeshType& m)
701 {
702 drawModeCapability = 0;
703
704 if (m.vertexNumber() > 0) {
705 drawModeCapability |= DRAW_MESH;
706
707 // -- Points --
708 drawModeCapability |= DRAW_POINTS;
709 drawModeCapability |= DRAW_POINTS_PIXEL;
710 drawModeCapability |= DRAW_POINTS_CIRCLE;
711 drawModeCapability |= DRAW_POINTS_COLOR_USER;
712
713 if constexpr (vcl::HasPerVertexColor<MeshType>) {
714 if (vcl::isPerVertexColorAvailable(m)) {
715 drawModeCapability |= DRAW_POINTS_COLOR_VERTEX;
716 }
717 }
718
719 if constexpr (vcl::HasColor<MeshType>) {
720 drawModeCapability |= DRAW_POINTS_COLOR_MESH;
721 }
722
723 // -- Surface and Wireframe --
724 if constexpr (vcl::HasFaces<MeshType>) {
725 if (m.faceNumber() > 0) {
726 drawModeCapability |= DRAW_SURF;
727 drawModeCapability |= DRAW_SURF_FLAT;
728 drawModeCapability |= DRAW_SURF_COLOR_USER;
729 drawModeCapability |= DRAW_WIREFRAME;
730 drawModeCapability |= DRAW_WIREFRAME_COLOR_USER;
731
732 if constexpr (vcl::HasColor<MeshType>) {
733 drawModeCapability |= DRAW_SURF_COLOR_MESH;
734 drawModeCapability |= DRAW_WIREFRAME_COLOR_MESH;
735 }
736
737 if constexpr (vcl::HasPerVertexNormal<MeshType>) {
738 if (vcl::isPerVertexNormalAvailable(m))
739 drawModeCapability |= DRAW_SURF_SMOOTH;
740 }
741
742 if constexpr (vcl::HasPerFaceColor<MeshType>) {
743 if (vcl::isPerFaceColorAvailable(m))
744 drawModeCapability |= DRAW_SURF_COLOR_FACE;
745 }
746
747 if constexpr (vcl::HasPerVertexColor<MeshType>) {
748 if (vcl::isPerVertexColorAvailable(m)) {
749 drawModeCapability |= DRAW_SURF_COLOR_VERTEX;
750 }
751 }
752
753 if constexpr (vcl::HasTexturePaths<MeshType>) {
754 if constexpr (vcl::HasPerVertexTexCoord<MeshType>) {
755 if (vcl::isPerVertexTexCoordAvailable(m))
756 drawModeCapability |= DRAW_SURF_TEX_VERTEX;
757 }
758
759 if constexpr (vcl::HasPerFaceWedgeTexCoords<MeshType>) {
760 if (vcl::isPerFaceWedgeTexCoordsAvailable(m))
761 drawModeCapability |= DRAW_SURF_TEX_WEDGE;
762 }
763 }
764 }
765 }
766
767 // -- Bounding Box --
768 if constexpr (vcl::HasBoundingBox<MeshType>) {
769 drawModeCapability |= DRAW_BOUNDINGBOX;
770 }
771 }
772
773 // make sure that the previous draw mode satisfies the new capabilites
774 drawMode &= drawModeCapability;
775 }
776
777 void setDefaultSettingsFromCapability()
778 {
779 drawMode = 0;
780
781 // default settings - ignored if not available
782 setPointCloudColorUserDefined();
783 setSurfaceColorUserDefined();
784 setWireframeColorUserDefined();
785
786 if (canBeVisible()) {
787 setVisibility(true);
788 if (canSurfaceBeVisible()) {
789 setSurfaceVisibility(true);
790 if (canSurfaceBeSmooth()) {
791 setSurfaceShadingSmooth();
792 }
793 else {
794 setSurfaceShadingFlat();
795 }
796 if (canSurfaceBeColoredPerVertex()) {
797 setSurfaceColorPerVertex();
798 }
799 else if (canSurfaceBeColoredPerFace()) {
800 setSurfaceColorPerFace();
801 }
802 else if (canSurfaceBeColoredPerWedgeTexcoords()) {
803 setSurfaceColorPerWedgeTexcoords();
804 }
805 else if (canSurfaceBeColoredPerVertexTexcoords()) {
806 setSurfaceColorPerVertexTexcoords();
807 }
808 else if (canSurfaceBeColoredPerMesh()) {
809 setSurfaceColorPerMesh();
810 }
811 }
812 else {
813 if (canPointCloudBeVisible()) {
814 setPointCloudVisibility(true);
815 if (canPointCloudBeColoredPerVertex()) {
816 setPointCloudColorPerVertex();
817 }
818 else if (canPointCloudBeColoredPerMesh()) {
819 setPointCloudColorPerMesh();
820 }
821 }
822 }
823 }
824 }
