33template<EdgeMeshConcept MeshType>
34void writePlyEdges(
35 std::ostream& file,
36 const PlyHeader& header,
37 const MeshType& mesh)
38{
39 using EdgeType = MeshType::EdgeType;
40
41 FileType format;
42 if (header.format() == ply::ASCII) {
43 format.isBinary = false;
44 }
45 else if (header.format() == ply::BINARY_BIG_ENDIAN) {
46 format.endian = std::endian::big;
47 }
48
49 // indices of vertices that do not consider deleted vertices
50 std::vector<uint> vIndices = mesh.vertexCompactIndices();
51
52 for (const EdgeType& e : mesh.edges()) {
53 for (const PlyProperty& p : header.edgeProperties()) {
54 bool hasBeenWritten = false;
55 if (p.name == ply::vertex1) {
56 io::writeProperty(
57 file, vIndices[mesh.index(e.vertex(0))], p.type, format);
58 hasBeenWritten = true;
59 }
60 if (p.name == ply::vertex2) {
61 io::writeProperty(
62 file, vIndices[mesh.index(e.vertex(1))], p.type, format);
63 hasBeenWritten = true;
64 }
65 if (!hasBeenWritten) {
66 // be sure to write something if the header declares some
67 // property that is not in the mesh
68 io::writeProperty(file, 0, p.type, format);
69 }
70 }
71 }
72}
