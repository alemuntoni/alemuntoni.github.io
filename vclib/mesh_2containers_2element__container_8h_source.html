169 uint addElement()
170 {
171 mVerticalCompVecTuple.resize(mElemVec.size() + 1);
172 if constexpr (comp::HasCustomComponents<T>)
173 mCustomCompVecMap.resize(mElemVec.size() + 1);
174
175 T* oldB = mElemVec.data();
176 mElemVec.emplace_back();
177 T* newB = mElemVec.data();
178 mElemNumber++;
179
180 mElemVec.back().setParentMesh(mParentMesh);
181 mElemVec.back().initVerticalComponents();
182
183 if (oldB != newB) {
184 setParentMeshPointers(mParentMesh);
185 mParentMesh->updateAllReferences(oldB);
186 }
187
188 return mElemVec.size() - 1;
189 }

199 uint addElements(uint size)
200 {
201 mVerticalCompVecTuple.resize(mElemVec.size() + size);
202 if constexpr (comp::HasCustomComponents<T>)
203 mCustomCompVecMap.resize(mElemVec.size() + size);
204
205 uint baseId = mElemVec.size();
206 T* oldB = mElemVec.data();
207 mElemVec.resize(mElemVec.size() + size);
208 T* newB = mElemVec.data();
209 mElemNumber += size;
210
211 for (uint i = baseId; i < mElemVec.size(); ++i) {
212 mElemVec[i].setParentMesh(mParentMesh);
213 mElemVec[i].initVerticalComponents();
214 }
215
216 if (oldB != newB) {
217 setParentMeshPointers(mParentMesh);
218 mParentMesh->updateAllReferences(oldB);
219 }
220
221 return baseId;
222 }

426 void append(const ElementContainer<T>& other)
427 {
428 using Comps = T::Components;
429
430 uint on = other.elementNumber();
431 uint n = elementContainerSize();
432 addElements(on);
433 for (uint i = 0; i < on; ++i) {
434 // copy everything from the other elements, also the (not updated)
435 // pointers:
436 element(n + i) = other.element(i);
437 element(n + i).setParentMesh(mParentMesh);
438 }
439 // importing also optional, vertical and custom components:
440 appendVerticalComponents(other, vComps());
441 appendCustomComponents(other);
442 }

878 template<typename OtherMesh>
879 void importFrom(const OtherMesh& m)
880 {
881 if constexpr (OtherMesh::template hasContainerOf<T>()) {
882 // get the container type of the other mesh for T - used to upcast
883 // othMesh
884 using Container = OtherMesh::template ContainerOf<T>::type;
885
886 const Container& c = (const Container&) m;
887
888 clearElements();
889
890 // recreate a container having the same number of elements as the
891 // other mesh. this call will also update the parentMesh pointers
892 // and sets the vertical components vectors in vcVecTuple
893 addElements(c.elementContainerSize());
894 unsigned int eid = 0;
895 for (const typename Container::ElementType& e : c.elements(false)) {
896 // import the components of the e, when they are available.
897 // note that will set also the deleted flag of e, therefore if e
898 // is deleted, then also element(eid) will be deleted.
899 element(eid).importFrom(e);
900 ++eid;
901 }
902 // set the number of elements (different from the container size)
903 mElemNumber = c.mElemNumber;
904 if constexpr (
905 comp::HasCustomComponents<T> &&
906 comp::HasCustomComponents<typename Container::ElementType>) {
907 mCustomCompVecMap = c.mCustomCompVecMap;
908 }
909 }

925 template<typename ElPtr, typename... Comps>
926 void updateReferencesOnComponents(
927 const std::vector<uint>& newIndices,
928 TypeWrapper<Comps...>)
929 {
930 (updateReferencesOnComponent<Comps, ElPtr>(newIndices), ...);
931 }
932
933 /*
934 * This function is called for each component of the element.
935 *
936 * Only if a component has references of the type ElPtr, then the
937 * updateReferences on each element will be executed.
938 *
939 * firstElementToProcess and offset are used only when an append operation
940 * has been executed. In this case, the firstElementToProcess is the index
941 * of the first element that has been appended, and offset is the number of
942 * ElPtr elements that have been appended (the offset that must be added
943 * to the newBase w.r.t. the oldBase that was the other container from which
944 * the elements have been copied).
945 */
946 template<typename Comp, typename ElPtr>
947 void updateReferencesOnComponent(
948 const ElPtr* oldBase,
949 uint firstElementToProcess = 0,
950 uint offset = 0)
951 {
952 if constexpr (comp::HasReferencesOfType<Comp, ElPtr>) {
953 // lambda to avoid code duplication
954 auto loop = [&]() {
955 for (uint i = firstElementToProcess; i < elementContainerSize();
956 i++) {
957 T& e = element(i);
958 if (!e.deleted()) {
959 e.Comp::updateReferences(oldBase, offset);
960 }
961 }
962 };
963
964 if constexpr (comp::HasOptionalReferencesOfType<Comp, ElPtr>) {
965 if (isOptionalComponentEnabled<Comp>()) {
966 loop();
967 }
968 }
969 else {
970 loop();
971 }
972 }
973 }
974
975 template<typename Comp, typename ElPtr>
976 void updateReferencesOnComponent(const std::vector<uint>& newIndices)
977 {
978 if constexpr (comp::HasReferencesOfType<Comp, ElPtr>) {
979 if constexpr (comp::HasOptionalReferencesOfType<Comp, ElPtr>) {
980 if (isOptionalComponentEnabled<Comp>()) {
981 for (T& e : elements()) {
982 e.Comp::updateReferences(newIndices);
983 }
984 }
985 }
986 else {
987 for (T& e : elements()) {
988 e.Comp::updateReferences(newIndices);
989 }
990 }
991 }

1002 template<typename Comp>
1003 void appendVerticalComponent(const ElementContainer<T>& other)
1004 {
1005 uint on = other.elementNumber();
1006 uint n = elementContainerSize() - on;
1007
1008 if (mVerticalCompVecTuple.template isComponentEnabled<Comp>() &&
1009 other.mVerticalCompVecTuple.template isComponentEnabled<Comp>()) {
1010 auto& vc = mVerticalCompVecTuple.template vector<Comp>();
1011 const auto& ovc =
1012 other.mVerticalCompVecTuple.template vector<Comp>();
1013
1014 for (uint i = 0; i < on; ++i) {
1015 vc[n + i] = ovc[i];
1016 }
1017 }

1020 void appendCustomComponents(const ElementContainer<T>& other)
1021 {
1022 if constexpr (comp::HasCustomComponents<T>) {
1023 uint on = other.elementNumber();
1024 uint n = elementContainerSize() - on;
1025
1026 std::vector<std::string> ccNames =
1027 mCustomCompVecMap.allComponentNames();
1028
1029 for (const std::string& name : ccNames) {
1030 for (uint i = 0; i < on; ++i) {
1031 mCustomCompVecMap.importSameCustomComponentFrom(
1032 n + i, i, name, other.mCustomCompVecMap);
1033 }
1034 }

1049 template<typename Comp, typename Cont2>
1050 void enableSameOptionalComponent(const Cont2& c2)
1051 {
1052 // if Comp is an optional component in This container
1053 if constexpr (comp::IsOptionalComponent<Comp>) {
1054 // if Comp is available in Cont2
1055 if constexpr (comp::HasComponentOfType<
1056 typename Cont2::ElementType,
1057 Comp::COMPONENT_ID>) {
1058 // if Comp is optional in Cont2
1059 if constexpr (comp::HasOptionalComponentOfType<
1060 typename Cont2::ElementType,
1061 Comp::COMPONENT_ID>) {
1062 // if Comp is enabled in Cont2, we enable it in this
1063 // container
1064 if (c2.template isOptionalComponentEnabled<
1065 Comp::COMPONENT_ID>()) {
1066 enableOptionalComponent<Comp::COMPONENT_ID>();
1067 }
1068 }
1069 // if Comp is not optional (but is available), we enable it in
1070 // this container
1071 else {
1072 enableOptionalComponent<Comp::COMPONENT_ID>();
1073 }
1074 }
