168 uint addElement()
169 {
170 mVerticalCompVecTuple.resize(mElemVec.size() + 1);
171 if constexpr (comp::HasCustomComponents<T>)
172 mCustomCompVecMap.resize(mElemVec.size() + 1);
173
174 T* oldB = mElemVec.data();
175 mElemVec.emplace_back();
176 T* newB = mElemVec.data();
177 mElemNumber++;
178
179 mElemVec.back().setParentMesh(mParentMesh);
180 mElemVec.back().initVerticalComponents();
181
182 if (oldB != newB) {
183 setParentMeshPointers(mParentMesh);
184 mParentMesh->updateAllReferences(oldB);
185 }
186
187 return mElemVec.size() - 1;
188 }

198 uint addElements(uint size)
199 {
200 mVerticalCompVecTuple.resize(mElemVec.size() + size);
201 if constexpr (comp::HasCustomComponents<T>)
202 mCustomCompVecMap.resize(mElemVec.size() + size);
203
204 uint baseId = mElemVec.size();
205 T* oldB = mElemVec.data();
206 mElemVec.resize(mElemVec.size() + size);
207 T* newB = mElemVec.data();
208 mElemNumber += size;
209
210 for (uint i = baseId; i < mElemVec.size(); ++i) {
211 mElemVec[i].setParentMesh(mParentMesh);
212 mElemVec[i].initVerticalComponents();
213 }
214
215 if (oldB != newB) {
216 setParentMeshPointers(mParentMesh);
217 mParentMesh->updateAllReferences(oldB);
218 }
219
220 return baseId;
221 }

425 void append(const ElementContainer<T>& other)
426 {
427 using Comps = T::Components;
428
429 uint on = other.elementNumber();
430 uint n = elementContainerSize();
431 addElements(on);
432 for (uint i = 0; i < on; ++i) {
433 // copy everything from the other elements, also the (not updated)
434 // pointers:
435 element(n + i) = other.element(i);
436 element(n + i).setParentMesh(mParentMesh);
437 }
438 // importing also optional, vertical and custom components:
439 appendVerticalComponents(other, vComps());
440 appendCustomComponents(other);

877 template<typename OtherMesh>
878 void importFrom(const OtherMesh& m)
879 {
880 if constexpr (OtherMesh::template hasContainerOf<T>()) {
881 // get the container type of the other mesh for T - used to upcast
882 // othMesh
883 using Container = OtherMesh::template ContainerOf<T>::type;
884
885 const Container& c = (const Container&) m;
886
887 clearElements();
888
889 // recreate a container having the same number of elements as the
890 // other mesh. this call will also update the parentMesh pointers
891 // and sets the vertical components vectors in vcVecTuple
892 addElements(c.elementContainerSize());
893 unsigned int eid = 0;
894 for (const typename Container::ElementType& e : c.elements(false)) {
895 // import the components of the e, when they are available.
896 // note that will set also the deleted flag of e, therefore if e
897 // is deleted, then also element(eid) will be deleted.
898 element(eid).importFrom(e);
899 ++eid;
900 }
901 // set the number of elements (different from the container size)
902 mElemNumber = c.mElemNumber;
903 if constexpr (
904 comp::HasCustomComponents<T> &&
905 comp::HasCustomComponents<typename Container::ElementType>)
906 {
907 mCustomCompVecMap = c.mCustomCompVecMap;
908 }
909 }

925 template<typename ElPtr, typename... Comps>
926 void updateReferencesOnComponents(

928 TypeWrapper<Comps...>)

932
933 /*
934 * This function is called for each component of the element.
935 *
936 * Only if a component has references of the type ElPtr, then the
937 * updateReferences on each element will be executed.
938 *
939 * firstElementToProcess and offset are used only when an append operation
940 * has been executed. In this case, the firstElementToProcess is the index
941 * of the first element that has been appended, and offset is the number of
942 * ElPtr elements that have been appended (the offset that must be added
943 * to the newBase w.r.t. the oldBase that was the other container from which
944 * the elements have been copied).
945 */
946 template<typename Comp, typename ElPtr>
947 void updateReferencesOnComponent(
948 const ElPtr* oldBase,
949 uint firstElementToProcess = 0,
950 uint offset = 0)
951 {
952 if constexpr (comp::HasReferencesOfType<Comp, ElPtr>) {
953 // lambda to avoid code duplication
954 auto loop = [&]() {
955 for (uint i = firstElementToProcess; i < elementContainerSize();
956 i++)
957 {
958 T& e = element(i);
959 if (!e.deleted()) {
960 e.Comp::updateReferences(oldBase, offset);
961 }
962 }
963 };
964
965 if constexpr (comp::HasOptionalReferencesOfType<Comp, ElPtr>) {
966 if (isOptionalComponentEnabled<Comp>()) {
967 loop();
968 }
969 }
970 else {
971 loop();
972 }
973 }
974 }

976 template<typename Comp, typename ElPtr>
977 void updateReferencesOnComponent(const std::vector<uint>& newIndices)
978 {
979 if constexpr (comp::HasReferencesOfType<Comp, ElPtr>) {
980 if constexpr (comp::HasOptionalReferencesOfType<Comp, ElPtr>) {
981 if (isOptionalComponentEnabled<Comp>()) {

983 e.Comp::updateReferences(newIndices);

1005 {
1006 uint on = other.elementNumber();

1008
1009 if (mVerticalCompVecTuple.template isComponentEnabled<Comp>() &&
1010 other.mVerticalCompVecTuple.template isComponentEnabled<Comp>())

1013 const auto& ovc =
1014 other.mVerticalCompVecTuple.template vector<Comp>();

1016 for (uint i = 0; i < on; ++i) {
1017 vc[n + i] = ovc[i];
