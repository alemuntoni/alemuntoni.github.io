36template<uint ELEM_ID, LoggerConcept LogType = NullLogger>
37void normalizeNoThrow(auto& elem, LogType& log = nullLogger)
38{
39 try {
40 elem.normal().normalize();
41 }
42 catch (const std::exception& e) {
43 log.log(
44 log.WARNING,
45 elementEnumString<ELEM_ID>() + " " + std::to_string(elem.index()) +
46 ": " + e.what());
47 }

132template<uint ELEM_ID, typename MScalar, LoggerConcept LogType = NullLogger>
133void multiplyPerElementNormalsByMatrix(
134 MeshConcept auto& mesh,
135 Matrix33<MScalar> mat,
136 bool removeScalingFromMatrix = true,
137 LogType& log = nullLogger)
138{
139 requirePerElementComponent<ELEM_ID, CompId::NORMAL>(mesh);
140
141 if (removeScalingFromMatrix) {
142 MScalar scaleX = std::sqrt(
143 mat(0, 0) * mat(0, 0) + mat(0, 1) * mat(0, 1) +
144 mat(0, 2) * mat(0, 2));
145 MScalar scaleY = std::sqrt(
146 mat(1, 0) * mat(1, 0) + mat(1, 1) * mat(1, 1) +
147 mat(1, 2) * mat(1, 2));
148 MScalar scaleZ = std::sqrt(
149 mat(2, 0) * mat(2, 0) + mat(2, 1) * mat(2, 1) +
150 mat(2, 2) * mat(2, 2));
151 for (int i = 0; i < 3; ++i) {
152 mat(0, i) /= scaleX;
153 mat(1, i) /= scaleY;
154 mat(2, i) /= scaleZ;
155 }
156 }
157
158 log.log(
159 0,
160 "Multiplying per-" + elementEnumString<ELEM_ID>() +
161 " normals by matrix...");
162
163 parallelFor(mesh.template elements<ELEM_ID>(), [&](auto& e) {
164 e.normal() *= mat;
165 });
166
167 log.log(
168 100, "Per-" + elementEnumString<ELEM_ID>() + " normals multiplied.");
169}

228template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
229void clearPerReferencedVertexNormals(MeshType& mesh, LogType& log = nullLogger)
230{
231 vcl::requirePerVertexNormal(mesh);
232
233 // function that is called for each container Cont
234 auto f = [&]<mesh::ElementContainerConcept Cont>() {
235 using Elem = typename Cont::ElementType; // the Element type of Cont
236 if constexpr (comp::HasVertexReferences<Elem>) { // if Elem has vertices
237 for (auto& e : mesh.template elements<Elem::ELEMENT_ID>()) {
238 for (auto* v : e.vertices()) {
239 v->normal().setZero();
240 }
241 }
242 }
243 };
244
245 log.log(0, "Clearing per-Vertex normals...");
246
247 // apply the function f to all the containers of the mesh
248 vcl::ForEachType<typename MeshType::Containers>::apply(f);
249
250 log.log(100, "Per-Vertex normals cleared.");
251}

381template<LoggerConcept LogType = NullLogger>
382void updatePerFaceNormals(
383 FaceMeshConcept auto& mesh,
384 bool normalize = true,
385 LogType& log = nullLogger)
386{
387 vcl::requirePerFaceNormal(mesh);
388
389 using FaceType = std::remove_reference_t<decltype(mesh)>::FaceType;
390 using ScalarType = FaceType::NormalType::ScalarType;
391
392 log.log(0, "Updating per-Face normals...");
393
394 parallelFor(mesh.faces(), [](auto& f) {
395 f.normal() = faceNormal(f).template cast<ScalarType>();
396 });
397
398 if (normalize) {
399 log.startNewTask(50, 100, "Normalizing per-Face normals...");
400 normalizePerFaceNormals(mesh, log);
401 log.endTask("Normalizing per-Face normals...");
402 }
403
404 log.log(100, "Per-Face normals updated.");
405}

579template<LoggerConcept LogType = NullLogger>
580void updatePerVertexNormalsAngleWeighted(
581 FaceMeshConcept auto& mesh,
582 bool normalize = true,
583 LogType& log = nullLogger)
584{
585 using VertexType = std::remove_reference_t<decltype(mesh)>::VertexType;
586 using NormalType = VertexType::NormalType;
587 using NScalarType = NormalType::ScalarType;
588
589 log.log(0, "Updating per-Vertex normals...");
590
591 log.startNewTask(0, 5, "Clearing per-Vertex normals...");
592 clearPerReferencedVertexNormals(mesh, log);
593 log.endTask("Clearing per-Vertex normals...");
594
595 log.log(5, "Updating per-Vertex normals...");
596
597 for (auto& f : mesh.faces()) {
598 auto n = faceNormal(f).template cast<NScalarType>();
599
600 for (uint i = 0; i < f.vertexNumber(); ++i) {
601 NormalType vec1 =
602 (f.vertexMod(i - 1)->coord() - f.vertexMod(i)->coord())
603 .normalized()
604 .template cast<NScalarType>();
605 NormalType vec2 =
606 (f.vertexMod(i + 1)->coord() - f.vertexMod(i)->coord())
607 .normalized()
608 .template cast<NScalarType>();
609
610 f.vertex(i)->normal() += n * vec1.angle(vec2);
611 }
612 }
613
614 if (normalize) {
615 log.startNewTask(95, 100, "Normalizing per-Vertex normals...");
616 normalizePerReferencedVertexNormals(mesh, log);
617 log.endTask("Normalizing per-Vertex normals...");
618 }
619
620 log.log(100, "Per-Vertex normals updated.");
621}

653template<LoggerConcept LogType = NullLogger>
654void updatePerVertexNormalsNelsonMaxWeighted(
655 FaceMeshConcept auto& mesh,
656 bool normalize = true,
657 LogType& log = nullLogger)
658{
659 using VertexType = std::remove_reference_t<decltype(mesh)>::VertexType;
660 using NScalarType = VertexType::NormalType::ScalarType;
661
662 log.log(0, "Updating per-Vertex normals...");
663
664 log.startNewTask(0, 5, "Clearing per-Vertex normals...");
665 clearPerReferencedVertexNormals(mesh, log);
666 log.endTask("Clearing per-Vertex normals...");
667
668 log.log(5, "Updating per-Vertex normals...");
669
670 for (auto& f : mesh.faces()) {
671 auto n = faceNormal(f).template cast<NScalarType>();
672
673 for (uint i = 0; i < f.vertexNumber(); ++i) {
674 NScalarType e1 =
675 (f.vertexMod(i - 1)->coord() - f.vertexMod(i)->coord())
676 .squaredNorm();
677 NScalarType e2 =
678 (f.vertexMod(i + 1)->coord() - f.vertexMod(i)->coord())
679 .squaredNorm();
680
681 f.vertex(i)->normal() += n / (e1 * e2);
682 }
683 }
684
685 if (normalize) {
686 log.startNewTask(95, 100, "Normalizing per-Vertex normals...");
687 normalizePerReferencedVertexNormals(mesh, log);
688 log.endTask("Normalizing per-Vertex normals...");
689 }
690
691 log.log(100, "Per-Vertex normals updated.");
692}
