40template<MeshConcept MeshType>
41void loadOffVertices(
42 MeshType& mesh,
43 std::ifstream& file,
44 const MeshInfo& fileInfo,
45 uint nv)
46{
47 using VertexType = MeshType::VertexType;
48
49 const uint nTexCoords = fileInfo.hasVertexTexCoords() ? 2 : 0;
50
51 mesh.addVertices(nv);
52 for (uint i = 0; i < nv; i++) {
53 VertexType& v = mesh.vertex(i);
54
55 vcl::Tokenizer tokens = internal::nextNonEmptyTokenizedLine(file);
56 vcl::Tokenizer::iterator token = tokens.begin();
57
58 // Read 3 vertex coordinates
59 for (unsigned int j = 0; j < 3; j++) {
60 // Read vertex coordinate
61 v.coord()[j] = io::internal::readDouble<double>(token);
62 }
63
64 if constexpr(vcl::HasPerVertexNormal<MeshType>) {
65 if (vcl::isPerVertexNormalAvailable(mesh) &&
66 fileInfo.hasVertexNormals())
67 {
68 // Read 3 normal coordinates
69 for (unsigned int j = 0; j < 3; j++) {
70 v.normal()[j] = io::internal::readDouble<double>(token);
71 }
72 }
73 }
74 // need to read and throw away data
75 else if (fileInfo.hasVertexNormals()){
76 for (unsigned int j = 0; j < 3; j++) {
77 io::internal::readDouble<double>(token);
78 }
79 }
80
81 const uint nReadComponents = token - tokens.begin();
82 const int nColorComponents =
83 (int) tokens.size() - nReadComponents - nTexCoords;
84
85 if constexpr(vcl::HasPerVertexColor<MeshType>) {
86 if (vcl::isPerVertexColorAvailable(mesh) &&
87 fileInfo.hasVertexColors())
88 {
89 if (nColorComponents != 1 && nColorComponents != 3 &&
90 nColorComponents != 4)
91 throw MalformedFileException(
92 "Wrong number of components in line.");
93 v.color() = off::loadColor(token, nColorComponents);
94 }
95 }
96 // need to read and throw away data
97 else if (fileInfo.hasVertexColors()){
98 if (nColorComponents != 1 && nColorComponents != 3 &&
99 nColorComponents != 4)
100 throw MalformedFileException(
101 "Wrong number of components in line.");
102 off::loadColor(token, nColorComponents);
103 }
104
105 if constexpr(vcl::HasPerVertexTexCoord<MeshType>) {
106 if (vcl::isPerVertexTexCoordAvailable(mesh) &&
107 fileInfo.hasVertexTexCoords())
108 {
109 // Read 2 tex coordinates
110 for (unsigned int j = 0; j < 2; j++) {
111 v.texCoord()[j] = io::internal::readDouble<double>(token);
112 }
113 }
114 }
115 // need to read and throw away data
116 else if (fileInfo.hasVertexTexCoords()){
117 for (unsigned int j = 0; j < 2; j++) {
118 io::internal::readDouble<double>(token);
119 }
120 }
121 }
122}

124template<FaceMeshConcept MeshType>
125void loadOffFaces(
126 MeshType& mesh,
127 std::ifstream& file,
128 MeshInfo& loadedInfo,
129 uint nf,
130 bool enableOptionalComponents)
131{
132 if constexpr (HasFaces<MeshType>) {
133 using FaceType = MeshType::FaceType;
134
135 mesh.reserveFaces(nf);
136 for (uint fid = 0; fid < nf; ++fid) {
137 vcl::Tokenizer tokens =
138 vcl::io::internal::nextNonEmptyTokenizedLine(file);
139 vcl::Tokenizer::iterator token = tokens.begin();
140 mesh.addFace();
141 FaceType& f = mesh.face(mesh.faceNumber() - 1);
142
143 // read vertex indices
144 uint fSize = io::internal::readUInt<uint>(token);
145 // contains the vertex ids of the actual face
146 std::vector<uint> vids;
147 vids.resize(fSize);
148 for (uint i = 0; i < fSize; ++i) {
149 vids[i] = io::internal::readUInt<uint>(token);
150 bool splitFace = false;
151 // we have a polygonal mesh
152 if constexpr (FaceType::VERTEX_NUMBER < 0) {
153 // need to resize to the right number of verts
154 f.resizeVertices(vids.size());
155 }
156 else if (FaceType::VERTEX_NUMBER != vids.size()) {
157 // we have faces with static sizes (triangles), but we are
158 // loading faces with number of verts > 3. Need to split the
159 // face we are loading in n faces!
160 splitFace = true;
161 }
162 if (!splitFace) { // classic load, no split needed
163 for (uint i = 0; i < vids.size(); ++i) {
164 if (vids[i] >= mesh.vertexNumber()) {
165 throw vcl::MalformedFileException(
166 "Bad vertex index for face " +
167 std::to_string(i));
168 }
169 f.vertex(i) = &mesh.vertex(vids[i]);
170 }
171 }
172 else { // split needed
173 addTriangleFacesFromPolygon(mesh, f, vids);
174 }
175 }
176
177 // read face color
178 if (token != tokens.end()) { // there are colors to read
179 if constexpr (HasPerFaceColor<MeshType>) {
180 if (isPerFaceColorAvailable(mesh) ||
181 (enableOptionalComponents &&
182 enableIfPerFaceColorOptional(mesh)))
183 {
184 loadedInfo.setFaceColors();
185 f.color() = off::loadColor(
186 token, tokens.size() - (token - tokens.begin()));
187 // in case the loaded polygon has been triangulated in
188 // the last n triangles
189 for (uint ff = mesh.index(f); ff < mesh.faceNumber();
190 ++ff)
191 {
192 mesh.face(ff).color() = f.color();
193 }
194 }
195 }
196 }
197 }
198 }
199 else { // mesh does not have face, read nf lines and throw them away
200 for (uint i = 0; i < nf; ++i)
201 vcl::io::internal::nextNonEmptyTokenizedLine(file);
202 }
203}

207template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
208void loadOff(
209 MeshType& m,
210 const std::string& filename,
211 MeshInfo& loadedInfo,
212 LogType& log = nullLogger,
213 bool enableOptionalComponents = true)
214{
215 std::ifstream file = internal::loadFileStream(filename);
216 uint nVertices, nFaces, nEdges;
217
218 if constexpr (HasName<MeshType>) {
219 m.name() = FileInfo::fileNameWithoutExtension(filename);
220 }
221
222 MeshInfo fileInfo; // data that needs to be read from the file
223
224 off::loadOffHeader(file, fileInfo, nVertices, nFaces, nEdges);
225 loadedInfo = fileInfo; // data that will be stored in the mesh!
226 if (enableOptionalComponents)
227 internal::enableOptionalComponents(loadedInfo, m);
228
229 internal::loadOffVertices(m, file, fileInfo, nVertices);
230 internal::loadOffFaces(m, file, fileInfo, nFaces, enableOptionalComponents);
231 if (enableOptionalComponents)
232 loadedInfo = fileInfo;
233 // internal::loadOffEdges(m, file, loadedInfo, nEdges);
234}
