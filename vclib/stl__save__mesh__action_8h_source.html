34class StlSaveMeshAction : public SaveMeshAction
35{
36public:
37 using SaveMeshAction::save;
38
39 std::string name() const override { return "Save Stl Mesh"; }
40
41 std::shared_ptr<Action> clone() const override
42 {
43 return std::make_shared<StlSaveMeshAction>(*this);
44 }
45
46 ParameterVector parameters() const override
47 {
48 ParameterVector params;
49
50 params.pushBack(BoolParameter(
51 "binary",
52 true,
53 "Binary",
54 "If true, the file will be saved as binary."));
55 params.pushBack(BoolParameter(
56 "magics_mode",
57 false,
58 "Magics Mode",
59 "Magics mode is a specific file format used by the Magics "
60 "software, which includes more information than the standard STL "
61 "format, like face colors. The magicsMode works only when the file "
62 "is saved as binary."));
63
64 return params;
65 }
66
67 std::vector<FileFormat> formats() const override
68 {
69 return {FileFormat("stl", "STL STereo Lithography interface format")};
70 }
71
72 MeshInfo formatCapability() const override { return stlFormatCapability(); }
73
74 void save(
75 const std::string& filename,
76 const MeshI& mesh,
77 const MeshInfo& info,
78 const ParameterVector& parameters,
79 AbstractLogger& log = logger()) const override
80 {
81 // transform saveStl to a lambda function
82 auto fun = [&](auto&& m) {
83 saveStl(m, filename, info, parameters, log);
84 };
85
86 callFunctionForSupportedMeshTypes(mesh, fun);
87 }
88
89private:
90 void saveStl(
91 const MeshConcept auto& mesh,
92 const std::string& filename,
93 const MeshInfo& info,
94 const ParameterVector& parameters,
95 AbstractLogger& log) const
96 {
97 vcl::SaveSettings settings;
98 settings.info = info;
99 settings.binary = parameters.get("binary")->boolValue();
100 settings.magicsMode = parameters.get("magics_mode")->boolValue();
101 vcl::saveStl(mesh, filename, settings, log);
102 }
103};
