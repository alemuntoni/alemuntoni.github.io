172 template<MeshConcept Mesh>
173 MeshInfo(const Mesh& m)
174 {
175 setVertices();
176 setVertexCoords(
177 true, getType<typename Mesh::VertexType::CoordType::ScalarType>());
178 if constexpr (vcl::HasPerVertexNormal<Mesh>) {
179 if (vcl::isPerVertexNormalAvailable(m)) {
180 setVertexNormals(
181 true,
182 getType<
183 typename Mesh::VertexType::NormalType::ScalarType>());
184 }
185 }
186 if constexpr (vcl::HasPerVertexColor<Mesh>) {
187 if (vcl::isPerVertexColorAvailable(m)) {
188 setVertexColors(true, UCHAR);
189 }
190 }
191 if constexpr (vcl::HasPerVertexQuality<Mesh>) {
192 if (vcl::isPerVertexQualityAvailable(m)) {
193 setVertexQuality(
194 true, getType<typename Mesh::VertexType::QualityType>());
195 }
196 }
197 if constexpr (vcl::HasPerVertexTexCoord<Mesh>) {
198 if (vcl::isPerVertexTexCoordAvailable(m)) {
199 setVertexTexCoords(
200 true,
201 getType<
202 typename Mesh::VertexType::TexCoordType::ScalarType>());
203 }
204 }
205 if constexpr (vcl::HasPerVertexCustomComponents<Mesh>) {
206 auto names = m.perVertexCustomComponentNames();
207 for (auto& name : names) {
208 DataType dt = getType(m.perVertexCustomComponentType(name));
209 if (dt != UNKNOWN) {
210 addVertexCustomComponent(name, dt);
211 }
212 }
213 }
214
215 if constexpr (vcl::HasFaces<Mesh>) {
216 setFaces();
217 setFaceVRefs();
218 if (vcl::HasTriangles<Mesh>) {
219 setTriangleMesh();
220 }
221 else if (vcl::HasQuads<Mesh>) {
222 setQuadMesh();
223 }
224 else {
225 setPolygonMesh();
226 }
227 if constexpr (vcl::HasPerFaceNormal<Mesh>) {
228 if (vcl::isPerFaceNormalAvailable(m)) {
229 setFaceNormals(
230 true,
231 getType<
232 typename Mesh::FaceType::NormalType::ScalarType>());
233 }
234 }
235 if constexpr (vcl::HasPerFaceColor<Mesh>) {
236 if (vcl::isPerFaceColorAvailable(m)) {
237 setFaceColors(true, UCHAR);
238 }
239 }
240 if constexpr (vcl::HasPerFaceQuality<Mesh>) {
241 if (vcl::isPerFaceQualityAvailable(m)) {
242 setFaceQuality(
243 true, getType<typename Mesh::FaceType::QualityType>());
244 }
245 }
246 if constexpr (vcl::HasPerFaceWedgeTexCoords<Mesh>) {
247 if (vcl::isPerFaceWedgeTexCoordsAvailable(m)) {
248 setFaceWedgeTexCoords(
249 true,
250 getType<typename Mesh::FaceType::WedgeTexCoordType::
251 ScalarType>());
252 }
253 }
254 if constexpr(vcl::HasPerFaceCustomComponents<Mesh>) {
255 auto names = m.perFaceCustomComponentNames();
256 for (auto& name : names) {
257 DataType dt = getType(m.perFaceCustomComponentType(name));
258 if (dt != UNKNOWN) {
259 addFaceCustomComponent(name, dt);
260 }
261 }
262 }
263 }
264
265 if constexpr (vcl::HasEdges<Mesh>) {
266 setEdges();
267 setEdgeVRefs();
268 // if constexpr (vcl::HasPerEdgeColor<Mesh>)
269 // if (vcl::isPerEdgeColorAvailable(m))
270 // setEdgeColors(true, UCHAR);
271 }
272
273 if constexpr (vcl::HasTexturePaths<Mesh>) {
274 if (m.textureNumber() > 0) {
275 setTextures(true);
276 }
277 }

631 MeshInfo intersect(const MeshInfo& info) const
632 {
633 MeshInfo res;
634 for (uint i = 0; i < NUM_ELEMENTS; ++i) {
635 res.elements[i] = elements[i] && info.elements[i];
636 for (uint j = 0; j < NUM_COMPONENTS; ++j) {
637 res.perElemComponents[i][j] =
638 perElemComponents[i][j] && info.perElemComponents[i][j];
639
640 if (res.perElemComponents[i][j]){
641 res.perElemComponentsType(i, j) =
642 perElemComponentsType(i, j);
643 }
644 }
645 }
646
647 if (type == info.type){
648 res.type = type;
649 }
650 res.perElemCustomComponents = perElemCustomComponents;
651
652 return res;
653 }
654
655 void reset()
656 {
657 elements.reset();
658 for (auto& comp : perElemComponents)
659 comp.reset();
660 perElemComponentsType.fill(UNKNOWN);
661
662 for (auto& v : perElemCustomComponents)
663 v.clear();
664
665 type = TRIANGLE_MESH;
666 }

677 template<typename T>
678 static DataType getType()
679 {
680 if constexpr (std::is_same_v<T, char>) return CHAR;
681 if constexpr (std::is_same_v<T, unsigned char>) return UCHAR;
682 if constexpr (std::is_same_v<T, short>) return SHORT;
683 if constexpr (std::is_same_v<T, unsigned short>) return USHORT;
684 if constexpr (std::is_same_v<T, int>) return INT;
685 if constexpr (std::is_same_v<T, uint>) return UINT;
686 if constexpr (std::is_integral_v<T>) return INT; // fallback to int
687 if constexpr (std::is_same_v<T, float>) return FLOAT;
688 if constexpr (std::is_same_v<T, double>) return DOUBLE;
689 if constexpr (std::is_floating_point_v<T>)
690 return FLOAT; // fallback to float
691 return UNKNOWN;
692 }
693
694 static DataType getType(std::type_index ti)
695 {
696 if (ti == typeid(char)) return CHAR;
697 if (ti == typeid(unsigned char)) return UCHAR;
698 if (ti == typeid(short)) return SHORT;
699 if (ti == typeid(unsigned short)) return USHORT;
700 if (ti == typeid(int)) return INT;
701 if (ti == typeid(uint)) return UINT;
702 if (ti == typeid(float)) return FLOAT;
703 if (ti == typeid(double)) return DOUBLE;
704 return UNKNOWN;
705 }
