39template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
40void readPlyTextures(
41 const PlyHeader& header,
42 MeshType& mesh,
43 LogType& log = nullLogger,
44 const LoadSettings& settings = LoadSettings())
45{
46 if constexpr (vcl::HasTexturePaths<MeshType>) {
47 for (const std::string& str : header.textureFileNames()) {
48 mesh.pushTexturePath(str);
49 }
50 }
51 if constexpr (vcl::HasTextureImages<MeshType>) {
52 for (const std::string& str : header.textureFileNames()) {
53 vcl::Texture t;
54 t.path() = str;
55 if (settings.loadTextureImages) {
56 bool b = t.image().load(mesh.meshBasePath() + str);
57 if (!b) {
58 log.log(LogType::WARNING, "Cannot load texture " + str);
59 }
60 }
61 mesh.pushTexture(t);
62 }
63 }
64}

66template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
67void writePlyTextures(
68 PlyHeader& header,
69 const MeshType& mesh,
70 LogType& log,
71 const SaveSettings& settings)
72{
73 if constexpr (vcl::HasTexturePaths<MeshType>) {
74 for (const std::string& str : mesh.texturePaths()) {
75 header.pushTextureFileName(str);
76 }
77 }
78 if constexpr (vcl::HasTextureImages<MeshType>) {
79 for (const vcl::Texture& t : mesh.textures()) {
80 header.pushTextureFileName(t.path());
81 if (settings.saveTextureImages) {
82 try {
83 t.image().save(mesh.meshBasePath() + t.path());
84 }
85 catch (const std::runtime_error& e) {
86 log.log(LogType::WARNING, e.what());
87 }
88 }
89 }
90 }
91}

93template<LoggerConcept LogType>
94void readPlyUnknownElement(
95 std::istream& file,
96 const PlyHeader& header,
97 PlyElement el,
98 LogType& log)
99{
100 log.startProgress("Reading unknown elements", el.numberElements);
101
102 if (header.format() == ply::ASCII) {
103 for (uint i = 0; i < el.numberElements; ++i) {
104 readAndTokenizeNextNonEmptyLine(file);
105 log.progress(i);
106 }
107 }
108 else {
109 for (uint i = 0; i < el.numberElements; ++i) {
110 for (const PlyProperty& p : el.properties) {
111 if (p.list) {
112 uint s = io::readPrimitiveType<int>(file, p.listSizeType);
113 for (uint i = 0; i < s; ++i)
114 io::readPrimitiveType<int>(file, p.type);
115 }
116 else {
117 io::readPrimitiveType<int>(file, p.type);
118 }
119 }
120 log.progress(i);
121 }
122 }
123
124 log.endProgress();
125}
