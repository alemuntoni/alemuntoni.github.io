48// returns a non-empty vector if the vertex container is not compact and the
49// user wants compact indices
50std::vector<uint> vertCompactIndices(const auto& mesh, bool wantCompact)
51{
52 std::vector<uint> vertCompIndices;
53
54 bool isCompact = mesh.vertexNumber() == mesh.vertexContainerSize();
55
56 if (wantCompact && !isCompact)
57 vertCompIndices = mesh.vertexCompactIndices();
58 return vertCompIndices;
59}
60
61// lambda to get the vertex index of a face (considering compact vertex indices)
62auto vIndexLambda(const auto& mesh, const std::vector<uint>& vertCompIndices)
63{
64 // lambda to get the vertex index of a face (considering compact indices)
65 auto vIndex = [&vertCompIndices](const auto& f, uint i) {
66 if (vertCompIndices.size() > 0)
67 return vertCompIndices[f.vertexIndex(i)];
68 else
69 return f.vertexIndex(i);
70 };
71
72 return vIndex;
73}

98template<MeshConcept MeshType>
99void vertexCoordsToBuffer(
100 const MeshType& mesh,
101 auto* buffer,
102 MatrixStorageType storage = MatrixStorageType::ROW_MAJOR)
103{
104 if (storage == MatrixStorageType::ROW_MAJOR) {
105 for (uint i = 0; const auto& c : mesh.vertices() | views::coords) {
106 buffer[i * 3 + 0] = c.x();
107 buffer[i * 3 + 1] = c.y();
108 buffer[i * 3 + 2] = c.z();
109 ++i;
110 }
111 }
112 else {
113 const uint VERT_NUM = mesh.vertexNumber();
114 for (uint i = 0; const auto& c : mesh.vertices() | views::coords) {
115 buffer[0 * VERT_NUM + i] = c.x();
116 buffer[1 * VERT_NUM + i] = c.y();
117 buffer[2 * VERT_NUM + i] = c.z();
118 ++i;
119 }
120 }

153template<FaceMeshConcept MeshType>
154void triangleIndicesToBuffer(
155 const MeshType& mesh,
156 auto* buffer,
157 MatrixStorageType storage = MatrixStorageType::ROW_MAJOR,
158 bool getIndicesAsIfContainerCompact = true)
159{
160 const std::vector<uint> vertCompIndices =
161 detail::vertCompactIndices(mesh, getIndicesAsIfContainerCompact);
162
163 // lambda to get the vertex index of a face (considering compact indices)
164 auto vIndex = detail::vIndexLambda(mesh, vertCompIndices);
165
166 if (storage == MatrixStorageType::ROW_MAJOR) {
167 for (uint i = 0; const auto& f : mesh.faces()) {
168 buffer[i * 3 + 0] = vIndex(f, 0);
169 buffer[i * 3 + 1] = vIndex(f, 1);
170 buffer[i * 3 + 2] = vIndex(f, 2);
171 ++i;
172 }
173 }
174 else {
175 const uint FACE_NUM = mesh.faceNumber();
176 for (uint i = 0; const auto& f : mesh.faces()) {
177 buffer[0 * FACE_NUM + i] = vIndex(f, 0);
178 buffer[1 * FACE_NUM + i] = vIndex(f, 1);
179 buffer[2 * FACE_NUM + i] = vIndex(f, 2);
180 ++i;
181 }
182 }

333template<FaceMeshConcept MeshType>
334void faceIndicesToBuffer(
335 const MeshType& mesh,
336 auto* buffer,
337 uint largestFaceSize,
338 MatrixStorageType storage = MatrixStorageType::ROW_MAJOR,
339 bool getIndicesAsIfContainerCompact = true)
340{
341 const std::vector<uint> vertCompIndices =
342 detail::vertCompactIndices(mesh, getIndicesAsIfContainerCompact);
343
344 // lambda to get the vertex index of a face (considering compact indices)
345 auto vIndex = detail::vIndexLambda(mesh, vertCompIndices);
346
347 if (storage == MatrixStorageType::ROW_MAJOR) {
348 for (uint i = 0; const auto& f : mesh.faces()) {
349 uint j = 0;
350 for (; j < f.vertexNumber(); ++j)
351 buffer[i * largestFaceSize + j] = vIndex(f, j);
352 for (; j < largestFaceSize; ++j) // remaining vertices set to -1
353 buffer[i * largestFaceSize + j] = -1;
354 ++i;
355 }
356 }
357 else {
358 const uint FACE_NUM = mesh.faceNumber();
359 for (uint i = 0; const auto& f : mesh.faces()) {
360 uint j = 0;
361 for (; j < f.vertexNumber(); ++j)
362 buffer[j * FACE_NUM + i] = vIndex(f, j);
363 for (; j < largestFaceSize; ++j) // remaining vertices set to -1
364 buffer[j * FACE_NUM + i] = -1;
365 ++i;
366 }
367 }

416template<FaceMeshConcept MeshType>
417void triangulatedFaceIndicesToBuffer(
418 const MeshType& mesh,
419 auto* buffer,
420 TriPolyIndexBiMap& indexMap = detail::indexMap,
421 MatrixStorageType storage = MatrixStorageType::ROW_MAJOR,
422 uint numTriangles = UINT_NULL,
423 bool getIndicesAsIfContainerCompact = true)
424{
425 const std::vector<uint> vertCompIndices =
426 detail::vertCompactIndices(mesh, getIndicesAsIfContainerCompact);
427
428 // lambda to get the vertex index of a face (considering compact indices)
429 auto vIndex = detail::vIndexLambda(mesh, vertCompIndices);
430
431 // there will be at least a triangle for each polygon
432 indexMap.clear();
433 indexMap.reserve(mesh.faceNumber(), mesh.faceContainerSize());
434
435 if constexpr (TriangleMeshConcept<MeshType>) {
436 // construct the indexMap, which maps each triangle to the face index
437 for (uint t = 0; const auto& f : mesh.faces()) {
438 // map the ith triangle to the f face
439 indexMap.insert(t, f.index());
440 ++t;
441 }
442
443 return triangleIndicesToBuffer(
444 mesh, buffer, storage, getIndicesAsIfContainerCompact);
445 }
446 else {
447 // if the user did not give the number of triangles, and the buffer
448 // storage is column major, we need to compute the number of resulting
449 // triangles
450 if (numTriangles == UINT_NULL &&
451 storage == MatrixStorageType::COLUMN_MAJOR &&
452 mesh.faceNumber() > 0) {
453 numTriangles = countTriangulatedTriangles(mesh);
454 }
455 for (uint t = 0; const auto& f : mesh.faces()) {
456 std::vector<uint> vind = vcl::earCut(f);
457
458 // for each triangle of the triangulation (t is the triangle index)
459 for (uint vi = 0; vi < vind.size(); vi += 3) {
460 // map the t-th triangle to the f polygonal face
461 indexMap.insert(t, f.index());
462
463 if (storage == MatrixStorageType::ROW_MAJOR) {
464 buffer[t * 3 + 0] = vIndex(f, vind[vi + 0]);
465 buffer[t * 3 + 1] = vIndex(f, vind[vi + 1]);
466 buffer[t * 3 + 2] = vIndex(f, vind[vi + 2]);
467 }
468 else {
469 buffer[0 * numTriangles + t] = vIndex(f, vind[vi + 0]);
470 buffer[1 * numTriangles + t] = vIndex(f, vind[vi + 1]);
471 buffer[2 * numTriangles + t] = vIndex(f, vind[vi + 2]);
472 }
473 ++t;
474 }
475 }
476 }
477}

532 buffer[1 * EDGE_NUM + i] = vIndex(e, 1);
