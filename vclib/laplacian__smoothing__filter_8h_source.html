33class LaplacianSmoothingFilter : public FilterMeshAction
34{
35public:
36 std::shared_ptr<Action> clone() const override

41 std::string name() const override { return "Laplacian Smoothing"; }
42
43 std::string description() const override
44 {
45 return "Laplacian smooth. Average each vertex position with weighted "
46 "positions of neighbour vertices.<br><b>"
47 "Laplacian Mesh Processing</b> by <i>Olga Sorkine</i>. "
48 "EUROGRAPHICS 2005<br>"
49 "<a href='http://dx.doi.org/10.2312/egst.20051044'>doi:10.2312/"
50 "egst.20051044</a>";
51 }
52
53 vcl::BitSet<uint> categories() const override
54 {
55 return vcl::BitSet<uint>({SMOOTHING});
56 }
57
58 MeshParamVector inputMeshParameters() const final { return {}; }
59
60 MeshParamVector inputOutputMeshParameters() const final
61 {
62 std::pair<MeshParameter, BitSet<short>> par;
63 par.first = MeshParameter("input_output", "Input/Output Mesh", "");
64 par.second = BitSet<short>({MeshIType::TRI_MESH, MeshIType::POLY_MESH});
65
66 return {par};
67 }
68
69 ParameterVector parameters() const override
70 {
71 ParameterVector params;

73 params.pushBack(UintParameter(
74 "smoothing_steps",
75 3,
76 "Smoothing steps",
77 "The number of times that the whole algorithm (normal smoothing + "
78 "vertex fitting) is "
79 "iterated."));
80 params.pushBack(BoolParameter(
81 "cotangent_weighting",
82 true,
83 "Cotangent weighting",
84 "Use cotangent weighting scheme for the averaging of the position. "
85 "Otherwise the simpler umbrella scheme (1 if the edge is present) "
86 "is used."));
87 params.pushBack(BoolParameter(
88 "only_selected",
89 false,
90 "Only selected",
91 "If checked, the filter is performed only on the selected area."));
92
93 return params;
94 }
95
96 virtual OutputValues applyFilter(
97 const MeshVector,
98 const std::vector<std::shared_ptr<MeshI>>& inputOutputMeshes,
99 MeshVector&,
100 const ParameterVector& parameters,
101 AbstractLogger& log = logger()) const override
102 {
103 BitSet<short> supportedMeshTypes =
104 inputOutputMeshParameters().front().second;
105
106 uint smoothingSteps = parameters.get("smoothing_steps")->uintValue();
107 bool cotangentWeighting =
108 parameters.get("cotangent_weighting")->boolValue();
109 bool onlySelected = parameters.get("only_selected")->boolValue();
110
111 std::shared_ptr<MeshI> mesh = inputOutputMeshes[0];
112
113 auto fun = [&](auto& mesh) {
114 laplacianSmooth(
115 mesh, smoothingSteps, cotangentWeighting, onlySelected);
116 };
117
118 callFunctionForSupportedInputOutputMeshTypes(
119 *mesh, supportedMeshTypes, fun);
120
121 return OutputValues();
122 }
123
124private:
125 void laplacianSmooth(
126 MeshConcept auto& mesh,
127 uint smoothingSteps,
128 bool cotangentWeighting,
129 bool onlySelected) const
130 {
131 vcl::laplacianSmoothing(
132 mesh, smoothingSteps, onlySelected, cotangentWeighting);
133
134 updateBoxAndNormals(mesh);
135 }
136};
