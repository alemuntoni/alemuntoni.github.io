31template<Point3Concept PointType, PlaneConcept PlaneType>
32auto pointPlaneDistance(
33 const PointType& p,
34 const PlaneType& plane,
35 bool signedDist = false)
36{
37 auto dist = plane.direction().dot(p) - plane.offset();
38 if (!signedDist)
39 dist = std::abs(dist);
40 return dist;
41}

43template<PointConcept PointType, SegmentConcept SegmentType>
44auto pointSegmentDistance(
45 const PointType& p,
46 const SegmentType& s,
47 PointType& closest) requires (PointType::DIM == SegmentType::DIM)
48{
49 using ScalarType = PointType::ScalarType;
50
51 ScalarType dist;
52
53 PointType dir = s.direction();
54 ScalarType esn = dir.squaredNorm();
55
56 if (esn < std::numeric_limits<ScalarType>::min()) {
57 closest = s.midPoint();
58 }
59 else {
60 ScalarType t = ((p - s.p0()).dot(dir)) / esn;
61 if (t < 0)
62 t = 0;
63 else if (t > 1)
64 t = 1;
65
66 closest = s.p0() * (1 - t) + s.p1() * t;
67 }
68 dist = p.dist(closest);
69 return dist;
70}

74 requires (PointType::DIM == SegmentType::DIM)
75{
76 PointType closest;
77 return pointSegmentDistance(p, s, closest);
