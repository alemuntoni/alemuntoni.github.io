33template<typename FaceType, bool CNST = false>
34class EdgeAdjFaceIterator
35{
36 using FT = std::conditional_t<CNST, FaceType, const FaceType>;
37
38 using VT = std::conditional_t<CNST,
39 typename FT::VertexType,
40 const typename FT::VertexType>;
41
42 FT* current = nullptr;
43 FT* end = nullptr;
44 VT* v0 = nullptr;
45 VT* v1 = nullptr;
46
47public:
48 using difference_type = ptrdiff_t;
49 using iterator_category = std::forward_iterator_tag;
50 using value_type = FT*;
51 using reference = FT*&;
52 using pointer = FT**;
53
54 EdgeAdjFaceIterator() = default;
55
56 EdgeAdjFaceIterator(FT& f, uint edge) :
57 current(&f), end(&f), v0(f.vertex(edge)), v1(f.vertexMod(edge + 1))
58 {
59 }
60
61 bool operator==(const EdgeAdjFaceIterator& oi) const
62 {
63 return current == oi.current && v0 == oi.v0 && v1 == oi.v1;
64 }
65
66 bool operator!=(const EdgeAdjFaceIterator& oi) const
67 {
68 return !(*this == oi);
69 }
70
71 EdgeAdjFaceIterator& operator++()
72 {
73 assert(current);
74 uint edge = current->indexOfEdge(v0, v1);
75 assert(edge != UINT_NULL);
76 current = current->adjFace(edge);
77 if (current == end || current == nullptr) {
78 current = nullptr;
79 v0 = nullptr;
80 v1 = nullptr;
81 }
82 return *this;
83 }
84
85 EdgeAdjFaceIterator operator++(int)
86 {
87 auto it = *this;
88 ++(*this);
89 return it;
90 }
91
92 reference operator*() const { return current; }
93
94 pointer operator->() const { return &current; }
95};
96
97template<typename FaceType>
98using ConstEdgeAdjFaceIterator = EdgeAdjFaceIterator<FaceType, true>;
