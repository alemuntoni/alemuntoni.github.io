39template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
40void savePly(
41 const MeshType& m,
42 const std::string& filename,
43 const MeshInfo& info,
44 LogType& log = nullLogger,
45 bool binary = true)
46{
47 MeshInfo meshInfo(m);
48
49 // make sure that the given info contains only components that are actually
50 // available in the mesh. meshInfo will contain the intersection between the
51 // components that the user wants to save and the components that are
52 // available in the mesh.
53 meshInfo = info.intersect(meshInfo);
54
55 ply::PlyHeader header(binary ? ply::BINARY : ply::ASCII, meshInfo);
56 header.setNumberVertices(m.vertexNumber());
57
58 if constexpr (vcl::HasFaces<MeshType>) {
59 if (header.hasFaces()) {
60 header.setNumberFaces(m.faceNumber());
61 }
62 }
63 if constexpr (vcl::HasEdges<MeshType>) {
64 if (header.hasEdges()) {
65 header.setNumberEdges(m.edgeNumber());
66 }
67 }
68 ply::saveTextures(header, m);
69
70 if (!header.isValid())
71 throw std::runtime_error("Ply Header not valid.");
72
73 std::ofstream fp = internal::saveFileStream(filename, "ply");
74
75 fp << header.toString();
76
77 ply::saveVertices(fp, header, m);
78
79 if constexpr (vcl::HasFaces<MeshType>) {
80 if (header.hasFaces()) {
81 ply::saveFaces(fp, header, m);
82 }
83 }
84
85 if constexpr (vcl::HasEdges<MeshType>) {
86 if (header.hasEdges()) {
87 ply::saveEdges(fp, header, m);
88 }
89 }
90
91 fp.close();
92}
