101template<typename T>
102T readProperty(std::ifstream& file, PropertyType type, bool isColor = false)
103{
104 T p;
105 switch (type) {
106 case CHAR: p = readChar<T>(file); break;
107 case UCHAR: p = readUChar<T>(file); break;
108 case SHORT: p = readShort<T>(file); break;
109 case USHORT: p = readUShort<T>(file); break;
110 case INT: p = readInt<T>(file); break;
111 case UINT: p = readUInt<T>(file); break;
112 case FLOAT: p = readFloat<T>(file, isColor); break;
113 case DOUBLE: p = readDouble<T>(file, isColor); break;
114 default: assert(0); p = 0;
115 }
116 // if I read a color that must be returned as a float or double
117 if (isColor && !std::is_integral<T>::value)
118 p = (float) p / 255.0;
119 return p;
120}

218template<typename T>
219T readProperty(
220 vcl::Tokenizer::iterator& token,
221 PropertyType type,
222 bool isColor = false)
223{
224 T p;
225 switch (type) {
226 case CHAR:
227 case UCHAR:
228 case SHORT:
229 case USHORT:
230 case INT:
231 case UINT: p = std::stoi(*token++); break;
232 case FLOAT:
233 case DOUBLE:
234 if (isColor) {
235 p = std::stod(*token++) * 255;
236 }
237 else {
238 p = std::stod(*token++);
239 }
240 break;
241 default: assert(0); p = 0;
242 }
243 // if I read a color that must be returned as a float or double
244 if (isColor && !std::is_integral<T>::value)
245 p = (float) p / 255.0;
246 return p;
247}
