73 bool load(const std::string& filename)
74 {
75 int w, h;
76 // we first load the data, then we copy it into our array2d, and then we
77 // free it.
78 unsigned char* tmp =
79 stbi_load(filename.c_str(), &w, &h, nullptr, 4); // force 4 channels
80 if (tmp) {
81 std::size_t size = w * h * 4;
82
83 img.resize(w, h);
84 std::copy(tmp, tmp + size, (unsigned char*)img.data());
85 stbi_image_free(tmp);
86 return true;
87 }
88 else {
89 return false;
90 }
91 }
92
93 void mirror(bool horizontal = false, bool vertical = true)
94 {
95 if (horizontal) {
96 for (uint i = 0; i < img.rows(); i++) {
97 std::reverse(img.data(i), img.data(i) + img.cols());
98 }
99 }
100 if (vertical) {
101 for (uint i = 0; i < img.rows() / 2; i++) {
102 uint mir = img.rows() - i - 1;
103 std::swap_ranges(
104 img.data(i), img.data(i) + img.cols(), img.data(mir));
105 }
106 }
107 }
108};
