38template<MeshConcept MeshType>
39class DrawableMeshBGFX : public AbstractDrawableMesh, public MeshType
40{
41 Box3d mBoundingBox;
42
43 MeshRenderBuffers<MeshType> mMRB;
44
45 bgfx::ProgramHandle mProgram =
46 Context::instance().programManager().getProgram(
47 VclProgram::DRAWABLE_MESH);
48
49 DrawableMeshUniforms mMeshUniforms;
50 mutable MeshRenderSettingsUniforms mMeshRenderSettingsUniforms;
51
52public:
53 using AbstractDrawableMesh::name;
54
55 DrawableMeshBGFX() = default;
56
57 DrawableMeshBGFX(const MeshType& mesh) :
58 AbstractDrawableMesh(mesh), MeshType(mesh)
59 {
60 updateBuffers();
61 }
62
63 ~DrawableMeshBGFX() = default;
64
65 void updateBuffers() override
66 {
67 if constexpr (HasName<MeshType>) {
68 AbstractDrawableMesh::name() = MeshType::name();
69 }
70
71 bool bbToInitialize = !vcl::HasBoundingBox<MeshType>;
72 if constexpr (vcl::HasBoundingBox<MeshType>) {
73 if (this->MeshType::boundingBox().isNull()) {
74 bbToInitialize = true;
75 }
76 else {
77 mBoundingBox =
78 this->MeshType::boundingBox().template cast<double>();
79 }
80 }
81
82 if (bbToInitialize) {
83 mBoundingBox = vcl::boundingBox(*this);
84 }
85
86 mMRB = MeshRenderBuffers<MeshType>(*this);
87 mMRS.setRenderCapabilityFrom(*this);
88 mMeshRenderSettingsUniforms.updateSettings(mMRS);
89 mMeshUniforms.update(*this);
90 }
91
92 void swap(DrawableMeshBGFX& other)
93 {
94 using std::swap;
95 AbstractDrawableMesh::swap(other);
96 MeshType::swap(other);
97 swap(mBoundingBox, other.mBoundingBox);
98 swap(mMRB, other.mMRB);
99 swap(mProgram, other.mProgram);
100 swap(mMeshUniforms, other.mMeshUniforms);
101 swap(mMeshRenderSettingsUniforms, other.mMeshRenderSettingsUniforms);
102 }
103
104 friend void swap(DrawableMeshBGFX& a, DrawableMeshBGFX& b) { a.swap(b); }
105
106 // DrawableObject implementation
107
108 void init() override {}
109
110 void draw(uint viewId) const override
111 {
112 if (bgfx::isValid(mProgram)) {
113 uint64_t state = 0 | BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A |
114 BGFX_STATE_WRITE_Z | BGFX_STATE_DEPTH_TEST_LEQUAL |
115 BGFX_STATE_BLEND_NORMAL;
116
117 if (mMRS.isSurfaceVisible()) {
118 mMRB.bindTextures(); // Bind textures before vertex buffers!!
119 mMRB.bindVertexBuffers(mMRS);
120 mMRB.bindIndexBuffers();
121 bindUniforms(VCL_MRS_DRAWING_SURFACE);
122
123 bgfx::setState(state);
124
125 bgfx::submit(viewId, mProgram);
126 }
127
128 if (mMRS.isWireframeVisible()) {
129 mMRB.bindVertexBuffers(mMRS);
130 mMRB.bindIndexBuffers(MeshBufferId::WIREFRAME);
131 bindUniforms(VCL_MRS_DRAWING_WIREFRAME);
132
133 bgfx::setState(state | BGFX_STATE_PT_LINES);
134
135 bgfx::submit(viewId, mProgram);
136 }
137
138 if (mMRS.isPointVisible()) {
139 mMRB.bindVertexBuffers(mMRS);
140 bindUniforms(VCL_MRS_DRAWING_POINTS);
141
142 bgfx::setState(state | BGFX_STATE_PT_POINTS);
143
144 bgfx::submit(viewId, mProgram);
145 }
146
147 if (mMRS.isEdgesVisible()) {
148 mMRB.bindVertexBuffers(mMRS);
149 mMRB.bindIndexBuffers(MeshBufferId::EDGES);
150 bindUniforms(VCL_MRS_DRAWING_EDGES);
151
152 bgfx::setState(state | BGFX_STATE_PT_LINES);
153
154 bgfx::submit(viewId, mProgram);
155 }
156 }
157 }
158
159 Box3d boundingBox() const override { return mBoundingBox; }
160
161 std::shared_ptr<DrawableObject> clone() const override
162 {
163 return std::make_shared<DrawableMeshBGFX>(*this);
164 }
165
166 void setVisibility(bool vis) override
167 {
168 AbstractDrawableMesh::setVisibility(vis);
169 mMeshRenderSettingsUniforms.updateSettings(mMRS);
170 }
171
172 void setRenderSettings(const MeshRenderSettings& rs) override
173 {
174 AbstractDrawableMesh::setRenderSettings(rs);
175 mMeshRenderSettingsUniforms.updateSettings(rs);
176 }
177
178private:
179 void bindUniforms(uint primitive) const
180 {
181 mMeshRenderSettingsUniforms.updatePrimitive(primitive);
182 mMeshRenderSettingsUniforms.bind();
183 mMeshUniforms.bind();
184 }
185};
