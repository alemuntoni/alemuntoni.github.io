147void loadOffHeader(
148 std::ifstream& file,
149 MeshInfo& fileInfo,
150 uint& nv,
151 uint& nf,
152 uint& ne)
153{
154 fileInfo.reset();
155 vcl::Tokenizer tokens = internal::nextNonEmptyTokenizedLine(file);
156 vcl::Tokenizer::iterator token = tokens.begin();
157 std::string header = *token;
158
159 // the OFF string is in the header go on parsing it.
160 if (header.rfind("OFF") != std::basic_string<char>::npos) {
161 for (int u = header.rfind("OFF") - 1; u >= 0; u--) {
162 if (header[u] == 'C')
163 fileInfo.setVertexColors();
164 else if (header[u] == 'N')
165 fileInfo.setVertexNormals();
166 else if (u > 0 && header[u - 1] == 'S' && header[u] == 'T')
167 fileInfo.setVertexTexCoords();
168 else if (header[u] == '4')
169 throw vcl::MalformedFileException(
170 "Unsupported Homogeneus components in OFF.");
171 else if (header[u] == 'n')
172 throw vcl::MalformedFileException(
173 "Unsupported High Dimension OFF.");
174 }
175 }
176 else
177 throw vcl::MalformedFileException("Missing OFF header in file.");
178
179 // If the file is slightly malformed and it has nvert and nface AFTER the
180 // OFF string instead of in the next line we manage it here...
181 if (tokens.size() == 1) {
182 tokens = internal::nextNonEmptyTokenizedLine(file);
183 token = tokens.begin();
184 }
185 else
186 ++token;
187
188 nv = io::internal::readUInt<uint>(token);
189 nf = io::internal::readUInt<uint>(token);
190 ne = io::internal::readUInt<uint>(token);
191 if (nv > 0)
192 fileInfo.setVertices();
193 if (nf > 0)
194 fileInfo.setFaces();
195 // if (ne > 0)
196 // loadedInfo.setEdges();
197}

199inline vcl::Color
200loadColor(vcl::Tokenizer::iterator& token, int nColorComponents)
201{
202 uint red, green, blue, alpha = 255;
203
204 if (nColorComponents == 1) {
205 uint k = io::internal::readUInt<uint>(token);
206
207 red = GEOMVIEW_COLOR_MAP[k][0]*255;
208 green = GEOMVIEW_COLOR_MAP[k][1]*255;
209 blue = GEOMVIEW_COLOR_MAP[k][2]*255;
210 alpha = GEOMVIEW_COLOR_MAP[k][3]*255;
211 }
212 else {
213 double r = io::internal::readDouble<double>(token);
214 double g = io::internal::readDouble<double>(token);
215 double b = io::internal::readDouble<double>(token);
216 double a = -1;
217 if (nColorComponents == 4) {
218 a = io::internal::readDouble<double>(token);
219 }
220 if (r > 1 || g > 1 || b > 1){
221 red = r;
222 green = g;
223 blue = b;
224 alpha = a != -1 ? a : alpha;
225 }
226 else {
227 red = r * 255;
228 green = g * 255;
229 blue = b * 255;
230 alpha = a != -1 ? a * 255 : alpha;
231 }
232 }
233 return vcl::Color(red, green, blue, alpha);
234}
