36template<MatrixConcept FMatrix>
37std::vector<uint> faceVertIndices(const FMatrix& faces, uint f)
38{
39 std::vector<uint> fVerts;
40
41 uint j = 0;
42 while (j < faces.cols() && faces(f, j) != -1 && faces(f, j) != UINT_NULL)
43 fVerts.push_back(faces(f, j));
44
45 return fVerts;
46}

48template<uint ELEM_ID, MeshConcept MeshType, MatrixConcept NMatrix>
49void importElementNormalsFromMatrix(
50 MeshType& mesh,
51 const NMatrix& normals)
52{
53 // The type of the normal of the element
54 using NormalType = MeshType::template ElementType<ELEM_ID>::NormalType;
55
56 if (normals.cols() != 3)
57 throw WrongSizeException(
58 "The input " + std::string(elementEnumString<ELEM_ID>()) +
59 " normal matrix must have 3 columns");
60
61 // matrix rows must be equal to the number of elements of the given type
62 if (normals.rows() != mesh.template number<ELEM_ID>())
63 throw WrongSizeException(
64 "The input normal matrix must have the same number of rows "
65 "as the number of " +
66 std::string(elementEnumString<ELEM_ID>()) + " element in the mesh");
67
68 enableIfPerElementComponentOptional<ELEM_ID, NORMAL>(mesh);
69 requirePerElementComponent<ELEM_ID, NORMAL>(mesh);
70
71 uint i = 0;
72 for (auto& e : mesh.template elements<ELEM_ID>()) {
73 e.normal() = NormalType(
74 normals(i, 0), normals(i, 1), normals(i, 2));
75 i++;
76 }
77}

79template<uint ELEM_ID, MeshConcept MeshType, MatrixConcept CMatrix>
80void importElementColorsFromMatrix(
81 MeshType& mesh,
82 const CMatrix& colors)
83{
84 using MatrixScalar = CMatrix::Scalar;
85
86 if (colors.cols() != 3 && colors.cols() != 4)
87 throw WrongSizeException(
88 "The input " + std::string(elementEnumString<ELEM_ID>()) +
89 " color matrix must have 3 or 4 columns");
90
91 // matrix rows must be equal to the number of elements of the given type
92 if (colors.rows() != mesh.template number<ELEM_ID>())
93 throw WrongSizeException(
94 "The input color matrix must have the same number of rows "
95 "as the number of " +
96 std::string(elementEnumString<ELEM_ID>()) + " element in the mesh");
97
98 enableIfPerElementComponentOptional<ELEM_ID, COLOR>(mesh);
99 requirePerElementComponent<ELEM_ID, COLOR>(mesh);
100
101 uint i = 0;
102 for (auto& e : mesh.template elements<ELEM_ID>()) {
103 // Matrix has colors in range 0-255
104 if constexpr (std::integral<MatrixScalar>) {
105 if (colors.cols() == 3) {
106 e.color() =
107 vcl::Color(colors(i, 0), colors(i, 1), colors(i, 2));
108 }
109 else {
110 e.color() = vcl::Color(
111 colors(i, 0), colors(i, 1), colors(i, 2), colors(i, 3));
112 }
113 }
114 else { // Matrix has colors in range 0-1
115 if (colors.cols() == 3) {
116 e.color() = vcl::Color(
117 colors(i, 0) * 255, colors(i, 1) * 255, colors(i, 2) * 255);
118 }
119 else {
120 e.color() = vcl::Color(
121 colors(i, 0) * 255,
122 colors(i, 1) * 255,
123 colors(i, 2) * 255,
124 colors(i, 3) * 255);
125 }
126 }
127
128 i++;
129 }
130}

318template<
319 MeshConcept MeshType,
320 MatrixConcept VMatrix,
321 MatrixConcept FMatrix = Eigen::MatrixX3i,
322 MatrixConcept EMatrix = Eigen::MatrixX2i,
323 MatrixConcept VNMatrix = Eigen::MatrixX3d,
324 MatrixConcept FNMatrix = Eigen::MatrixX3d>
325void importMeshFromMatrices(
326 MeshType& mesh,
327 const VMatrix& vertices,
328 const FMatrix& faces = FMatrix(),
329 const EMatrix& edges = EMatrix(),
330 const VNMatrix& vertexNormals = VNMatrix(),
331 const FNMatrix& faceNormals = FNMatrix())
332{
333 mesh.clear();
334 mesh.disableAllOptionalComponents();
335
336 importVerticesFromMatrix(mesh, vertices);
337
338 if constexpr (HasPerVertexNormal<MeshType>) {
339 if (vertexNormals.rows() > 0) {
340 importVertexNormalsFromMatrix(mesh, vertexNormals);
341 }
342 }
343
344 if constexpr (HasFaces<MeshType>) {
345 if (faces.rows() > 0)
346 importFacesFromMatrix(mesh, faces);
347
348 if constexpr (HasPerFaceNormal<MeshType>) {
349 if (faceNormals.rows() > 0) {
350 importFaceNormalsFromMatrix(mesh, faceNormals);
351 }
352 }
353 }
354
355 if constexpr (HasEdges<MeshType>) {
356 if (edges.rows() > 0)
357 importEdgesFromMatrix(mesh, edges);
358 }
359}

399template<MeshConcept MeshType, MatrixConcept VMatrix>
400void importVerticesFromMatrix(
401 MeshType& mesh,
402 const VMatrix& vertices,
403 bool clearBeforeSet = true)
404{
405 using CoordType = MeshType::VertexType::CoordType;
406
407 if (vertices.cols() != 3)
408 throw WrongSizeException("The input vertex matrix must have 3 columns");
409
410 if (clearBeforeSet) {
411 mesh.clearVertices();
412 mesh.resizeVertices(vertices.rows());
413 }
414 else {
415 if (vertices.rows() != mesh.vertexNumber()) {
416 throw WrongSizeException(
417 "The input vertex matrix has a different number of rows than "
418 "the number of vertices of the mesh");
419 }
420 }
421
422
423 uint i = 0;
424 for (auto& v : mesh.vertices()) {
425 v.coord() = CoordType(vertices(i, 0), vertices(i, 1), vertices(i, 2));
426 i++;
427 }
428}

430template<FaceMeshConcept MeshType, MatrixConcept FMatrix>
431void importFacesFromMatrix(
432 MeshType& mesh,
433 const FMatrix& faces,
434 bool clearBeforeSet = true)
435{
436 if (clearBeforeSet) {
437 mesh.clearFaces();
438 mesh.resizeFaces(faces.rows());
439 }
440 else {
441 if (faces.rows() != mesh.faceNumber()) {
442 throw WrongSizeException(
443 "The input face matrix has a different number of rows "
444 "than the number of faces of the mesh.");
445 }
446 }
447
448 if constexpr (HasPolygons<MeshType>) {
449 uint i = 0;
450 for (auto& f : mesh.faces()) {
451 uint vertexNumber = 0;
452
453 // count the number of vertices of the face
454 while (vertexNumber < faces.cols() &&
455 faces(i, vertexNumber) != -1 &&
456 faces(i, vertexNumber) != UINT_NULL)
457 vertexNumber++;
458
459 f.resizeVertices(vertexNumber);
460
461 for (uint j = 0; j < vertexNumber; ++j)
462 f.vertex(j) = &mesh.vertex(faces(i, j));
463 }
464 }
465 else { // the vertex number of mesh faces is fixed
466 using FaceType = MeshType::FaceType;
467
468 constexpr int VN = FaceType::VERTEX_NUMBER;
469 if (faces.cols() == VN) { // faces of matrix and mesh have same size
470
471 uint i = 0;
472 for (auto& f : mesh.faces()) {
473 for (uint j = 0; j < VN; ++j)
474 f.vertex(j) = &mesh.vertex(faces(i, j));
475 }
476 }
477 else { // faces of matrix and mesh have different sizes
478 // matrix cols is higher than 3 (polygons), but we have a triangle
479 // mesh and we can triangulate the faces
480 if constexpr (VN == 3) {
481 if (!clearBeforeSet) {
482 throw WrongSizeException(
483 "Cannot import the input face matrix into the mesh "
484 "without clearing the face container first "
485 "(need to perform a triangulation to import polygons "
486 "in a triangle mesh, and this operation that does not "
487 "guarantee a predefined number of faces is required).");
488 }
489 mesh.reserveFaces(faces.rows());
490 for (uint i = 0; i < faces.rows(); ++i) {
491 std::vector<uint> fVertIndices =
492 internal::faceVertIndices(faces, i);
493
494 addTriangleFacesFromPolygon(mesh, fVertIndices);
495 }
496 }
497 else {
498 // no triangulation available because VN != 3, we don't
499 // know how to import the faces and we throw an exception
500 throw WrongSizeException(
501 "The input face matrix has a different number of columns "
502 "than the number of vertices of the mesh faces.");
503 }
504 }
505 }
506}

508template<EdgeMeshConcept MeshType, MatrixConcept EMatrix>
509void importEdgesFromMatrix(
510 MeshType& mesh,
511 const EMatrix& edges,
512 bool clearBeforeSet = true)
513{
514 if (edges.cols() != 2)
515 throw WrongSizeException(
516 "The input edge matrix must have 2 columns");
517
518 if (clearBeforeSet) {
519 mesh.clearEdges();
520 mesh.resizeEdges(edges.rows());
521 }
522 else {
523 if (edges.rows() != mesh.edgeNumber()) {
524 throw WrongSizeException(
525 "The input edge matrix has a different number of rows than "
526 "the number of edges of the mesh");
527 }
528 }
529
530 uint i = 0;
531 for (auto& e : mesh.edges()) {
532 e.vertex(0) = &mesh.vertex(edges(i, 0));
533 e.vertex(1) = &mesh.vertex(edges(i, 1));
534 i++;
535 }
536}
