197 bool overlap(const Box<PointT>& b) const
198 {
199 // Check if any dimension of the boxes does not overlap.
200 for (uint i = 0; i < PointT::DIM; ++i) {
201 if (b.mMin[i] >= mMax[i] || b.mMax[i] <= mMin[i])
202 return false;
203 }
204 // All dimensions overlap, the boxes collide.
205 return true;

257 auto volume() const
258 {
259 // Initialize volume as the width of the first dimension.

322 Box<PointT> intersection(const Box<PointT>& b) const
323 {
324 Box<PointT> res = *this;
325 for (uint i = 0; i < PointT::DIM; ++i) {
326 if (mMin[i] < b.mMin[i]) {
327 // set the minimum point to the larger value
328 res.mMin[i] = b.mMin[i];
329 }
330 if (mMax[i] > b.mMax[i]) {
331 // set the maximum point to the smaller value
332 res.mMax[i] = b.mMax[i];
333 }
334 if (res.mMin[i] > res.mMax[i]) {
335 // if the minimum point is larger than the maximum point, the
336 // box is null
337 res.setNull();
338 return res;
339 }
340 }
341 return res;
342 }

368 void add(const PointT& p)
369 {
370 if (isNull()) {
371 // If the box was null, set both the minimum and maximum points to
372 // the given point
373 *this = Box(p);
374 }
375 else {
376 // Expand the current box to include the added point
377 mMin = vcl::min(mMin, p);
378 mMax = vcl::max(mMax, p);
379 }
380 }
381
394 template<typename Scalar>
395 void add(const PointT& p, Scalar radius)
396 {
397 if (isNull()) {
398 // If the box was null, initialize it with the minimum and maximum
399 // points obtained by subtracting and adding the radius to the given
400 // point
401 *this = Box(p - radius, p + radius);
402 }
403 else {
404 // Expand the current box to include the added point and all points
405 // within the radius around it
406 mMin = vcl::min(mMin, p - radius);
407 mMax = vcl::max(mMax, p + radius);
408 }
409 }
410
419 void add(const Box<PointT>& b)
420 {
421 // Adding a null box should do nothing
422 if (!b.isNull()) {
423 if (isNull()) {
424 *this = b; // if the current box is null, set it equal to b
425 }
426 else {
427 // expand this box to contain the minimum and maximum point of b
428 add(b.mMin);
429 add(b.mMax);
430 }
431 }
432 }
