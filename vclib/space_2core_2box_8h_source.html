168 bool isInside(const PointT& p) const

199 bool overlap(const Box<PointT>& b) const
200 {
201 // Check if any dimension of the boxes does not overlap.
202 for (uint i = 0; i < PointT::DIM; ++i) {
203 if (b.mMin[i] >= mMax[i] || b.mMax[i] <= mMin[i])
204 return false;
205 }
206 // All dimensions overlap, the boxes collide.
207 return true;
208 }

259 auto volume() const
260 {
261 // Initialize volume as the width of the first dimension.
262 auto vol = mMax[0] - mMin[0];
263 // Multiply by the width of each dimension.
264 for (uint i = 1; i < PointT::DIM; ++i) {

324 Box<PointT> intersection(const Box<PointT>& b) const
325 {
326 Box<PointT> res = *this;
327 for (uint i = 0; i < PointT::DIM; ++i) {
328 if (mMin[i] < b.mMin[i]) {
329 // set the minimum point to the larger value
330 res.mMin[i] = b.mMin[i];
331 }
332 if (mMax[i] > b.mMax[i]) {
333 // set the maximum point to the smaller value
334 res.mMax[i] = b.mMax[i];
335 }
336 if (res.mMin[i] > res.mMax[i]) {
337 // if the minimum point is larger than the maximum point, the
338 // box is null
339 res.setNull();
340 return res;
341 }
342 }
343 return res;
344 }
345
353 void setNull()
354 {
355 mMin.setConstant(
356 std::numeric_limits<typename PointT::ScalarType>::max());
357 mMax.setConstant(
358 std::numeric_limits<typename PointT::ScalarType>::lowest());
359 }
360
370 void add(const PointT& p)
371 {
372 if (isNull()) {
373 // If the box was null, set both the minimum and maximum points to
374 // the given point
375 *this = Box(p);
376 }
377 else {
378 // Expand the current box to include the added point
379 mMin = vcl::min(mMin, p);
380 mMax = vcl::max(mMax, p);
381 }
382 }
383
396 template<typename Scalar>
397 void add(const PointT& p, Scalar radius)
398 {
399 if (isNull()) {
400 // If the box was null, initialize it with the minimum and maximum
401 // points obtained by subtracting and adding the radius to the given
402 // point
403 *this = Box(p - radius, p + radius);
404 }
405 else {
406 // Expand the current box to include the added point and all points
407 // within the radius around it
408 mMin = vcl::min(mMin, p - radius);
409 mMax = vcl::max(mMax, p + radius);
410 }
411 }
412
421 void add(const Box<PointT>& b)
422 {
423 // Adding a null box should do nothing
424 if (!b.isNull()) {
425 if (isNull()) {
426 *this = b; // if the current box is null, set it equal to b
427 }
428 else {
429 // expand this box to contain the minimum and maximum point of b
430 add(b.mMin);
431 add(b.mMax);
432 }
433 }
434 }
435
441 void translate(const PointT& p)
442 {
443 mMin += p; // translate the minimum point by p
444 mMax += p; // translate the maximum point by p
445 }

473 bool operator==(const Box<PointT>& b) const
474 {
475 return mMin == b.mMin && mMax == b.mMax;
476 }
477
484 bool operator!=(const Box<PointT>& b) const { return !(*this == b); }
485};
