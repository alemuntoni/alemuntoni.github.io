48template<MeshConcept MeshType>
49void loadMaterials(
50 std::map<std::string, obj::Material>& materialMap,
51 MeshType& mesh,
52 const std::string& mtllib)
53{
54 std::ifstream file = loadFileStream(mtllib);
55 std::string matName;
56 obj::Material mat;
57
58 do {
59 vcl::Tokenizer tokens = nextNonEmptyTokenizedLineNoThrow(file);
60 if (file) {
61 // counter for texture images, used when mesh has no texture files
62 uint nt = 0;
63 vcl::Tokenizer::iterator token = tokens.begin();
64 std::string header = *token++;
65 if (header == "newmtl"){
66 if (!matName.empty())
67 materialMap[matName] = mat;
68 mat = obj::Material();
69 matName = *token;
70 }
71 if (header == "Ka") {
72 if (tokens.size() >= 4) {
73 if (*token != "spectral" && *token != "xyz") {
74 mat.Ka.x() = readFloat<float>(token);
75 mat.Ka.y() = readFloat<float>(token);
76 mat.Ka.z() = readFloat<float>(token);
77 }
78 }
79 }
80 if (header == "Kd") {
81 if (tokens.size() >= 4) {
82 if (*token != "spectral" && *token != "xyz") {
83 mat.Kd.x() = readFloat<float>(token);
84 mat.Kd.y() = readFloat<float>(token);
85 mat.Kd.z() = readFloat<float>(token);
86 mat.hasColor = true;
87 }
88 }
89 }
90 if (header == "Ks") {
91 if (tokens.size() >= 4) {
92 if (*token != "spectral" && *token != "xyz") {
93 mat.Ks.x() = readFloat<float>(token);
94 mat.Ks.y() = readFloat<float>(token);
95 mat.Ks.z() = readFloat<float>(token);
96 }
97 }
98 }
99 if (header == "d") {
100 if ((*token)[0] == '-') token++;
101 mat.d = readFloat<float>(token);
102 }
103 if (header == "Tr") {
104 if ((*token)[0] == '-') token++;
105 mat.d = 1 - readFloat<float>(token);
106 }
107 if (header == "Ns") {
108 mat.Ns = readFloat<float>(token);
109 }
110 if (header == "illum") {
111 mat.illum = readFloat<int>(token);
112 }
113 if (header == "map_Kd") {
114 // need to manage args
115 while ((*token)[0] == '-') {
116 if (*token == "-o" || *token == "-s" || *token == "-t"){
117 // ignore the argument and the three values
118 ++token; ++token; ++token; ++token;
119 }
120 if (*token == "-mm"){
121 // ignore the argument and the two values
122 ++token; ++token; ++token;
123 }
124 if (*token == "-blendu" || *token == "-blendv" ||
125 *token == "-cc" || *token == "-clamp" ||
126 *token == "-texres")
127 {
128 // ignore the argument and the value
129 ++token; ++token;
130 }
131 }
132 mat.map_Kd = *token;
133 mat.hasTexture = true;
134 if constexpr (HasTexturePaths<MeshType>) {
135 mat.mapId = mesh.textureNumber();
136 mesh.pushTexturePath(mat.map_Kd);
137 }
138 else {
139 mat.mapId = nt++;
140 }
141 }
142 }
143 } while(file);
144 if (!matName.empty())
145 materialMap[matName] = mat;
146}

148template<MeshConcept MeshType>
149void loadVertexCoord(
150 MeshType& m,
151 vcl::Tokenizer::iterator& token,
152 MeshInfo& loadedInfo,
153 const vcl::Tokenizer& tokens,
154 const obj::Material& currentMaterial,
155 bool enableOptionalComponents)
156{
157 // first, need to set that I'm loading vertices
158 if (m.vertexNumber() == 0) {
159 loadedInfo.setVertices();
160 loadedInfo.setVertexCoords();
161 }
162 uint vid = m.addVertex();
163 for (uint i = 0; i < 3; ++i) {
164 m.vertex(vid).coord()[i] = internal::readDouble<double>(token);
165 }
166 if constexpr (HasPerVertexColor<MeshType>){
167 if (vid == 0) {
168 // if the current material has a valid color, of the file stores the
169 // vertex color in the non-standard way (color values after the
170 // coordinates)
171 if (currentMaterial.hasColor || tokens.size() > 6) {
172 if (enableOptionalComponents) {
173 enableIfPerVertexColorOptional(m);
174 loadedInfo.setVertexColors();
175 }
176 else {
177 if (isPerVertexColorAvailable(m))
178 loadedInfo.setVertexColors();
179 }
180 }
181 }
182 if (loadedInfo.hasVertexColors()) {
183 // the file has the nonstandard way to store vertex colors, after
184 // the coords...
185 if (tokens.size() > 6) {
186 m.vertex(vid).color().setRedF(
187 internal::readFloat<float>(token));
188 m.vertex(vid).color().setGreenF(
189 internal::readFloat<float>(token));
190 m.vertex(vid).color().setBlueF(
191 internal::readFloat<float>(token));
192 }
193 else if (currentMaterial.hasColor) {
194 m.vertex(vid).color() = currentMaterial.color();
195 }
196 }
197 }

200template<MeshConcept MeshType>
201void loadVertexNormal(
202 MeshType& m,
203 internal::NormalsMap<MeshType>& mapNormalsCache,
204 uint vn,
205 vcl::Tokenizer::iterator& token,
206 MeshInfo& loadedInfo,
207 bool enableOptionalComponents)
208{
209 using NormalType = MeshType::VertexType::NormalType;
210
211 // first, need to check if I can store normals in the mesh
212 if (vn == 0) {
213 if (enableOptionalComponents) {
214 enableIfPerVertexNormalOptional(m);
215 loadedInfo.setVertexNormals();
216 }
217 else {
218 if (isPerVertexNormalAvailable(m))
219 loadedInfo.setVertexNormals();
220 }
221 }
222 if (loadedInfo.hasVertexNormals()) {
223 // read the normal
224 NormalType n;
225 for (uint i = 0; i < 3; ++i) {
226 n[i] = internal::readDouble<typename NormalType::ScalarType>(token);
227 }
228 // I can store the normal in its vertex
229 if (m.vertexNumber() > vn) {
230 m.vertex(vn).normal() = n;
231 }
232 // read the normal and save it in the cache map, because we still don't
233 // have read the vertex corresponding to the current normal
234 else {
235 mapNormalsCache[vn] = n;
236 }
237 }

240template<FaceMeshConcept MeshType>
241void loadFace(
242 MeshType& m,
243 MeshInfo& loadedInfo,
244 const vcl::Tokenizer& tokens,
245 const std::vector<vcl::TexCoordd>& wedgeTexCoords,
246 const obj::Material& currentMaterial,
247 bool enableOptionalComponents)
248{
249 using FaceType = MeshType::FaceType;
250
251 std::vector<uint> vids;
252 std::vector<uint> wids;
253
254 // actual read - load vertex indices and texcoords indices, if present
255 vcl::Tokenizer::iterator token = tokens.begin();
256 ++token;
257 vids.resize(tokens.size()-1);
258 wids.reserve(tokens.size()-1);
259 for (uint i = 0; i < tokens.size()-1; ++i) {
260 vcl::Tokenizer subt(*token, '/', false);
261 auto t = subt.begin();
262 vids[i] = readUInt<uint>(t) - 1;
263 if (subt.size() > 1) {
264 if (!t->empty()) {
265 wids.push_back(readUInt<uint>(t) - 1);
266 }
267 }
268 ++token;
269 }
270
271 // add the face
272 uint fid = m.addFace();
273 FaceType& f = m.face(fid);
274
275 // check if we need to split the face we read into triangles
276 bool splitFace = false;
277 // we have a polygonal mesh, no need to split
278 if constexpr (FaceType::VERTEX_NUMBER < 0) {
279 // need to resize to the right number of verts
280 f.resizeVertices(tokens.size()-1);
281 }
282 else if (FaceType::VERTEX_NUMBER != tokens.size()-1) {
283 // we have faces with static sizes (triangles), but we are loading faces
284 // with number of verts > 3. Need to split the face we are loading in n
285 // faces!
286 splitFace = true;
287 }
288
289 // create the face in the mesh, for now we manage only vertex indices
290 if (!splitFace) { // no need to split face case
291 for (uint i = 0; i < vids.size(); ++i) {
292 if (vids[i] >= m.vertexNumber()) {
293 throw vcl::MalformedFileException(
294 "Bad vertex index for face " + std::to_string(fid));
295 }
296 f.vertex(i) = &m.vertex(vids[i]);
297 }
298 }
299 else { // split needed
300 addTriangleFacesFromPolygon(m, f, vids);
301 }
302
303 // color
304 if (HasPerFaceColor<MeshType>){
305 // if the first face, we need to check if I can store colors
306 if (fid == 0) {
307 // if the current material has no color, we assume that the file has
308 // no face color
309 if (currentMaterial.hasColor) {
310 if (enableOptionalComponents) {
311 enableIfPerFaceColorOptional(m);
312 loadedInfo.setFaceColors();
313 }
314 else {
315 if (isPerFaceColorAvailable(m))
316 loadedInfo.setFaceColors();
317 }
318 }
319 }
320 if (loadedInfo.hasFaceColors()) {
321 if (currentMaterial.hasColor) {
322 // in case the loaded polygon has been triangulated in the last
323 // n triangles of mesh
324 for (uint ff = fid; ff < m.faceNumber(); ++ff) {
325 m.face(ff).color() = currentMaterial.color();
326 }
327 }
328 }
329 }
330
331 // wedge coords
332 if constexpr(HasPerFaceWedgeTexCoords<MeshType>) {
333 // first, need to check if I can store wedge texcoords in the mesh
334 if (fid == 0) {
335 // if the current face has the right number of wedge texcoords, we
336 // assume that we can load wedge texcoords
337 if (wids.size() == vids.size()) {
338 if (enableOptionalComponents) {
339 enableIfPerFaceWedgeTexCoordsOptional(m);
340 loadedInfo.setFaceWedgeTexCoords();
341 }
342 else {
343 if (isPerFaceWedgeTexCoordsAvailable(m))
344 loadedInfo.setFaceWedgeTexCoords();
345 }
346 }
347 }
348 if (loadedInfo.hasFaceWedgeTexCoords()) {
349 if (wids.size() == vids.size()) {
350 if (!splitFace) { // there wasn't a triangulation of the face
351 // it is safe to assign each wedge texcoord to its position
352 // in the face
353 for (uint i = 0; i < wids.size(); ++i) {
354 if (wids[i] >= wedgeTexCoords.size()) {
355 throw vcl::MalformedFileException(
356 "Bad texcoord index for face " +
357 std::to_string(fid));
358 }
359 f.wedgeTexCoord(i) =
360 wedgeTexCoords[wids[i]]
361 .cast<typename FaceType::WedgeTexCoordType::
362 ScalarType>();
363 if (currentMaterial.hasTexture){
364 f.textureIndex() = currentMaterial.mapId;
365 }
366 }
367 }
368 else {
369 // take read texcoords and map them in the triangulated
370 // faces for each face of the triangulation of the polygon
371 for (uint ff = fid; ff < m.faceNumber(); ++ff) {
372 FaceType& f = m.face(ff);
373 // for each vertex of the face
374 for (uint i = 0; i < f.vertexNumber(); ++i){
375 uint vid = m.index(f.vertex(i));
376 // find the position of the vertex in the vids array
377 auto it = std::find(vids.begin(), vids.end(), vid);
378 assert(it != vids.end());
379 uint pos = it - vids.begin();
380 // check that the texcoord id is valid
381 if (wids[pos] >= wedgeTexCoords.size()) {
382 throw vcl::MalformedFileException(
383 "Bad texcoord index for face " +
384 std::to_string(fid));
385 }
386 // set the wedge texcoord in the same position of
387 // the vertex
388 f.wedgeTexCoord(i) =
389 wedgeTexCoords[wids[pos]]
390 .cast<typename FaceType::WedgeTexCoordType::
391 ScalarType>();
392 if (currentMaterial.hasTexture){
393 f.textureIndex() = currentMaterial.mapId;
394 }
395 }
396 }
397 }
398 }
399 }
400 }

405template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
406void loadObj(
407 MeshType& m,
408 const std::string& filename,
409 MeshInfo& loadedInfo,
410 LogType& log = nullLogger,
411 bool enableOptionalComponents = true)
412{
413 std::ifstream file = internal::loadFileStream(filename);
414 // save normals if they can't be stored directly into vertices
415 internal::NormalsMap<MeshType> mapNormalsCache;
416 uint vn = 0; // number of vertex normals read
417 // save array of texcoords, that are stored later (into wedges when loading
418 // faces or into vertices as a fallback)
419 std::vector<vcl::TexCoordd> texCoords;
420
421 // map of materials loaded
422 std::map<std::string, obj::Material> materialMap;
423 obj::Material currentMaterial; // the current material, set by 'usemtl'
424
425 // some obj files do not declare the material file name with mtllib, but
426 // they assume that material file has the same name of the obj file.
427 // Therefore, we first load this file if it exists.
428 std::string stdmtlfile = FileInfo::pathWithoutFileName(filename) +
429 FileInfo::fileNameWithoutExtension(filename) +
430 ".mtl";
431 try {
432 internal::loadMaterials(materialMap, m, stdmtlfile);
433 }
434 catch(vcl::CannotOpenFileException){
435 // nothing to do, this file was missing but was a fallback for some type
436 // of files...
437 }
438
439 if constexpr (HasTexturePaths<MeshType>) {
440 m.meshBasePath() = FileInfo::pathWithoutFileName(filename);
441 }
442
443 if constexpr (HasName<MeshType>) {
444 m.name() = FileInfo::fileNameWithoutExtension(filename);
445 }
446
447 // cycle that reads line by line
448 do {
449 vcl::Tokenizer tokens =
450 internal::nextNonEmptyTokenizedLineNoThrow(file);
451 if (file) {
452 vcl::Tokenizer::iterator token = tokens.begin();
453 std::string header = *token++;
454 if (header == "mtllib") { // material file
455 std::string mtlfile =
456 FileInfo::pathWithoutFileName(filename) + *token;
457 internal::loadMaterials(materialMap, m, mtlfile);
458 }
459 // use a new material - change currentMaterial
460 if (header == "usemtl") {
461 std::string matname = *token;
462 auto it = materialMap.find(matname);
463 if (it != materialMap.end()) {
464 currentMaterial = it->second;
465 }
466 }
467 // read vertex (and for some non-standard obj files, also vertex
468 // color)
469 if (header == "v") {
470 internal::loadVertexCoord(
471 m,
472 token,
473 loadedInfo,
474 tokens,
475 currentMaterial,
476 enableOptionalComponents);
477 }
478 // read vertex normal (and save in vn how many normals we read)
479 if constexpr(HasPerVertexNormal<MeshType>) {
480 if (header == "vn") {
481 internal::loadVertexNormal(
482 m,
483 mapNormalsCache,
484 vn,
485 token,
486 loadedInfo,
487 enableOptionalComponents);
488 vn++;
489 }
490 }
491 // read texcoords and save them in the vector of texcoords, we will
492 // store them in the mesh later
493 if constexpr (
494 HasPerVertexTexCoord<MeshType> ||
495 HasPerFaceWedgeTexCoords<MeshType>)
496 {
497 if (header == "vt") {
498 // save the texcoord for later
499 TexCoordd tf;
500 for (uint i = 0; i < 2; ++i) {
501 tf[i] = internal::readDouble<double>(token);
502 }
503 texCoords.push_back(tf);
504 }
505 }
506 // read faces and manage:
507 // - color
508 // - eventual texcoords
509 // - possibility to split polygonal face into several triangles
510 if constexpr (HasFaces<MeshType>) {
511 if (header == "f") {
512 internal::loadFace(
513 m,
514 loadedInfo,
515 tokens,
516 texCoords,
517 currentMaterial,
518 enableOptionalComponents);
519 }
520 }
521 }
522 } while (file);
523
524 if constexpr (HasPerVertexNormal<MeshType>) {
525 // set all vertex normals that have not been stored in vertices
526 for (const auto& p : mapNormalsCache) {
527 if (p.first < m.vertexNumber()) {
528 m.vertex(p.first).normal() = p.second;
529 }
530 }
531 }
532 if constexpr (HasPerVertexTexCoord<MeshType>) {
533 using VertexType = MeshType::VertexType;
534 if (!loadedInfo.hasFaceWedgeTexCoords()) {
535 // we can set the loaded texCoords to vertices, also if they are not
536 // supported in obj
537 if (texCoords.size() == m.vertexNumber()) {
538 if (enableOptionalComponents) {
539 enableIfPerVertexTexCoordOptional(m);
540 loadedInfo.setVertexTexCoords();
541 }
542 else {
543 if (isPerVertexTexCoordAvailable(m))
544 loadedInfo.setVertexTexCoords();
545 }
546 if (loadedInfo.hasVertexTexCoords()) {
547 uint i = 0;
548 for (VertexType& v : m.vertices()) {
549 v.texCoord() =
550 texCoords[i++]
551 .cast<typename VertexType::TexCoordType::
552 ScalarType>();
553 }
554 }
555 }
556 }
557 }
558}
