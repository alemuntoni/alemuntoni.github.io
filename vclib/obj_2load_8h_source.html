48template<MeshConcept MeshType>
49void loadObjMaterials(
50 std::map<std::string, ObjMaterial>& materialMap,
51 MeshType& mesh,
52 std::istream& stream,
53 MeshInfo& loadedInfo,
54 const LoadSettings& settings)
55{
56 std::string matName;
57 ObjMaterial mat;
58
59 do {
60 vcl::Tokenizer tokens = readAndTokenizeNextNonEmptyLineNoThrow(stream);
61 if (stream) {
62 // counter for texture images, used when mesh has no texture files
63 uint nt = 0;
64 vcl::Tokenizer::iterator token = tokens.begin();
65 std::string header = *token++;
66 if (header == "newmtl") {
67 if (!matName.empty())
68 materialMap[matName] = mat;
69 mat = ObjMaterial();
70 matName = *token;
71 }
72 if (header == "Ka") {
73 if (tokens.size() >= 4) {
74 if (*token != "spectral" && *token != "xyz") {
75 mat.Ka.x() = io::readFloat<float>(token);
76 mat.Ka.y() = io::readFloat<float>(token);
77 mat.Ka.z() = io::readFloat<float>(token);
78 }
79 }
80 }
81 if (header == "Kd") {
82 if (tokens.size() >= 4) {
83 if (*token != "spectral" && *token != "xyz") {
84 mat.Kd.x() = io::readFloat<float>(token);
85 mat.Kd.y() = io::readFloat<float>(token);
86 mat.Kd.z() = io::readFloat<float>(token);
87 mat.hasColor = true;
88 }
89 }
90 }
91 if (header == "Ks") {
92 if (tokens.size() >= 4) {
93 if (*token != "spectral" && *token != "xyz") {
94 mat.Ks.x() = io::readFloat<float>(token);
95 mat.Ks.y() = io::readFloat<float>(token);
96 mat.Ks.z() = io::readFloat<float>(token);
97 }
98 }
99 }
100 if (header == "d") {
101 if ((*token)[0] == '-')
102 token++;
103 mat.d = io::readFloat<float>(token);
104 }
105 if (header == "Tr") {
106 if ((*token)[0] == '-')
107 token++;
108 mat.d = 1 - io::readFloat<float>(token);
109 }
110 if (header == "Ns") {
111 mat.Ns = io::readFloat<float>(token);
112 }
113 if (header == "illum") {
114 mat.illum = io::readFloat<int>(token);
115 }
116 if (header == "map_Kd") {
117 // need to manage args
118 while ((*token)[0] == '-') {
119 if (*token == "-o" || *token == "-s" || *token == "-t") {
120 // ignore the argument and the three values
121 ++token;
122 ++token;
123 ++token;
124 ++token;
125 }
126 if (*token == "-mm") {
127 // ignore the argument and the two values
128 ++token;
129 ++token;
130 ++token;
131 }
132 if (*token == "-blendu" || *token == "-blendv" ||
133 *token == "-cc" || *token == "-clamp" ||
134 *token == "-texres")
135 {
136 // ignore the argument and the value
137 ++token;
138 ++token;
139 }
140 }
141 mat.map_Kd = *token;
142 mat.hasTexture = true;
143 if constexpr (HasTexturePaths<MeshType>) {
144 loadedInfo.setTextures();
145 mat.mapId = mesh.textureNumber();
146 mesh.pushTexturePath(mat.map_Kd);
147 }
148 else if constexpr (HasTextureImages<MeshType>) {
149 loadedInfo.setTextures();
150 mat.mapId = mesh.textureNumber();
151 mesh.pushTexture(mat.map_Kd);
152 }
153 else {
154 mat.mapId = nt++;
155 }
156 }
157 }
158 } while (stream);
159 if (!matName.empty())
160 materialMap[matName] = mat;
161}

175template<MeshConcept MeshType>
176void readObjVertex(
177 MeshType& m,
178 vcl::Tokenizer::iterator& token,
179 MeshInfo& loadedInfo,
180 const vcl::Tokenizer& tokens,
181 const ObjMaterial& currentMaterial,
182 const LoadSettings& settings)
183{
184 // first, need to set that I'm loading vertices
185 if (m.vertexNumber() == 0) {
186 loadedInfo.setVertices();
187 loadedInfo.setVertexCoords();
188 }
189 uint vid = m.addVertex();
190 for (uint i = 0; i < 3; ++i) {
191 m.vertex(vid).coord()[i] = io::readDouble<double>(token);
192 }
193 if constexpr (HasPerVertexColor<MeshType>) {
194 if (vid == 0) {
195 // if the current material has a valid color, of the file stores the
196 // vertex color in the non-standard way (color values after the
197 // coordinates)
198 if (currentMaterial.hasColor || tokens.size() > 6) {
199 if (settings.enableOptionalComponents) {
200 enableIfPerVertexColorOptional(m);
201 loadedInfo.setVertexColors();
202 }
203 else {
204 if (isPerVertexColorAvailable(m))
205 loadedInfo.setVertexColors();
206 }
207 }
208 }
209 if (loadedInfo.hasVertexColors()) {
210 // the file has the nonstandard way to store vertex colors, after
211 // the coords...
212 if (tokens.size() > 6) {
213 m.vertex(vid).color().setRedF(io::readFloat<float>(token));
214 m.vertex(vid).color().setGreenF(io::readFloat<float>(token));
215 m.vertex(vid).color().setBlueF(io::readFloat<float>(token));
216 }
217 else if (currentMaterial.hasColor) {
218 m.vertex(vid).color() = currentMaterial.color();
219 }
220 }
221 }
222}

224template<MeshConcept MeshType>
225void readObjVertexNormal(
226 MeshType& m,
227 detail::ObjNormalsMap<MeshType>& mapNormalsCache,
228 uint vn,
229 vcl::Tokenizer::iterator& token,
230 MeshInfo& loadedInfo,
231 const LoadSettings& settings)
232{
233 using NormalType = MeshType::VertexType::NormalType;
234
235 // first, need to check if I can store normals in the mesh
236 if (vn == 0) {
237 if (settings.enableOptionalComponents) {
238 enableIfPerVertexNormalOptional(m);
239 loadedInfo.setVertexNormals();
240 }
241 else {
242 if (isPerVertexNormalAvailable(m))
243 loadedInfo.setVertexNormals();
244 }
245 }
246 if (loadedInfo.hasVertexNormals()) {
247 // read the normal
248 NormalType n;
249 for (uint i = 0; i < 3; ++i) {
250 n[i] = io::readDouble<typename NormalType::ScalarType>(token);
251 }
252 // I can store the normal in its vertex
253 if (m.vertexNumber() > vn) {
254 m.vertex(vn).normal() = n;
255 }
256 // read the normal and save it in the cache map, because we still don't
257 // have read the vertex corresponding to the current normal
258 else {
259 mapNormalsCache[vn] = n;
260 }
261 }

264template<FaceMeshConcept MeshType>
265void readObjFace(
266 MeshType& m,
267 MeshInfo& loadedInfo,
268 const vcl::Tokenizer& tokens,
269 const std::vector<vcl::TexCoordd>& wedgeTexCoords,
270 const ObjMaterial& currentMaterial,
271 const LoadSettings& settings)
272{
273 using FaceType = MeshType::FaceType;
274
275 std::vector<uint> vids;
276 std::vector<uint> wids;
277
278 loadedInfo.updateMeshType(tokens.size() - 1);
279
280 // actual read - load vertex indices and texcoords indices, if present
281 vcl::Tokenizer::iterator token = tokens.begin();
282 ++token;
283 vids.resize(tokens.size() - 1);
284 wids.reserve(tokens.size() - 1);
285 for (uint i = 0; i < tokens.size() - 1; ++i) {
286 vcl::Tokenizer subt(*token, '/', false);
287 auto t = subt.begin();
288 vids[i] = io::readUInt<uint>(t) - 1;
289 if (subt.size() > 1) {
290 if (!t->empty()) {
291 wids.push_back(io::readUInt<uint>(t) - 1);
292 }
293 }
294 ++token;
295 }
296
297 // add the face
298 uint fid = m.addFace();
299 FaceType& f = m.face(fid);
300
301 // check if we need to split the face we read into triangles
302 bool splitFace = false;
303 // we have a polygonal mesh, no need to split
304 if constexpr (FaceType::VERTEX_NUMBER < 0) {
305 // need to resize to the right number of verts
306 f.resizeVertices(tokens.size() - 1);
307 }
308 else if (FaceType::VERTEX_NUMBER != tokens.size() - 1) {
309 // we have faces with static sizes (triangles), but we are loading faces
310 // with number of verts > 3. Need to split the face we are loading in n
311 // faces!
312 splitFace = true;
313 }
314
315 // create the face in the mesh, for now we manage only vertex indices
316 if (!splitFace) { // no need to split face case
317 for (uint i = 0; i < vids.size(); ++i) {
318 if (vids[i] >= m.vertexNumber()) {
319 throw vcl::MalformedFileException(
320 "Bad vertex index for face " + std::to_string(fid));
321 }
322 f.setVertex(i, vids[i]);
323 }
324 }
325 else { // split needed
326 addTriangleFacesFromPolygon(m, f, vids);
327 }
328
329 // color
330 if (HasPerFaceColor<MeshType>) {
331 // if the first face, we need to check if I can store colors
332 if (fid == 0) {
333 // if the current material has no color, we assume that the file has
334 // no face color
335 if (currentMaterial.hasColor) {
336 if (settings.enableOptionalComponents) {
337 enableIfPerFaceColorOptional(m);
338 loadedInfo.setFaceColors();
339 }
340 else {
341 if (isPerFaceColorAvailable(m))
342 loadedInfo.setFaceColors();
343 }
344 }
345 }
346 if (loadedInfo.hasFaceColors()) {
347 if (currentMaterial.hasColor) {
348 // in case the loaded polygon has been triangulated in the last
349 // n triangles of mesh
350 for (uint ff = fid; ff < m.faceNumber(); ++ff) {
351 m.face(ff).color() = currentMaterial.color();
352 }
353 }
354 }
355 }
356
357 // wedge coords
358 if constexpr (HasPerFaceWedgeTexCoords<MeshType>) {
359 // first, need to check if I can store wedge texcoords in the mesh
360 if (fid == 0) {
361 // if the current face has the right number of wedge texcoords, we
362 // assume that we can load wedge texcoords
363 if (wids.size() == vids.size()) {
364 if (settings.enableOptionalComponents) {
365 enableIfPerFaceWedgeTexCoordsOptional(m);
366 loadedInfo.setFaceWedgeTexCoords();
367 }
368 else {
369 if (isPerFaceWedgeTexCoordsAvailable(m))
370 loadedInfo.setFaceWedgeTexCoords();
371 }
372 }
373 }
374 if (loadedInfo.hasFaceWedgeTexCoords()) {
375 if (wids.size() == vids.size()) {
376 if (!splitFace) { // there wasn't a triangulation of the face
377 // it is safe to assign each wedge texcoord to its position
378 // in the face
379 for (uint i = 0; i < wids.size(); ++i) {
380 if (wids[i] >= wedgeTexCoords.size()) {
381 throw vcl::MalformedFileException(
382 "Bad texcoord index for face " +
383 std::to_string(fid));
384 }
385 f.wedgeTexCoord(i) =
386 wedgeTexCoords[wids[i]]
387 .cast<typename FaceType::WedgeTexCoordType::
388 ScalarType>();
389 if (currentMaterial.hasTexture) {
390 f.textureIndex() = currentMaterial.mapId;
391 }
392 }
393 }
394 else {
395 // take read texcoords and map them in the triangulated
396 // faces for each face of the triangulation of the polygon
397 for (uint ff = fid; ff < m.faceNumber(); ++ff) {
398 FaceType& f = m.face(ff);
399 // for each vertex of the face
400 for (uint i = 0; i < f.vertexNumber(); ++i) {
401 uint vid = m.index(f.vertex(i));
402 // find the position of the vertex in the vids array
403 auto it = std::find(vids.begin(), vids.end(), vid);
404 assert(it != vids.end());
405 uint pos = it - vids.begin();
406 // check that the texcoord id is valid
407 if (wids[pos] >= wedgeTexCoords.size()) {
408 throw vcl::MalformedFileException(
409 "Bad texcoord index for face " +
410 std::to_string(fid));
411 }
412 // set the wedge texcoord in the same position of
413 // the vertex
414 f.wedgeTexCoord(i) =
415 wedgeTexCoords[wids[pos]]
416 .cast<typename FaceType::WedgeTexCoordType::
417 ScalarType>();
418 if (currentMaterial.hasTexture) {
419 f.textureIndex() = currentMaterial.mapId;
420 }
421 }
422 }
423 }
424 }
425 }
426 }

448template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
449void loadObj(
450 MeshType& m,
451 std::istream& inputObjStream,
452 const std::vector<std::istream*>& inputMtlStreams,
453 MeshInfo& loadedInfo,
454 const std::string& filename = "",
455 bool ignoreMtlLib = false,
456 LogType& log = nullLogger,
457 const LoadSettings& settings = LoadSettings())
458{
459 loadedInfo.clear();
460
461 // save normals if they can't be stored directly into vertices
462 detail::ObjNormalsMap<MeshType> mapNormalsCache;
463 uint vn = 0; // number of vertex normals read
464 // save array of texcoords, that are stored later (into wedges when loading
465 // faces or into vertices as a fallback)
466 std::vector<vcl::TexCoordd> texCoords;
467
468 // map of materials loaded
469 std::map<std::string, detail::ObjMaterial> materialMap;
470
471 // load materials from the material files, if any
472 for (auto* stream : inputMtlStreams) {
473 detail::loadObjMaterials(materialMap, m, *stream, loadedInfo, settings);
474 }
475
476 // the current material, set by 'usemtl'
477 detail::ObjMaterial currentMaterial;
478
479 if constexpr (HasTexturePaths<MeshType> || HasTextureImages<MeshType>) {
480 m.meshBasePath() = FileInfo::pathWithoutFileName(filename);
481 }
482
483 if constexpr (HasName<MeshType>) {
484 m.name() = FileInfo::fileNameWithoutExtension(filename);
485 }
486
487 inputObjStream.seekg(0, inputObjStream.end);
488 std::size_t fsize = inputObjStream.tellg();
489 inputObjStream.seekg(0, inputObjStream.beg);
490 log.startProgress("Loading OBJ file", fsize);
491
492 // cycle that reads line by line
493 do {
494 vcl::Tokenizer tokens =
495 readAndTokenizeNextNonEmptyLineNoThrow(inputObjStream);
496 if (inputObjStream) {
497 vcl::Tokenizer::iterator token = tokens.begin();
498 std::string header = *token++;
499 if (header == "mtllib" && !ignoreMtlLib) { // material file
500 // we load the material file if they are not ignored
501 std::string mtlfile =
502 FileInfo::pathWithoutFileName(filename) + *token;
503 try {
504 detail::loadObjMaterials(
505 materialMap, m, mtlfile, loadedInfo, settings);
506 }
507 catch (vcl::CannotOpenFileException) {
508 log.log(
509 LogType::WARNING,
510 "Cannot open material file " + mtlfile);
511 }
512 }
513 // use a new material - change currentMaterial
514 if (header == "usemtl") {
515 std::string matname = *token;
516 auto it = materialMap.find(matname);
517 if (it != materialMap.end()) {
518 currentMaterial = it->second;
519 }
520 else { // material not found - warning
521 log.log(
522 LogType::WARNING,
523 "Material " + matname + " not found.");
524 }
525 }
526 // read vertex (and for some non-standard obj files, also vertex
527 // color)
528 if (header == "v") {
529 detail::readObjVertex(
530 m, token, loadedInfo, tokens, currentMaterial, settings);
531 }
532 // read vertex normal (and save in vn how many normals we read)
533 if constexpr (HasPerVertexNormal<MeshType>) {
534 if (header == "vn") {
535 detail::readObjVertexNormal(
536 m, mapNormalsCache, vn, token, loadedInfo, settings);
537 vn++;
538 }
539 }
540 // read texcoords and save them in the vector of texcoords, we will
541 // store them in the mesh later
542 if constexpr (
543 HasPerVertexTexCoord<MeshType> ||
544 HasPerFaceWedgeTexCoords<MeshType>)
545 {
546 if (header == "vt") {
547 // save the texcoord for later
548 TexCoordd tf;
549 for (uint i = 0; i < 2; ++i) {
550 tf[i] = io::readDouble<double>(token);
551 }
552 texCoords.push_back(tf);
553 }
554 }
555 // read faces and manage:
556 // - color
557 // - eventual texcoords
558 // - possibility to split polygonal face into several triangles
559 if constexpr (HasFaces<MeshType>) {
560 if (header == "f") {
561 detail::readObjFace(
562 m,
563 loadedInfo,
564 tokens,
565 texCoords,
566 currentMaterial,
567 settings);
568 }
569 }
570 log.progress(inputObjStream.tellg());
571 }
572 } while (inputObjStream);
573
574 if constexpr (HasPerVertexNormal<MeshType>) {
575 // set all vertex normals that have not been stored in vertices
576 for (const auto& p : mapNormalsCache) {
577 if (p.first < m.vertexNumber()) {
578 m.vertex(p.first).normal() = p.second;
579 }
580 }
581 }
582 if constexpr (HasPerVertexTexCoord<MeshType>) {
583 using VertexType = MeshType::VertexType;
584 if (!loadedInfo.hasFaceWedgeTexCoords()) {
585 // we can set the loaded texCoords to vertices, also if they are not
586 // supported in obj
587 if (texCoords.size() == m.vertexNumber()) {
588 if (settings.enableOptionalComponents) {
589 enableIfPerVertexTexCoordOptional(m);
590 loadedInfo.setVertexTexCoords();
591 }
592 else {
593 if (isPerVertexTexCoordAvailable(m))
594 loadedInfo.setVertexTexCoords();
595 }
596 if (loadedInfo.hasVertexTexCoords()) {
597 uint i = 0;
598 for (VertexType& v : m.vertices()) {
599 v.texCoord() =
600 texCoords[i++]
601 .cast<typename VertexType::TexCoordType::
602 ScalarType>();
603 }
604 }
605 }
606 }
607 }
608
609 if constexpr (HasTextureImages<MeshType>) {
610 if (settings.loadTextureImages) {
611 for (vcl::Texture& texture : m.textures()) {
612 bool b =
613 texture.image().load(m.meshBasePath() + texture.path());
614 if (!b) {
615 log.log(
616 LogType::WARNING,
617 "Cannot load texture " + texture.path());
618 }
619 }
620 }
621 }
622
623 log.endProgress();
624}

819template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
820void loadObj(
821 MeshType& m,
822 const std::string& filename,
823 MeshInfo& loadedInfo,
824 LogType& log = nullLogger,
825 const LoadSettings& settings = LoadSettings())
826{
827 std::ifstream file = openInputFileStream(filename);
828
829 // some obj files do not declare the material file name with mtllib, but
830 // they assume that material file has the same name of the obj file.
831 // Therefore, we first load this file if it exists.
832 std::string stdmtlfile = FileInfo::pathWithoutFileName(filename) +
833 FileInfo::fileNameWithoutExtension(filename) +
834 ".mtl";
835
836 std::ifstream f;
837 std::vector<std::istream*> mtlStreams;
838 try {
839 f = openInputFileStream(stdmtlfile);
840 mtlStreams.push_back(&f);
841 }
842 catch (vcl::CannotOpenFileException) {
843 // nothing to do, this file was missing, but this was a fallback for
844 // some type of files...
845 }
846
847 detail::loadObj(
848 m, file, mtlStreams, loadedInfo, filename, false, log, settings);
849}
