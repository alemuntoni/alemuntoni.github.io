399 {
400 if constexpr (HasAdjacentVertices<Element>) {
401 if (isAdjacentVerticesAvailableOn(e)) {
402 // from static/dynamic to dynamic size: need to resize first,

404 resizeAdjVertices(e.adjVerticesNumber());
405 importPtrsFrom(e, base, ebase);

411 template<typename Element, typename ElVType>
412 void importPtrsFrom(const Element& e, Vertex* base, const ElVType* ebase)
413 {
414 if (ebase != nullptr && base != nullptr) {
415 for (uint i = 0; i < e.adjVerticesNumber(); ++i) {
416 if (e.adjVertex(i) != nullptr) {
417 adjVertex(i) = base + (e.adjVertex(i) - ebase);
418 }
419 }
420 }
