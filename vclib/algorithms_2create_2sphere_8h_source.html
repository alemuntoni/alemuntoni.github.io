125template<FaceMeshConcept MeshType>
126MeshType
127createSphereUV(const SphereConcept auto& sp, uint parallels, uint meridians)
128{
129 using VertexType = MeshType::VertexType;
130 using CoordType = VertexType::CoordType;
131 using Facetype = MeshType::FaceType;
132
133 MeshType mesh;
134
135 mesh.addVertex(CoordType(0, 1, 0));
136 for (uint j = 0; j < parallels - 1; ++j) {
137 const double polar = M_PI * double(j + 1) / double(parallels);
138 const double sp = std::sin(polar);
139 const double cp = std::cos(polar);
140 for (uint i = 0; i < meridians; ++i) {
141 const double azimuth = 2.0 * M_PI * double(i) / double(meridians);
142 const double sa = std::sin(azimuth);
143 const double ca = std::cos(azimuth);
144 const double x = sp * ca;
145 const double y = cp;
146 const double z = sp * sa;
147 mesh.addVertex(CoordType(x, y, z));
148 }
149 }
150 mesh.addVertex(CoordType(0, -1, 0));
151
152 for (uint i = 0; i < meridians; ++i) {
153 VertexType* v = &mesh.vertex(0);
154 VertexType* a = &mesh.vertex(i + 1);
155 VertexType* b = &mesh.vertex((i + 1) % meridians + 1);
156 mesh.addFace(0, b, a);
157 }
158
159 for (uint32_t j = 0; j < parallels - 2; ++j) {
160 uint32_t aStart = j * meridians + 1;
161 uint32_t bStart = (j + 1) * meridians + 1;
162 for (uint32_t i = 0; i < meridians; ++i) {
163 VertexType* a = &mesh.vertex(aStart + i);
164 VertexType* a1 = &mesh.vertex(aStart + (i + 1) % meridians);
165 VertexType* b = &mesh.vertex(bStart + i);
166 VertexType* b1 = &mesh.vertex(bStart + (i + 1) % meridians);
167 if constexpr (HasTriangles<MeshType>) {
168 mesh.addFace(a, a1, b1);
169 mesh.addFace(b1, b, a);
170 }
171 else {
172 mesh.addFace(a, a1, b1, b);
173 }
174 }
175 }
176
177 for (uint32_t i = 0; i < meridians; ++i) {
178 VertexType* a = &mesh.vertex(i + meridians * (parallels - 2) + 1);
179 VertexType* b =
180 &mesh.vertex((i + 1) % meridians + meridians * (parallels - 2) + 1);
181 VertexType* v = &mesh.vertex(mesh.vertexNumber() - 1);
182 mesh.addFace(v, a, b);
183 }
184
185 vcl::scale(mesh, sp.radius());
186 vcl::translate(mesh, sp.center());
187
188 return mesh;
189}

204template<FaceMeshConcept MeshType>
205MeshType
206createSphereNormalizedCube(const SphereConcept auto& sp, uint divisions)
207{
208 using VertexType = MeshType::VertexType;
209 using CoordType = VertexType::CoordType;
210 using Facetype = MeshType::FaceType;
211 using ScalarType = CoordType::ScalarType;
212
213 MeshType mesh;
214
215 const double step = 1.0 / double(divisions);
216 const CoordType step3(step, step, step);
217
218 for (uint face = 0; face < 6; ++face) {
219 const CoordType origin =
220 internal::cts::origins[face].cast<ScalarType>();
221 const CoordType right = internal::cts::rights[face].cast<ScalarType>();
222 const CoordType up = internal::cts::ups[face].cast<ScalarType>();
223 for (uint j = 0; j < divisions + 1; ++j) {
224 const CoordType j3(j, j, j);
225 for (uint i = 0; i < divisions + 1; ++i) {
226 const CoordType i3(i, i, i);
227 const CoordType p =
228 origin + step3.mul(i3.mul(right) + j3.mul(up));
229
230 mesh.addVertex(p.normalized());
231 }
232 }
233 }
234
235 const uint k = divisions + 1;
236 for (uint face = 0; face < 6; ++face) {
237 for (uint j = 0; j < divisions; ++j) {
238 const bool bottom = j < (divisions / 2);
239 for (uint i = 0; i < divisions; ++i) {
240 const bool left = i < (divisions / 2);
241 VertexType* a = &mesh.vertex((face * k + j) * k + i);
242 VertexType* b = &mesh.vertex((face * k + j) * k + i + 1);
243 VertexType* c = &mesh.vertex((face * k + j + 1) * k + i);
244 VertexType* d = &mesh.vertex((face * k + j + 1) * k + i + 1);
245
246 if constexpr (vcl::HasTriangles<MeshType>) {
247 if (bottom ^ left) {
248 mesh.addFace(a, c, b);
249 mesh.addFace(c, d, b);
250 }
251 else {
252 mesh.addFace(a, c, d);
253 mesh.addFace(a, d, b);
254 }
255 }
256 else {
257 mesh.addFace(a, c, d, b);
258 }
259 }
260 }
261 }
262
263 vcl::scale(mesh, sp.radius());
264 vcl::translate(mesh, sp.center());
265
266 return mesh;
267}

282template<FaceMeshConcept MeshType>
283MeshType
284createSphereSpherifiedCube(const SphereConcept auto& sp, uint divisions)
285{
286 using VertexType = MeshType::VertexType;
287 using CoordType = VertexType::CoordType;
288 using ScalarType = CoordType::ScalarType;
289 using Facetype = MeshType::FaceType;
290
291 MeshType mesh;
292
293 const double step = 1.0 / double(divisions);
294 const CoordType step3(step, step, step);
295
296 for (uint face = 0; face < 6; ++face) {
297 const CoordType origin =
298 internal::cts::origins[face].cast<ScalarType>();
299 const CoordType right = internal::cts::rights[face].cast<ScalarType>();
300 const CoordType up = internal::cts::ups[face].cast<ScalarType>();
301 for (uint j = 0; j < divisions + 1; ++j) {
302 const CoordType j3(j, j, j);
303 for (uint i = 0; i < divisions + 1; ++i) {
304 const CoordType i3(i, i, i);
305 const CoordType p =
306 origin + step3.mul(i3.mul(right) + j3.mul(up));
307 const CoordType p2 = p.mul(p);
308 const CoordType n(
309 p.x() * std::sqrt(
310 1.0 - 0.5 * (p2.y() + p2.z()) +
311 p2.y() * p2.z() / 3.0),
312 p.y() * std::sqrt(
313 1.0 - 0.5 * (p2.z() + p2.x()) +
314 p2.z() * p2.x() / 3.0),
315 p.z() * std::sqrt(
316 1.0 - 0.5 * (p2.x() + p2.y()) +
317 p2.x() * p2.y() / 3.0));
318 mesh.addVertex(n);
319 }
320 }
321 }
322
323 const uint k = divisions + 1;
324 for (uint face = 0; face < 6; ++face) {
325 for (uint j = 0; j < divisions; ++j) {
326 const bool bottom = j < (divisions / 2);
327 for (uint i = 0; i < divisions; ++i) {
328 const bool left = i < (divisions / 2);
329 VertexType* a = &mesh.vertex((face * k + j) * k + i);
330 VertexType* b = &mesh.vertex((face * k + j) * k + i + 1);
331 VertexType* c = &mesh.vertex((face * k + j + 1) * k + i);
332 VertexType* d = &mesh.vertex((face * k + j + 1) * k + i + 1);
333 if constexpr (vcl::HasTriangles<MeshType>) {
334 if (bottom ^ left) {
335 mesh.addFace(a, c, b);
336 mesh.addFace(c, d, b);
337 }
338 else {
339 mesh.addFace(a, c, d);
340 mesh.addFace(a, d, b);
341 }
342 }
343 else {
344 mesh.addFace(a, c, d, b);
345 }
346 }
347 }
348 }
349
350 vcl::scale(mesh, sp.radius());
351 vcl::translate(mesh, sp.center());
352
353 return mesh;
354}

369template<FaceMeshConcept MeshType>
370MeshType createSphereIcosahedron(const SphereConcept auto& sp, uint divisions)
371{
372 using VertexType = MeshType::VertexType;
373 using CoordType = VertexType::CoordType;
374 using FaceType = MeshType::FaceType;
375
376 MeshType mesh = vcl::createIcosahedron<MeshType>(true);
377
378 for (uint d = 0; d < divisions; d++) {
379 uint nf = mesh.faceNumber();
380 for (uint f = 0; f < nf; f++) {
381 FaceType& f0 = mesh.face(f);
382 VertexType& v0 = *f0.vertex(0);
383 VertexType& v1 = *f0.vertex(1);
384 VertexType& v2 = *f0.vertex(2);
385 uint v1id = mesh.index(v1);
386 uint v2id = mesh.index(v2);
387
388 CoordType pa = (v0.coord() + v1.coord());
389 pa.normalize();
390 CoordType pb = (v1.coord() + v2.coord());
391 pb.normalize();
392 CoordType pc = (v2.coord() + v0.coord());
393 pc.normalize();
394 uint vaid = mesh.addVertex(pa);
395 uint vbid = mesh.addVertex(pb);
396 uint vcid = mesh.addVertex(pc);
397
398 f0.vertex(1) = &mesh.vertex(vaid);
399 f0.vertex(2) = &mesh.vertex(vcid);
400 mesh.addFace(vaid, v1id, vbid);
401 mesh.addFace(vcid, vbid, v2id);
402 mesh.addFace(vaid, vbid, vcid);
403 }
404 }
405
406 vcl::removeDuplicatedVertices(mesh);
407
408 vcl::scale(mesh, sp.radius());
409 vcl::translate(mesh, sp.center());
410
411 return mesh;
412}

429template<FaceMeshConcept MeshType>
430MeshType createSphere(
431 const SphereConcept auto& sp,
432 const CreateSphereArgs& args = CreateSphereArgs())
433{
434 MeshType m;
435 switch (args.mode) {
436 case CreateSphereArgs::UV:
437 m = createSphereUV<MeshType>(sp, args.parallels, args.meridians);
438 break;
439 case CreateSphereArgs::NORMALIZED_CUBE:
440 m = createSphereNormalizedCube<MeshType>(sp, args.divisions);
441 break;
442 case CreateSphereArgs::SPHERIFIED_CUBE:
443 m = createSphereSpherifiedCube<MeshType>(sp, args.divisions);
444 break;
445 case CreateSphereArgs::ICOSAHEDRON:
446 m = createSphereIcosahedron<MeshType>(sp, args.divisions);
447 break;
448 }
449
450 return m;
451}
