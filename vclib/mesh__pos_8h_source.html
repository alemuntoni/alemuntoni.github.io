105 static bool isValid(const FaceType* f, const VertexType* v, short e)
106 {
107 if (f == nullptr || v == nullptr || e < 0)
108 return false;
109 if (!comp::isAdjacentFacesAvailableOn(*f))
110 return false;
111 return (ushort) e < f->vertexNumber() &&
112 (v == f->vertex(e) || v == f->vertexMod(e + 1));
113 }

180 bool flipFace()
181 {
182 const FaceType* nf = f->adjFace(e);
183 if (nf != nullptr){
184 e = nf->indexOfAdjFace(f);
185 f = nf;
186 return true;
187 }
188 else {
189 return false;
190 }
191 }

197 void flipVertex()
198 {
199 if (f->vertexMod(e) == v) {
200 v = f->vertexMod(e+1);
201 }
202 else {
203 v = f->vertexMod(e);
204 }
205 }
206
211 void flipEdge()
212 {
213 if (f->vertexMod(e+1) == v) {
214 e = (e + 1) % (short)f->vertexNumber();
215 }
216 else {
217 short n = f->vertexNumber();
218 e = ((e - 1) % n + n) % n; // be sure to get the right index of the previous edge
219 }
220 }

233 void nextEdgeAdjacentToV()
234 {
235 flipEdge();

237 }

247 bool nextEdgeOnBorderAdjacentToV()
248 {
249 MeshPos<FaceType> startPos = *this;
250 do {
251 nextEdgeAdjacentToV();
252 } while (*this != startPos && !isEdgeOnBorder());
253 return (*this != startPos);
254 }

264 uint numberOfAdjacentFacesToV() const
265 {
266 bool onBorder = false;
267 uint count = countAdjacentFacesToV(onBorder);
268 // if we visited a border, it means that we visited the all faces
269 // adjacent to v twice to reach the same starting MeshPos.
270 if (onBorder)
271 count /= 2;
272 return count;
273 }

275 bool operator==(const MeshPos& op) const
276 {
277 return f == op.f && v == op.v && e == op.e;
278 }
279
280 bool operator!=(const MeshPos& op) const
281 {
282 return !(*this == op);
283 }
284
285 bool operator<(const MeshPos& op) const
286 {
287 if (f == op.f){
288 if (e == op.e)
289 return v < op.v;
290 else
291 return e < op.e;
292 }
293 else {
294 return f < op.f;
295 }
296 }
297
298private:
299 uint countAdjacentFacesToV(bool& onBorder) const
300 {
301 uint count = 0;
302 onBorder = false;
303 // start from this MeshPos
304 MeshPos<FaceType> mp = *this;
305 do {
306 // go to the next edge in the star of v (if face is on border, will
307 // change just edge)
308 mp.nextEdgeAdjacentToV();
309 ++count;
310 // if we are visiting a border, it means that we will start to
311 // navigate in the opposite sense as we were before.
312 if (mp.isEdgeOnBorder()) // flag that we visited a border
313 onBorder = true;
314 }
315 // we end when we will reach the same pos (same triplet f-v-e) of start
316 while (mp != *this);
317 return count;
318 }
