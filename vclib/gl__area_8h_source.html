75 void fitScene()
76 {
77 Box3d bb = fullBB();
78 Point3d sceneCenter = bb.center();
79 double sceneRadius = bb.diagonal() / 2;
80
81 setSceneCenter(
82 qglviewer::Vec(sceneCenter.x(), sceneCenter.y(), sceneCenter.z()));
83 setSceneRadius(sceneRadius);

87protected:

91 obj->draw();
92 }
93
94private:
95 uint firstVisibleObject() const
96 {
97 uint i = 0;
98
99 // if the current object i is not visible, check the next one
100 while (i < drawList->size() && !drawList->at(i).isVisible())
101 i++;
102
103 return i;
104 }
105
106 vcl::Box3d fullBB() const
107 {
108 Box3d bb(Point3d(-1,-1,-1), Point3d(1,1,1));
109 if (drawList->size() > 0) {
110 uint i = firstVisibleObject();
111
112 if (i < drawList->size()) {
113 Point3d sc = drawList->at(i).center();
114 bb.min() = sc - drawList->at(i).radius();
115 bb.max() = sc + drawList->at(i).radius();
116
117 for (i = i+1; i < drawList->size(); i++) { //rest of the list
118 Point3d sc = drawList->at(i).center();
119 Point3d tmp = sc - drawList->at(i).radius();
120 bb.min() = vcl::min(bb.min(), tmp);
121 tmp = sc + drawList->at(i).radius();
122 bb.max() = vcl::max(bb.max(), tmp);
123 }
124 }
125 }
126 return bb;
127 }
128};
