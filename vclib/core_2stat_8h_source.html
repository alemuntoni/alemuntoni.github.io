38template<Point3Concept PointType>
39auto covarianceMatrixOfPointCloud(const std::vector<PointType>& pointVec)
40{
41 Matrix33<typename PointType::ScalarType> m;
42 m.setZero();
43 PointType barycenter =
44 Polygon<PointType>::barycenter(pointVec.begin(), pointVec.end());
45
46 // compute covariance matrix
47 for (const PointType& p : pointVec) {
48 PointType e = p - barycenter;
49 m += e.outerProduct(e);
50 }
51 return m;
52}

60template<Point3Concept PointType>
61auto weightedCovarianceMatrixOfPointCloud(
62 const std::vector<PointType>& pointVec,
63 const std::vector<typename PointType::ScalarType>& weights)
64{
65 Matrix33<typename PointType::ScalarType> m;
66 m.setZero();
67 PointType barycenter = Polygon<PointType>::weightedBarycenter(
68 pointVec.begin(), pointVec.end(), weights.begin());
69
70 // compute covariance matrix
71 typename PointType::ScalarType wsum = 0;
72 for (uint i = 0; i < pointVec.size(); ++i) {
73 PointType e = (pointVec[i] - barycenter) * weights[i];
74 m += e.outerProduct(e);
75 wsum += weights[i];
76 }
77 return m / wsum;
78}
