55inline int poissonRatioOfUniformsInteger(double L, std::mt19937& gen)
56{
57 // constants
58 const double SHAT1 = 2.943035529371538573; // 8/e
59 const double SHAT2 = 0.8989161620588987408; // 3-sqrt(12/e)
60
61 double u; // uniform random
62 double lf; // ln(f(x))
63 double x; // real sample
64 int k; // integer sample
65
66 double pois_a = L + 0.5; // hat center
67 int mode = (int) L; // mode
68 double pois_g = std::log(L);
69 double pois_f0 = mode * pois_g - lnOfFactorial(mode); // value at mode
70 double pois_h = std::sqrt(SHAT1 * (L + 0.5)) + SHAT2; // hat width
71 double pois_bound = (int) (pois_a + 6.0 * pois_h); // safety-bound
72
73 std::uniform_real_distribution<double> unif(0, 1);
74
75 while (1) {
76 u = unif(gen);
77 if (u == 0)
78 continue; // avoid division by 0
79 x = pois_a + pois_h * (unif(gen) - 0.5) / u;
80 if (x < 0 || x >= pois_bound)
81 continue; // reject if outside valid range
82 k = (int) (x);
83 lf = k * pois_g - lnOfFactorial(k) - pois_f0;
84 if (lf >= u * (4.0 - u) - 3.0)
85 break; // quick acceptance
86 if (u * (u - lf) > 1.0)
87 continue; // quick rejection
88 if (2.0 * log(u) <= lf)
89 break; // final acceptance
90 }
91 return k;
92}

114inline int poissonRandomNumber(double lambda, std::mt19937& gen)
115{
116 if (lambda > 50)
117 return poissonRatioOfUniformsInteger(lambda, gen);
118
119 std::uniform_real_distribution<double> unif(0, 1);
120
121 double L = exp(-lambda);
122 int k = 0;
123 double p = 1.0;
124 do {
125 k = k + 1;
126 p = p * unif(gen);
127 } while (p > L);
128
129 return k - 1;
130}

147template<Point3Concept PointType>
148PointType randomTriangleBarycentricCoordinate(std::mt19937& gen)
149{
150 using ScalarType = PointType::ScalarType;
151
152 PointType interp;
153 std::uniform_real_distribution<ScalarType> unif(0, 1);
154
155 interp[1] = unif(gen);
156 interp[2] = unif(gen);
157 if (interp[1] + interp[2] > 1.0) {
158 interp[1] = 1.0 - interp[1];
159 interp[2] = 1.0 - interp[2];
160 }
161
162 interp[0] = 1.0 - (interp[1] + interp[2]);
163 return interp;
164}

174template<typename ScalarType>
175std::vector<ScalarType> randomPolygonBarycentricCoordinate(
176 uint polySize,
177 std::mt19937& gen)
178{
179 std::vector<ScalarType> barCoord(polySize);
180 ScalarType sum = 0;
181
182 std::uniform_real_distribution<ScalarType> unif(0, 100);
183
184 for (uint i = 0; i < polySize; i++) {
185 barCoord[i] = unif(gen);
186 sum += barCoord[i];
187 }
188 for (uint i = 0; i < polySize; i++) {
189 barCoord[i] /= sum;
190 }
191 return barCoord;
192}
