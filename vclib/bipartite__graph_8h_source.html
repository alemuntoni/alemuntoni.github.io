42template<class T1, class T2>
43class BipartiteGraph
44{
45protected:
46 std::map<T1, unsigned int> mapL;
47 std::map<T2, unsigned int> mapR;
48
49 std::vector<UndirectedNode<T1>> nodesL;
50 std::vector<UndirectedNode<T2>> nodesR;

52 std::set<unsigned int> unusedLNodes;
53 std::set<unsigned int> unusedRNodes;

55public:

59 using RightNodeType = UndirectedNode<T2>;

66 using LeftNodeView = vcl::View<LeftNodeIterator>;

150 bool addLeftNode(const T1& info)
151 {
152 if (mapL.find(info) == mapL.end()) {
153 if (unusedLNodes.size() == 0) {
154 mapL[info] = (unsigned int) nodesL.size();
155 nodesL.emplace_back(info);
156 }
157 else {
158 unsigned int id = *(unusedLNodes.begin());
159 unusedLNodes.erase(unusedLNodes.begin());
160 mapL[info] = id;
161 nodesL[id] = UndirectedNode<T1>(info);
162 }
163 return true;
164 }
165 else
166 return false;
167 }
168
175 bool addRightNode(const T2& info)
176 {
177 if (mapR.find(info) == mapR.end()) {
178 if (unusedRNodes.size() == 0) {
179 mapR[info] = (unsigned int) nodesR.size();
180 nodesR.emplace_back(info);
181 }
182 else {
183 unsigned int id = *(unusedRNodes.begin());
184 unusedRNodes.erase(unusedRNodes.begin());
185 mapR[info] = id;
186 nodesR[id] = UndirectedNode<T2>(info);
187 }
188 return true;
189 }
190 else
191 return false;
192 }
193
199 bool deleteLeftNode(const T1& lNode)
200 {
201 if (clearAdjacencesLeftNode(lNode)) {
202 unusedLNodes.insert(mapL[lNode]);
203 mapL.erase(lNode);
204 return true;
205 }
206 else
207 return false;
208 }
209
215 bool deleteRightNode(const T2& rNode)
216 {
217 if (clearAdjacencesRightNode(rNode)) {
218 unusedRNodes.insert(mapR[rNode]);
219 mapR.erase(rNode);
220 return true;
221 }
222 else
223 return false;

233 bool addArc(const T1& lNode, const T2& rNode)
234 {
235 try {
236 int uid = getIdLeftNode(lNode);
237 int vid = getIdRightNode(rNode);
238 assert((unsigned int) uid < nodesL.size());
239 assert((unsigned int) vid < nodesR.size());
240 nodesL[uid].addAdjacent(vid);
241 nodesR[vid].addAdjacent(uid);
242 return true;
243 }
244 catch (...) {
245 return false;
246 }
247 }
248
256 bool deleteArc(const T1& lNode, const T2& rNode)
257 {
258 try {
259 int uid = getIdLeftNode(lNode);
260 int vid = getIdRightNode(rNode);
261 assert((unsigned int) uid < nodesL.size());
262 assert((unsigned int) vid < nodesR.size());
263 nodesL[uid].deleteAdjacent(vid);
264 nodesR[vid].deleteAdjacent(uid);
265 return true;
266 }
267 catch (...) {
268 return false;
269 }
270 }

278 bool clearAdjacencesLeftNode(const T1& lNode)
279 {
280 try {
281 int uid = getIdLeftNode(lNode);
282 for (unsigned int adj : nodesL[uid]) {
283 nodesR[adj].deleteAdjacent(uid);
284 }
285 nodesL[uid].clearAdjacentNodes();
286 return true;
287 }
288 catch (...) {
289 return false;
290 }
291 }
292
299 bool clearAdjacencesRightNode(const T2& rNode)
300 {
301 try {
302 int vid = getIdRightNode(rNode);
303 for (unsigned int adj : nodesR[vid]) {
304 nodesL[adj].deleteAdjacent(vid);
305 }
306 nodesR[vid].clearAdjacentNodes();
307 return true;
308 }
309 catch (...) {
310 return false;
311 }
312 }

347 return true;
348 }

350 return false;
351 }
352 }
