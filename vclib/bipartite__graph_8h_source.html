41template<class T1, class T2>
42class BipartiteGraph
43{
44protected:

149 bool addLeftNode(const T1& info)
150 {
151 if (mMapL.find(info) == mMapL.end()) {
152 if (mUnusedLNodes.size() == 0) {
153 mMapL[info] = (unsigned int) mNodesL.size();
154 mNodesL.emplace_back(info);
155 }
156 else {
157 unsigned int id = *(mUnusedLNodes.begin());
158 mUnusedLNodes.erase(mUnusedLNodes.begin());
159 mMapL[info] = id;
160 mNodesL[id] = UndirectedNode<T1>(info);
161 }
162 return true;
163 }
164 else
165 return false;
166 }
167
174 bool addRightNode(const T2& info)
175 {
176 if (mMapR.find(info) == mMapR.end()) {
177 if (mUnusedRNodes.size() == 0) {
178 mMapR[info] = (unsigned int) mNodesR.size();
179 mNodesR.emplace_back(info);
180 }
181 else {
182 unsigned int id = *(mUnusedRNodes.begin());
183 mUnusedRNodes.erase(mUnusedRNodes.begin());
184 mMapR[info] = id;
185 mNodesR[id] = UndirectedNode<T2>(info);
186 }
187 return true;
188 }
189 else
190 return false;
191 }
192
198 bool deleteLeftNode(const T1& lNode)
199 {
200 if (clearAdjacencesLeftNode(lNode)) {
201 mUnusedLNodes.insert(mMapL[lNode]);
202 mMapL.erase(lNode);
203 return true;
204 }
205 else
206 return false;
207 }
208
214 bool deleteRightNode(const T2& rNode)
215 {
216 if (clearAdjacencesRightNode(rNode)) {
217 mUnusedRNodes.insert(mMapR[rNode]);
218 mMapR.erase(rNode);
219 return true;
220 }
221 else
222 return false;
223 }

232 bool addArc(const T1& lNode, const T2& rNode)
233 {
234 try {
235 int uid = getIdLeftNode(lNode);
236 int vid = getIdRightNode(rNode);
237 assert((unsigned int) uid < mNodesL.size());
238 assert((unsigned int) vid < mNodesR.size());
239 mNodesL[uid].addAdjacent(vid);
240 mNodesR[vid].addAdjacent(uid);
241 return true;
242 }
243 catch (...) {
244 return false;
245 }
246 }
247
255 bool deleteArc(const T1& lNode, const T2& rNode)
256 {
257 try {
258 int uid = getIdLeftNode(lNode);
259 int vid = getIdRightNode(rNode);
260 assert((unsigned int) uid < mNodesL.size());
261 assert((unsigned int) vid < mNodesR.size());
262 mNodesL[uid].deleteAdjacent(vid);
263 mNodesR[vid].deleteAdjacent(uid);
264 return true;
265 }
266 catch (...) {
267 return false;
268 }
269 }
270
277 bool clearAdjacencesLeftNode(const T1& lNode)
278 {
279 try {
280 int uid = getIdLeftNode(lNode);
281 for (unsigned int adj : mNodesL[uid]) {
282 mNodesR[adj].deleteAdjacent(uid);
283 }
284 mNodesL[uid].clearAdjacentNodes();
285 return true;
286 }
287 catch (...) {
288 return false;
289 }
290 }
291
298 bool clearAdjacencesRightNode(const T2& rNode)
299 {
300 try {
301 int vid = getIdRightNode(rNode);
302 for (unsigned int adj : mNodesR[vid]) {
303 mNodesL[adj].deleteAdjacent(vid);
304 }
305 mNodesR[vid].clearAdjacentNodes();
306 return true;
307 }
308 catch (...) {
309 return false;
310 }
311 }

349 return false;
350 }
