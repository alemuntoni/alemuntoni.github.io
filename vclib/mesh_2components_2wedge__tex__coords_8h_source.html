309 // Component interface function
310 template <typename Element>
311 void importFrom(const Element& e)
312 {
313 if constexpr (HasWedgeTexCoords<Element>) {
314 if (isWedgeTexCoordsAvailableOn(e)) {
315 if constexpr(N > 0) {
316 // same static size
317 if constexpr (N == Element::WEDGE_TEX_COORD_NUMBER) {
318 importWedgeTexCoordsFrom(e);
319 }
320 // from dynamic to static, but dynamic size == static size
321 else if constexpr (Element::WEDGE_TEX_COORD_NUMBER < 0){
322 if (e.vertexNumber() == N){
323 importWedgeTexCoordsFrom(e);
324 }
325 }
326 else {
327 // do not import in this case: cannot import from
328 // dynamic size != static size
329 }
330 }
331 else {
332 // from static/dynamic to dynamic size: need to resize
333 // first, then import
334 resize(e.vertexNumber());
335 importWedgeTexCoordsFrom(e);
336 }
337 }
338 }
339 }
