36struct ParameterType
37{
38 enum Enum {
39 // native parameters - the std::any mValue member is one of these types
40 INT,
41 UINT,
42 SCALAR,
43 BOOL,
44 STRING,
45 COLOR,
46 POINT3,
47
48 // special parameters - they store native parameters, but they add
49 // additional information or functionalities
50 ENUM, // enum is an int that can be converted to a string
51 USCALAR, // uscalar is a scalar that is guaranteed to be positive
52
53 MESH, // mesh is a special parameter that stores the index of a mesh
54
55 COUNT
56 };
57};

197 void checkParameterType(ParameterType::Enum t) const
198 {
199 if (nativeType() != nativeType(t)) {
