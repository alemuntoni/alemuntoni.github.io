113 template<MeshConcept MeshType>
114 KDTree(
115 const MeshType& m,
116 uint pointsPerCell = 16,
117 uint maxDepth = 64,
118 bool balanced = false)
119 requires (std::is_same_v<
120 typename MeshType::VertexType::CoordType,
121 PointType>)
122 :
123 points(m.vertexNumber()), indices(m.vertexNumber()),
124 pointsPerCell(pointsPerCell), maxDepth(maxDepth)
125 {
126 using VertexType = MeshType::VertexType;
127
128 uint i = 0;
129 for (const VertexType& v : m.vertices()) {
130 points[i] = v.coord();
131 indices[i] = m.index(v);
132 i++;
133 }
134 nodes.resize(1);
135 nodes.back().leaf = 0;
136
137 depth = createTree(0, 0, points.size(), 1, balanced);
138 }

146 uint nearestNeighborIndex(
147 const PointType& queryPoint,
148 Scalar& dist = dummyScalar) const
149 {
150 std::vector<QueryNode> mNodeStack(depth + 1);
151 mNodeStack[0].nodeId = 0;
152 mNodeStack[0].sq = 0.;
153 unsigned int count = 1;
154
155 int minIndex = indices.size() / 2;
156 Scalar minDist = queryPoint.squaredDist(points[minIndex]);
157 minIndex = indices[minIndex];
158
159 while (count) {
160 QueryNode& qnode = mNodeStack[count - 1];
161 const Node& node = nodes[qnode.nodeId];
162
163 if (qnode.sq < minDist) {
164 if (node.leaf) {
165 --count; // pop
166 uint end = node.start + node.size;
167 for (uint i = node.start; i < end; ++i) {
168 Scalar pointSquareDist =
169 queryPoint.squaredDist(points[i]);
170 if (pointSquareDist < minDist) {
171 minDist = pointSquareDist;
172 minIndex = indices[i];
173 }
174 }
175 }
176 else {
177 // replace the stack top by the farthest and push the closest
178 Scalar new_off = queryPoint[node.dim] - node.splitValue;
179 if (new_off < 0.) {
180 mNodeStack[count].nodeId = node.firstChildId;
181 qnode.nodeId = node.firstChildId + 1;
182 }
183 else {
184 mNodeStack[count].nodeId = node.firstChildId + 1;
185 qnode.nodeId = node.firstChildId;
186 }
187 mNodeStack[count].sq = qnode.sq;
188 qnode.sq = new_off * new_off;
189 ++count;
190 }
191 }
192 else {
193 // pop
194 --count;
195 }
196 }
197 dist = std::sqrt(minDist);
198 return minIndex;
199 }

225 std::vector<uint> kNearestNeighborsIndices(
226 const PointType& queryPoint,
227 uint k,
228 std::vector<Scalar>& distances = dummyScalars) const
229 {
230 struct P
231 {
232 P(const PointType& p, int i) : p(p), i(i) {}
233 PointType p;
234 uint i;
235 };
236
237 struct Comparator
238 {
239 const PointType qp;
240 Comparator(const PointType& qp) : qp(qp) {}
241 bool operator()(const P& p1, const P& p2)
242 {
243 if (qp.squaredDist(p1.p) > qp.squaredDist(p2.p))
244 return true;
245 return false;
246 }
247 };
248
249 Comparator cmp(queryPoint);
250
251 std::priority_queue<P, std::vector<P>, Comparator> neighborQueue(cmp);
252
253 std::vector<QueryNode> mNodeStack(depth + 1);
254 mNodeStack[0].nodeId = 0;
255 mNodeStack[0].sq = 0.;
256 unsigned int count = 1;
257
258 while (count) {
259 // we select the last node (AABB) inserted in the stack
260 QueryNode& qnode = mNodeStack[count - 1];
261
262 // while going down the tree qnode.nodeId is the nearest sub-tree,
263 // otherwise, in backtracking, qnode.nodeId is the other sub-tree
264 // that will be visited iff the actual nearest node is further than
265 // the split distance.
266 const Node& node = nodes[qnode.nodeId];
267
268 // if the distance is less than the top of the max-heap, it could be
269 // one of the k-nearest neighbours
270 if (neighborQueue.size() < k ||
271 qnode.sq < queryPoint.squaredDist(neighborQueue.top().p)) {
272 // when we arrive to a leaf
273 if (node.leaf) {
274 --count; // pop of the leaf
275
276 // end is the index of the last element of the leaf in
277 // mPoints
278 unsigned int end = node.start + node.size;
279 // adding the element of the leaf to the heap
280 for (unsigned int i = node.start; i < end; ++i)
281 neighborQueue.push(P(points[i], indices[i]));
282 }
283 // otherwise, if we're not on a leaf
284 else {
285 // the new offset is the distance between the searched point
286 // and the actual split coordinate
287 Scalar new_off = queryPoint[node.dim] - node.splitValue;
288
289 // left sub-tree
290 if (new_off < 0.) {
291 mNodeStack[count].nodeId = node.firstChildId;
292 // in the father's nodeId we save the index of the other
293 // sub-tree (for backtracking)
294 qnode.nodeId = node.firstChildId + 1;
295 }
296 // right sub-tree (same as above)
297 else {
298 mNodeStack[count].nodeId = node.firstChildId + 1;
299 qnode.nodeId = node.firstChildId;
300 }
301 // distance is inherited from the father (while descending
302 // the tree it's equal to 0)
303 mNodeStack[count].sq = qnode.sq;
304 // distance of the father is the squared distance from the
305 // split plane
306 qnode.sq = new_off * new_off;
307 ++count;
308 }
309 }
310 else {
311 // pop
312 --count;
313 }
314 }
315 distances.clear();
316 std::vector<uint> res;
317
318 uint i = 0;
319 while (!neighborQueue.empty() && i < k) {
320 res.push_back(neighborQueue.top().i);
321 distances.push_back(queryPoint.dist(neighborQueue.top().p));
322 neighborQueue.pop();
323 i++;
324 }
325 return res;
326 }

350 std::vector<uint> neighborsIndicesInDistance(
351 const PointType& queryPoint,
352 Scalar dist,
353 std::vector<Scalar>& distances = dummyScalars) const
354 {
355 std::vector<uint> queryPoints;
356 distances.clear();
357 std::vector<QueryNode> mNodeStack(depth + 1);
358 mNodeStack[0].nodeId = 0;
359 mNodeStack[0].sq = 0.;
360 unsigned int count = 1;
361
362 Scalar squareDist = dist * dist;
363 while (count) {
364 QueryNode& qnode = mNodeStack[count - 1];
365 const Node& node = nodes[qnode.nodeId];
366
367 if (qnode.sq < squareDist) {
368 if (node.leaf) {
369 --count; // pop
370 unsigned int end = node.start + node.size;
371 for (unsigned int i = node.start; i < end; ++i) {
372 Scalar pointSquareDist =
373 queryPoint.squareDist(points[i]);
374 if (pointSquareDist < squareDist) {
375 queryPoints.push_back(indices[i]);
376 distances.push_back(queryPoint.dist(points[i]));
377 }
378 }
379 }
380 else {
381 // replace the stack top by the farthest and push the
382 // closest
383 Scalar new_off = queryPoint[node.dim] - node.splitValue;
384 if (new_off < 0.) {
385 mNodeStack[count].nodeId = node.firstChildId;
386 qnode.nodeId = node.firstChildId + 1;
387 }
388 else {
389 mNodeStack[count].nodeId = node.firstChildId + 1;
390 qnode.nodeId = node.firstChildId;
391 }
392 mNodeStack[count].sq = qnode.sq;
393 qnode.sq = new_off * new_off;
394 ++count;
395 }
396 }
397 else {
398 // pop
399 --count;
400 }
401 }
402 return queryPoints;
403 }

442 uint
443 createTree(uint nodeId, uint start, uint end, uint level, bool balanced)
444 {
445 // select the first node
446 Node& node = nodes[nodeId];
447 Box<PointType> aabb;
448
449 // putting all the points in the bounding box
450 aabb.add(points[start]);
451 for (uint i = start + 1; i < end; ++i)
452 aabb.add(points[i]);
453
454 // bounding box diagonal
455 PointType diag = aabb.max() - aabb.min();
456
457 // the split "dim" is the dimension of the box with the biggest value
458 uint dim = 0;
459 Scalar max = std::numeric_limits<double>::lowest();
460 for (uint i = 0; i < PointType::DIM; ++i) {
461 if (diag[i] > max) {
462 max = diag[i];
463 dim = i;
464 }
465 }
466
467 node.dim = dim;
468 // we divide the points using the median value along the "dim" dimension
469 if (balanced) {
470 std::vector<Scalar> tempVector;
471 for (uint i = start + 1; i < end; ++i)
472 tempVector.push_back(points[i][dim]);
473 std::sort(tempVector.begin(), tempVector.end());
474 node.splitValue = (tempVector[tempVector.size() / 2.0] +
475 tempVector[tempVector.size() / 2.0 + 1]) /
476 2.0;
477 }
478 // we divide the bounding box in 2 partitions, considering the average
479 // of the "dim" dimension
480 else {
481 node.splitValue = Scalar(0.5 * (aabb.max()[dim] + aabb.min()[dim]));
482 }
483
484 // midId is the index of the first element in the second partition
485 unsigned int midId = split(start, end, dim, node.splitValue);
486
487 node.firstChildId = nodes.size();
488 nodes.resize(nodes.size() + 2);
489 bool flag = (midId == start) || (midId == end);
490 int leftLevel, rightLevel;
491
492 // left child
493 unsigned int childId = nodes[nodeId].firstChildId;
494 Node& childL = nodes[childId];
495 if (flag || (midId - start) <= pointsPerCell || level >= maxDepth) {
496 childL.leaf = 1;
497 childL.start = start;
498 childL.size = midId - start;
499 leftLevel = level;
500 }
501 else {
502 childL.leaf = 0;
503 leftLevel = createTree(childId, start, midId, level + 1, balanced);
504 }
505
506 // right child
507 childId = nodes[nodeId].firstChildId + 1;
508 Node& childR = nodes[childId];
509 if (flag || (end - midId) <= pointsPerCell || level >= maxDepth) {
510 childR.leaf = 1;
511 childR.start = midId;
512 childR.size = end - midId;
513 rightLevel = level;
514 }
515 else {
516 childR.leaf = 0;
517 rightLevel = createTree(childId, midId, end, level + 1, balanced);
518 }
519
520 if (leftLevel > rightLevel)
521 return leftLevel;
522 return rightLevel;
523 }
524
531 uint split(uint start, uint end, uint dim, Scalar splitValue)
532 {
533 uint l, r;
534 for (l = start, r = end - 1; l < r; ++l, --r) {
535 while (l < end && points[l][dim] < splitValue)
536 l++;
537 while (r >= start && points[r][dim] >= splitValue)
538 r--;
539 if (l > r)
540 break;
541 std::swap(points[l], points[r]);
542 std::swap(indices[l], indices[r]);
543 }
544
545 // returns the index of the first element on the second part
546 return (points[l][dim] < splitValue ? l + 1 : l);
547 }
