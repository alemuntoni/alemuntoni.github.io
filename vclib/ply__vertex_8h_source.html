42template<MeshConcept MeshType, VertexConcept VertexType, typename Stream>
43void loadVertexProperty(
44 Stream& file,
45 MeshType& mesh,
46 VertexType& v,
47 ply::Property p)
48{
49 bool hasBeenRead = false;
50 if (p.name >= ply::x && p.name <= ply::z) {
51 using Scalar = VertexType::CoordType::ScalarType;
52 int a = p.name - ply::x;
53 v.coord()[a] = io::internal::readProperty<Scalar>(file, p.type);
54 hasBeenRead = true;
55 }
56 if (p.name >= ply::nx && p.name <= ply::nz) {
57 if constexpr (vcl::HasPerVertexNormal<MeshType>) {
58 if (vcl::isPerVertexNormalAvailable(mesh)) {
59 using Scalar = VertexType::NormalType::ScalarType;
60 int a = p.name - ply::nx;
61 v.normal()[a] =
62 io::internal::readProperty<Scalar>(file, p.type);
63 hasBeenRead = true;
64 }
65 }
66 }
67 if (p.name >= ply::red && p.name <= ply::alpha) {
68 if constexpr (vcl::HasPerVertexColor<MeshType>) {
69 if (vcl::isPerVertexColorAvailable(mesh)) {
70 int a = p.name - ply::red;
71 v.color()[a] =
72 io::internal::readProperty<unsigned char>(file, p.type);
73 hasBeenRead = true;
74 }
75 }
76 }
77 if (p.name == ply::quality) {
78 if constexpr (vcl::HasPerVertexQuality<MeshType>) {
79 using QualityType = VertexType::QualityType;
80 if (vcl::isPerVertexQualityAvailable(mesh)) {
81 v.quality() =
82 io::internal::readProperty<QualityType>(file, p.type);
83 hasBeenRead = true;
84 }
85 }
86 }
87 if (p.name >= ply::texture_u && p.name <= ply::texture_v) {
88 if constexpr (vcl::HasPerVertexTexCoord<MeshType>) {
89 using Scalar = VertexType::TexCoordType::ScalarType;
90 if (vcl::isPerVertexTexCoordAvailable(mesh)) {
91 int a = p.name - ply::texture_u;
92 v.texCoord()[a] =
93 io::internal::readProperty<Scalar>(file, p.type);
94 hasBeenRead = true;
95 }
96 }
97 }
98 if (p.name == ply::unknown) {
99 if constexpr (vcl::HasPerVertexCustomComponents<MeshType>) {
100 if (mesh.hasPerVertexCustomComponent(p.unknownPropertyName)){
101 io::internal::readCustomComponent(
102 file, v, p.unknownPropertyName, p.type);
103 hasBeenRead = true;
104 }
105 }
106 }
107 if (!hasBeenRead) {
108 if (p.list) {
109 uint s = io::internal::readProperty<int>(file, p.listSizeType);
110 for (uint i = 0; i < s; ++i)
111 io::internal::readProperty<int>(file, p.type);
112 }
113 else {
114 io::internal::readProperty<int>(file, p.type);
115 }
116 }
117}

119template<VertexConcept VertexType, MeshConcept MeshType>
120void loadVertexTxt(
121 std::ifstream& file,
122 VertexType& v,
123 MeshType& mesh,
124 const std::list<ply::Property>& vertexProperties)
125{
126 vcl::Tokenizer spaceTokenizer =
127 io::internal::nextNonEmptyTokenizedLine(file);
128 vcl::Tokenizer::iterator token = spaceTokenizer.begin();
129 for (const ply::Property& p : vertexProperties) {
130 if (token == spaceTokenizer.end()){
131 throw vcl::MalformedFileException("Unexpected end of line.");
132 }
133 loadVertexProperty(token, mesh, v, p);
134 }
135}

151template <MeshConcept MeshType>
152void saveVertices(
153 std::ofstream& file,
154 const PlyHeader& header,
155 const MeshType& mesh)
156{
157 using VertexType = MeshType::VertexType;
158
159 bool bin = header.format() == ply::BINARY;
160 for(const VertexType& v : mesh.vertices()) {
161 for (const ply::Property& p : header.vertexProperties()) {
162 bool hasBeenWritten = false;
163 if (p.name >= ply::x && p.name <= ply::z) {
164 io::internal::writeProperty(
165 file, v.coord()[p.name - ply::x], p.type, bin);
166 hasBeenWritten = true;
167 }
168 if (p.name >= ply::nx && p.name <= ply::nz) {
169 if constexpr (vcl::HasPerVertexNormal<MeshType>) {
170 io::internal::writeProperty(
171 file, v.normal()[p.name - ply::nx], p.type, bin);
172 hasBeenWritten = true;
173 }
174 }
175 if (p.name >= ply::red && p.name <= ply::alpha) {
176 if constexpr (vcl::HasPerVertexColor<MeshType>) {
177 io::internal::writeProperty(
178 file, v.color()[p.name - ply::red], p.type, bin);
179 hasBeenWritten = true;
180 }
181 }
182 if (p.name == ply::quality) {
183 if constexpr (vcl::HasPerVertexQuality<MeshType>) {
184 io::internal::writeProperty(file, v.quality(), p.type, bin);
185 hasBeenWritten = true;
186 }
187 }
188 if (p.name >= ply::texture_u && p.name <= ply::texture_v) {
189 if constexpr (vcl::HasPerVertexTexCoord<MeshType>) {
190 const uint a = p.name - ply::texture_u;
191 io::internal::writeProperty(
192 file, v.texCoord()[a], p.type, bin);
193 hasBeenWritten = true;
194 }
195 }
196 if (p.name == ply::unknown) {
197 if constexpr (vcl::HasPerVertexCustomComponents<MeshType>) {
198 if (mesh.hasPerVertexCustomComponent(p.unknownPropertyName))
199 {
200 io::internal::writeCustomComponent(
201 file, v, p.unknownPropertyName, p.type, bin);
202 hasBeenWritten = true;
203 }
204 }
205 }
206 if (!hasBeenWritten){
207 // be sure to write something if the header declares some
208 // property that is not in the mesh
209 io::internal::writeProperty(file, 0, p.type, bin);
210 }
211 }
212 if (!bin)
213 file << std::endl;
214 }
215}

217template <MeshConcept MeshType>
218void loadVertices(
219 std::ifstream& file,
220 const PlyHeader& header,
221 MeshType& m)
222{
223 m.addVertices(header.numberVertices());
224
225 for(uint vid = 0; vid < header.numberVertices(); ++vid) {
226 auto& v = m.vertex(vid);
227 if(header.format() == ply::ASCII) {
228 internal::loadVertexTxt(file, v, m, header.vertexProperties());
229 }
230 else if(header.format() == ply::BINARY) {
231 internal::loadVertexBin(file, v, m, header.vertexProperties());
232 }
233 }
234}
