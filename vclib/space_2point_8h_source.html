541 void orthoBase(Point& u, Point& v) const requires (N == 3)
542 {
543 const double locEps = double(1e-7);
544
545 Point<Scalar, 3> up(0, 1, 0);
546 u = cross(up);
547
548 double len = u.norm();
549 if (len < locEps) {
550 if (std::abs(p[0]) < std::abs(p[1])) {
551 if (std::abs(p[0]) < std::abs(p[2]))
552 up = Point<Scalar, 3>(1, 0, 0); // x is the min
553 else
554 up = Point<Scalar, 3>(0, 0, 1); // z is the min
555 }
556 else {
557 if (std::abs(p[1]) < std::abs(p[2]))
558 up = Point<Scalar, 3>(0, 1, 0); // y is the min
559 else
560 up = Point<Scalar, 3>(0, 0, 1); // z is the min
561 }
562 u = cross(up);
563 }
564 v = cross(u);
565 }

809 Point operator*(const Eigen::Matrix<Scalar, N + 1, N + 1>& m) const
810 requires (N == 3)
811 {
812 Eigen::Matrix<Scalar, 1, N> s;
813
814 s(0) = m(0, 0) * p(0) + m(0, 1) * p(1) + m(0, 2) * p(2) + m(0, 3);
815 s(1) = m(1, 0) * p(0) + m(1, 1) * p(1) + m(1, 2) * p(2) + m(1, 3);
816 s(2) = m(2, 0) * p(0) + m(2, 1) * p(1) + m(2, 2) * p(2) + m(2, 3);
817
818 Scalar w = p(0) * m(3, 0) + p(1) * m(3, 1) + p(2) * m(3, 2) + m(3, 3);
819 if (w != 0)
820 s = s / w;
821 return Point(s);
822 }
