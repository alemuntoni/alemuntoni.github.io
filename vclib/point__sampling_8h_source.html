75template<SamplerConcept SamplerType, MeshConcept MeshType>
76SamplerType allVerticesPointSampling(
77 const MeshType& m,
78 std::vector<uint>& birthVertices,
79 bool onlySelected = false)
80{
81 using VertexType = MeshType::VertexType;
82
83 SamplerType sampler;
84
85 // Determine the number of vertices to sample
86 uint n = onlySelected ? vertexSelectionNumber(m) : m.vertexNumber();
87
88 // Reserve space in the sampler and birthVertices vectors
89 sampler.reserve(n);
90 birthVertices.reserve(n);
91 birthVertices.clear();
92
93 // Loop through all the vertices in the mesh and add them to the sampler if
94 // they are selected
95 for (const VertexType& v : m.vertices()) {
96 if (!onlySelected || v.selected()) {
97 sampler.add(v);
98 birthVertices.push_back(m.index(v));
99 }
100 }
101 return sampler;
102}

233template<SamplerConcept SamplerType, MeshConcept MeshType>
234SamplerType vertexUniformPointSampling(
235 const MeshType& m,
236 uint nSamples,
237 std::vector<uint>& birthVertices,
238 bool onlySelected = false,
239 bool deterministic = false)
240{
241 uint vn = onlySelected ? vcl::vertexSelectionNumber(m) : m.vertexNumber();
242
243 if (nSamples >= vn) {
244 return allVerticesPointSampling<SamplerType>(
245 m, birthVertices, onlySelected);
246 }
247
248 SamplerType ps;
249
250 // Reserve space in the sampler and birthVertices vectors
251 ps.reserve(vn);
252 birthVertices.reserve(vn);
253 birthVertices.clear();
254
255 std::uniform_int_distribution<uint> dist(0, m.vertexContainerSize() - 1);
256
257 std::random_device rd;
258 std::mt19937 gen(rd());
259 if (deterministic)
260 gen = std::mt19937(0);
261
262 std::vector<bool> visited(m.vertexContainerSize(), false);
263 uint nVisited = 0;
264
265 while (nVisited < nSamples) {
266 uint vi = dist(gen);
267 if (!m.vertex(vi).deleted() && !visited[vi] &&
268 (!onlySelected || m.vertex(vi).selected())) {
269 visited[vi] = true;
270 nVisited++;
271 ps.add(m.vertex(vi));
272 birthVertices.push_back(vi);
273 }
274 }
275
276 return ps;
277}

697template<SamplerConcept SamplerType, FaceMeshConcept MeshType>
698SamplerType vertexAreaWeightedPointSampling(
699 const MeshType& m,
700 uint nSamples,
701 bool deterministic = false)
702{
703 using VertexType = MeshType::VertexType;
704 using ScalarType = VertexType::ScalarType;
705 using FaceType = MeshType::FaceType;
706
707 std::vector<ScalarType> weights(m.vertexContainerSize(), 0);
708 std::vector<uint> cnt(m.vertexContainerSize(), 0);
709
710 // for each vertex, store in weights the adjacent faces area and their
711 // number
712 for (const FaceType& f : m.faces()) {
713 ScalarType area = vcl::faceArea(f);
714 for (const VertexType* v : f.vertices()) {
715 weights[m.index(v)] += area;
716 cnt[m.index(v)]++;
717 }
718 }
719
720 // divide each area sum by the number of adjacent faces
721 for (uint i = 0; i < weights.size(); i++) {
722 if (cnt[i] > 0)
723 weights[i] /= cnt[i];
724 }
725
726 // use these weights to create a sapler
727 return vertexWeightedPointSampling<SamplerType>(
728 m, weights, nSamples, deterministic);
729}

778template<SamplerConcept SamplerType, FaceMeshConcept MeshType>
779SamplerType montecarloPointSampling(
780 const MeshType& m,
781 uint nSamples,
782 std::vector<uint>& birthFaces,
783 bool deterministic = false)
784{
785 using VertexType = MeshType::VertexType;
786 using ScalarType = VertexType::CoordType::ScalarType;
787 using FaceType = MeshType::FaceType;
788 using Interval = std::pair<ScalarType, const FaceType*>;
789
790 vcl::FirstElementPairComparator<Interval> comparator;
791 SamplerType sampler;
792
793 // Reserve space in the sampler and birthFaces vectors
794 sampler.reserve(nSamples);
795 birthFaces.reserve(nSamples);
796 birthFaces.clear();
797
798 std::uniform_real_distribution<ScalarType> dist(0, 1);
799
800 std::random_device rd;
801 std::mt19937 gen(rd());
802 if (deterministic)
803 gen = std::mt19937(0);
804
805 std::vector<Interval> intervals(m.faceNumber());
806 uint i = 0;
807 ScalarType area = 0;
808 for (const FaceType& f : m.faces()) {
809 area += vcl::faceArea(f);
810 intervals[i] = std::make_pair(area, &f);
811 i++;
812 }
813
814 ScalarType meshArea = intervals.back().first;
815 for (uint i = 0; i < nSamples; i++) {
816 ScalarType val = meshArea * dist(gen);
817 // lower_bound returns the furthermost iterator i in [first, last) such
818 // that, for every iterator j in [first, i), *j < value. E.g. An
819 // iterator pointing to the first element "not less than" val, or end()
820 // if every element is less than val.
821 typename std::vector<Interval>::iterator it = std::lower_bound(
822 intervals.begin(),
823 intervals.end(),
824 std::make_pair(val, nullptr),
825 comparator);
826
827 sampler.add(
828 *it->second,
829 vcl::randomPolygonBarycentricCoordinate<ScalarType>(
830 it->second->vertexNumber(), gen));
831 birthFaces.push_back(it->second->index());
832 }
833
834 return sampler;
835}

864template<SamplerConcept SamplerType, FaceMeshConcept MeshType>
865SamplerType stratifiedMontecarloPointSampling(
866 const MeshType& m,
867 uint nSamples,
868 bool deterministic = false)
869{
870 using FaceType = MeshType::FaceType;
871 using ScalarType = SamplerType::ScalarType;
872
873 SamplerType ps;
874
875 std::random_device rd;
876 std::mt19937 gen(rd());
877 if (deterministic)
878 gen = std::mt19937(0);
879
880 double area = vcl::surfaceArea(m);
881 double samplePerAreaUnit = nSamples / area;
882 // Montecarlo sampling.
883 double floatSampleNum = 0.0;
884
885 for(const FaceType& f : m.faces()) {
886 // compute # samples in the current face (taking into account of the
887 // remainders)
888 floatSampleNum += vcl::faceArea(f) * samplePerAreaUnit;
889 int faceSampleNum = (int) floatSampleNum;
890 // for every sample p_i in T...
891 for(int i=0; i < faceSampleNum; i++)
892 ps.add(
893 f,
894 vcl::randomPolygonBarycentricCoordinate<ScalarType>(
895 f.vertexNumber(), gen));
896 floatSampleNum -= (double) faceSampleNum;
897 }
898
899 return ps;
900}

958template<
959 SamplerConcept SamplerType,
960 FaceMeshConcept MeshType,
961 typename ScalarType>
962SamplerType vertexWeightedMontecarloPointSampling(
963 const MeshType& m,
964 const std::vector<ScalarType>& weights,
965 uint nSamples,
966 double variance,
967 bool deterministic = false)
968{
969 using FaceType = MeshType::FaceType;
970
971 // lambda function to compute radius weighted area of a face
972 auto weightedArea = [](const FaceType& f,
973 const MeshType& m,
974 const std::vector<ScalarType>& r) -> ScalarType {
975 ScalarType averageQ = 0;
976 for (uint i = 0; i < f.vertexNumber(); ++i)
977 averageQ += r[m.index(f.vertex(i))];
978
979 averageQ /= f.vertexNumber();
980 return averageQ * averageQ * vcl::faceArea(f);
981 };
982
983 SamplerType ps;
984
985 std::random_device rd;
986 std::mt19937 gen(rd());
987 if (deterministic)
988 gen = std::mt19937(0);
989
990 std::vector<ScalarType> radius =
991 vcl::vertexRadiusFromWeights(m, weights, 1.0, variance, true);
992
993 ScalarType wArea = 0;
994 for(const FaceType& f : m.faces())
995 wArea += weightedArea(f, m , radius);
996
997 ScalarType samplePerAreaUnit = nSamples / wArea;
998 // Montecarlo sampling.
999 double floatSampleNum = 0.0;
1000 for (const FaceType& f : m.faces()) {
1001 // compute # samples in the current face (taking into account of the
1002 // remainders)
1003 floatSampleNum += weightedArea(f, m, radius) * samplePerAreaUnit;
1004 uint faceSampleNum = (uint) floatSampleNum;
1005
1006 // for every sample p_i in T...
1007 for (uint i = 0; i < faceSampleNum; i++)
1008 ps.add(
1009 f,
1010 vcl::randomPolygonBarycentricCoordinate<ScalarType>(
1011 f.vertexNumber(), gen));
1012
1013 floatSampleNum -= (double) faceSampleNum;
1014 }
1015
1016 return ps;
1017}
