228 uint8_t hsvHue() const
229 {
230 uint8_t rgbMin, rgbMax;
231 uint8_t h;
232
233 rgbMin = x() < y() ? (x() < z() ? x() : z()) : (y() < z() ? y() : z());
234 rgbMax = x() > y() ? (x() > z() ? x() : z()) : (y() > z() ? y() : z());
235
236 if (rgbMax == 0) {
237 return 0;
238 }
239
240 if (255 * long(rgbMax - rgbMin) / rgbMax == 0) {
241 return 0;
242 }
243
244 if (rgbMax == x())
245 h = 0 + 43 * (y() - z()) / (rgbMax - rgbMin);
246 else if (rgbMax == y())
247 h = 85 + 43 * (z() - x()) / (rgbMax - rgbMin);
248 else
249 h = 171 + 43 * (x() - y()) / (rgbMax - rgbMin);
250
251 return h;
252 }

258 uint8_t hsvSaturation() const
259 {

261
262 rgbMin = x() < y() ? (x() < z() ? x() : z()) : (y() < z() ? y() : z());
263 rgbMax = x() > y() ? (x() > z() ? x() : z()) : (y() > z() ? y() : z());

265 if (rgbMax == 0) {

471 void setHsv(uint8_t h, uint8_t s, uint8_t v, uint8_t alpha = 255)
472 {
473 w() = alpha;
474 if (s == 0) {
475 x() = v;
476 y() = v;
477 z() = v;
478 }
479 else {
480 h = (h / 360.0) * 255;
481 uint8_t region, remainder, p, q, t;
482 region = h / 43;
483 remainder = (h - (region * 43)) * 6;
484
485 p = (v * (255 - s)) >> 8;
486 q = (v * (255 - ((s * remainder) >> 8))) >> 8;
487 t = (v * (255 - ((s * (255 - remainder)) >> 8))) >> 8;
488
489 switch (region) {
490 case 0:
491 x() = v;
492 y() = t;
493 z() = p;
494 break;
495 case 1:
496 x() = q;
497 y() = v;
498 z() = p;
499 break;
500 case 2:
501 x() = p;
502 y() = v;
503 z() = t;
504 break;
505 case 3:
506 x() = p;
507 y() = q;
508 z() = v;
509 break;
510 case 4:
511 x() = t;
512 y() = p;
513 z() = v;
514 break;
515 default:
516 x() = v;
517 y() = p;
518 z() = q;
519 break;
520 }
521 }
522 }

681inline Color colorLerp(const Color& c0, const Color& c1, float value)
682{
683 Color c;
684 if (value < 0) // out of range - left
685 value = 0;
686 else if (value > 1) // out of range - right
687 value = 1;
688 for (uint i = 0; i < 4; i++)
689 c(i) = c1(i) * value + c0(i) * (1 - value);
690 return c;
691}

757inline Color colorFromIntervalParula(float value)
758{
759 if (value < 0) // out of range - left
760 value = 0;
761 else if (value > 1) // out of range - right
762 value = 1;
763
764 static uint paruVal[9] = {
765 0x271680ff,
766 0x0363e1ff,
767 0x1485d4ff,
768 0x06a7c6ff,
769 0x38b99eff,
770 0x92bf73ff,
771 0xd9ba56ff,
772 0xfcce2eff,
773 0xfffa0aff};
774
775 int ind = int(floor(value * 8.0f));
776 float div = (value * 8.0f - ind);
777
778 if (div < 0)
779 div = 0;
780 else if (div > 1)
781 div = 1;
782
783 return colorLerp(
784 (Color::ColorABGR) paruVal[ind],
785 (Color::ColorABGR) paruVal[ind + 1],
786 div);
787}

928inline std::vector<Color> colorScattering(
929 uint n,
930 float sat = 0.3,
931 float val = 0.9)
932{
933 std::vector<Color> scattering;
934 scattering.reserve(n);
935
936 for (uint v = 0; v < n; ++v) {
937 uint value = v, m = n;
938 uint b = 0;
939
940 for (uint k = 1; k < n; k <<= 1) {
941 if (value << 1 >= m) {
942 b += k;
943 value -= (m + 1) >> 1;
944 m >>= 1;
945 }
946 else {
947 m = (m + 1) >> 1;
948 }
949 }
950
951 Color rc;
952 rc.setHsvF(float(b) / float(n), sat, val);
953 scattering.push_back(rc);
954 }
955 return scattering;
956}
