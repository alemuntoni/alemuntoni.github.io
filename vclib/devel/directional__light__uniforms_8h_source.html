48class DirectionalLightUniforms
49{
50 float mDir[4] = {0.0, 0.0, 1.0, 0.0}; // just first 3 components are used
51 float mCol[4] = {1.0, 1.0, 1.0, 1.0}; // just first 3 components are used
52
53 Uniform mLightDirUniform =
54 Uniform("u_lightDirPack", bgfx::UniformType::Vec4);
55 Uniform mLightColorUniform =
56 Uniform("u_lightColorPack", bgfx::UniformType::Vec4);
57
58public:
59 DirectionalLightUniforms() {}
60
65 template<typename S>
66 void updateLight(const vcl::DirectionalLight<S>& light)
67 {
68 mDir[0] = light.direction().x();
69 mDir[1] = light.direction().y();
70 mDir[2] = light.direction().z();
71
72 mCol[0] = light.color().redF();
73 mCol[1] = light.color().greenF();
74 mCol[2] = light.color().blueF();
75 // light color alpha is not used
76 }
77
78 void bind() const
79 {
80 mLightDirUniform.bind(mDir);
81 mLightColorUniform.bind(mCol);
82 }
83};
