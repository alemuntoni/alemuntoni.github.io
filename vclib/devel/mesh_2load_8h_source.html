94template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
95void load(
96 MeshType& m,
97 const std::string& filename,
98 MeshInfo& loadedInfo,
99 LogType& log = nullLogger,
100 const LoadSettings& settings = LoadSettings())
101{
102 FileFormat ff = FileInfo::fileFormat(filename);
103
104 loadedInfo.clear();
105
106 if (ff == objFileFormat()) {
107 loadObj(m, filename, loadedInfo, log, settings);
108 }
109 else if (ff == offFileFormat()) {
110 loadOff(m, filename, loadedInfo, log, settings);
111 }
112 else if (ff == plyFileFormat()) {
113 loadPly(m, filename, loadedInfo, log, settings);
114 }
115 else if (ff == stlFileFormat()) {
116 loadStl(m, filename, loadedInfo, log, settings);
117 }
118#ifdef VCLIB_WITH_TINYGLTF
119 else if (ff == gltfFileFormat()) {
120 loadGltf(m, filename, loadedInfo, log, settings);
121 }
122#endif
123 else {
124 throw UnknownFileFormatException(ff.extensions().front());
125 }
126}
