37template<uint ELEM_ID, LoggerConcept LogType = NullLogger>
38void normalizeNoThrow(auto& elem, LogType& log = nullLogger)
39{
40 try {
41 elem.normal().normalize();
42 }
43 catch (const std::exception& e) {
44 log.log(
45 log.WARNING_LOG,
46 elementEnumString<ELEM_ID>() + " " + std::to_string(elem.index()) +
47 ": " + e.what());
48 }

94template<uint ELEM_ID, LoggerConcept LogType = NullLogger>
95void normalizePerElementNormals(
96 MeshConcept auto& mesh,
97 LogType& log = nullLogger)
98{
99 requirePerElementComponent<ELEM_ID, CompId::NORMAL>(mesh);
100
101 log.log(
102 0, "Normalizing per-" + elementEnumString<ELEM_ID>() + " normals...");
103
104 // define a lambda that, for each element, normalizes the normal
105 auto normalize = [&](auto& elem) {
106 detail::normalizeNoThrow<ELEM_ID>(elem, log);
107 };
108
109 parallelFor(mesh.template elements<ELEM_ID>(), normalize);
110
111 log.log(
112 100, "Per-" + elementEnumString<ELEM_ID>() + " normals normalized.");
113}

217template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
218void clearPerReferencedVertexNormals(MeshType& mesh, LogType& log = nullLogger)
219{
220 requirePerVertexNormal(mesh);
221
222 // function that is called for each container Cont of the mesh
223 auto f = [&]<mesh::ElementContainerConcept Cont>() {
224 using Elem = typename Cont::ElementType; // the Element type of Cont
225 if constexpr (comp::HasVertexReferences<Elem>) { // if Elem has vertices
226 for (auto& e : mesh.template elements<Elem::ELEMENT_ID>()) {
227 for (auto* v : e.vertices()) {
228 v->normal().setZero();
229 }
230 }
231 }
232 };
233
234 log.log(0, "Clearing per-Vertex normals...");
235
236 // apply the function f to all the containers of the mesh
237 ForEachType<typename MeshType::Containers>::apply(f);
238
239 log.log(100, "Per-Vertex normals cleared.");
240}

374template<LoggerConcept LogType = NullLogger>
375void updatePerFaceNormals(
376 FaceMeshConcept auto& mesh,
377 bool normalize = true,
378 LogType& log = nullLogger)
379{
380 requirePerFaceNormal(mesh);
381
382 using FaceType = RemoveRef<decltype(mesh)>::FaceType;
383 using ScalarType = FaceType::NormalType::ScalarType;
384
385 log.log(0, "Updating per-Face normals...");
386
387 parallelFor(mesh.faces(), [](auto& f) {
388 f.normal() = faceNormal(f).template cast<ScalarType>();
389 });
390
391 if (normalize) {
392 log.startNewTask(50, 100, "Normalizing per-Face normals...");
393 normalizePerFaceNormals(mesh, log);
394 log.endTask("Normalizing per-Face normals...");
395 }
396
397 log.log(100, "Per-Face normals updated.");
398}

418void updatePerVertexNormals(
419 FaceMeshConcept auto& mesh,

421 LogType& log = nullLogger)
422{
423 using VertexType = RemoveRef<decltype(mesh)>::VertexType;
424 using NScalar = VertexType::NormalType::ScalarType;

427
428 log.startNewTask(0, 20, "Clearing per-Vertex normals...");
429 clearPerReferencedVertexNormals(mesh, log);
430 log.endTask("Clearing per-Vertex normals...");
431
432 log.log(20, "Updating per-Vertex normals...");
433
434 for (auto& f : mesh.faces()) {
435 for (auto* v : f.vertices()) {
436 v->normal() += faceNormal(f).template cast<NScalar>();
437 }
438 }
439
440 if (normalize) {
441 log.startNewTask(80, 100, "Normalizing per-Vertex normals...");
442 normalizePerReferencedVertexNormals(mesh, log);
443 log.endTask("Normalizing per-Vertex normals...");
444 }
445
446 log.log(100, "Per-Vertex normals updated.");

572template<LoggerConcept LogType = NullLogger>
573void updatePerVertexNormalsAngleWeighted(
574 FaceMeshConcept auto& mesh,
575 bool normalize = true,
576 LogType& log = nullLogger)
577{
578 using VertexType = RemoveRef<decltype(mesh)>::VertexType;
579 using NormalType = VertexType::NormalType;
580 using NScalarType = NormalType::ScalarType;
581
582 log.log(0, "Updating per-Vertex normals...");
583
584 log.startNewTask(0, 5, "Clearing per-Vertex normals...");
585 clearPerReferencedVertexNormals(mesh, log);
586 log.endTask("Clearing per-Vertex normals...");
587
588 log.log(5, "Updating per-Vertex normals...");
589
590 for (auto& f : mesh.faces()) {
591 auto n = faceNormal(f).template cast<NScalarType>();
592
593 for (uint i = 0; i < f.vertexNumber(); ++i) {
594 NormalType vec1 =
595 (f.vertexMod(i - 1)->position() - f.vertexMod(i)->position())
596 .normalized()
597 .template cast<NScalarType>();
598 NormalType vec2 =
599 (f.vertexMod(i + 1)->position() - f.vertexMod(i)->position())
600 .normalized()
601 .template cast<NScalarType>();
602
603 f.vertex(i)->normal() += n * vec1.angle(vec2);
604 }
605 }
606
607 if (normalize) {
608 log.startNewTask(95, 100, "Normalizing per-Vertex normals...");
609 normalizePerReferencedVertexNormals(mesh, log);
610 log.endTask("Normalizing per-Vertex normals...");
611 }
612
613 log.log(100, "Per-Vertex normals updated.");
614}

646template<LoggerConcept LogType = NullLogger>
647void updatePerVertexNormalsNelsonMaxWeighted(
648 FaceMeshConcept auto& mesh,
649 bool normalize = true,
650 LogType& log = nullLogger)
651{
652 using VertexType = RemoveRef<decltype(mesh)>::VertexType;
653 using NScalarType = VertexType::NormalType::ScalarType;
654
655 log.log(0, "Updating per-Vertex normals...");
656
657 log.startNewTask(0, 5, "Clearing per-Vertex normals...");
658 clearPerReferencedVertexNormals(mesh, log);
659 log.endTask("Clearing per-Vertex normals...");
660
661 log.log(5, "Updating per-Vertex normals...");
662
663 for (auto& f : mesh.faces()) {
664 auto n = faceNormal(f).template cast<NScalarType>();
665
666 for (uint i = 0; i < f.vertexNumber(); ++i) {
667 NScalarType e1 =
668 (f.vertexMod(i - 1)->position() - f.vertexMod(i)->position())
669 .squaredNorm();
670 NScalarType e2 =
671 (f.vertexMod(i + 1)->position() - f.vertexMod(i)->position())
672 .squaredNorm();
673
674 f.vertex(i)->normal() += n / (e1 * e2);
675 }
676 }
677
678 if (normalize) {
679 log.startNewTask(95, 100, "Normalizing per-Vertex normals...");
680 normalizePerReferencedVertexNormals(mesh, log);
681 log.endTask("Normalizing per-Vertex normals...");
682 }
683
684 log.log(100, "Per-Vertex normals updated.");
685}
