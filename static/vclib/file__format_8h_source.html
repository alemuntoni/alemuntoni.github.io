94 auto operator<=>(const FileFormat& other) const
95 {
96 for (const auto& ext : mExtensions) {
97 if (other.matchExtension(ext)) {
98 return std::strong_ordering::equal;
99 }
100 }
101
102 return mExtensions[0] <=> other.mExtensions[0];
103 }

105 bool operator==(const FileFormat& other) const
106 {
107 return *this <=> other == std::strong_ordering::equal;

110private:

112 {
113 if (extension.front() == '.') {
