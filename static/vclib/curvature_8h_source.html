50template<FaceMeshConcept MeshType, LoggerConcept LogType = NullLogger>
51void updatePrincipalCurvatureTaubin95(MeshType& m, LogType& log)
52{
53 vcl::requirePerVertexPrincipalCurvature(m);
54 vcl::requirePerVertexAdjacentFaces(m);
55 vcl::requirePerFaceAdjacentFaces(m);
56
57 using VertexType = MeshType::VertexType;
58 using CoordType = VertexType::CoordType;
59 using NormalType = VertexType::NormalType;
60 using ScalarType = CoordType::ScalarType;
61 using FaceType = MeshType::FaceType;
62
63 // Auxiliary data structure
64 struct AdjVertex {
65 const VertexType* vert;
66 double doubleArea;
67 bool isBorder;
68 };
69
70 if constexpr (vcl::isLoggerValid<LogType>()) {
71 log.log(0, "Updating per vertex normals...");
72 }
73
74 vcl::updatePerVertexNormalsAngleWeighted(m);
75 vcl::normalizePerVertexNormals(m);
76
77 if constexpr (vcl::isLoggerValid<LogType>()) {
78 log.log(5, "Computing per vertex curvature...");
79 // log every 5%, starting from 5% to 100%
80 log.startProgress("", m.vertexNumber(), 5, 5, 100);
81 }
82
83 for (VertexType& v : m.vertices()) {
84 std::vector<ScalarType> weights;
85 std::vector<AdjVertex> vertices;
86
87 vcl::MeshPos<FaceType> pos(v.adjFace(0), &v);
88 const VertexType* firstVertex = pos.adjVertex();
89 const VertexType* tmpVertex;
90 float totalDoubleAreaSize = 0;
91
92 // compute the area of each triangle around the central vertex as well
93 // as their total area
94 do {
95 pos.nextEdgeAdjacentToV();
96 tmpVertex = pos.adjVertex();
97 AdjVertex adjV;
98 adjV.isBorder = pos.edgeOnBorder();
99 adjV.vert = tmpVertex;
100 adjV.doubleArea = vcl::faceArea(*pos.face()) * 2;
101 totalDoubleAreaSize += adjV.doubleArea;
102 vertices.push_back(adjV);
103 } while (tmpVertex != firstVertex);
104
105 weights.reserve(vertices.size());
106
107 for (int i = 0; i < vertices.size(); ++i) {
108 if (vertices[i].isBorder) {
109 weights.push_back(vertices[i].doubleArea / totalDoubleAreaSize);
110 }
111 else {
112 weights.push_back(
113 0.5f *
114 (vertices[i].doubleArea +
115 vertices[(i - 1) % vertices.size()].doubleArea) /
116 totalDoubleAreaSize);
117 }
118 assert(weights.back() < 1.0f);
119 }
120
121 // compute I-NN^t to be used for computing the T_i's
122 Matrix33<ScalarType> Tp;
123
124 NormalType n = v.normal();
125 for (int i = 0; i < 3; ++i)
126 Tp(i, i) = 1.0f - std::pow(n[i], 2);
127 Tp(0, 1) = Tp(1, 0) = -1.0f * (n[0] * n[1]);
128 Tp(1, 2) = Tp(2, 1) = -1.0f * (n[1] * n[2]);
129 Tp(0, 2) = Tp(2, 0) = -1.0f * (n[0] * n[2]);
130
131 // for all neighbors vi compute the directional curvatures k_i and the
132 // T_i compute M by summing all w_i k_i T_i T_i^t
133 Matrix33<ScalarType> tempMatrix;
134 Matrix33<ScalarType> M = Matrix33<ScalarType>::Zero();
135 for (size_t i = 0; i < vertices.size(); ++i) {
136 CoordType edge = (v.coord() - vertices[i].vert->coord());
137 float curvature =
138 (2.0f * (v.normal().dot(edge))) / edge.squaredNorm();
139 CoordType t(Tp * edge.eigenVector().transpose());
140 t.normalize();
141 tempMatrix = t.outerProduct(t);
142 M += tempMatrix * weights[i] * curvature ;
143 }
144
145 // compute vector W for the Householder matrix
146 CoordType w;
147 CoordType e1(1.0f, 0.0f, 0.0f);
148 if ((e1 - v.normal()).squaredNorm() > (e1 + v.normal()).squaredNorm())
149 w = e1 - v.normal();
150 else
151 w = e1 + v.normal();
152 w.normalize();
153
154 // compute the Householder matrix I - 2WW^t
155 Matrix33<ScalarType> Q = Matrix33<ScalarType>::Identity();
156 tempMatrix = w.outerProduct(w);
157 Q -= tempMatrix * 2.0f;
158
159 // compute matrix Q^t M Q
160 Matrix33<ScalarType> QtMQ = (Q.transpose() * M * Q);
161
162 Eigen::Matrix<ScalarType, 1, 3> T1 = Q.col(1);
163 Eigen::Matrix<ScalarType, 1, 3> T2 = Q.col(2);
164
165 // find sin and cos for the Givens rotation
166 ScalarType s, c;
167 // Gabriel Taubin hint and Valentino Fiorin impementation
168 ScalarType alpha = QtMQ(1, 1) - QtMQ(2, 2);
169 ScalarType beta = QtMQ(2, 1);
170
171 ScalarType h[2];
172 ScalarType delta =
173 std::sqrt(4.0f * std::pow(alpha, 2) + 16.0f * std::pow(beta, 2));
174 h[0] = (2.0f * alpha + delta) / (2.0f * beta);
175 h[1] = (2.0f * alpha - delta) / (2.0f * beta);
176
177 ScalarType t[2];
178 ScalarType bestC, bestS;
179 ScalarType minError = std::numeric_limits<ScalarType>::infinity();
180 for (uint i=0; i < 2; i++) {
181 delta = std::sqrt(std::pow(h[i], 2) + 4.0f);
182 t[0] = (h[i] + delta) / 2.0f;
183 t[1] = (h[i] - delta) / 2.0f;
184
185 for (uint j=0; j<2; j++) {
186 ScalarType squaredT = std::pow(t[j], 2);
187 ScalarType denominator = 1.0f + squaredT;
188
189 s = (2.0f * t[j]) / denominator;
190 c = (1 - squaredT) / denominator;
191
192 ScalarType approximation =
193 c * s * alpha + (std::pow(c, 2) - std::pow(s, 2)) * beta;
194 ScalarType angleSimilarity =
195 std::abs(std::acos(c) / std::asin(s));
196 ScalarType error =
197 std::abs(1.0f - angleSimilarity) + std::abs(approximation);
198 if (error < minError) {
199 minError = error;
200 bestC = c;
201 bestS = s;
202 }
203 }
204 }
205 c = bestC;
206 s = bestS;
207
208 Eigen::Matrix2f minor2x2;
209 Eigen::Matrix2f S;
210
211 // diagonalize M
212 minor2x2(0, 0) = QtMQ(1, 1);
213 minor2x2(0, 1) = QtMQ(1, 2);
214 minor2x2(1, 0) = QtMQ(2, 1);
215 minor2x2(1, 1) = QtMQ(2, 2);
216
217 S(0, 0) = S(1, 1) = c;
218 S(0, 1) = s;
219 S(1, 0) = -1.0f * s;
220
221 Eigen::Matrix2f StMS = S.transpose() * minor2x2 * S;
222
223 // compute curvatures and curvature directions
224 ScalarType principalCurv1 = (3.0f * StMS(0,0)) - StMS(1,1);
225 ScalarType principalCurv2 = (3.0f * StMS(1,1)) - StMS(0,0);
226
227 Eigen::Matrix<ScalarType, 1, 3> principalDir1 = T1 * c - T2 * s;
228 Eigen::Matrix<ScalarType, 1, 3> principalDir2 = T1 * s + T2 * c;
229
230 v.principalCurvature().maxDir() = principalDir1;
231 v.principalCurvature().minDir() = principalDir2;
232 v.principalCurvature().maxValue() = principalCurv1;
233 v.principalCurvature().minValue() = principalCurv2;
234
235 if constexpr (vcl::isLoggerValid<LogType>()) {
236 log.progress(m.index(v));
237 }
238 }
239
240 if constexpr (vcl::isLoggerValid<LogType>()) {
241 log.endProgress();
242 log.log(100, "Per vertex curvature computed.");
243 }
244}

258template<FaceMeshConcept MeshType, LoggerConcept LogType = NullLogger>
259void updatePrincipalCurvaturePCA(
260 MeshType& m,
261 typename MeshType::VertexType::CoordType::ScalarType radius,
262 bool montecarloSampling = true,
263 LogType& log = nullLogger)
264{
265 using VertexType = MeshType::VertexType;
266 using CoordType = VertexType::CoordType;
267 using ScalarType = CoordType::ScalarType;
268 using NormalType = VertexType::NormalType;
269 using FaceType = MeshType::FaceType;
270
271 using VGrid = vcl::StaticGrid3<VertexType*, ScalarType>;
272 using VGridIterator = VGrid::ConstIterator;
273
274 VGrid pGrid;
275 ScalarType area;
276
277 if constexpr (vcl::isLoggerValid<LogType>()) {
278 log.log(0, "Updating per vertex normals...");
279 }
280
281 vcl::updatePerVertexNormalsAngleWeighted(m);
282 vcl::normalizePerVertexNormals(m);
283
284 if constexpr (vcl::isLoggerValid<LogType>()) {
285 log.log(0, "Computing per vertex curvature...");
286 log.startProgress("", m.vertexNumber());
287 }
288
289 if (montecarloSampling) {
290 area = vcl::surfaceArea(m);
291 pGrid = VGrid(m.vertices() | views::addrOf);
292 pGrid.build();
293 }
294
295 vcl::parallelFor(m.vertices(), [&](VertexType& v){
296 //for (VertexType& v : m.vertices()) {
297 vcl::Matrix33<ScalarType> A, eigenvectors;
298 CoordType bp, eigenvalues;
299 if (montecarloSampling) {
300 vcl::Sphere s(v.coord(), radius);
301 std::vector<VGridIterator> vec = pGrid.valuesInSphere(s);
302 std::vector<CoordType> points;
303 points.reserve(vec.size());
304 for (const auto& it : vec){
305 points.push_back(it->second->coord());
306 }
307 A = vcl::covarianceMatrixOfPointCloud(points);
308 A *= area * area / 1000;
309 }
310 else {
311 Sphere<ScalarType> sph(v.coord(), radius);
312 MeshType tmpMesh = meshSphereIntersection(m, sph);
313 A = covarianceMatrixOfMesh(tmpMesh);
314 }
315 Eigen::SelfAdjointEigenSolver<vcl::Matrix33<ScalarType>> eig(A);
316 eigenvalues = CoordType(eig.eigenvalues());
317 eigenvectors = eig.eigenvectors(); // eigenvector are stored as columns.
318 // get the estimate of curvatures from eigenvalues and eigenvectors
319 // find the 2 most tangent eigenvectors (by finding the one closest to
320 // the normal)
321 uint best = 0;
322 ScalarType bestv = std::abs(
323 v.normal().dot(CoordType(eigenvectors.col(0).normalized())));
324 for(uint i = 1 ; i < 3; ++i){
325 ScalarType prod = std::abs(
326 v.normal().dot(CoordType(eigenvectors.col(i).normalized())));
327 if(prod > bestv){
328 bestv = prod;
329 best = i;
330 }
331 }
332 v.principalCurvature().maxDir() =
333 (eigenvectors.col((best + 1) % 3).normalized());
334 v.principalCurvature().minDir() =
335 (eigenvectors.col((best + 2) % 3).normalized());
336
337 vcl::Matrix33<ScalarType> rot;
338 ScalarType angle;
339 angle = acos(v.principalCurvature().maxDir().dot(v.normal()));
340
341 rot = vcl::rotationMatrix<vcl::Matrix33<ScalarType>>(
342 v.principalCurvature().maxDir().cross(v.normal()),
343 -(M_PI * 0.5 - angle));
344
345 v.principalCurvature().maxDir() = rot * v.principalCurvature().maxDir();
346
347 angle = acos(v.principalCurvature().minDir().dot(v.normal()));
348
349 rot = vcl::rotationMatrix<vcl::Matrix33<ScalarType>>(
350 v.principalCurvature().minDir().cross(v.normal()),
351 -(M_PI * 0.5 - angle));
352
353 v.principalCurvature().minDir() = rot * v.principalCurvature().minDir();
354
355 // computes the curvature values
356 const ScalarType r5 = std::pow(radius, 5);
357 const ScalarType r6 = r5 * radius;
358 v.principalCurvature().maxValue() =
359 (2.0 / 5.0) *
360 (4.0 * M_PI * r5 + 15 * eigenvalues[(best + 2) % 3] -
361 45.0 * eigenvalues[(best + 1) % 3]) /
362 (M_PI * r6);
363 v.principalCurvature().minValue() =
364 (2.0 / 5.0) *
365 (4.0 * M_PI * r5 + 15 * eigenvalues[(best + 1) % 3] -
366 45.0 * eigenvalues[(best + 2) % 3]) /
367 (M_PI * r6);
368 if (v.principalCurvature().maxValue() <
369 v.principalCurvature().minValue())
370 {
371 std::swap(
372 v.principalCurvature().minValue(),
373 v.principalCurvature().maxValue());
374 std::swap(
375 v.principalCurvature().minDir(),
376 v.principalCurvature().maxDir());
377 }
378
379 if constexpr (vcl::isLoggerValid<LogType>()) {
380 log.progress(m.index(v));
381 }
382 //}
383 });
384
385 if constexpr (vcl::isLoggerValid<LogType>()) {
386 log.endProgress();
387 log.log(100, "Per vertex curvature computed.");
388 }
389}

392void updatePrincipalCurvature(MeshType& m, LogType& log = nullLogger)

395
396 updatePrincipalCurvatureTaubin95(m, log);

399template<FaceMeshConcept MeshType, LoggerConcept LogType = NullLogger>
400void updatePrincipalCurvature(
401 MeshType& m,
402 VCLibPrincipalCurvatureAlgorithm alg = VCL_PRINCIPAL_CURVATURE_TAUBIN95,
403 LogType& log = nullLogger)
404{
405 vcl::requirePerVertexPrincipalCurvature(m);
406
407 double radius;
408 switch (alg) {
409 case VCL_PRINCIPAL_CURVATURE_TAUBIN95:
410 updatePrincipalCurvatureTaubin95(m, log);
411 break;
412 case VCL_PRINCIPAL_CURVATURE_PCA:
413 radius = vcl::boundingBox(m).diagonal() * 0.1;
414 updatePrincipalCurvaturePCA(m, radius, true, log);
415 }
416}
