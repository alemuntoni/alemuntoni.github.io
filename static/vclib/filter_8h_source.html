101template<MeshConcept InMeshType, MeshConcept OutMeshType = InMeshType>
102OutMeshType perVertexMeshFilter(
103 const InMeshType& m,
104 Range auto&& vertexFilterRng,
105 bool saveBirthIndicesInCustomComponent = true)
106{
107 OutMeshType res;
108 res.enableSameOptionalComponentsOf(m);
109
110 // enable the custom component birthVertex
111 if constexpr (vcl::HasPerVertexCustomComponents<OutMeshType>) {
112 if (saveBirthIndicesInCustomComponent) {
113 res.template addPerVertexCustomComponent<uint>("birthVertex");
114 }
115 }
116
117 for (const auto& [birthV, filter] :
118 std::views::zip(m.vertices(), vertexFilterRng))
119 {
120 if (filter) {
121 uint v = res.addVertex();
122 // import all the components from the input mesh
123 res.vertex(v).importFrom(birthV);
124 // set the birth vertex
125 if constexpr (vcl::HasPerVertexCustomComponents<OutMeshType>) {
126 if (saveBirthIndicesInCustomComponent) {
127 res.vertex(v).template customComponent<uint>(
128 "birthVertex") = m.index(birthV);
129 }
130 }
131 }
132 }
133
134 return res;
135}

211template<FaceMeshConcept InMeshType, FaceMeshConcept OutMeshType = InMeshType>
212OutMeshType perFaceMeshFilter(
213 const InMeshType& m,
214 Range auto&& faceFilterRng,
215 bool saveBirthIndicesInCustomComponent = true)
216{
217 using InVertexType = InMeshType::VertexType;
218
219 OutMeshType res;
220 res.enableSameOptionalComponentsOf(m);
221
222 // enable the custom component birthVertex
223 if constexpr (vcl::HasPerVertexCustomComponents<OutMeshType>) {
224 if (saveBirthIndicesInCustomComponent) {
225 res.template addPerVertexCustomComponent<uint>("birthVertex");
226 }
227 }
228
229 // enable the custom component birthFace
230 if constexpr (vcl::HasPerFaceCustomComponents<OutMeshType>) {
231 if (saveBirthIndicesInCustomComponent) {
232 res.template addPerFaceCustomComponent<uint>("birthFace");
233 }
234 }
235
236 std::vector<uint> vertexMapping(m.vertexContainerSize(), UINT_NULL);
237
238 for (const auto& [birthF, filter] :
239 std::views::zip(m.faces(), faceFilterRng))
240 {
241 if (filter) {
242 std::vector<uint> verts(birthF.vertexNumber(), UINT_NULL);
243 uint vi = 0; // incremented with vertices of the face
244 // set all the vertex indices in the verts vector
245 // two cases here:
246 // - the ith vertex of the face has been already added, we need just
247 // to take its id in the out mesh from the vertexMapping vector
248 // - the ith vertex of the face has not been added: we need to add
249 // it and import all its components, and update the
250 // vertexMappingVector
251 for (const InVertexType* v : birthF.vertices()) {
252 // the vertex has not already added
253 if (vertexMapping[m.index(v)] == UINT_NULL) {
254 // add the vertex to the out mesh
255 uint ov = res.addVertex();
256 // import all the components from the input mesh
257 res.vertex(ov).importFrom(*v);
258 if constexpr (vcl::HasPerVertexCustomComponents<
259 OutMeshType>)
260 {
261 // set the birth vertex
262 if (saveBirthIndicesInCustomComponent) {
263 res.vertex(ov).template customComponent<uint>(
264 "birthVertex") = m.index(v);
265 }
266 }
267 vertexMapping[m.index(v)] = ov;
268 verts[vi] = ov;
269 }
270 else {
271 verts[vi] = vertexMapping[m.index(v)];
272 }
273 ++vi;
274 }
275
276 // now all the vertices of the face are in the out mesh, we can add
277 // the actual face
278 uint f = res.addFace(verts.begin(), verts.end());
279 // import all the components from the input mesh
280 res.face(f).importFrom(birthF);
281 if constexpr (vcl::HasPerFaceCustomComponents<OutMeshType>) {
282 // set the birth face
283 if (saveBirthIndicesInCustomComponent) {
284 res.face(f).template customComponent<uint>("birthFace") =
285 m.index(birthF);
286 }
287 }
288 }
289 }
290 return res;
291}
