35template<MeshConcept OutMeshType, uint ELEM_ID, MeshConcept InMeshType>
36OutMeshType perElementMeshFilter(
37 const InMeshType& m,
38 Range auto&& elemFilterRng,
39 bool saveBirthIndicesInCustomComponent = true)
40{
41 using OutElemType = OutMeshType::template ElementType<ELEM_ID>;
42
43 std::string ccname = "birth" + vcl::elementEnumString<ELEM_ID>();
44
45 OutMeshType res;
46 res.enableSameOptionalComponentsOf(m);
47
48 // enable the custom component birth element
49 if constexpr (vcl::comp::HasCustomComponents<OutElemType>) {
50 if (saveBirthIndicesInCustomComponent) {
51 res.template addPerElementCustomComponent<ELEM_ID, uint>(ccname);
52 }
53 }
54
55 for (const auto& [birthV, filter] :
56 std::views::zip(m.template elements<ELEM_ID>(), elemFilterRng))
57 {
58 if (filter) {
59 uint v = res.template add<ELEM_ID>();
60 // import all the components from the input mesh
61 res.template element<ELEM_ID>(v).importFrom(birthV, false);
62 // set the birth element
63 if constexpr (vcl::comp::HasCustomComponents<OutElemType>) {
64 if (saveBirthIndicesInCustomComponent) {
65 res.template element<ELEM_ID>(v)
66 .template customComponent<uint>(ccname) =
67 m.index(birthV);
68 }
69 }
70 }
71 }
72
73 return res;
74}

90template<MeshConcept OutMeshType, uint ELEM_ID, MeshConcept InMeshType>
91OutMeshType perElementMeshFilterWithVRefs(
92 const InMeshType& m,
93 Range auto&& elemFilterRng,
94 bool saveBirthIndicesInCustomComponent = true)
95{
96 using InVertexType = InMeshType::VertexType;
97 using OutElemType = OutMeshType::template ElementType<ELEM_ID>;
98
99 std::string ccname = "birth" + vcl::elementEnumString<ELEM_ID>();
100
101 OutMeshType res;
102 res.enableSameOptionalComponentsOf(m);
103
104 // enable the custom component birthVertex
105 if constexpr (vcl::HasPerVertexCustomComponents<OutMeshType>) {
106 if (saveBirthIndicesInCustomComponent) {
107 res.template addPerVertexCustomComponent<uint>("birthVertex");
108 }
109 }
110
111 // enable the custom component birth element
112 if constexpr (vcl::comp::HasCustomComponents<OutElemType>) {
113 if (saveBirthIndicesInCustomComponent) {
114 res.template addPerElementCustomComponent<ELEM_ID, uint>(ccname);
115 }
116 }
117
118 std::vector<uint> vertexMapping(m.vertexContainerSize(), UINT_NULL);
119
120 for (const auto& [birthF, filter] :
121 std::views::zip(m.template elements<ELEM_ID>(), elemFilterRng))
122 {
123 if (filter) {
124 std::vector<uint> verts(birthF.vertexNumber(), UINT_NULL);
125 uint vi = 0; // incremented with vertices of the element
126 // set all the vertex indices in the verts vector
127 // two cases here:
128 // - the ith vertex of the face has been already added, we need just
129 // to take its id in the out mesh from the vertexMapping vector
130 // - the ith vertex of the face has not been added: we need to add
131 // it and import all its components, and update the
132 // vertexMappingVector
133 for (const InVertexType* v : birthF.vertices()) {
134 // the vertex has not already added
135 if (vertexMapping[m.index(v)] == UINT_NULL) {
136 // add the vertex to the out mesh
137 uint ov = res.addVertex();
138 // import all the components from the input mesh
139 res.vertex(ov).importFrom(*v, false);
140 if constexpr (vcl::HasPerVertexCustomComponents<
141 OutMeshType>)
142 {
143 // set the birth vertex
144 if (saveBirthIndicesInCustomComponent) {
145 res.vertex(ov).template customComponent<uint>(
146 "birthVertex") = m.index(v);
147 }
148 }
149 vertexMapping[m.index(v)] = ov;
150 verts[vi] = ov;
151 }
152 else {
153 verts[vi] = vertexMapping[m.index(v)];
154 }
155 ++vi;
156 }
157
158 // now all the vertices of the elements are in the out mesh, we can
159 // add the actual element
160 uint f = res.template add<ELEM_ID>();
161 // import all the components from the input mesh
162 res.template element<ELEM_ID>(f).importFrom(birthF, false);
163
164 if constexpr (OutElemType::VERTEX_NUMBER < 0) {
165 res.template element<ELEM_ID>(f).resizeVertices(verts.size());
166 }
167
168 res.template element<ELEM_ID>(f).setVertices(verts);
169
170 if constexpr (vcl::comp::HasCustomComponents<OutElemType>) {
171 // set the birth elements
172 if (saveBirthIndicesInCustomComponent) {
173 res.template element<ELEM_ID>(f)
174 .template customComponent<uint>(ccname) =
175 m.index(birthF);
176 }
177 }
178 }
179 }
180 return res;
181}

584template<EdgeMeshConcept OutMeshType, FaceMeshConcept InMeshType>
585OutMeshType perFaceEdgeMeshFilter(
586 const InMeshType& m,
587 const std::function<bool(const typename InMeshType::FaceType&, uint)>&
588 faceEdgeFilter,
589 bool dontDuplicateEdges = true,
590 bool saveBirthIndicesInCustomComponent = true)
591{
592 using InVertexType = InMeshType::VertexType;
593 using OutEdgeType = OutMeshType::EdgeType;
594
595 OutMeshType res;
596 // todo: enable only per vertex same optional components of m
597 res.enableSameOptionalComponentsOf(m);
598
599 // enable the custom component birthVertex
600 if constexpr (vcl::HasPerVertexCustomComponents<OutMeshType>) {
601 if (saveBirthIndicesInCustomComponent) {
602 res.template addPerVertexCustomComponent<uint>("birthVertex");
603 }
604 }
605
606 std::vector<uint> vertexMapping(m.vertexContainerSize(), UINT_NULL);
607
608 std::set<std::pair<uint, uint>, UnorderedPairComparator<uint>>
609 unorderedEdges;
610
611 for (const auto& f : m.faces()) {
612 for (uint ei = 0; ei < f.vertexNumber(); ++ei) {
613 if (faceEdgeFilter(f, ei)) {
614 std::array<uint, 2> verts = {UINT_NULL, UINT_NULL};
615 for (uint i = 0; i < 2; ++i) {
616 const InVertexType* v = f.vertexMod(ei + i);
617 if (vertexMapping[m.index(v)] == UINT_NULL) {
618 uint ov = res.addVertex();
619 res.vertex(ov).importFrom(*v, false);
620 if constexpr (vcl::HasPerVertexCustomComponents<
621 OutMeshType>)
622 {
623 if (saveBirthIndicesInCustomComponent) {
624 res.vertex(ov).template customComponent<uint>(
625 "birthVertex") = m.index(v);
626 }
627 }
628 vertexMapping[m.index(v)] = ov;
629 verts[i] = ov;
630 }
631 else {
632 verts[i] = vertexMapping[m.index(v)];
633 }
634 }
635
636 std::pair<uint, uint> ep {verts[0], verts[1]};
637 if (!dontDuplicateEdges || !unorderedEdges.contains(ep)) {
638 uint e = res.addEdge(verts[0], verts[1]);
639 unorderedEdges.insert(ep);
640 }
641 }
642 }
643 }
644
645 return res;
646}
