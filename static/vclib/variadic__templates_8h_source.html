98template<typename T, typename U, typename... Us>
99constexpr uint indexInTypePack()
100{
101 if constexpr (std::is_same_v<T, U>) {
102 return 0;
103 }
104 else {
105 if constexpr (sizeof...(Us)) {
106 // there is at least another type to check
107 constexpr uint res =
108 indexInTypePack<T, Us...>(); // look in the rest of the types
109 if constexpr (res == UINT_NULL) // not found in the rest
110 return UINT_NULL;
111 else
112 return 1 + res; // found
113 }
114 else { // not found
115 return UINT_NULL;
116 }
117 }
118}

129template<typename U, typename... Us>
130uint indexInTypePack(std::type_index ti)
131{
132 if (ti == typeid(U)) {
133 return 0;
134 }
135 else {
136 if constexpr (sizeof...(Us)) {
137 // there is at least another type to check
138 uint res = indexInTypePack<Us...>(ti);
139 if (res == UINT_NULL)
140 return UINT_NULL;
141 else
142 return 1 + res;
143 }
144 else { // not found
145 return UINT_NULL;
146 }
147 }
148}
