96template<typename T, typename U, typename... Us>
97constexpr uint indexInTypePack()
98{
99 if constexpr (std::is_same_v<T, U>) {
100 return 0;
101 }
102 else {
103 if constexpr (sizeof...(Us)) {
104 // there is at least another type to check
105 constexpr uint res =
106 indexInTypePack<T, Us...>(); // look in the rest of the types
107 if constexpr (res == UINT_NULL) // not found in the rest
108 return UINT_NULL;
109 else
110 return 1 + res; // found
111 }
112 else { // not found
113 return UINT_NULL;
114 }
115 }
116}

127template<typename U, typename... Us>
128uint indexInTypePack(std::type_index ti)
129{
130 if (ti == typeid(U)) {
131 return 0;
132 }
133 else {
134 if constexpr (sizeof...(Us)) {
135 // there is at least another type to check
136 uint res = indexInTypePack<Us...>(ti);
137 if (res == UINT_NULL)
138 return UINT_NULL;
139 else
140 return 1 + res;
141 }
142 else { // not found
143 return UINT_NULL;
144 }
145 }
146}
