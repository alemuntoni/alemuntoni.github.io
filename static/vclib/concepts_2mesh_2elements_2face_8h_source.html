143template<typename T>
144concept FaceConcept =
145 T::ELEMENT_TYPE == FACE &&
146 (face::IsDerivedFromFace<T>::value || face::IsAFace<T>::value) &&
147 face::HasBitFlags<T> &&
148 face::HasVertexPointers<T> &&
149 (T::VERTEX_NUMBER < 0 || T::VERTEX_NUMBER >= 3) &&
150 (!face::HasTriangleBitFlags<T> || T::VERTEX_NUMBER == 3) &&
151 comp::SanityCheckAdjacentEdges<T> &&
152 comp::SanityCheckAdjacentFaces<T> &&
153 comp::SanityCheckWedgeColors<T> &&
154 comp::SanityCheckWedgeTexCoords<T>;
155
156template<typename T>
157concept TriangleFaceConcept =
158 T::VERTEX_NUMBER == 3 && FaceConcept<T>;
159
170template<typename T>
171concept PolygonFaceConcept =
172 T::VERTEX_NUMBER < 0 && FaceConcept<T>;
