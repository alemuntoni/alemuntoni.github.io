195private:
196 bool insertInCell(const KeyType& k, const ValueType& v)
197 {
198 if constexpr (AllowDuplicates) {
199 map.emplace(k, v);
200 return true;
201 }
202 else {
203 auto range = map.equal_range(k);
204 bool found = false;
205 for (Iterator ci = range.first; ci != range.second && !found; ++ci)
206 {
207 if (ci->second == v) {
208 found = true;
209 }
210 }
211 if (!found)
212 map.emplace(k, v);
213 return !found;
214 }
215 }
216
217 bool eraseInCell(const KeyType& k, const ValueType& v)
218 {
219 bool found = false;
220
221 std::pair<Iterator, Iterator> range = map.equal_range(k);
222 for(Iterator ci = range.first; ci != range.second; ++ci) {
223 if (ci->second == v) {
224 found = true;
225 map.erase(ci);
226 if constexpr (!AllowDuplicates) {
227 return true;
228 }
229 }
230 }
231 return found;
232 }
