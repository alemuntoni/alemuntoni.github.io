75inline double perlinNoiseGrad(int hash, double x, double y, double z)
76{
77 int h = hash & 15; // CONVERT LO 4 BITS OF HASH CODE
78 double u = h < 8 ? x : y, // INTO 12 GRADIENT DIRECTIONS.
79 v = h < 4 ? y :
80 h == 12 || h == 14 ? x :
81 z;
82 return ((h & 1) == 0 ? u : -u) + ((h & 2) == 0 ? v : -v);
83}

90inline double perlinNoiseFade(double t)
91{
92 return t * t * t * (t * (t * 6 - 15) + 10);

108inline double perlinNoise(double x, double y, double z)
109{
110 int X = (int) floor(x) & 255, // FIND UNIT CUBE THAT
111 Y = (int) floor(y) & 255, // CONTAINS POINT.
112 Z = (int) floor(z) & 255;
113 x -= floor(x); // FIND RELATIVE X,Y,Z
114 y -= floor(y); // OF POINT IN CUBE.
115 z -= floor(z);
116 double u = detail::perlinNoiseFade(x), // COMPUTE FADE CURVES
117 v = detail::perlinNoiseFade(y), // FOR EACH OF X,Y,Z.
118 w = detail::perlinNoiseFade(z);
119 int A = detail::perlinNoiseP(X) + Y, AA = detail::perlinNoiseP(A) + Z,
120 AB = detail::perlinNoiseP(A + 1) + Z, // HASH COORDINATES OF
121 B = detail::perlinNoiseP(X + 1) + Y, BA = detail::perlinNoiseP(B) + Z,
122 BB = detail::perlinNoiseP(B + 1) + Z; // THE 8 CUBE CORNERS,
123
124 return detail::perlinNoiseLerp(
125 w,
126 detail::perlinNoiseLerp(
127 v,
128 detail::perlinNoiseLerp(
129 u,
130 detail::perlinNoiseGrad(
131 detail::perlinNoiseP(AA), x, y, z), // AND ADD
132 detail::perlinNoiseGrad(
133 detail::perlinNoiseP(BA), x - 1, y, z)), // BLENDED
134 detail::perlinNoiseLerp(
135 u,
136 detail::perlinNoiseGrad(
137 detail::perlinNoiseP(AB), x, y - 1, z), // RESULTS
138 detail::perlinNoiseGrad(
139 detail::perlinNoiseP(BB), x - 1, y - 1, z))), // FROM 8
140 detail::perlinNoiseLerp(
141 v,
142 detail::perlinNoiseLerp(
143 u,
144 detail::perlinNoiseGrad(
145 detail::perlinNoiseP(AA + 1), x, y, z - 1), // CORNERS
146 detail::perlinNoiseGrad(
147 detail::perlinNoiseP(BA + 1), x - 1, y, z - 1)), // OF
148 // CUBE
149 detail::perlinNoiseLerp(
150 u,
151 detail::perlinNoiseGrad(
152 detail::perlinNoiseP(AB + 1), x, y - 1, z - 1),
153 detail::perlinNoiseGrad(
154 detail::perlinNoiseP(BB + 1), x - 1, y - 1, z - 1))));
155}
