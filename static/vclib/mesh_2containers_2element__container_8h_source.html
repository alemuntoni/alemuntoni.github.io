159 uint addElement()
160 {
161 vcVecTuple.resize(vec.size() + 1);
162 if constexpr (comp::HasCustomComponents<T>)
163 ccVecMap.resize(vec.size() + 1);
164
165 T* oldB = vec.data();
166 vec.emplace_back();
167 T* newB = vec.data();
168 en++;
169
170 vec.back().setParentMesh(parentMesh);
171 vec.back().initVerticalComponents();
172
173 if (oldB != newB) {
174 setParentMeshPointers(parentMesh);
175 parentMesh->updateAllPointers(oldB, newB);
176 }
177
178 return vec.size() - 1;
179 }
180
189 uint addElements(uint size)
190 {
191 vcVecTuple.resize(vec.size() + size);
192 if constexpr (comp::HasCustomComponents<T>)
193 ccVecMap.resize(vec.size() + size);
194
195 uint baseId = vec.size();
196 T* oldB = vec.data();
197 vec.resize(vec.size() + size);
198 T* newB = vec.data();
199 en += size;
200
201 for (uint i = baseId; i < vec.size(); ++i) {
202 vec[i].setParentMesh(parentMesh);
203 vec[i].initVerticalComponents();
204 }
205
206 if (oldB != newB) {
207 setParentMeshPointers(parentMesh);
208 parentMesh->updateAllPointers(oldB, newB);
209 }
210
211 return baseId;
212 }

543 {

729 template<typename OtherMesh>
730 void importFrom(const OtherMesh& m)
731 {
732 if constexpr (OtherMesh::template hasContainerOf<T>()) {
733 // get the container type of the other mesh for T - used to upcast
734 // othMesh
735 using Container = OtherMesh::template ContainerOf<T>::type;
736
737 const Container& c = (const Container&)m;
738
739 clearElements();
740
741 // recreate a container having the same number of elements as the
742 // other mesh. this call will also update the parentMesh pointers
743 // and sets the vertical components vectors in vcVecTuple
744 addElements(c.elementContainerSize());
745 unsigned int eid = 0;
746 for (const typename Container::ElementType& e : c.elements(false)) {
747 // import the components of the e, when they are available.
748 // note that will set also the deleted flag of e, therefore if e
749 // is deleted, then also element(eid) will be deleted.
750 element(eid).importFrom(e);
751 ++eid;
752 }
753 // set the number of elements (different from the container size)
754 en = c.en;
755 if constexpr (
756 comp::HasCustomComponents<T> &&
757 comp::HasCustomComponents<typename Container::ElementType>)
758 {
759 ccVecMap = c.ccVecMap;
760 }
761 }

774 template<typename OtherMesh, typename OtherElement>
775 void importPointersFrom(const OtherMesh& othMesh, OtherElement* base)
776 {
777 // We need to be sure that the other mesh has two containers (that can
778 // be the same, see later):
779 // - the one of Elements of same type as T
780 // - the one of Elements of same type as OtherElement (the pointers
781 // that we are actually importing on this Container of T elements)
782 // Note that OtherElement may be the same of T (e.g. Vertex[T] has
783 // pointers of other Vertices[OtherElement]) or different (e.g.
784 // Vertex[T] has pointers of Faces[OtherElement])
785 if constexpr (
786 OtherMesh::template hasContainerOf<T>() &&
787 OtherMesh::template hasContainerOf<OtherElement>())
788 {
789 // get the containe type of the other mesh for MyBase - used for get
790 // the base pointer
791 using OthBaseContainer =
792 OtherMesh::template ContainerOf<OtherElement>::type;
793 // get the container type of the other mesh for T - used to upcast
794 // othMesh
795 using OthTContainer = OtherMesh::template ContainerOf<T>::type;
796
797 // get the container base of the other mesh, that we use to import
798 // pointers
799 const auto* cbase = othMesh.OthBaseContainer::vec.data();
800
801 // upcast the other mesh to the container and import the pointers
802 // from the OthTContainer
803 importPointersFromContainer(
804 (const OthTContainer&) othMesh, base, cbase);
805 }

876 /*
877 * This function is called for each component of the element.
878 *
879 * Only if a component has references of the type ElPtr, then the
880 * updatePointers on each element will be executed
881 */
882 template<typename Comp, typename ElPtr>
883 void updatePointersOnComponent(const ElPtr* oldBase, const ElPtr* newBase)
884 {
885 if constexpr (comp::HasPointersOfType<Comp, ElPtr>) {
886 if constexpr (comp::HasOptionalPointersOfType<Comp, ElPtr>) {
887 if(isOptionalComponentEnabled<Comp>()) {
888 for (T& e : elements()) {
889 e.Comp::updatePointers(oldBase, newBase);
890 }
891 }
892 }
893 else {
894 for (T& e : elements()) {
895 e.Comp::updatePointers(oldBase, newBase);
896 }
897 }
898 }
899 }

957 template<typename Comp, typename Cont2>
958 void enableSameOptionalComponent(const Cont2& c2)
959 {
960 // if Comp is an optional component in This container
961 if constexpr (comp::IsOptionalComponent<Comp>) {
962 // if Comp is available in Cont2
963 if constexpr (comp::HasComponentOfType<
964 typename Cont2::ElementType,
965 Comp::COMPONENT_ID>)
966 {
967 // if Comp is optional in Cont2
968 if constexpr (comp::HasOptionalComponentOfType<
969 typename Cont2::ElementType,
970 Comp::COMPONENT_ID>)
971 {
972 // if Comp is enabled in Cont2, we enable it in this
973 // container
974 if (c2.template isOptionalComponentEnabled<
975 Comp::COMPONENT_ID>())
976 {
977 enableOptionalComponent<Comp::COMPONENT_ID>();
978 }
979 }
980 // if Comp is not optional (but is available), we enable it in
981 // this container
982 else {
983 enableOptionalComponent<Comp::COMPONENT_ID>();
984 }
985 }
