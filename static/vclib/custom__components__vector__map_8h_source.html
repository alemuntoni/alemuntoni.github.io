70template<>
71class CustomComponentsVectorMap<true>
72{
73 // the actual map containing, for each name of a custom component, the
74 // vector of values (a value for each element(vertex/face...) of the mesh)
75 std::unordered_map<std::string, std::vector<std::any>> map;
76 // map that stores if some custom components need to be initialized
77 // this happens when we resize the container of an element: we need to
78 // resize also the vectors of the custom components, but we cannot
79 // initialize them with the right type (because we don't know it in resize
80 // phase). The initialization will be made when the type is known (get of
81 // the custom component vector).
82 mutable std::unordered_map<std::string, bool> needToInitialize;
83 // types used for each custom component
84 std::unordered_map<std::string, std::type_index> compType;
85
86public:
90 void clear(){
91 map.clear();
92 needToInitialize.clear();
93 compType.clear();
94 }
95
100 void reserve(uint size)
101 {
102 for (auto& p : map) {
103 p.second.reserve(size);
104 }
105 }
106
112 void resize(uint size)
113 {
114 for (auto& p : map) {
115 // At this call, we don't know statically the types of each custom
116 // component, therefore we cannot initialize them (std::any of each
117 // vector will be in an invalid state). Therefore, we mark all the
118 // resized custom components as 'needToInitialize'. Initiaization
119 // will be made just on the uninitialized values of the vectors at
120 // the first access of each custom component.
121 if (p.second.size() < size)
122 needToInitialize.at(p.first) = true;
123 p.second.resize(size);
124 }
125 }
126
139 void compact(const std::vector<uint>& newIndices)
140 {
141 for (auto& p : map) {
142 vcl::compactVector(p.second, newIndices);
143 }
144 }

278 template<typename CompType>
279 const std::vector<std::any>&
280 componentVector(const std::string& compName) const
281 {
282 checkComponentType<CompType>(compName);
283 std::vector<std::any>& v =
284 const_cast<std::vector<std::any>&>(map.at(compName));
285 // if the component needs to be initialized (e.g. we made a resize)
286 if (needToInitialize.at(compName)) {
287 for (std::any& a : v) {
288 if (!a.has_value())
289 a = CompType();
290 }
291 needToInitialize.at(compName) = false;
292 }
293 return v;
294 }

333 throw BadCustomComponentTypeException(
