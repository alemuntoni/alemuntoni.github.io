81 PlyHeader(const std::string& filename, std::ifstream& file)
82 {
83 clear();
84 if (file.is_open()) {
85 std::string line;
86 std::getline(file, line);
87 str::removeWindowsNewLine(line);
88 if (line.compare(0, 3, "ply") == 0) {
89 bool error = false;
90 bool firstElement = true;
91 std::string headerLine;
92 ply::Element element;
93 do {
94 vcl::Tokenizer spaceTokenizer =
95 vcl::io::internal::nextNonEmptyTokenizedLine(file);
96 if (!error) {
97 str::removeWindowsNewLine(line);
98 vcl::Tokenizer::iterator token = spaceTokenizer.begin();
99 headerLine = *token;
100 if (headerLine == "format") {
101 token++;
102 if (*token == "ascii")
103 frmt = ply::ASCII;
104 else if (
105 *token == "binary_big_endian" ||
106 *token == "binary_little_endian" ||
107 *token == "binary")
108 frmt = ply::BINARY;
109 }
110 // reading a comment, may be a texture file...
111 else if (headerLine == "comment") {
112 token++;
113 if (token != spaceTokenizer.end()){
114 if (vcl::str::containsCaseInsensitive(
115 *token, "texture"))
116 {
117 ++token;
118 if (token != spaceTokenizer.end()){
119 std::string textName = *token;
120 auto it = vcl::str::findCaseInsensitive(
121 textName, "<this>");
122 if (it != textName.end()) {
123 uint pos = it-textName.begin();
124 std::string fn = vcl::FileInfo::
125 fileNameWithoutExtension(
126 filename);
127 textName =
128 textName.substr(0, pos) + fn +
129 textName.substr(
130 pos + 6, textName.size());
131 }
132 textureFiles.push_back(textName);
133 }
134 }
135 }
136 }
137 // I am reading a new element
138 else if (headerLine == "element") {
139 // if it is not the first element to read, it means
140 // that the previous one needs to be saved
141 if (!firstElement) {
142 // index of each element type in elements vector
143 if (element.type == ply::VERTEX)
144 vertElemPos = (long int) elements.size();
145 if (element.type == ply::FACE)
146 faceElemPos = (long int) elements.size();
147 if (element.type == ply::EDGE)
148 edgeElemPos = (long int) elements.size();
149 if (element.type == ply::TRISTRIP)
150 trisElemPos = (long int) elements.size();
151 elements.push_back(element);
152 element = ply::Element();
153 }
154 element = readElement(spaceTokenizer);
155 firstElement = false;
156 }
157 else if (headerLine == "property") {
158 ply::Property p = readProperty(spaceTokenizer);
159 element.properties.push_back(p);
160 }
161 // save the last element
162 else if (headerLine == "end_header") {
163 if (element.type == ply::VERTEX)
164 vertElemPos = (long int) elements.size();
165 if (element.type == ply::FACE)
166 faceElemPos = (long int) elements.size();
167 if (element.type == ply::EDGE)
168 edgeElemPos = (long int) elements.size();
169 if (element.type == ply::TRISTRIP)
170 trisElemPos = (long int) elements.size();
171 elements.push_back(element);
172 }
173 }
174 } while (!error && headerLine != "end_header");
175 valid = !error && hasVertices();
176 }
177 }

196 MeshInfo getInfo() const
197 {
198 MeshInfo mod;
199 // x, y, z, nx, ny, nz, red, green, blue, alpha, vertex_indices
200
201 if (vertElemPos >= 0) {
202 mod.setVertices();
203 for (const Property& p : elements[vertElemPos].properties) {
204 switch (p.name) {
205 case ply::x:
206 case ply::y:
207 case ply::z: mod.setVertexCoords(); break;
208 case ply::nx:
209 case ply::ny:
210 case ply::nz: mod.setVertexNormals(); break;
211 case ply::red:
212 case ply::green:
213 case ply::blue:
214 case ply::alpha: mod.setVertexColors(); break;
215 case ply::quality: mod.setVertexQuality(); break;
216 case ply::texture_u: mod.setVertexTexCoords(); break;
217 case ply::unknown:
218 if (p.type <= PropertyType::DOUBLE) {
219 mod.addVertexCustomComponent(
220 p.unknownPropertyName, (MeshInfo::DataType) p.type);
221 }
222 default: break;
223 }
224 }
225 }
226 if (faceElemPos >= 0) {
227 mod.setFaces();
228 for (const Property& p : elements[faceElemPos].properties) {
229 switch (p.name) {
230 case ply::vertex_indices: mod.setFaceVRefs(); break;
231 case ply::nx:
232 case ply::ny:
233 case ply::nz: mod.setFaceNormals(); break;
234 case ply::red:
235 case ply::green:
236 case ply::blue:
237 case ply::alpha: mod.setFaceColors(); break;
238 case ply::quality: mod.setFaceQuality(); break;
239 case ply::texcoord: mod.setFaceWedgeTexCoords(); break;
240 case ply::unknown:
241 if (p.type <= PropertyType::DOUBLE) {
242 mod.addFaceCustomComponent(
243 p.unknownPropertyName, (MeshInfo::DataType) p.type);
244 }
245 default: break;
246 }
247 }
248 }
249 if (trisElemPos >= 0) {
250 mod.setFaces();
251 for (const Property& p : elements[trisElemPos].properties) {
252 switch (p.name) {
253 case ply::vertex_indices: mod.setFaceVRefs(); break;
254 case ply::nx:
255 case ply::ny:
256 case ply::nz: mod.setFaceNormals(); break;
257 case ply::red:
258 case ply::green:
259 case ply::blue:
260 case ply::alpha: mod.setFaceColors(); break;
261 case ply::quality: mod.setFaceQuality(); break;
262 case ply::texcoord: mod.setFaceWedgeTexCoords(); break;
263 default: break;
264 }
265 }
266 }
267 if (textureFiles.size() > 0) {
268 mod.setTextures(true);
269 }
270 return mod;
271 }

366 void setInfo(
367 const MeshInfo& info,
368 std::vector<std::string> textureFileNames = std::vector<std::string>(),
369 bool binary = true)
370 {
371 clear();
372 frmt = binary ? BINARY : ASCII;
373 valid = true;
374 textureFiles = textureFileNames;
375 if (info.hasVertices()) {
376 vertElemPos = nextElementID++;
377 ply::Element vElem;
378 vElem.type = ply::VERTEX;
379 if (info.hasVertexCoords()) {
380 ply::Property px, py, pz;
381 px.name = x;
382 px.type = (PropertyType)info.vertexCoordsType();
383 py.name = y;
384 py.type = (PropertyType)info.vertexCoordsType();
385 pz.name = z;
386 pz.type = (PropertyType)info.vertexCoordsType();
387 vElem.properties.push_back(px);
388 vElem.properties.push_back(py);
389 vElem.properties.push_back(pz);
390 }
391 if (info.hasVertexNormals()) {
392 ply::Property vnx, vny, vnz;
393 vnx.name = nx;
394 vnx.type = (PropertyType)info.vertexNormalsType();
395 vny.name = ny;
396 vny.type = (PropertyType)info.vertexNormalsType();
397 vnz.name = nz;
398 vnz.type = (PropertyType)info.vertexNormalsType();
399 vElem.properties.push_back(vnx);
400 vElem.properties.push_back(vny);
401 vElem.properties.push_back(vnz);
402 }
403 if (info.hasVertexColors()) {
404 ply::Property vcr, vcg, vcb, vca;
405 vcr.name = red;
406 vcr.type = (PropertyType)info.vertexColorsType();
407 vcg.name = green;
408 vcg.type = (PropertyType)info.vertexColorsType();
409 vcb.name = blue;
410 vcb.type = (PropertyType)info.vertexColorsType();
411 vca.name = alpha;
412 vca.type = (PropertyType)info.vertexColorsType();
413 vElem.properties.push_back(vcr);
414 vElem.properties.push_back(vcg);
415 vElem.properties.push_back(vcb);
416 vElem.properties.push_back(vca);
417 }
418 if (info.hasVertexQuality()) {
419 ply::Property vs;
420 vs.name = quality;
421 vs.type = (PropertyType)info.vertexQualityType();
422 vElem.properties.push_back(vs);
423 }
424 if (info.hasVertexTexCoords()) {
425 ply::Property tcu, tcv, tcn;
426 tcu.name = texture_u;
427 tcu.type = (PropertyType)info.vertexTexCoordsType();
428 tcv.name = texture_v;
429 tcv.type = (PropertyType)info.vertexTexCoordsType();
430 tcn.name = texnumber;
431 tcn.type = io::internal::USHORT;
432 vElem.properties.push_back(tcu);
433 vElem.properties.push_back(tcv);
434 vElem.properties.push_back(tcn);
435 }
436 if (info.hasVertexCustomComponents()) {
437 for (const auto& cc : info.vertexCustomComponents()) {
438 if (cc.type <= MeshInfo::DOUBLE) {
439 ply::Property pp;
440 pp.name = unknown;
441 pp.unknownPropertyName = cc.name;
442 pp.type = (PropertyType)cc.type;
443 vElem.properties.push_back(pp);
444 }
445 }
446 }
447 elements.push_back(vElem);
448 }
449 if (info.hasFaces()) {
450 faceElemPos = nextElementID++;
451 ply::Element fElem;
452 fElem.type = ply::FACE;
453 if (info.hasFaceVRefs()) {
454 ply::Property vids;
455 vids.list = true;
456 vids.name = vertex_indices;
457 vids.type = io::internal::UINT;
458 vids.listSizeType = io::internal::UCHAR;
459 fElem.properties.push_back(vids);
460 }
461 if (info.hasFaceNormals()) {
462 ply::Property fnx, fny, fnz;
463 fnx.name = nx;
464 fnx.type = (PropertyType)info.faceNormalsType();
465 fny.name = ny;
466 fny.type = (PropertyType)info.faceNormalsType();
467 fnz.name = nz;
468 fnz.type = (PropertyType)info.faceNormalsType();
469 fElem.properties.push_back(fnx);
470 fElem.properties.push_back(fny);
471 fElem.properties.push_back(fnz);
472 }
473 if (info.hasFaceColors()) {
474 ply::Property fcr, fcg, fcb, fca;
475 fcr.name = red;
476 fcr.type = (PropertyType)info.faceColorsType();
477 fcg.name = green;
478 fcg.type = (PropertyType)info.faceColorsType();
479 fcb.name = blue;
480 fcb.type = (PropertyType)info.faceColorsType();
481 fca.name = alpha;
482 fca.type = (PropertyType)info.faceColorsType();
483 fElem.properties.push_back(fcr);
484 fElem.properties.push_back(fcg);
485 fElem.properties.push_back(fcb);
486 fElem.properties.push_back(fca);
487 }
488 if (info.hasFaceQuality()) {
489 ply::Property fs;
490 fs.name = quality;
491 fs.type = (PropertyType)info.faceQualityType();
492 fElem.properties.push_back(fs);
493 }
494 if (info.hasFaceWedgeTexCoords()) {
495 ply::Property tc, tn;
496 tc.list = true;
497 tc.listSizeType = io::internal::UCHAR;
498 tc.name = texcoord;
499 tc.type = (PropertyType)info.faceWedgeTexCoordsType();
500 tn.name = texnumber;
501 tn.type = io::internal::USHORT;
502 fElem.properties.push_back(tc);
503 fElem.properties.push_back(tn);
504 }
505 if (info.hasFaceCustomComponents()) {
506 for (const auto& cc : info.faceCustomComponents()) {
507 if (cc.type <= MeshInfo::DOUBLE) {
508 ply::Property pp;
509 pp.name = unknown;
510 pp.unknownPropertyName = cc.name;
511 pp.type = (PropertyType)cc.type;
512 fElem.properties.push_back(pp);
513 }
514 }
515 }
516 elements.push_back(fElem);
517 }
518 if (info.hasEdges()) {
519 edgeElemPos = nextElementID++;
520 ply::Element eElem;
521 eElem.type = ply::EDGE;
522 if (info.hasEdgeVRefs()) {
523 ply::Property v1;
524 v1.name = vertex1;
525 v1.type = io::internal::UINT;
526 eElem.properties.push_back(v1);
527 ply::Property v2;
528 v2.name = vertex2;
529 v2.type = io::internal::UINT;
530 eElem.properties.push_back(v2);
531 }
532 elements.push_back(eElem);
533 }

536 std::string toString() const
537 {
538 std::string s;
539
540 s += "ply\nformat ";
541 s += (frmt == ASCII ? "ascii 1.0\n" : "binary_little_endian 1.0\n");
542 s += "comment Generated by vclib\n";
543 for (const std::string& str : textureFiles) {
544 s += "comment TextureFile " + str + "\n";
545 }
546 for (const Element& e : elements) {
547 s += "element ";
548 switch (e.type) {
549 case VERTEX:
550 s += "vertex " + std::to_string(e.numberElements) + "\n";
551 break;
552 case FACE:
553 s += "face " + std::to_string(e.numberElements) + "\n";
554 break;
555 case EDGE:
556 s += "edge " + std::to_string(e.numberElements) + "\n";
557 break;
558 case TRISTRIP:
559 s += "tristrips " + std::to_string(e.numberElements) + "\n";
560 break;
561 case MATERIAL:
562 s += "material " + std::to_string(e.numberElements) + "\n";
563 break;
564 case OTHER:
565 s += e.unknownElementType + " " +
566 std::to_string(e.numberElements) + "\n";
567 break;
568 }
569 for (const Property& p : e.properties) {
570 s += "property ";
571 if (p.list) {
572 s += "list ";
573 s += typeToString(p.listSizeType) + " ";
574 }
575 s += typeToString(p.type) + " ";
576 if (p.name == unknown)
577 s += p.unknownPropertyName + "\n";
578 else
579 s += nameToString(p.name) + "\n";
580 }
581 }
582 s += "end_header\n";
583
584 return s;
585 }

594 ply::Element readElement(const vcl::Tokenizer& lineTokenizer) const
595 {
596 ply::Element e;
597 vcl::Tokenizer::iterator token = lineTokenizer.begin();
598 std::string s = *(++token);
599 if (s == "vertex") {
600 e.type = ply::VERTEX;
601 e.numberElements = std::stoi(*(++token));
602 }
603 else if (s == "face") {
604 e.type = ply::FACE;
605 e.numberElements = std::stoi(*(++token));
606 }
607 else if (s == "edge") {
608 e.type = ply::EDGE;
609 e.numberElements = std::stoi(*(++token));
610 }
611 else if (s == "tristrips") {
612 e.type =ply::TRISTRIP;
613 e.numberElements = std::stoi(*(++token));
614 }
615 else {
616 e.type = ply::OTHER;
617 e.numberElements = std::stoi(*(++token));
618 e.unknownElementType = s;
619 }
620 return e;
621 }

623 ply::Property readProperty(const vcl::Tokenizer& lineTokenizer) const
624 {
625 ply::Property p;
626 vcl::Tokenizer::iterator token = lineTokenizer.begin();
627 std::string type = *(++token);
628 if (type == "list") {
629 p.list = true;
630 std::string typeSize = *(++token);
631 std::string typeData = *(++token);
632 std::string name = *(++token);
633 p.listSizeType = stringToType(typeSize);
634 p.type = stringToType(typeData);
635 p.name = stringToName(name);
636 if (p.name == unknown)
637 p.unknownPropertyName = name;
638 }
639 else {
640 p.list = false;
641 std::string name = *(++token);
642 p.type = stringToType(type);
643 p.name = stringToName(name);
644 if (p.name == unknown)
645 p.unknownPropertyName = name;
646 }
647
648 return p;
649 }
