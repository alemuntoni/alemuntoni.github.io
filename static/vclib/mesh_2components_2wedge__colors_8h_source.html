263 void importFrom(const Element& e)
264 {
265 if constexpr (HasWedgeColors<Element>) {
266 if (isWedgeColorsAvailableOn(e)) {
267 if constexpr(N > 0) {
268 // same static size
269 if constexpr (N == Element::WEDGE_COLOR_NUMBER) {
270 importWedgeColorsFrom(e);
271 }
272 // from dynamic to static, but dynamic size == static size

278 else {

281 }
282 }
