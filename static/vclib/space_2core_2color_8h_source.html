212 uint8_t hsvHue() const
213 {
214 uint8_t rgbMin, rgbMax;
215 uint8_t h;
216
217 rgbMin = x() < y() ? (x() < z() ? x() : z()) : (y() < z() ? y() : z());
218 rgbMax = x() > y() ? (x() > z() ? x() : z()) : (y() > z() ? y() : z());
219
220 if (rgbMax == 0) {
221 return 0;
222 }
223
224 if (255 * long(rgbMax - rgbMin) / rgbMax == 0) {
225 return 0;
226 }
227
228 if (rgbMax == x())
229 h = 0 + 43 * (y() - z()) / (rgbMax - rgbMin);
230 else if (rgbMax == y())
231 h = 85 + 43 * (z() - x()) / (rgbMax - rgbMin);
232 else
233 h = 171 + 43 * (x() - y()) / (rgbMax - rgbMin);
234
235 return h;
236 }

242 uint8_t hsvSaturation() const
243 {
244 uint8_t rgbMin, rgbMax;
245
246 rgbMin = x() < y() ? (x() < z() ? x() : z()) : (y() < z() ? y() : z());
247 rgbMax = x() > y() ? (x() > z() ? x() : z()) : (y() > z() ? y() : z());
248
249 if (rgbMax == 0) {
250 return 0;
251 }
252
253 return 255 * long(rgbMax - rgbMin) / rgbMax;
254 }

450 void setHsv(uint h, uint8_t s, uint8_t v, uint8_t alpha = 255)
451 {
452 w() = alpha;
453 if (s == 0) {
454 x() = v;
455 y() = v;
456 z() = v;
457 }
458 else {
459 h = (h / 360.0) * 255;
460 uint8_t region, remainder, p, q, t;
461 region = h / 43;
462 remainder = (h - (region * 43)) * 6;
463
464 p = (v * (255 - s)) >> 8;
465 q = (v * (255 - ((s * remainder) >> 8))) >> 8;
466 t = (v * (255 - ((s * (255 - remainder)) >> 8))) >> 8;
467
468 switch (region) {
469 case 0:
470 x() = v;
471 y() = t;
472 z() = p;
473 break;
474 case 1:
475 x() = q;
476 y() = v;
477 z() = p;
478 break;
479 case 2:
480 x() = p;
481 y() = v;
482 z() = t;
483 break;
484 case 3:
485 x() = p;
486 y() = q;
487 z() = v;
488 break;
489 case 4:
490 x() = t;
491 y() = p;
492 z() = v;
493 break;
494 default:
495 x() = v;
496 y() = p;
497 z() = q;
498 break;
499 }
500 }
501 }

643inline Color colorLerp(const Color& c0, const Color& c1, float value)
644{
645 Color c;
646 if (value < 0) // out of range - left
647 value = 0;
648 else if (value > 1) // out of range - right
649 value = 1;
650 for (uint i = 0; i < 4; i++)
651 c(i) = c1(i) * value + c0(i) * (1 - value);
652 return c;
653}

719inline Color colorFromIntervalParula(float value)
720{
721 if (value < 0) // out of range - left
722 value = 0;
723 else if (value > 1) // out of range - right
724 value = 1;
725
726 static uint paruVal[9] = {
727 0x271680ff,
728 0x0363e1ff,
729 0x1485d4ff,
730 0x06a7c6ff,
731 0x38b99eff,
732 0x92bf73ff,
733 0xd9ba56ff,
734 0xfcce2eff,
735 0xfffa0aff};
736
737 int ind = int(floor(value * 8.0f));
738 float div = (value * 8.0f - ind);
739
740 if (div < 0)
741 div = 0;
742 else if (div > 1)
743 div = 1;
744
745 return colorLerp(
746 (Color::ColorABGR) paruVal[ind],
747 (Color::ColorABGR) paruVal[ind + 1],
748 div);
749}

889inline std::vector<Color> colorScattering(
890 uint n,
891 float sat = 0.3,
892 float val = 0.9)
893{
894 std::vector<Color> scattering;
895 scattering.reserve(n);
896
897 for (uint v = 0; v < n; ++v) {
898 uint value = v, m = n;
899 uint b = 0;
900
901 for (uint k = 1; k < n; k <<= 1) {
902 if (value << 1 >= m) {
903 b += k;
904 value -= (m + 1) >> 1;
905 m >>= 1;
906 }
907 else {
908 m = (m + 1) >> 1;
909 }
910 }
911
912 Color rc;
913 rc.setHsvF(float(b) / float(n), sat, val);
914 scattering.push_back(rc);
915 }
916 return scattering;
917}
