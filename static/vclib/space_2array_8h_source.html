460 template<typename... Sizes>
461 void conservativeResize(Sizes... s) requires (sizeof...(s) == N)
462 {
463 std::size_t newSizes[N] = {static_cast<std::size_t>(s)...};
464 std::size_t newTotalSize = 1;
465 for (uint i = 0; i < N; i++)
466 newTotalSize *= newSizes[i];
467 std::vector<T> newVector(newTotalSize);
468
469 for (std::size_t i = 0; i < v.size(); i++) {
470 std::array<std::size_t, N> indices = reverseIndex(i);
471 bool outOfBound = false;
472 for (std::size_t j = 0; j < N; j++)
473 if (indices[j] >= newSizes[j] || indices[j] >= sizes[j])
474 outOfBound = true;
475 if (!outOfBound) {
476 newVector[getIndex(indices.data(), newSizes)] = v[i];
477 }
478 }
479
480 for (uint i = 0; i < sizes.size(); i++) {
481 sizes[i] = newSizes[i];
482 }
483
484 v = std::move(newVector);
485 }

579 std::array<std::size_t, N> reverseIndex(uint index)
580 {
581 std::array<std::size_t, N> indices;
582 for (long int i = N - 1; i >= 0; i--) {
583 indices[i] = index % sizes[i];

602 void initializeNestedLists(vcl::NestedInitializerLists<T, N> values)
603 {
604 std::list<std::size_t> szs =
605 NestedInitializerListsProcessor<T, N>::maxDimensionsLevels(values);
606
607 uint i = 0;
608 size_t totalSize = 1;
609 for (std::size_t s : szs) {
610 totalSize *= s;
611 sizes[i++] = s;
612 }
613 v.resize(totalSize);
614
615 typename std::vector<T>::iterator iterator = v.begin();
616 NestedInitializerListsProcessor<T, N>::processElements(
617 values, [&iterator](T value) { *(iterator++) = value; }, szs);
