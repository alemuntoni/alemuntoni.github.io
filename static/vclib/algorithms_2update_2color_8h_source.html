149template<FaceMeshConcept MeshType>
150void setPerVertexColorFromFaceColor(MeshType& m)
151{
152 vcl::requirePerVertexColor(m);
153 vcl::requirePerFaceColor(m);
154
155 using VertexType = MeshType::VertexType;
156 using FaceType = MeshType::FaceType;
157
158 std::vector<internal::ColorAvgInfo> csi(m.vertexContainerSize());
159
160 for (const FaceType& f : m.faces()) {
161 for (const VertexType* v : f.vertices()) {
162 csi[m.index(v)].c += v->color();
163 csi[m.index(v)].cnt++;
164 }
165 }
166
167 for (VertexType& v : m.vertices()) {
168 v.color() = csi[m.index(v)].c / csi[m.index(v)].cnt;
169 }
170}

188template<FaceMeshConcept MeshType>
189void setPerFaceColorFromVertexColor(MeshType& m)
190{
191 vcl::requirePerVertexColor(m);
192 vcl::requirePerFaceColor(m);
193
194 using VertexType = MeshType::VertexType;
195 using FaceType = MeshType::FaceType;
196
197 for (FaceType& f : m.faces()) {
198 vcl::Point4<uint> avg(0, 0, 0, 0);
199 for (const VertexType* v : f.vertices()) {
200 avg += v->color().template cast<uint>();
201 }
202 avg /= f.vertexNumber();
203 f.color() = avg.cast<uint8_t>();
204 }
205}

232void setPerVertexColorFromQuality(

328template<FaceMeshConcept MeshType>
329void setPerVertexColorFromFaceBorderFlag(
330 MeshType& m,
331 Color borderColor = vcl::Color::Blue,
332 Color internalColor = vcl::Color::White,
333 Color mixColor = vcl::Color::Cyan)
334{
335 vcl::requirePerVertexColor(m);
336
337 using FaceType = MeshType::FaceType;
338
339 const vcl::Color baseColor = Color::Green;
340
341 setPerVertexColor(m, baseColor);
342
343 for (FaceType& f : m.faces()) {
344 for (uint i = 0; i < f.vertexNumber(); ++i) {
345 if (f.edgeOnBorder(i)) {
346 if (f.vertex(i).color() == baseColor)
347 f.vertex(i).color() = borderColor;
348 if (f.vertex(i).color() == internalColor)
349 f.vertex(i).color() = mixColor;
350 if (f.vertexMod(i + 1).color() == baseColor)
351 f.vertexMod(i + 1).color() = borderColor;
352 if (f.vertexMod(i + 1).color() == internalColor)
353 f.vertexMod(i + 1).color() = mixColor;
354 }
355 else {
356 if (f.vertex(i).color() == baseColor)
357 f.vertex(i).color() = internalColor;
358 if (f.vertex(i).color() == borderColor)
359 f.vertex(i).color() = mixColor;
360 if (f.vertexMod(i + 1).color() == baseColor)
361 f.vertexMod(i + 1).color() = internalColor;
362 if (f.vertexMod(i + 1).color() == borderColor)
363 f.vertexMod(i + 1).color() = mixColor;
364 }
365 }
366 }

454template<FaceMeshConcept MeshType>
455void setPerFaceColorScattering(
456 MeshType& m,
457 uint nColors = 50,
458 bool checkFauxEdges = true)
459{
460 vcl::requirePerFaceColor(m);
461
462 using FaceType = MeshType::FaceType;
463
464 Color baseColor = Color::Black;
465 setPerFaceColor(m, baseColor);
466
467 std::vector<Color> vc = colorScattering(nColors);
468
469 for (FaceType& f : m.faces()) {
470 if (f.color() == baseColor) {
471 f.color() = vc[m.index(f) % nColors];
472 }
473 if constexpr (HasPerFaceAdjacentFaces<MeshType>) {
474 if (checkFauxEdges && isPerFaceAdjacentFacesAvailable(m)) {
475 for (uint i = 0; i < f.vertexNumber(); ++i) {
476 if (f.edgeFaux(i)) {
477 assert(f.adjFace(i) != nullptr);
478 f.adjFace(i)->color = f.color();
479 }
480 }
481 }
482 }
483 }

507template<MeshConcept MeshType, PointConcept PointType>
508void setPerVertexColorPerlinNoise(
509 MeshType& m,
510 PointType period,
511 PointType offset = PointType(0, 0, 0),
512 bool onSelected = false)
513{
514 vcl::requirePerVertexColor(m);
515
516 using VertexType = MeshType::VertexType;
517
518 PointType p[3];
519
520 for (VertexType& v : m.vertices()) {
521 if (!onSelected || v.selected()) {
522 p[0] = (v.coord() / period[0]) + offset;
523 p[1] = (v.coord() / period[1]) + offset;
524 p[2] = (v.coord() / period[2]) + offset;
525 v.color() = Color(
526 127 + 128.0 * perlinNoise(p[0][0], p[0][1], p[0][2]),
527 127 + 128.0 * perlinNoise(p[1][0], p[1][1], p[1][2]),
528 127 + 128.0 * perlinNoise(p[2][0], p[2][1], p[2][2]),
529 255);
530 }
531 }

552template<MeshConcept MeshType, PointConcept PointType>
553void setPerVertexPerlinColor(
554 MeshType& m,
555 double period,
556 PointType offset = PointType(0, 0, 0),
557 Color color1 = Color::Black,
558 Color color2 = Color::White,
559 bool onSelected = false)
560{
561 vcl::requirePerVertexColor(m);
562
563 using VertexType = MeshType::VertexType;
564
565 for (VertexType& v : m.vertices()) {
566 if (!onSelected || v.selected()) {
567 PointType p = v.coord() / period + offset;
568
569 double factor = (perlinNoise(p[0], p[1], p[2]) + 1.0) / 2.0;
570
571 int rr = (color1[0] * factor) + (color2[0] * (1.0 - factor));
572 int gg = (color1[1] * factor) + (color2[1] * (1.0 - factor));
573 int bb = (color1[2] * factor) + (color2[2] * (1.0 - factor));
574 int aa = (color1[3] * factor) + (color2[3] * (1.0 - factor));
575
576 v.color() = Color(rr, gg, bb, aa);
577 }
578 }
