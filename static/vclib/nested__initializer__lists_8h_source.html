40// recursive case: a nested list is a list of L-1 levels
41template<typename T, std::size_t L>
42struct NestedInitializerListsTraits
43{
44 using type = std::initializer_list<
45 typename NestedInitializerListsTraits<T, L - 1>::type>;
46};
47
48// base case: no list, only the type T
49template<typename T>
50struct NestedInitializerListsTraits<T, 0>
51{
52 using type = T;
53};
54
55} // namespace internal

70// Recursive part.
71
78template<typename T, std::size_t L>
79class NestedInitializerListsProcessor
80{
81public:
105 static std::list<size_t>
106 maxDimensionsLevels(NestedInitializerLists<T, L> values)
107 {
108 std::list<size_t> final;
109 bool first = true;
110 for (auto nested : values) {
111 std::list<size_t> l =
112 NestedInitializerListsProcessor<T, L - 1>::maxDimensionsLevels(
113 nested);
114
115 if (first) {
116 first = false;
117 for (size_t s : l)
118 final.push_back(s);
119 }
120 else {
121 assert(l.size() == final.size());
122 std::list<size_t>::iterator it = l.begin();
123 for (size_t& s : final) {
124 if (s < *it)
125 s = *it;
126 ++it;
127 }
128 }
129 }
130 final.push_front(values.size());
131 return final;
132 }
133
156 template<typename T_Function>
157 static void
158 processElements(NestedInitializerLists<T, L> values, T_Function function)
159 {
160 for (auto nested : values) {
161 NestedInitializerListsProcessor<T, L - 1>::processElements(
162 nested, function);
163 }
164 }
165
193 template<typename T_Function>
194 static void processElements(
195 NestedInitializerLists<T, L> values,
196 T_Function function,
197 std::list<size_t> sizes)
198 {
199 size_t curr_size = sizes.front();
200 sizes.pop_front();
201 for (auto nested : values) {
202 NestedInitializerListsProcessor<T, L - 1>::processElements(
203 nested, function, sizes);
204 }
205
206 // 0 in left values
207 if (values.size() < curr_size) {
208 std::size_t count = 1;
209 for (size_t s : sizes)
210 count *= s;
211
212 while (count-- > 0)
213 function(static_cast<T>(0));
214 }
215 }
216};

220template<typename T>
221class NestedInitializerListsProcessor<T, 1>
222{
223public:
224 using InitializerList = vcl::NestedInitializerLists<T, 1>;
225
226 static std::list<size_t> maxDimensionsLevels(InitializerList values)
227 {
228 std::list<size_t> dim;
229 dim.push_back(values.size());
230 return dim;
231 }
232
233 template<typename T_Function>
234 static void processElements(
235 InitializerList values,
236 T_Function function,
237 std::list<size_t> sizes)
238 {
239 size_t row_size = 1;
240 for (size_t s : sizes)
241 row_size *= s;
242 std::for_each(values.begin(), values.end(), function);
243
244 // 0 in left values
245 if (values.size() < row_size) {
246 std::size_t count = row_size - values.size();
247 while (count-- > 0) {
248 function(static_cast<T>(0));
249 }
250 }
251 }
252
253 template<typename T_Function>
254 static void processElements(InitializerList values, T_Function function)
255 {
256 std::for_each(values.begin(), values.end(), function);
257 }
258};
