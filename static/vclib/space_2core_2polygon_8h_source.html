284 template<typename Iterator>
285 static PointT normal(Iterator begin, Iterator end) requires (
286 std::is_same_v<typename Iterator::value_type, PointT> &&
287 PointT::DIM == 3)
288 {
289 // compute the sum of normals for each triplet of consecutive points
290 PointT sum;
291 sum.setZero();
292 for (auto i = begin; i != end; ++i) {
293 auto i1 = i;
294 ++i1;
295 if (i1 == end)
296 i1 = begin;
297
298 auto i2 = i1;
299 ++i2;
300 if (i2 == end)
301 i2 = begin;
302
303 sum += (*i1 - *i).cross(*i2 - *i);
304 }
305 sum.normalize();
306 return sum;
307 }

334 template<typename Iterator>
335 static PointT barycenter(Iterator begin, Iterator end)
336 requires (std::is_same_v<typename Iterator::value_type, PointT>)
337 {
338 PointT bar;
339 bar.setZero();
340
341 uint cnt = 0;
342 for (; begin != end; ++begin) {
343 bar += *begin;
344 ++cnt;
345 }
346
347 assert(cnt);
348
349 return bar / cnt;
350 }

382 template<typename Iterator, typename WIterator>
383 static PointT weightedBarycenter(
384 Iterator begin,
385 Iterator end,
386 WIterator wBegin)
387 requires (std::is_same_v<typename Iterator::value_type, PointT>)
388 {
389 using ScalarType = WIterator::value_type;
390
391 PointT bar;
392 bar.setZero();
393 ScalarType wSum = 0;
394
395 for (; begin != end; ++begin, ++wBegin) {
396 bar += *begin * *wBegin;
397 wSum += *wBegin;
398 }
399
400 return bar / wSum;
401 }

440 template<typename Iterator>
441 static ScalarType perimeter(Iterator begin, Iterator end)
442 requires (std::is_same_v<typename Iterator::value_type, PointT>)
443 {
444 using Scalar = PointType::ScalarType;
445
446 Scalar per = 0;
447 for (auto i = begin; i != end; ++i) {
448 const PointT& p0 = *i;
449 auto i1 = i;
450 ++i;
451 if (i1 == end)
452 i1 = begin;
453 const PointT& p1 = *i1;
454 per += p0.dist(p1);
455 }
456 return per;
457 }

488 template<typename Iterator>
489 static ScalarType area(Iterator begin, Iterator end)
490 requires (std::is_same_v<typename Iterator::value_type, PointT>)
491 {
492 using Scalar = PointType::ScalarType;
493
494 PointType bar = barycenter(begin, end);
495 Scalar area = 0;
496 for (auto i = begin; i != end; ++i) {
497 const PointT& p0 = *i;
498 auto i1 = i;
499 ++i1;
500 if (i1 == end)
501 i1 = begin;
502 const PointT& p1 = *i1;
503
504 area += Triangle<PointT>::area(p0, p1, bar);
505 }
506 return area;
507 }
