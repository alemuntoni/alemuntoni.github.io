123 template<typename Iterator>
124 static PointT normal(Iterator begin, Iterator end) requires (
125 std::is_same_v<typename Iterator::value_type, PointT> &&
126 PointT::DIM == 3)
127 {
128 // compute the sum of normals for each triplet of consecutive points
129 PointT sum;
130 sum.setZero();
131 for (auto i = begin; i != end; ++i) {
132 auto i1 = i;
133 ++i1;
134 if (i1 == end)
135 i1 = begin;
136
137 auto i2 = i1;
138 ++i2;
139 if (i2 == end)
140 i2 = begin;
141
142 sum += (*i1 - *i).cross(*i2 - *i);
143 }
144 sum.normalize();
145 return sum;
146 }

166 template<typename Iterator>
167 static PointT barycenter(Iterator begin, Iterator end)
168 requires (std::is_same_v<typename Iterator::value_type, PointT>)
169 {
170 PointT bar;
171 bar.setZero();
172
173 uint cnt = 0;
174 for (; begin != end; ++begin) {
175 bar += *begin;
176 ++cnt;
177 }
178
179 assert(cnt);
180
181 return bar / cnt;
182 }

207 template<typename Iterator, typename WIterator>
208 static PointT weightedBarycenter(
209 Iterator begin,
210 Iterator end,
211 WIterator wbegin)
212 requires (std::is_same_v<typename Iterator::value_type, PointT>)
213 {
214 using ScalarType = WIterator::value_type;
215
216 PointT bar;
217 bar.setZero();
218 ScalarType wsum = 0;
219
220 for (; begin != end; ++begin, ++wbegin) {
221 bar += *begin * *wbegin;
222 wsum += *wbegin;
223 }
224
225 return bar / wsum;

256 template<typename Iterator>
257 static ScalarType perimeter(Iterator begin, Iterator end)
258 requires (std::is_same_v<typename Iterator::value_type, PointT>)
259 {
260 using Scalar = PointType::ScalarType;
261
262 Scalar per = 0;
263 for (auto i = begin; i != end; ++i) {
264 const PointT& p0 = *i;
265 auto i1 = i;
266 ++i;
267 if (i1 == end)
268 i1 = begin;
269 const PointT& p1 = *i1;
270 per += p0.dist(p1);
271 }
272 return per;

297 template<typename Iterator>
298 static ScalarType area(Iterator begin, Iterator end)
299 requires (std::is_same_v<typename Iterator::value_type, PointT>)
300 {
301 using Scalar = PointType::ScalarType;
302
303 PointType bar = barycenter(begin, end);
304 Scalar area = 0;
305 for (auto i = begin; i != end; ++i) {
306 const PointT& p0 = *i;
307 auto i1 = i;
308 ++i1;
309 if (i1 == end)
310 i1 = begin;
311 const PointT& p1 = *i1;
312
313 area += Triangle<PointT>::area(p0, p1, bar);
314 }
315 return area;
316 }

353 template<typename Iterator>
354 static std::vector<uint> earCut(Iterator begin, Iterator end) requires (
355 std::is_same_v<typename Iterator::value_type, PointT> &&
356 PointT::DIM == 2)
357 {
358 using Scalar = PointT::ScalarType;
359
360 // Convert the polygon represented as a sequence of vertices to a vector
361 // of vectors of points. This is necessary because the earcut library
362 // expects the polygon to be represented as a vector of contours, where
363 // each contour is a vector of points. In this case, there is only one
364 // contour, which is represented as a vector of points.
365 std::vector<std::vector<Point2<Scalar>>> poly;
366 poly.emplace_back(begin, end);
367
368 // Use the earcut library to triangulate the polygon and return the
369 // result.
370 return mapbox::earcut<uint>(poly);
371 }

406 template<typename Iterator>
407 static std::vector<uint> earCut(Iterator begin, Iterator end) requires (
408 std::is_same_v<typename Iterator::value_type, PointT> &&
409 PointT::DIM == 3)
410 {
411 using Scalar = PointType::ScalarType;
412
413 // Calculate the normal vector of the polygon and an orthonormal basis
414 // for the plane containing the polygon.
415 PointType n = normal(begin, end);
416 PointType u, v;
417 n.orthoBase(u, v);
418
419 // Project each vertex onto the plane defined by the orthonormal basis.
420 std::vector<Point2<Scalar>> poly2D;
421
422 if constexpr (std::is_same_v<
423 typename Iterator::iterator_category,
424 std::random_access_iterator_tag>)
425 {
426 poly2D.reserve(std::distance(begin, end));
427 }
428
429 for (auto i = begin; i != end; ++i) {
430 // project i-th polygon in a 2D plane
431 poly2D.emplace_back(*i * u, *i * v);
432 }
433
434 // Use the ear-cut algorithm to triangulate the polygon in the 2D plane
435 // and return the result.
436 return Polygon<Point2<Scalar>>::earCut(poly2D.begin(), poly2D.end());
437 }
