434 edgeFaux(0) = true;

448 int exportToVCGFlags() const
449 {
450 int f = 0;
451 if (visited())
452 f &= 0x00000010;
453 if (selected())
454 f &= 0x00000020;
455 if (edgeOnBorder(0))
456 f &= 0x00000040;
457 if (edgeOnBorder(1))
458 f &= 0x00000080;
459 if (edgeOnBorder(2))
460 f &= 0x00000100;
461 if (edgeSelected(0))
462 f &= 0x00008000;
463 if (edgeSelected(1))
464 f &= 0x00010000;
465 if (edgeSelected(2))
466 f &= 0x00020000;
467 if (edgeFaux(0))
468 f &= 0x00040000;
469 if (edgeFaux(1))
470 f &= 0x00080000;
471 if (edgeFaux(2))
472 f &= 0x00100000;
473 return f;

482 // Component interface function
483 template<typename Element>
484 void importFrom(const Element& e)
485 {
486 if constexpr (HasBitFlags<Element>) {
487 resetBitFlags();
488 if constexpr (HasPolygonBitFlags<Element>) {
489 flags() = e.flags();
490 edgeFlags() = e.edgeFlags();
491 }
492 else {
493 // BitFlags
494 deletedBit() = e.deleted();
495 selected() = e.selected();
496 visited() = e.visited();
497 const uint UM = std::min(USER_BITS_NUMBER, e.USER_BITS_NUMBER);
498 for (uint i = 0; i < UM; ++i)
499 userBit(i) = e.userBit(i);
500 if constexpr (HasTriangleBitFlags<Element>) {
501 // TriangleBitFlags
502 for (uint i = 0; i < 3; ++i) {
503 edgeOnBorder(i) = e.edgeOnBorder(i);
504 edgeSelected(i) = e.edgeSelected(i);
505 edgeVisited(i) = e.edgeVisited(i);
506 edgeFaux(i) = e.edgeFaux(i);
507 }
508 }
509 }
510 }
