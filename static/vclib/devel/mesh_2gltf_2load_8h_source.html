37template<Matrix44Concept MatrixType, LoggerConcept LogType>
38void loadGltfMeshesWhileTraversingNodes(
39 const tinygltf::Model& model,
40 auto& currentMeshIt,
41 auto& currentInfoIt,
42 MatrixType currentMatrix,
43 uint currentNode,
44 LogType& log,
45 const LoadSettings& settings)
46{
47 currentMatrix =
48 currentMatrix * gltfCurrentNodeMatrix<MatrixType>(model, currentNode);
49
50 if (model.nodes[currentNode].mesh >= 0) {
51 int meshid = model.nodes[currentNode].mesh;
52 gltfLoadMesh(
53 *currentMeshIt,
54 *currentInfoIt,
55 model.meshes[meshid],
56 model,
57 currentMatrix,
58 log,
59 settings);
60
61 ++currentMeshIt;
62 ++currentInfoIt;
63 }
64 // for each child
65 for (int c : model.nodes[currentNode].children) {
66 if (c >= 0) { // if it is valid
67 // visit child
68 loadGltfMeshesWhileTraversingNodes(
69 model,
70 currentMeshIt,
71 currentInfoIt,
72 currentMatrix,
73 c,
74 log,
75 settings);
76 }
77 }
78}

80template<MeshConcept MeshType, LoggerConcept LogType>
81void loadGltf(
82 tinygltf::Model& model,
83 std::vector<MeshType>& meshes,
84 std::vector<MeshInfo>& infos,
85 LogType& log,
86 const LoadSettings& settings = LoadSettings())
87{
88 using ScalarType = MeshType::ScalarType;
89
90 using Matrix44m = std::conditional_t<
91 HasTransformMatrix<MeshType>,
92 typename MeshType::TransformMatrixType,
93 Matrix44<ScalarType>>;
94
95 uint nMeshes = gltfNumberMeshes(model);
96
97 Matrix44m identityMatrix = Matrix44m::Identity();
98
99 meshes.resize(nMeshes);
100 infos.resize(nMeshes);
101
102 log.startProgress("Reading meshes", nMeshes);
103
104 auto mit = meshes.begin();
105 auto iit = infos.begin();
106
107 for (unsigned int s = 0; s < model.scenes.size(); ++s) {
108 const tinygltf::Scene& scene = model.scenes[s];
109 for (unsigned int n = 0; n < scene.nodes.size(); ++n) {
110 loadGltfMeshesWhileTraversingNodes(
111 model, mit, iit, identityMatrix, scene.nodes[n], log, settings);
112
113 log.progress(s * scene.nodes.size() + n);
114 }
115 }
116
117 log.endProgress();
118}

122template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
123void loadGltf(
124 MeshType& m,
125 const std::string& filename,
126 MeshInfo& loadedInfo,
127 LogType& log = nullLogger,
128 const LoadSettings& settings = LoadSettings())
129{
130 tinygltf::TinyGLTF loader;
131 tinygltf::Model model;
132
133 std::string err;
134 std::string warn;
135
136 // Determine the file format from the filename extension
137 std::string ext = toLower(FileInfo::extension(filename));
138
139 bool ret = false;
140
141 if (ext == ".gltf") {
142 ret = loader.LoadASCIIFromFile(&model, &err, &warn, filename);
143 }
144 else if (ext == ".glb") {
145 ret = loader.LoadBinaryFromFile(&model, &err, &warn, filename);
146 }
147 else {
148 throw UnknownFileFormatException(ext);
149 }
150
151 if (!ret) {
152 throw CannotOpenFileException(
153 "Failed to load glTF file: " + filename + "\n" + err);
154 }
155
156 if (!warn.empty()) {
157 log.log(
158 "Warnings while loading glTF file: " + warn,
159 LogType::LogLevel::WARNING_LOG);
160 }
161
162 // Load the mesh data from the model
163 std::vector<MeshType> meshes;
164 std::vector<MeshInfo> infos;
165 detail::loadGltf(model, meshes, infos, log, settings);
166
167 m = std::move(meshes.front());
168 loadedInfo = std::move(infos.front());
169 for (std::size_t i = 1; i < meshes.size(); ++i) {
170 m.append(std::move(meshes[i]));
171 // todo manage infos
172 }
173}
