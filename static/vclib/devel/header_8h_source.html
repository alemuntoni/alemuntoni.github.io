74 PlyHeader(std::istream& file, const std::string& filename = "")
75 {
76 clear();
77
78 std::string line;
79 std::getline(file, line);
80 removeCarriageReturn(line);
81 if (line.compare(0, 3, "ply") == 0) {
82 bool error = false;
83 bool firstElement = true;
84 std::string headerLine;
85 PlyElement element;
86 do {
87 Tokenizer spaceTokenizer =
88 readAndTokenizeNextNonEmptyLine(file);
89 if (!error) {
90 Tokenizer::iterator token = spaceTokenizer.begin();
91 headerLine = *token;
92 if (headerLine == "format") {
93 token++;
94 if (*token == "ascii")
95 mFormat = ply::ASCII;
96 else if (
97 *token == "binary_little_endian" ||
98 *token == "binary")
99 mFormat = ply::BINARY_LITTLE_ENDIAN;
100 else if (*token == "binary_big_endian")
101 mFormat = ply::BINARY_BIG_ENDIAN;
102 }
103 // reading a comment, may be a texture file...
104 else if (headerLine == "comment") {
105 token++;
106 if (token != spaceTokenizer.end()) {
107 if (containsCaseInsensitive(*token, "texture")) {
108 ++token;
109 if (token != spaceTokenizer.end()) {
110 std::string textName = *token;
111 auto it =
112 findCaseInsensitive(textName, "<this>");
113 if (it != textName.end()) {
114 uint pos = it - textName.begin();
115 std::string fn =
116 FileInfo::fileNameWithoutExtension(
117 filename);
118 textName =
119 textName.substr(0, pos) + fn +
120 textName.substr(
121 pos + 6, textName.size());
122 }
123 mTextureFiles.push_back(textName);
124 }
125 }
126 }
127 }
128 // I am reading a new element
129 else if (headerLine == "element") {
130 // if it is not the first element to read, it means that
131 // the previous one needs to be saved
132 if (!firstElement) {
133 // index of each element type in elements vector
134 if (element.type == ply::VERTEX)
135 mVertElemPos = mElements.size();
136 if (element.type == ply::FACE)
137 mFaceElemPos = mElements.size();
138 if (element.type == ply::EDGE)
139 mEdgeElemPos = mElements.size();
140 if (element.type == ply::TRISTRIP)
141 mTriStripElemPos = mElements.size();
142 mElements.push_back(element);
143 element = PlyElement();
144 }
145 element = readElement(spaceTokenizer);
146 firstElement = false;
147 }
148 else if (headerLine == "property") {
149 PlyProperty p = readProperty(spaceTokenizer);
150 element.properties.push_back(p);
151 }
152 // save the last element
153 else if (headerLine == "end_header") {
154 if (element.type == ply::VERTEX)
155 mVertElemPos = mElements.size();
156 if (element.type == ply::FACE)
157 mFaceElemPos = mElements.size();
158 if (element.type == ply::EDGE)
159 mEdgeElemPos = mElements.size();
160 if (element.type == ply::TRISTRIP)
161 mTriStripElemPos = mElements.size();
162 mElements.push_back(element);
163 }
164 }
165 } while (!error && headerLine != "end_header");
166 mValid = !error && hasVertices();
167 }
168 }

187 MeshInfo getInfo() const
188 {
189 MeshInfo mod;
190 // x, y, z, nx, ny, nz, red, green, blue, alpha, vertex_indices
191
192 if (mVertElemPos != UINT_NULL) {
193 mod.setVertices();
194 for (const PlyProperty& p : mElements[mVertElemPos].properties) {
195 switch (p.name) {
196 case ply::x:
197 case ply::y:
198 case ply::z: mod.setPerVertexPosition(); break;
199 case ply::nx:
200 case ply::ny:
201 case ply::nz: mod.setPerVertexNormal(); break;
202 case ply::red:
203 case ply::green:
204 case ply::blue:
205 case ply::alpha: mod.setPerVertexColor(); break;
206 case ply::quality: mod.setPerVertexQuality(); break;
207 case ply::texture_u: mod.setPerVertexTexCoord(); break;
208 case ply::unknown:
209 if (p.type <= ply::PropertyType::DOUBLE) {
210 mod.addPerVertexCustomComponent(
211 p.unknownPropertyName, (MeshInfo::DataType) p.type);
212 }
213 default: break;
214 }
215 }
216 }
217 if (mFaceElemPos != UINT_NULL) {
218 mod.setFaces();
219 for (const PlyProperty& p : mElements[mFaceElemPos].properties) {
220 switch (p.name) {
221 case ply::vertex_indices:
222 mod.setPerFaceVertexReferences();
223 break;
224 case ply::nx:
225 case ply::ny:
226 case ply::nz: mod.setPerFaceNormal(); break;
227 case ply::red:
228 case ply::green:
229 case ply::blue:
230 case ply::alpha: mod.setPerFaceColor(); break;
231 case ply::quality: mod.setPerFaceQuality(); break;
232 case ply::texcoord: mod.setPerFaceWedgeTexCoords(); break;
233 case ply::unknown:
234 if (p.type <= ply::PropertyType::DOUBLE) {
235 mod.addPerFaceCustomComponent(
236 p.unknownPropertyName, (MeshInfo::DataType) p.type);
237 }
238 default: break;
239 }
240 }
241 }
242 if (mTriStripElemPos != UINT_NULL) {
243 mod.setFaces();
244 for (const PlyProperty& p :
245 mElements[mTriStripElemPos].properties) {
246 switch (p.name) {
247 case ply::vertex_indices:
248 mod.setPerFaceVertexReferences();
249 break;
250 case ply::nx:
251 case ply::ny:
252 case ply::nz: mod.setPerFaceNormal(); break;
253 case ply::red:
254 case ply::green:
255 case ply::blue:
256 case ply::alpha: mod.setPerFaceColor(); break;
257 case ply::quality: mod.setPerFaceQuality(); break;
258 case ply::texcoord: mod.setPerFaceWedgeTexCoords(); break;
259 default: break;
260 }
261 }
262 }
263 if (mEdgeElemPos != UINT_NULL) {
264 mod.setEdges();
265 for (const PlyProperty& p : mElements[mEdgeElemPos].properties) {
266 switch (p.name) {
267 case ply::vertex_indices:
268 mod.setPerEdgeVertexReferences();
269 break;
270 case ply::nx:
271 case ply::ny:
272 case ply::nz: mod.setPerEdgeNormal(); break;
273 case ply::red:
274 case ply::green:
275 case ply::blue:
276 case ply::alpha: mod.setPerEdgeColor(); break;
277 case ply::quality: mod.setPerEdgeQuality(); break;
278 default: break;
279 }
280 }
281 }
282 if (mTextureFiles.size() > 0) {
283 mod.setTextures(true);
284 }
285 return mod;
286 }

378 void setInfo(
379 const MeshInfo& info,
380 std::vector<std::string> textureFileNames = std::vector<std::string>(),
381 ply::Format format = ply::BINARY_LITTLE_ENDIAN)
382 {
383 clear();
384 mFormat = format;
385 mValid = true;
386 mTextureFiles = textureFileNames;
387 if (info.hasVertices()) {
388 mVertElemPos = mElements.size();
389 PlyElement vElem;
390 vElem.type = ply::VERTEX;
391 if (info.hasPerVertexPosition()) {
392 PlyProperty px, py, pz;
393 px.name = ply::x;
394 px.type = info.perVertexPositionType();
395 py.name = ply::y;
396 py.type = info.perVertexPositionType();
397 pz.name = ply::z;
398 pz.type = info.perVertexPositionType();
399 vElem.properties.push_back(px);
400 vElem.properties.push_back(py);
401 vElem.properties.push_back(pz);
402 }
403 if (info.hasPerVertexNormal()) {
404 PlyProperty vnx, vny, vnz;
405 vnx.name = ply::nx;
406 vnx.type = info.perVertexNormalType();
407 vny.name = ply::ny;
408 vny.type = info.perVertexNormalType();
409 vnz.name = ply::nz;
410 vnz.type = info.perVertexNormalType();
411 vElem.properties.push_back(vnx);
412 vElem.properties.push_back(vny);
413 vElem.properties.push_back(vnz);
414 }
415 if (info.hasPerVertexColor()) {
416 PlyProperty vcr, vcg, vcb, vca;
417 vcr.name = ply::red;
418 vcr.type = info.perVertexColorType();
419 vcg.name = ply::green;
420 vcg.type = info.perVertexColorType();
421 vcb.name = ply::blue;
422 vcb.type = info.perVertexColorType();
423 vca.name = ply::alpha;
424 vca.type = info.perVertexColorType();
425 vElem.properties.push_back(vcr);
426 vElem.properties.push_back(vcg);
427 vElem.properties.push_back(vcb);
428 vElem.properties.push_back(vca);
429 }
430 if (info.hasPerVertexQuality()) {
431 PlyProperty vs;
432 vs.name = ply::quality;
433 vs.type = info.perVertexQualityType();
434 vElem.properties.push_back(vs);
435 }
436 if (info.hasPerVertexTexCoord()) {
437 PlyProperty tcu, tcv, tcn;
438 tcu.name = ply::texture_u;
439 tcu.type = info.perVertexTexCoordType();
440 tcv.name = ply::texture_v;
441 tcv.type = info.perVertexTexCoordType();
442 tcn.name = ply::texnumber;
443 tcn.type = PrimitiveType::USHORT;
444 vElem.properties.push_back(tcu);
445 vElem.properties.push_back(tcv);
446 vElem.properties.push_back(tcn);
447 }
448 if (info.hasPerVertexCustomComponents()) {
449 for (const auto& cc : info.perVertexCustomComponents()) {
450 if (cc.type <= PrimitiveType::DOUBLE) {
451 PlyProperty pp;
452 pp.name = ply::unknown;
453 pp.unknownPropertyName = cc.name;
454 pp.type = cc.type;
455 vElem.properties.push_back(pp);
456 }
457 }
458 }
459 mElements.push_back(vElem);
460 }
461 if (info.hasFaces()) {
462 mFaceElemPos = mElements.size();
463 PlyElement fElem;
464 fElem.type = ply::FACE;
465 if (info.hasPerFaceVertexReferences()) {
466 PlyProperty vids;
467 vids.list = true;
468 vids.name = ply::vertex_indices;
469 vids.type = PrimitiveType::UINT;
470 vids.listSizeType = PrimitiveType::UCHAR;
471 fElem.properties.push_back(vids);
472 }
473 if (info.hasPerFaceNormal()) {
474 PlyProperty fnx, fny, fnz;
475 fnx.name = ply::nx;
476 fnx.type = info.perFaceNormalType();
477 fny.name = ply::ny;
478 fny.type = info.perFaceNormalType();
479 fnz.name = ply::nz;
480 fnz.type = info.perFaceNormalType();
481 fElem.properties.push_back(fnx);
482 fElem.properties.push_back(fny);
483 fElem.properties.push_back(fnz);
484 }
485 if (info.hasPerFaceColor()) {
486 PlyProperty fcr, fcg, fcb, fca;
487 fcr.name = ply::red;
488 fcr.type = info.perFaceColorType();
489 fcg.name = ply::green;
490 fcg.type = info.perFaceColorType();
491 fcb.name = ply::blue;
492 fcb.type = info.perFaceColorType();
493 fca.name = ply::alpha;
494 fca.type = info.perFaceColorType();
495 fElem.properties.push_back(fcr);
496 fElem.properties.push_back(fcg);
497 fElem.properties.push_back(fcb);
498 fElem.properties.push_back(fca);
499 }
500 if (info.hasPerFaceQuality()) {
501 PlyProperty fs;
502 fs.name = ply::quality;
503 fs.type = (ply::PropertyType) info.perFaceQualityType();
504 fElem.properties.push_back(fs);
505 }
506 if (info.hasPerFaceWedgeTexCoords()) {
507 PlyProperty tc, tn;
508 tc.list = true;
509 tc.listSizeType = PrimitiveType::UCHAR;
510 tc.name = ply::texcoord;
511 tc.type = (ply::PropertyType) info.perFaceWedgeTexCoordsType();
512 tn.name = ply::texnumber;
513 tn.type = PrimitiveType::USHORT;
514 fElem.properties.push_back(tc);
515 fElem.properties.push_back(tn);
516 }
517 if (info.hasPerFaceCustomComponents()) {
518 for (const auto& cc : info.perFaceCustomComponents()) {
519 if (cc.type <= PrimitiveType::DOUBLE) {
520 PlyProperty pp;
521 pp.name = ply::unknown;
522 pp.unknownPropertyName = cc.name;
523 pp.type = (ply::PropertyType) cc.type;
524 fElem.properties.push_back(pp);
525 }
526 }
527 }
528 mElements.push_back(fElem);
529 }
530 if (info.hasEdges()) {
531 mEdgeElemPos = mElements.size();
532 PlyElement eElem;
533 eElem.type = ply::EDGE;
534 if (info.hasPerEdgeVertexReferences()) {
535 PlyProperty v1;
536 v1.name = ply::vertex1;
537 v1.type = PrimitiveType::UINT;
538 eElem.properties.push_back(v1);
539 PlyProperty v2;
540 v2.name = ply::vertex2;
541 v2.type = PrimitiveType::UINT;
542 eElem.properties.push_back(v2);
543 }
544 mElements.push_back(eElem);
545 }
546 }

548 std::string toString() const
549 {
550 std::string s;
551
552 s += "ply\nformat ";
553
554 switch (mFormat) {
555 case ply::ASCII: s += "ascii 1.0\n"; break;
556 case ply::BINARY_BIG_ENDIAN: s += "binary_big_endian 1.0\n"; break;
557 default: s += "binary_little_endian 1.0\n"; break;
558 }
559
560 s += "comment Generated by vclib\n";
561 for (const std::string& str : mTextureFiles) {
562 s += "comment TextureFile " + str + "\n";
563 }
564 for (const PlyElement& e : mElements) {
565 s += "element ";
566 switch (e.type) {
567 case ply::VERTEX:
568 s += "vertex " + std::to_string(e.numberElements) + "\n";
569 break;
570 case ply::FACE:
571 s += "face " + std::to_string(e.numberElements) + "\n";
572 break;
573 case ply::EDGE:
574 s += "edge " + std::to_string(e.numberElements) + "\n";
575 break;
576 case ply::TRISTRIP:
577 s += "tristrips " + std::to_string(e.numberElements) + "\n";
578 break;
579 case ply::MATERIAL:
580 s += "material " + std::to_string(e.numberElements) + "\n";
581 break;
582 case ply::OTHER:
583 s += e.unknownElementType + " " +
584 std::to_string(e.numberElements) + "\n";
585 break;
586 }
587 for (const PlyProperty& p : e.properties) {
588 s += "property ";
589 if (p.list) {
590 s += "list ";
591 s += typeToString(p.listSizeType) + " ";
592 }
593 s += typeToString(p.type) + " ";
594 if (p.name == ply::unknown)
595 s += p.unknownPropertyName + "\n";
596 else
597 s += nameToString(p.name) + "\n";
598 }
599 }
600 s += "end_header\n";
601
602 return s;
603 }

611private:
612 PlyElement readElement(const Tokenizer& lineTokenizer) const
613 {
614 PlyElement e;
615 Tokenizer::iterator token = lineTokenizer.begin();
616 std::string s = *(++token);
617 if (s == "vertex") {
618 e.type = ply::VERTEX;
619 e.numberElements = std::stoi(*(++token));
620 }
621 else if (s == "face") {
622 e.type = ply::FACE;
623 e.numberElements = std::stoi(*(++token));
624 }
625 else if (s == "edge") {
626 e.type = ply::EDGE;
627 e.numberElements = std::stoi(*(++token));
628 }
629 else if (s == "tristrips") {
630 e.type = ply::TRISTRIP;
631 e.numberElements = std::stoi(*(++token));
632 }
633 else {
634 e.type = ply::OTHER;
635 e.numberElements = std::stoi(*(++token));
636 e.unknownElementType = s;
637 }
638 return e;
639 }
640
641 PlyProperty readProperty(const Tokenizer& lineTokenizer) const
642 {
643 PlyProperty p;
644 Tokenizer::iterator token = lineTokenizer.begin();
645 std::string type = *(++token);
646 if (type == "list") {
647 p.list = true;
648 std::string typeSize = *(++token);
649 std::string typeData = *(++token);
650 std::string name = *(++token);
651 p.listSizeType = stringToType(typeSize);
652 p.type = stringToType(typeData);
653 p.name = stringToName(name);
654 if (p.name == ply::unknown)
655 p.unknownPropertyName = name;
656 }
657 else {
658 p.list = false;
659 std::string name = *(++token);
660 p.type = stringToType(type);
661 p.name = stringToName(name);
662 if (p.name == ply::unknown)
663 p.unknownPropertyName = name;
664 }
665
666 return p;
667 }

669 ply::PropertyName stringToName(const std::string& name) const
670 {
671 ply::PropertyName pn = ply::unknown;
672 if (name == "x")
673 pn = ply::x;
674 if (name == "y")
675 pn = ply::y;
676 if (name == "z")
677 pn = ply::z;
678 if (name == "nx")
679 pn = ply::nx;
680 if (name == "ny")
681 pn = ply::ny;
682 if (name == "nz")
683 pn = ply::nz;
684 if (name == "red")
685 pn = ply::red;
686 if (name == "green")
687 pn = ply::green;
688 if (name == "blue")
689 pn = ply::blue;
690 if (name == "alpha")
691 pn = ply::alpha;
692 if (name == "quality" || name == "scalar")
693 pn = ply::quality;
694 if (name == "texture_u" || name == "s" || name == "u")
695 pn = ply::texture_u;
696 if (name == "texture_v" || name == "t" || name == "v")
697 pn = ply::texture_v;
698 if (name == "texnumber")
699 pn = ply::texnumber;
700 if (name == "vertex_indices")
701 pn = ply::vertex_indices;
702 if (name == "texcoord")
703 pn = ply::texcoord;
704 if (name == "vertex1")
705 pn = ply::vertex1;
706 if (name == "vertex2")
707 pn = ply::vertex2;
708
709 return pn;
710 }

712 ply::PropertyType stringToType(const std::string& type) const
713 {
714 ply::PropertyType pt = ply::PropertyType::UCHAR;
715 if (type == "char")
716 pt = ply::PropertyType::CHAR;
717 if (type == "uchar")
718 pt = ply::PropertyType::UCHAR;
719 if (type == "short")
720 pt = ply::PropertyType::SHORT;
721 if (type == "ushort")
722 pt = ply::PropertyType::USHORT;
723 if (type == "int")
724 pt = ply::PropertyType::INT;
725 if (type == "uint")
726 pt = ply::PropertyType::UINT;
727 if (type == "float")
728 pt = ply::PropertyType::FLOAT;
729 if (type == "double")
730 pt = ply::PropertyType::DOUBLE;
731 return pt;
732 }

758
759 std::string typeToString(ply::PropertyType t) const
760 {
761 switch (t) {
762 case ply::PropertyType::CHAR: return "char";
763 case ply::PropertyType::UCHAR: return "uchar";
764 case ply::PropertyType::SHORT: return "short";
