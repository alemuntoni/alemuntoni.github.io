623 for (uint i = 0; i < adjFacesNumber(); ++i) {

629 {
630 if constexpr (N < 0) {

633 Base::resize(n);

635 for (uint i = 0; i < adjFacesNumber(); ++i) {
636 uint afi;
637 vcl::deserialize(is, afi);
638 setAdjFace(i, afi);
639 }

722/* importFrom function */
723
724template<
725 bool STORE_INDICES,
726 typename Face,
727 int N,
728 bool TTVN,
729 typename ParentElemType,
730 bool VERT,
731 bool OPT>
732template<typename Element>
733void AdjacentFaces<STORE_INDICES, Face, N, TTVN, ParentElemType, VERT, OPT>::
734 importFrom(const Element& e, bool importRefs)
735{
736 if (importRefs) {
737 if constexpr (HasAdjacentFaces<Element>) {
738 if (isAdjacentFacesAvailableOn(e)) {
739 if constexpr (N > 0) {
740 // same static size
741 if constexpr (N == Element::ADJ_FACE_NUMBER) {
742 importIndicesFrom(e);
743 }
744 // from dynamic to static, but dynamic size == static
745 // size
746 else if constexpr (Element::ADJ_FACE_NUMBER < 0) {
747 if (e.adjFacesNumber() == N) {
748 importIndicesFrom(e);
749 }
750 }
751 else {
752 // do not import in this case: cannot import from
753 // dynamic size != static size
754 }
755 }
756 else {
757 // from static/dynamic to dynamic size: need to resize
758 // first, then import
759 Base::resize(e.adjFacesNumber());
760 importIndicesFrom(e);
761 }
762 }
763 }
764 }
