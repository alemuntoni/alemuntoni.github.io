48template<Triangle3Concept TriangleType, Point3Concept PointType>
49auto halfSpaceDeterminant(const TriangleType& triangle, const PointType& point)
50{
51 return (triangle.point(1) - triangle.point(0))
52 .cross(triangle.point(2) - triangle.point(0))
53 .dot(point - triangle.point(0));
54}

69template<FaceConcept FaceType, Point3Concept PointType>
70auto halfSpaceDeterminant(const FaceType& face, const PointType& point)
71{
72 // - if it is a triangle, compute the determinant using the triangle
73 // - if it is a polygon, use its normal to compute the determinant
74 if constexpr (TriangleFaceConcept<FaceType>) {
75 return halfSpaceDeterminant(
76 face.vertex(0)->position(),
77 face.vertex(1)->position(),
78 face.vertex(2)->position(),
79 point);
80 }
81 else {
82 if (face.vertexNumber() == 3) {
83 return halfSpaceDeterminant(
84 face.vertex(0)->position(),
85 face.vertex(1)->position(),
86 face.vertex(2)->position(),
87 point);
88 }
89 else {
90 PointType n = faceNormal(face);
91 return n.dot(point - face.vertex(0)->position());
92 }
93 }
94}
