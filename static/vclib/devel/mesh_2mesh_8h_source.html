285 // will need it to update all the pointers of this mesh

287 NumberOfTypes<typename Mesh<Args...>::Containers>::value;
288 std::array<const void*, N_CONTAINERS> othBases =
289 Mesh<Args...>::getContainerBases(oth);

291 // update all the pointers contained on each container
292 // use the base pointer of each container of oth to update all the
293 // pointers in this mesh. Each pointer of oth that was copied in this
294 // mesh, will be updated computing its offset wrt the base of oth, and
295 // then adding that offset to the new base pointer of this mesh
296 (updateReferencesOfContainerType<Args>(*this, othBases), ...);

409 void append(const Mesh& m)
410 {
411 // for each container of this mesh, save its size
412 // will need it to update all the pointers/indices of this mesh
413 constexpr uint N_CONTAINERS =
414 NumberOfTypes<typename Mesh<Args...>::Containers>::value;
415 std::array<std::size_t, N_CONTAINERS> sizes =
416 Mesh<Args...>::getContainerSizes(*this);
417
418 // for each container of the other mesh m, save its bases
419 // will need it to update all the pointers of this mesh
420 std::array<const void*, N_CONTAINERS> bases =
421 Mesh<Args...>::getContainerBases(m);
422
423 // call the append function for each container
424 (appendContainer<Args>(m), ...);
425
426 // update all the pointers/indices contained on each container
427 (updateReferencesOfContainerTypeAfterAppend<Args>(*this, bases, sizes),
428 ...);
429
430 // manage transform matrix
431 if constexpr (HasTransformMatrix<Mesh<Args...>>) {
432 using Matrixtype = typename Mesh<Args...>::TransformMatrixType;
433
434 Matrixtype matrix = this->transformMatrix();
435 matrix = matrix.inverse().eval();
436 matrix *= m.transformMatrix();
437
438 if (matrix != Matrixtype::Identity()) {
439 (updatePosAndNormalsOfContainerTypeAfterAppend<Args>(
440 *this, sizes, matrix),
441 ...);
442 }
443 }
444
445 if constexpr (mesh::HasTexturePaths<Mesh<Args...>>) {
446 uint nTextures = this->textureNumber();
447
448 std::vector<uint> mapping(m.textureNumber());
449
450 for (uint i = 0; i < m.textureNumber(); ++i) {
451 uint tpi = this->indexOfTexturePath(m.texturePath(i));
452
453 if (tpi == UINT_NULL) {
454 if constexpr (mesh::HasTextureImages<Mesh<Args...>>) {
455 this->pushTexture(m.texture(i));
456 }
457 else {
458 this->pushTexturePath(m.texturePath(i));
459 }
460 mapping[i] = nTextures++;
461 }
462 else {
463 mapping[i] = tpi;
464 }
465 }
466
467 if (nTextures > 0 || mapping.size() > 0) {
468 (updateTextureIndicesOfContainerTypeAfterAppend<Args>(
469 *this, sizes, mapping),
470 ...);
471 }
472 }
473 }

489 template<typename OtherMeshType>
490 void importFrom(const OtherMeshType& m)
491 {
492 // This function will first:
493 // Call, for each Container and Component of the mesh, its importFrom
494 // function. In case of containers, it first creates the same number of
495 // elements in the container, and then calls the importFrom function for
496 // each new element.
497
498 (Args::importFrom(m), ...);
499
500 // Set to all element containers their parent mesh (this)
501 updateAllParentMeshPointers();
502
503 if constexpr (mesh::HasFaceContainer<Mesh<Args...>>) {
504 // Now I need to manage imports between different types of meshes
505 // (same type of meshes are already managed from importFrom member
506 // function).
507 //
508 // Generally speaking, Polygon meshes can import from any other type
509 // of mesh. We need to take care when this mesh has static vertex
510 // references number in the face container (VERTEX_NUMBER >= 3).
511 //
512 // The follwing case don't need to be managed:
513 // - import polygon mesh from triangle mesh
514 //
515 // I can manage the following cases:
516 // - import triangle mesh from polygon mesh: need triangulation
517 //
518 // I cannot manage the follwing cases:
519 // - import static non-triangle mesh from polygon mesh or from a
520 // mesh with different VERTEX_NUMBER
521
522 // in case of import from poly to triangle mesh, I need to manage
523 // triangulation of polygons and create additional triangle faces
524 // for each of the imported polygons. This function statically
525 // asserts that the import can be done.
526 using FaceContainer = typename Mesh<Args...>::FaceContainer;
527 FaceContainer::manageImportTriFromPoly(m);
528 }
529 }

535 void swap(Mesh& m2)
536 {
537 Mesh<Args...>& m1 = *this;
538
539 constexpr uint N_CONTAINERS =
540 NumberOfTypes<typename Mesh<Args...>::Containers>::value;
541 static_assert(N_CONTAINERS != 0);
542
543 // container bases of each container for m1 and m2
544 // we save the bases of the containers before swap
545 std::array<const void*, N_CONTAINERS> m1Bases =
546 Mesh<Args...>::getContainerBases(m1);
547 std::array<const void*, N_CONTAINERS> m2Bases =
548 Mesh<Args...>::getContainerBases(m2);
549
550 // actual swap of all the containers and the components of the mesh
551 // using pack expansion: swap will be called for each of the containers
552 // (or components!) that compose the Mesh
553 using std::swap;
554 (swap((Args&) m1, (Args&) m2), ...);
555
556 // Set to all elements their parent mesh
557 m1.updateAllParentMeshPointers();
558 m2.updateAllParentMeshPointers();
559
560 // update all the refs to m1 and m2: old base of m1 is now "old
561 // base" of m2, and viceversa
562 (updateReferencesOfContainerType<Args>(m1, m2Bases), ...);
563 (updateReferencesOfContainerType<Args>(m2, m1Bases), ...);
564 }
