44template<FaceConcept FaceType>
45typename FaceType::VertexType::PositionType faceNormal(const FaceType& f)
46{
47 using PositionType = FaceType::VertexType::PositionType;
48 if constexpr (TriangleFaceConcept<FaceType>) {
49 return Triangle<PositionType>::normal(
50 f.vertex(0)->position(),
51 f.vertex(1)->position(),
52 f.vertex(2)->position());
53 }
54 else {
55 if (f.vertexNumber() == 3) {
56 return Triangle<PositionType>::normal(
57 f.vertex(0)->position(),
58 f.vertex(1)->position(),
59 f.vertex(2)->position());
60 }
61 else {
62 return Polygon<PositionType>::normal(
63 f.vertices() | views::positions);
64 }
65 }
66}

177template<FaceConcept FaceType>
178auto faceAngleOnVertexRad(const FaceType& f, uint vi)
179{
180 const auto& p0 = f.vertex(vi)->position();
181 const auto& p1 = f.vertexMod((int) vi + 1)->position();
182 const auto& p2 = f.vertexMod((int) vi - 1)->position();
183 return (p2 - p0).angle(p1 - p0);
184}
