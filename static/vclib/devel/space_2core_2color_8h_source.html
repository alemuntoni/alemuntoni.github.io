228 uint8_t hsvHue() const
229 {
230 uint8_t rgbMin, rgbMax;
231 uint8_t h;
232
233 rgbMin = x() < y() ? (x() < z() ? x() : z()) : (y() < z() ? y() : z());
234 rgbMax = x() > y() ? (x() > z() ? x() : z()) : (y() > z() ? y() : z());
235
236 if (rgbMax == 0) {
237 return 0;
238 }
239
240 if (255 * long(rgbMax - rgbMin) / rgbMax == 0) {
241 return 0;
242 }
243
244 if (rgbMax == x())
245 h = 0 + 43 * (y() - z()) / (rgbMax - rgbMin);
246 else if (rgbMax == y())
247 h = 85 + 43 * (z() - x()) / (rgbMax - rgbMin);
248 else
249 h = 171 + 43 * (x() - y()) / (rgbMax - rgbMin);
250
251 return h;
252 }

471 void setHsv(uint8_t h, uint8_t s, uint8_t v, uint8_t alpha = 255)
472 {
473 w() = alpha;
474 if (s == 0) {
475 x() = v;
476 y() = v;
477 z() = v;
478 }
479 else {
480 h = (h / 360.0) * 255;
481 uint8_t region, remainder, p, q, t;
482 region = h / 43;
483 remainder = (h - (region * 43)) * 6;
484
485 p = (v * (255 - s)) >> 8;
486 q = (v * (255 - ((s * remainder) >> 8))) >> 8;
487 t = (v * (255 - ((s * (255 - remainder)) >> 8))) >> 8;
488
489 switch (region) {
490 case 0:
491 x() = v;
492 y() = t;
493 z() = p;
494 break;
495 case 1:
496 x() = q;
497 y() = v;
498 z() = p;
499 break;
500 case 2:
501 x() = p;
502 y() = v;
503 z() = t;
504 break;
505 case 3:
506 x() = p;
507 y() = q;
508 z() = v;
509 break;
510 case 4:
511 x() = t;
512 y() = p;
513 z() = v;
514 break;
515 default:
516 x() = v;
517 y() = p;
518 z() = q;
519 break;
520 }
521 }
522 }

664inline Color colorLerp(const Color& c0, const Color& c1, float value)
665{
666 Color c;
667 if (value < 0) // out of range - left
668 value = 0;
669 else if (value > 1) // out of range - right
670 value = 1;
671 for (uint i = 0; i < 4; i++)
672 c(i) = c1(i) * value + c0(i) * (1 - value);
673 return c;
674}

740inline Color colorFromIntervalParula(float value)
741{
742 if (value < 0) // out of range - left
743 value = 0;
744 else if (value > 1) // out of range - right
745 value = 1;
746
747 static uint paruVal[9] = {
748 0x271680ff,
749 0x0363e1ff,
750 0x1485d4ff,
751 0x06a7c6ff,
752 0x38b99eff,
753 0x92bf73ff,
754 0xd9ba56ff,
755 0xfcce2eff,
756 0xfffa0aff};
757
758 int ind = int(floor(value * 8.0f));
759 float div = (value * 8.0f - ind);
760
761 if (div < 0)
762 div = 0;
763 else if (div > 1)
764 div = 1;
765
766 return colorLerp(
767 (Color::ColorABGR) paruVal[ind],
768 (Color::ColorABGR) paruVal[ind + 1],
769 div);
770}

911inline std::vector<Color> colorScattering(
912 uint n,
913 float sat = 0.3,
914 float val = 0.9)
915{
916 std::vector<Color> scattering;
917 scattering.reserve(n);
918
919 for (uint v = 0; v < n; ++v) {
920 uint value = v, m = n;
921 uint b = 0;
922
923 for (uint k = 1; k < n; k <<= 1) {
924 if (value << 1 >= m) {
925 b += k;
926 value -= (m + 1) >> 1;
927 m >>= 1;
928 }
929 else {
930 m = (m + 1) >> 1;
931 }
932 }
933
934 Color rc;
935 rc.setHsvF(float(b) / float(n), sat, val);
936 scattering.push_back(rc);
937 }
938 return scattering;
939}
