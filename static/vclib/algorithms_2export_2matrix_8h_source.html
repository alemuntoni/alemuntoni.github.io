56template<MatrixConcept Matrix, MeshConcept MeshType>
57Matrix vertexMatrix(const MeshType& mesh)
58{
59 Matrix V(mesh.vertexNumber(), 3);
60
61 uint i = 0;
62 for (const auto& v : mesh.vertices()) {
63 for (uint j = 0; j < 3; ++j) {
64 V(i, j) = v.coord()[j];
65 }
66 ++i;
67 }

102template<MatrixConcept Matrix, FaceMeshConcept MeshType>
103Matrix faceMatrix(const MeshType& mesh)
104{
105 vcl::requireVertexContainerCompactness(mesh);
106
107 Matrix F(mesh.faceNumber(), 3);
108
109 uint i = 0;
110 for (const auto& f : mesh.faces()){
111 // check if this face is greater than the cols of the matrix
112 if (f.vertexNumber() > F.cols()) { // need to resize
113 uint oldCols = F.cols(); // save old cols number
114 F.conservativeResize(F.rows(), f.vertexNumber());
115 // need to set to -1 all the previous rows that have been resized
116 for (uint k = 0; k < i; ++k){
117 for (uint j = oldCols; j < F.cols(); ++j)
118 F(k, j) = -1;
119 }
120 }
121 uint j = 0;
122 for (const auto* v : f.vertices()){
123 F(i, j) = mesh.index(v);
124 j++;
125 }
126 for (; j < F.cols(); ++j) // remaining vertices set to -1
127 F(i, j ) = -1;
128 ++i; // go to next face/row
129 }
130 return F;
131}

206template<MatrixConcept Matrix, EdgeMeshConcept MeshType>
207Matrix edgeMatrix(const MeshType &mesh)
208{
209 vcl::requireVertexContainerCompactness(mesh);
210
211 Matrix E(mesh.edgeNumber(), 2);
212
213 uint i = 0;
214 for (const auto& e : mesh.edges()){
215 E(i, 0) = mesh.index(e.vertex(0));
216 E(i, 1) = mesh.index(e.vertex(1));
217 ++i; // go to next edge/row
218 }
219 return E;
220}
