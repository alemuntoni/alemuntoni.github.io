149template<MeshConcept MeshType>
150void normalizePerVertexQuality(
151 MeshType& m,
152 typename MeshType::VertexType::QualityType minS = 0,
153 typename MeshType::VertexType::QualityType maxS = 1)
154{
155 vcl::requirePerVertexQuality(m);
156
157 using VertexType = MeshType::VertexType;
158 using QualityType = VertexType::QualityType;
159
160 QualityType range = maxS - minS;
161 std::pair<QualityType, QualityType> p = vertexQualityMinMax(m);
162
163 for (VertexType& v : m.vertices()) {
164 v.quality() =
165 minS + range * ((v.quality() - p.first) / (p.second - p.first));
166 }
167}

319template<MeshConcept MeshType>
320void setPerVertexQualityFromPrincipalCurvatureShapeIndex(MeshType& m)
321{
322 vcl::requirePerVertexQuality(m);
323 vcl::requirePerVertexPrincipalCurvature(m);
324
325 using VertexType = MeshType::VertexType;
326 using ScalarType = VertexType::PrincipalCurvatureType::ScalarType;
327
328 for (VertexType& v : m.vertices()) {
329 ScalarType k1 = v.principalCurvature().maxValue();
330 ScalarType k2 = v.principalCurvature().minValue();
331 if (k1 < k2) std::swap(k1, k2);
332 v.quality() = (2.0 / M_PI) * std::atan2(k1 + k2, k1 - k2);
333 }
334}

347template<MeshConcept MeshType>
348void setPerVertexQualityFromPrincipalCurvatureCurvedness(MeshType& m)
349{
350 vcl::requirePerVertexQuality(m);
351 vcl::requirePerVertexPrincipalCurvature(m);
352
353 using VertexType = MeshType::VertexType;
354 using ScalarType = VertexType::PrincipalCurvatureType::ScalarType;
355
356 for (VertexType& v : m.vertices()) {
357 ScalarType k1 = v.principalCurvature().maxValue();
358 ScalarType k2 = v.principalCurvature().minValue();
359
360 v.quality() = std::sqrt((k1 * k1 + k2 * k2) / 2.0);
361 }
362}
