162 uint cell(uint d, const Scalar& s) const
163 {
164 if (s < mBBox.min()(d))
165 return 0;
166 if (s > mBBox.max()(d))
167 return cellNumber(d) - 1;
168 Scalar t = s - mBBox.min()(d);

188 BBoxType cellBox(const CellCoord& c) const
189 {
190 BBoxType b;
191
192 Point<Scalar, N> p;
193 for (size_t i = 0; i < DIM; ++i)
194 p(i) = (Scalar) c(i) * cellLength(i);
195 p += mBBox.min();
196 b.min() = p;
197 b.max() = (p + cellLengths());

200 }

251template<PointConcept PointType>
252Point<uint, PointType::DIM> bestGridSize(
253 const PointType& lengths,
254 uint nElements)
255{
256 using Scalar = PointType::ScalarType;
257
258 static const int DIM = PointType::DIM;
259
260 const uint mincells = 1; // Numero minimo di celle
261 const Scalar GFactor = 1; // GridEntry = NumElem*GFactor
262 const Scalar diag = lengths.norm(); // Diagonale del box
263 const Scalar eps = diag * 1e-4; // Fattore di tolleranza
264 const uint ncell = nElements * GFactor; // Calcolo numero di voxel
265
266 Point<uint, PointType::DIM> sizes;
267 sizes.setConstant(mincells);
268
269 bool sanityCheckLengths = true;
270 uint lessEpsDimNumber = 0;
271 std::array<bool, DIM> isLessEps;
272
273 for (uint i = 0; i < DIM; i++) {
274 sanityCheckLengths = sanityCheckLengths && lengths(i) > 0.0;
275 if (lengths(i) < eps) {
276 lessEpsDimNumber++;
277 isLessEps[i] = true;
278 }
279 }
280 uint greaterEpsDimNumber = DIM - lessEpsDimNumber;
281
282 if (nElements > 0 && sanityCheckLengths) {
283 // no lenghts less than epsilon - standard computation for sizes
284 if (greaterEpsDimNumber == DIM) {
285 Scalar product = 1;
286 for (uint i = 0; i < DIM; i++)
287 product *= lengths(i);
288
289 Scalar k = std::pow((ncell / product), (1.0 / DIM));
290
291 for (uint i = 0; i < DIM; i++)
292 sizes(i) = int(lengths(i) * k);
293 }
294 // at least one lenght is less than epsilon
295 else {
296 for (uint i = 0; i < DIM; i++) {
297 if (isLessEps[i]) { // the ith dimension is less than epsilon
298 sizes(i) = 1;
299 }
300 else {
301 // compute the product between all the dimensions that are
302 // not i and not less than epsilon
303 Scalar product = 1;
304 for (uint j = 0; j < DIM; j++)
305 if (j != i && !isLessEps[j])
306 product *= lengths(j);
307
308 // ith dimension size
309 sizes(i) = std::pow(
310 (ncell * lengths(i) / product),
311 (1.0 / greaterEpsDimNumber));
312 }
313 }
314 }
315
316 for (uint i = 0; i < DIM; i++)
317 sizes(i) = std::max(sizes(i), (uint) 1);
318 }
319
320 return sizes;
321}
