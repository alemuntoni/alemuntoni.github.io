194 BBoxType b;
195
196 Point<Scalar, N> p;
197 for (size_t i = 0; i < DIM; ++i)
198 p(i) = (Scalar)c(i)*cellLength(i);
199 p += bbox.min();

255template<PointConcept PointType>
256Point<uint, PointType::DIM>
257bestGridSize(const PointType& lengths, uint nElements)
258{
259 using Scalar = PointType::ScalarType;
260
261 static const int DIM = PointType::DIM;
262
263 const uint mincells = 1; // Numero minimo di celle
264 const Scalar GFactor = 1; // GridEntry = NumElem*GFactor
265 const Scalar diag = lengths.norm(); // Diagonale del box
266 const Scalar eps = diag * 1e-4; // Fattore di tolleranza
267 const uint ncell = nElements * GFactor; // Calcolo numero di voxel
268
269 Point<uint, PointType::DIM> sizes;
270 sizes.setConstant(mincells);
271
272 bool sanityCheckLengths = true;
273 uint lessEpsDimNumber = 0;
274 std::array<bool, DIM> isLessEps;
275
276 for (uint i = 0; i < DIM; i++) {
277 sanityCheckLengths = sanityCheckLengths && lengths(i) > 0.0;
278 if (lengths(i) < eps) {
279 lessEpsDimNumber++;
280 isLessEps[i] = true;
281 }
282 }
283 uint greaterEpsDimNumber = DIM - lessEpsDimNumber;
284
285 if (nElements > 0 && sanityCheckLengths) {
286 // no lenghts less than epsilon - standard computation for sizes
287 if (greaterEpsDimNumber == DIM) {
288 Scalar product = 1;
289 for (uint i = 0; i < DIM; i++)
290 product *= lengths(i);
291
292 Scalar k = std::pow((ncell/product), (1.0/DIM));
293
294 for (uint i = 0; i < DIM; i++)
295 sizes(i) = int(lengths(i) * k);
296 }
297 // at least one lenght is less than epsilon
298 else {
299 for (uint i = 0; i < DIM; i++) {
300 if (isLessEps[i]) { // the ith dimension is less than epsilon
301 sizes(i) = 1;
302 }
303 else {
304 // compute the product between all the dimensions that are
305 // not i and not less than epsilon
306 Scalar product = 1;
307 for (uint j = 0; j < DIM; j++)
308 if (j != i && !isLessEps[j])
309 product *= lengths(j);
310
311 // ith dimension size
312 sizes(i) = std::pow(
313 (ncell * lengths(i) / product),
314 (1.0 / greaterEpsDimNumber));
315 }
316 }
317 }
318
319 for (uint i = 0; i < DIM; i++)
320 sizes(i) = std::max(sizes(i), (uint)1);
321 }
322
323 return sizes;
324}
