35template<MatrixConcept FMatrix>
36std::vector<uint> faceVertIndices(const FMatrix& faces, uint f)
37{
38 std::vector<uint> fVerts;
39
40 uint j = 0;
41 while (j < faces.cols() && faces(f, j) != -1 && faces(f, j) != UINT_NULL)
42 fVerts.push_back(faces(f, j));
43
44 return fVerts;
45}

47template<uint ELEM_ID, MeshConcept MeshType, MatrixConcept NMatrix>
48void importElementNormalsFromMatrix(MeshType& mesh, const NMatrix& normals)
49{
50 // The type of the normal of the element
51 using NormalType = MeshType::template ElementType<ELEM_ID>::NormalType;
52
53 if (normals.cols() != 3)
54 throw WrongSizeException(
55 "The input " + elementEnumString<ELEM_ID>() +
56 " normal matrix must have 3 columns");
57
58 // matrix rows must be equal to the number of elements of the given type
59 if (normals.rows() != mesh.template number<ELEM_ID>())
60 throw WrongSizeException(
61 "The input normal matrix must have the same number of rows "
62 "as the number of " +
63 elementEnumString<ELEM_ID>() + " element in the mesh");
64
65 enableIfPerElementComponentOptional<ELEM_ID, CompId::NORMAL>(mesh);
66 requirePerElementComponent<ELEM_ID, CompId::NORMAL>(mesh);
67
68 uint i = 0;
69 for (auto& e : mesh.template elements<ELEM_ID>()) {
70 e.normal() = NormalType(normals(i, 0), normals(i, 1), normals(i, 2));
71 i++;
72 }
73}

75template<uint ELEM_ID, MeshConcept MeshType, MatrixConcept CMatrix>
76void importElementColorsFromMatrix(MeshType& mesh, const CMatrix& colors)
77{
78 using MatrixScalar = CMatrix::Scalar;
79
80 if (colors.cols() != 3 && colors.cols() != 4)
81 throw WrongSizeException(
82 "The input " + elementEnumString<ELEM_ID>() +
83 " color matrix must have 3 or 4 columns");
84
85 // matrix rows must be equal to the number of elements of the given type
86 if (colors.rows() != mesh.template number<ELEM_ID>())
87 throw WrongSizeException(
88 "The input color matrix must have the same number of rows "
89 "as the number of " +
90 elementEnumString<ELEM_ID>() + " element in the mesh");
91
92 enableIfPerElementComponentOptional<ELEM_ID, CompId::COLOR>(mesh);
93 requirePerElementComponent<ELEM_ID, CompId::COLOR>(mesh);
94
95 uint i = 0;
96 for (auto& e : mesh.template elements<ELEM_ID>()) {
97 // Matrix has colors in range 0-255
98 if constexpr (std::integral<MatrixScalar>) {
99 if (colors.cols() == 3) {
100 e.color() =
101 vcl::Color(colors(i, 0), colors(i, 1), colors(i, 2));
102 }
103 else {
104 e.color() = vcl::Color(
105 colors(i, 0), colors(i, 1), colors(i, 2), colors(i, 3));
106 }
107 }
108 else { // Matrix has colors in range 0-1
109 if (colors.cols() == 3) {
110 e.color() = vcl::Color(
111 colors(i, 0) * 255, colors(i, 1) * 255, colors(i, 2) * 255);
112 }
113 else {
114 e.color() = vcl::Color(
115 colors(i, 0) * 255,
116 colors(i, 1) * 255,
117 colors(i, 2) * 255,
118 colors(i, 3) * 255);
119 }
120 }
121
122 i++;
123 }
124}

307template<
308 MeshConcept MeshType,
309 MatrixConcept VMatrix,
310 MatrixConcept FMatrix = Eigen::MatrixX3i,
311 MatrixConcept EMatrix = Eigen::MatrixX2i,
312 MatrixConcept VNMatrix = Eigen::MatrixX3d,
313 MatrixConcept FNMatrix = Eigen::MatrixX3d>
314void importMeshFromMatrices(
315 MeshType& mesh,
316 const VMatrix& vertices,
317 const FMatrix& faces = FMatrix(),
318 const EMatrix& edges = EMatrix(),
319 const VNMatrix& vertexNormals = VNMatrix(),
320 const FNMatrix& faceNormals = FNMatrix())
321{
322 mesh.clear();
323 mesh.disableAllOptionalComponents();
324
325 importVerticesFromMatrix(mesh, vertices);
326
327 if constexpr (HasPerVertexNormal<MeshType>) {
328 if (vertexNormals.rows() > 0) {
329 importVertexNormalsFromMatrix(mesh, vertexNormals);
330 }
331 }
332
333 if constexpr (HasFaces<MeshType>) {
334 if (faces.rows() > 0)
335 importFacesFromMatrix(mesh, faces);
336
337 if constexpr (HasPerFaceNormal<MeshType>) {
338 if (faceNormals.rows() > 0) {
339 importFaceNormalsFromMatrix(mesh, faceNormals);
340 }
341 }
342 }
343
344 if constexpr (HasEdges<MeshType>) {
345 if (edges.rows() > 0)
346 importEdgesFromMatrix(mesh, edges);
347 }
348}

388template<MeshConcept MeshType, MatrixConcept VMatrix>
389void importVerticesFromMatrix(
390 MeshType& mesh,
391 const VMatrix& vertices,
392 bool clearBeforeSet = true)
393{
394 using CoordType = MeshType::VertexType::CoordType;
395
396 if (vertices.cols() != 3)
397 throw WrongSizeException("The input vertex matrix must have 3 columns");
398
399 if (clearBeforeSet) {
400 mesh.clearVertices();
401 mesh.resizeVertices(vertices.rows());
402 }
403 else {
404 if (vertices.rows() != mesh.vertexNumber()) {
405 throw WrongSizeException(
406 "The input vertex matrix has a different number of rows than "
407 "the number of vertices of the mesh");
408 }
409 }
410
411 uint i = 0;
412 for (auto& v : mesh.vertices()) {
413 v.coord() = CoordType(vertices(i, 0), vertices(i, 1), vertices(i, 2));
414 i++;
415 }
416}

418template<FaceMeshConcept MeshType, MatrixConcept FMatrix>
419void importFacesFromMatrix(
420 MeshType& mesh,
421 const FMatrix& faces,
422 bool clearBeforeSet = true)
423{
424 if (clearBeforeSet) {
425 mesh.clearFaces();
426 mesh.resizeFaces(faces.rows());
427 }
428 else {
429 if (faces.rows() != mesh.faceNumber()) {
430 throw WrongSizeException(
431 "The input face matrix has a different number of rows "
432 "than the number of faces of the mesh.");
433 }
434 }
435
436 if constexpr (HasPolygons<MeshType>) {
437 uint i = 0;
438 for (auto& f : mesh.faces()) {
439 uint vertexNumber = 0;
440
441 // count the number of vertices of the face
442 while (vertexNumber < faces.cols() &&
443 faces(i, vertexNumber) != -1 &&
444 faces(i, vertexNumber) != UINT_NULL)
445 vertexNumber++;
446
447 f.resizeVertices(vertexNumber);
448
449 for (uint j = 0; j < vertexNumber; ++j)
450 f.setVertex(j, faces(i, j));
451 ++i;
452 }
453 }
454 else { // the vertex number of mesh faces is fixed
455 using FaceType = MeshType::FaceType;
456
457 constexpr int VN = FaceType::VERTEX_NUMBER;
458 if (faces.cols() == VN) { // faces of matrix and mesh have same size
459 uint i = 0;
460 for (auto& f : mesh.faces()) {
461 for (uint j = 0; j < VN; ++j)
462 f.setVertex(j, faces(i, j));
463 ++i;
464 }
465 }
466 else { // faces of matrix and mesh have different sizes
467 // matrix cols is higher than 3 (polygons), but we have a triangle
468 // mesh and we can triangulate the faces
469 if constexpr (VN == 3) {
470 if (!clearBeforeSet) {
471 throw WrongSizeException(
472 "Cannot import the input face matrix into the mesh "
473 "without clearing the face container first "
474 "(need to perform a triangulation to import polygons "
475 "in a triangle mesh, and this operation that does not "
476 "guarantee a predefined number of faces is required).");
477 }
478 mesh.reserveFaces(faces.rows());
479 for (uint i = 0; i < faces.rows(); ++i) {
480 std::vector<uint> fVertIndices =
481 detail::faceVertIndices(faces, i);
482
483 addTriangleFacesFromPolygon(mesh, fVertIndices);
484 }
485 }
486 else {
487 // no triangulation available because VN != 3, we don't
488 // know how to import the faces and we throw an exception
489 throw WrongSizeException(
490 "The input face matrix has a different number of columns "
491 "than the number of vertices of the mesh faces.");
492 }
493 }
494 }
495}

497template<EdgeMeshConcept MeshType, MatrixConcept EMatrix>
498void importEdgesFromMatrix(
499 MeshType& mesh,
500 const EMatrix& edges,
501 bool clearBeforeSet = true)
502{
503 if (edges.cols() != 2)
504 throw WrongSizeException("The input edge matrix must have 2 columns");
505
506 if (clearBeforeSet) {
507 mesh.clearEdges();
508 mesh.resizeEdges(edges.rows());
509 }
510 else {
511 if (edges.rows() != mesh.edgeNumber()) {
512 throw WrongSizeException(
513 "The input edge matrix has a different number of rows than "
514 "the number of edges of the mesh");
515 }
516 }
517
518 uint i = 0;
519 for (auto& e : mesh.edges()) {
520 e.setVertex(0, edges(i, 0));
521 e.setVertex(1, edges(i, 1));
522 i++;
523 }
524}
