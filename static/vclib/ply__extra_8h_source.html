56inline void readUnknownElements(
57 std::ifstream& file,
58 const PlyHeader& header,
59 ply::Element el)
60{
61 if (header.format() == ply::ASCII) {
62 for (uint i = 0; i < el.numberElements; ++i) {
63 io::internal::nextNonEmptyTokenizedLine(file);
64 }
65 }
66 else {
67 for (uint i = 0; i < el.numberElements; ++i) {
68 for (const Property& p : el.properties) {
69 if (p.list) {
70 uint s =
71 io::internal::readProperty<int>(file, p.listSizeType);
72 for (uint i = 0; i < s; ++i)
73 io::internal::readProperty<int>(file, p.type);
74 }
75 else {
76 io::internal::readProperty<int>(file, p.type);
77 }
78 }
79 }
80 }
