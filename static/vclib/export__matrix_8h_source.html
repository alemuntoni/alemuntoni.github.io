73template<MatrixConcept Matrix, MeshConcept MeshType>
74Matrix vertexCoordsMatrix(const MeshType& mesh)
75{
76 Matrix vM(mesh.vertexNumber(), 3);
77
78 MatrixStorageType stg = MatrixStorageType::ROW_MAJOR;
79
80 // Eigen matrices can be column major
81 if constexpr (EigenMatrixConcept<Matrix>) {
82 if constexpr (!Matrix::IsRowMajor) {
83 stg = MatrixStorageType::COLUMN_MAJOR;
84 }
85 }
86
87 vertexCoordsToBuffer(mesh, vM.data(), stg);
88
89 return vM;
90}
