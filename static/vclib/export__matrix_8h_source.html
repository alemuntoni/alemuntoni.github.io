55template<MatrixConcept Matrix, MeshConcept MeshType>
56Matrix vertexMatrix(const MeshType& mesh)
57{
58 Matrix V(mesh.vertexNumber(), 3);
59
60 uint i = 0;
61 for (const auto& v : mesh.vertices()) {
62 for (uint j = 0; j < 3; ++j) {

65 ++i;
66 }

101template<MatrixConcept Matrix, FaceMeshConcept MeshType>
102Matrix faceMatrix(const MeshType& mesh)
103{
104 vcl::requireVertexContainerCompactness(mesh);
105
106 Matrix F(mesh.faceNumber(), 3);
107
108 uint i = 0;
109 for (const auto& f : mesh.faces()) {
110 // check if this face is greater than the cols of the matrix
111 if (f.vertexNumber() > F.cols()) { // need to resize
112 uint oldCols = F.cols(); // save old cols number
113 F.conservativeResize(F.rows(), f.vertexNumber());
114 // need to set to -1 all the previous rows that have been resized
115 for (uint k = 0; k < i; ++k) {
116 for (uint j = oldCols; j < F.cols(); ++j)
117 F(k, j) = -1;
118 }
119 }
120 uint j = 0;
121 for (const auto* v : f.vertices()) {
122 F(i, j) = mesh.index(v);
123 j++;
124 }
125 for (; j < F.cols(); ++j) // remaining vertices set to -1
126 F(i, j) = -1;
127 ++i; // go to next face/row
128 }
129 return F;
130}

205template<MatrixConcept Matrix, EdgeMeshConcept MeshType>
206Matrix edgeMatrix(const MeshType& mesh)
207{
208 vcl::requireVertexContainerCompactness(mesh);
209
210 Matrix E(mesh.edgeNumber(), 2);
211
212 uint i = 0;
213 for (const auto& e : mesh.edges()) {
214 E(i, 0) = mesh.index(e.vertex(0));
215 E(i, 1) = mesh.index(e.vertex(1));
216 ++i; // go to next edge/row
217 }
218 return E;
219}
