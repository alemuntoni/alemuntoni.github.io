129/* Detector function to check if a class has AdjacentFaces enabled */
130
131template <typename T>
132bool isAdjacentFacesEnabledOn(const T& element)
133{
134 if constexpr (HasOptionalAdjacentFaces<T>) {
135 return element.isAdjFacesEnabled();
136 }
137 else {
138 return HasAdjacentFaces<T>;
139 }
