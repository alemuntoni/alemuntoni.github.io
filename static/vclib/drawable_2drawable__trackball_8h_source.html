38class DrawableTrackBall : public DrawableObject
39{
40 inline static const uint N_POINTS = 128;
41 inline static const auto TRACKBALL_DATA =
42 createTrackBall<float, uint16_t>(1.0, N_POINTS);
43
44 bool mVisible = true;
45
46 VertexBuffer mVertexCoordsBuffer;
47 IndexBuffer mEdgeIndexBuffer;
48
49 // TODO: can we be sure that this is called after the context initialization
50 // triggered by a window?
51 bgfx::ProgramHandle mProgram =
52 Context::instance().programManager().getProgram(
53 VclProgram::DRAWABLE_TRACKBALL);
54
55 DrawableTrackballUniforms mUniforms;
56
57 vcl::Matrix44f mTransform = vcl::Matrix44f::Identity();
58
59public:
60 DrawableTrackBall()
61 {
62 mUniforms.setNumberOfVerticesPerAxis(N_POINTS);
63
64 createBuffers();
65 }
66
67 DrawableTrackBall(const DrawableTrackBall& other) :
68 mVisible(other.mVisible), mProgram(other.mProgram),
69 mUniforms(other.mUniforms), mTransform(other.mTransform)
70 {
71 // copy all the members that can be copied, and then re-create the
72 // buffers
73 createBuffers();
74 }
75
76 // default move constructor - buffers can be moved
77 DrawableTrackBall(DrawableTrackBall&& other) = default;
78
79 // default destructor - buffers are destroyed by their destructor
80 ~DrawableTrackBall() = default;
81
88 void swap(DrawableTrackBall& other)
89 {
90 using std::swap;
91 swap(mVisible, other.mVisible);
92 swap(mVertexCoordsBuffer, other.mVertexCoordsBuffer);
93 swap(mEdgeIndexBuffer, other.mEdgeIndexBuffer);
94 swap(mProgram, other.mProgram);
95 swap(mUniforms, other.mUniforms);
96 swap(mTransform, other.mTransform);
97 }
98
99 friend void swap(DrawableTrackBall& a, DrawableTrackBall& b) { a.swap(b); }
100
106 void updateDragging(bool isDragging) { mUniforms.setDragging(isDragging); }
107
108 void setTransform(const vcl::Matrix44f& mtx) { mTransform = mtx; }
109
110 // copy and swap idiom
111 DrawableTrackBall& operator=(DrawableTrackBall other)
112 {
113 swap(other);
114 return *this;
115 }
116
117 // DrawableObject interface
118
119 void draw(uint viewId) const override
120 {
121 if (isVisible()) {
122 if (bgfx::isValid(mProgram)) {
123 bgfx::setState(
124 0 | BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_Z |
125 BGFX_STATE_DEPTH_TEST_LESS | BGFX_STATE_PT_LINES |
126 BGFX_STATE_BLEND_NORMAL);
127
128 mVertexCoordsBuffer.bind(0);
129 mEdgeIndexBuffer.bind();
130
131 bgfx::setTransform(mTransform.data());
132
133 mUniforms.bind();
134
135 bgfx::submit(viewId, mProgram);
136 }
137 }
138 }
139
140 Box3d boundingBox() const override { return Box3d(); }
141
142 std::shared_ptr<DrawableObject> clone() const override
143 {
144 return std::make_shared<DrawableTrackBall>(*this);
145 }
146
147 bool isVisible() const override { return mVisible; }
148
149 void setVisibility(bool vis) override { mVisible = vis; }
150
151private:
152 void createBuffers()
153 {
154 // vertex buffer
155 mVertexCoordsBuffer.set(
156 TRACKBALL_DATA.first.data(),
157 TRACKBALL_DATA.first.size() * 3,
158 bgfx::Attrib::Position,
159 3,
160 PrimitiveType::FLOAT);
161
162 // edge index buffer
163 mEdgeIndexBuffer.set(
164 TRACKBALL_DATA.second.data(), TRACKBALL_DATA.second.size(), false);
165 }
166};
