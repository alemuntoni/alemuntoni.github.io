41template<VertexConcept VertexType, MeshConcept MeshType>
42ObjMaterial objMaterialFromVertex(const VertexType& v, const MeshInfo& fi)
43{
44 ObjMaterial mat;
45 if constexpr (HasPerVertexColor<MeshType>) {
46 if (fi.hasVertexColors()) {
47 mat.hasColor = true;
48 mat.Kd.x() = v.color().redF();
49 mat.Kd.y() = v.color().greenF();
50 mat.Kd.z() = v.color().blueF();
51 }
52 }
53 return mat;
54}

56template<FaceConcept FaceType, MeshConcept MeshType>
57ObjMaterial objMaterialFromFace(
58 const FaceType& f,
59 const MeshType& m,
60 const MeshInfo& fi)
61{
62 ObjMaterial mat;
63 if constexpr (HasPerFaceColor<MeshType>) {
64 if (fi.hasFaceColors()) {
65 mat.hasColor = true;
66 mat.Kd.x() = f.color().redF();
67 mat.Kd.y() = f.color().greenF();
68 mat.Kd.z() = f.color().blueF();
69 }
70 }
71 if constexpr (HasPerFaceWedgeTexCoords<MeshType>) {
72 if (fi.hasFaceWedgeTexCoords()) {
73 mat.hasTexture = true;
74 if constexpr (HasTexturePaths<MeshType>) {
75 mat.map_Kd = m.texturePath(f.textureIndex());
76 }
77 if constexpr (HasTextureImages<MeshType>) {
78 mat.map_Kd = m.texture(f.textureIndex()).path();
79 }
80 }
81 }
82 return mat;
83}

85template<
86 ElementConcept ElementType,
87 MeshConcept MeshType,
88 LoggerConcept LogType = NullLogger>
89void writeElementObjMaterial(
90 const ElementType& e,
91 const MeshType& m,
92 const MeshInfo& fi,
93 ObjMaterial& lastMaterial,
94 std::map<ObjMaterial, std::string>& materialMap,
95 std::ostream& fp,
96 std::ostream& mtlfp,
97 const SaveSettings& settings,
98 LogType& log = nullLogger)
99{
100 ObjMaterial mat;
101 constexpr bool EL_IS_VERTEX = ElementType::ELEMENT_ID == ElemId::VERTEX;
102 constexpr bool EL_IS_FACE = ElementType::ELEMENT_ID == ElemId::FACE;
103
104 if constexpr (EL_IS_VERTEX) {
105 mat = objMaterialFromVertex<typename MeshType::VertexType, MeshType>(
106 e, fi);
107 }
108 if constexpr (EL_IS_FACE) {
109 mat = objMaterialFromFace(e, m, fi);
110 }
111 if (!mat.isEmpty()) {
112 static const std::string MATERIAL_PREFIX = "MATERIAL_";
113 std::string mname; // name of the material of the vertex
114 auto it = materialMap.find(mat);
115 if (it == materialMap.end()) { // if it is a new material
116 // add the new material to the map
117 mname = MATERIAL_PREFIX + std::to_string(materialMap.size());
118 materialMap[mat] = mname;
119 // save the material in the mtl file
120 mtlfp << "newmtl " << mname << std::endl;
121 mtlfp << mat << std::endl;
122 if constexpr (HasTextureImages<MeshType>) {
123 if (settings.saveTextureImages && mat.hasTexture) {
124 // we need to save the texture image
125 // first, get the index of the texture: 0 if vertex,
126 // textureIndex if face
127 uint textureIndex = 0;
128 if constexpr (EL_IS_FACE) {
129 textureIndex = e.textureIndex();
130 }
131 const Texture& t = m.texture(textureIndex);
132 try {
133 t.image().save(m.meshBasePath() + mat.map_Kd);
134 }
135 catch (const std::runtime_error& e) {
136 log.log(LogType::WARNING, e.what());
137 }
138 }
139 }
140 }
141 else { // get the name of the material
142 mname = it->second;
143 }
144 // if the material of the vertex is different from the last used, need
145 // to add usemtl
146 if (mat != lastMaterial) {
147 lastMaterial = mat;
148 fp << "usemtl " << mname << std::endl;
149 }
150 }
151}

153template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
154void saveObj(
155 const MeshType& m,
156 const std::string& filename,
157 std::ostream& fp,
158 std::ostream* mtlfp,
159 bool saveMtlFile,
160 LogType& log = nullLogger,
161 const SaveSettings& settings = SaveSettings())
162{
163 MeshInfo meshInfo(m);
164
165 // make sure that the given info contains only components that are actually
166 // available in the mesh. meshInfo will contain the intersection between the
167 // components that the user wants to save and the components that are
168 // available in the mesh.
169 if (!settings.info.isEmpty())
170 meshInfo = settings.info.intersect(meshInfo);
171
172 // if the mesh has both vertex and wedge texcords, will be saved just wedges
173 // because obj does not allow to save them both. In any case, also vertex
174 // texcoords will result saved as wedge texcoords in the final file.
175 if (meshInfo.hasVertexTexCoords() && meshInfo.hasFaceWedgeTexCoords()) {
176 meshInfo.setVertexTexCoords(false);
177 }
178
179 std::ofstream mtlftmp;
180 std::map<detail::ObjMaterial, std::string> materialMap;
181
182 bool useMtl =
183 meshInfo.hasVertexColors() || meshInfo.hasFaceColors() ||
184 (meshInfo.hasTextures() &&
185 (meshInfo.hasVertexTexCoords() || meshInfo.hasFaceWedgeTexCoords()));
186 if (useMtl) {
187 if (saveMtlFile) {
188 std::string mtlFileName =
189 FileInfo::fileNameWithExtension(filename) + ".mtl";
190
191 mtlftmp = openOutputFileStream(
192 FileInfo::pathWithoutFileName(filename) + mtlFileName);
193 mtlfp = &mtlftmp;
194
195 fp << "mtllib ./" << mtlFileName << std::endl;
196 }
197 else if (mtlfp == nullptr) {
198 useMtl = false;
199 }
200 }
201
202 detail::ObjMaterial lastMaterial;
203
204 // vertices
205 using VertexType = MeshType::VertexType;
206 for (const VertexType& v : m.vertices()) {
207 if (useMtl) { // mtl management
208 detail::writeElementObjMaterial<VertexType, MeshType>(
209 v,
210 m,
211 meshInfo,
212 lastMaterial,
213 materialMap,
214 fp,
215 *mtlfp,
216 settings,
217 log);
218 }
219 fp << "v ";
220 io::writeDouble(fp, v.coord().x(), false);
221 io::writeDouble(fp, v.coord().y(), false);
222 io::writeDouble(fp, v.coord().z(), false);
223 fp << std::endl;
224
225 if constexpr (HasPerVertexNormal<MeshType>) {
226 if (meshInfo.hasVertexNormals()) {
227 fp << "vn ";
228 io::writeDouble(fp, v.normal().x(), false);
229 io::writeDouble(fp, v.normal().y(), false);
230 io::writeDouble(fp, v.normal().z(), false);
231 fp << std::endl;
232 }
233 }
234 if constexpr (HasPerVertexTexCoord<MeshType>) {
235 if (meshInfo.hasVertexTexCoords()) {
236 fp << "vt ";
237 io::writeFloat(fp, v.texCoord().u(), false);
238 io::writeFloat(fp, v.texCoord().v(), false);
239 fp << std::endl;
240 }
241 }
242 }
243
244 // faces
245 if constexpr (vcl::HasFaces<MeshType>) {
246 using VertexType = MeshType::VertexType;
247 using FaceType = MeshType::FaceType;
248
249 // indices of vertices that do not consider deleted vertices
250 std::vector<uint> vIndices = m.vertexCompactIndices();
251
252 uint wedgeTexCoord = 1;
253 for (const FaceType& f : m.faces()) {
254 if (useMtl) { // mtl management
255 detail::writeElementObjMaterial(
256 f,
257 m,
258 meshInfo,
259 lastMaterial,
260 materialMap,
261 fp,
262 *mtlfp,
263 settings,
264 log);
265 }
266 if constexpr (HasPerFaceWedgeTexCoords<MeshType>) {
267 if (meshInfo.hasFaceWedgeTexCoords()) {
268 using WedgeTexCoordType = FaceType::WedgeTexCoordType;
269 for (const WedgeTexCoordType wt : f.wedgeTexCoords()) {
270 fp << "vt ";
271 io::writeFloat(fp, wt.u(), false);
272 io::writeFloat(fp, wt.v(), false);
273 fp << std::endl;
274 }
275 }
276 }
277
278 fp << "f ";
279 for (const VertexType* v : f.vertices()) {
280 fp << vIndices[m.index(v)] + 1;
281 if constexpr (HasPerVertexTexCoord<MeshType>) {
282 // we wrote texcoords along with vertices, each texcoord has
283 // the same index of its vertex
284 if (meshInfo.hasVertexTexCoords()) {
285 fp << "/" << vIndices[m.index(v)] + 1;
286 }
287 }
288 if constexpr (HasPerFaceWedgeTexCoords<MeshType>) {
289 // we wrote texcoords before the face; indices are
290 // consecutive and wedge coords are the same of the number
291 // of vertices of the face
292 if (meshInfo.hasFaceWedgeTexCoords()) {
293 fp << "/" << wedgeTexCoord++;
294 }
295 }
296 fp << " ";
297 }
298 fp << std::endl;
299 }
300 }
301}
