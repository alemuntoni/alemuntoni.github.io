41template<VertexConcept VertexType, MeshConcept MeshType>
42obj::Material materialFromVertex(const VertexType& v, const MeshInfo& fi)
43{
44 obj::Material mat;
45 if constexpr (HasPerVertexColor<MeshType>) {
46 if (fi.hasVertexColors()) {
47 mat.hasColor = true;
48 mat.Kd.x() = v.color().redF();
49 mat.Kd.y() = v.color().greenF();
50 mat.Kd.z() = v.color().blueF();
51 }
52 }
53 return mat;
54}

56template<FaceConcept FaceType, MeshConcept MeshType>
57obj::Material
58materialFromFace(const FaceType& f, const MeshType& m, const MeshInfo& fi)
59{
60 obj::Material mat;
61 if (fi.hasFaceColors()) {
62 mat.hasColor = true;
63 mat.Kd.x() = f.color().redF();
64 mat.Kd.y() = f.color().greenF();
65 mat.Kd.z() = f.color().blueF();
66 }
67 if constexpr (HasPerFaceWedgeTexCoords<MeshType>) {
68 if (fi.hasFaceWedgeTexCoords()) {
69 mat.hasTexture = true;
70 mat.map_Kd = m.texturePath(f.textureIndex());
71 }
72 }
73 return mat;
74}

76template<typename ElementType, MeshConcept MeshType>
77void writeElementMaterial(
78 const ElementType& e,
79 const MeshType& m,
80 const MeshInfo& fi,
81 obj::Material& lastMaterial,
82 std::map<obj::Material, std::string>& materialMap,
83 std::ofstream& fp,
84 std::ofstream& mtlfp)
85{
86 obj::Material mat;
87 if constexpr (std::is_same<ElementType, typename MeshType::VertexType>::
88 value)
89 mat =
90 materialFromVertex<typename MeshType::VertexType, MeshType>(e, fi);
91 if constexpr(HasFaces<MeshType>)
92 if constexpr (std::is_same<ElementType, typename MeshType::FaceType>::
93 value)
94 mat = materialFromFace(e, m, fi);
95 if (!mat.isEmpty()) {
96 static const std::string MATERIAL_PREFIX = "MATERIAL_";
97 std::string mname; // name of the material of the vertex
98 auto it = materialMap.find(mat);
99 if (it == materialMap.end()) { // if it is a new material
100 // add the new material to the map
101 mname = MATERIAL_PREFIX + std::to_string(materialMap.size());
102 materialMap[mat] = mname;
103 // save the material in the mtl file
104 mtlfp << "newmtl " << mname << std::endl;
105 mtlfp << mat << std::endl;
106 }
107 else { // get the name of the material
108 mname = it->second;
109 }
110 // if the material of the vertex is different from the last used, need
111 // to add usemtl
112 if (mat != lastMaterial) {
113 lastMaterial = mat;
114 fp << "usemtl " << mname << std::endl;
115 }
116 }
117}

121template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
122void saveObj(
123 const MeshType& m,
124 const std::string& filename,
125 const MeshInfo& info,
126 LogType& log = nullLogger)
127{
128 MeshInfo meshInfo(m);
129
130 // make sure that the given info contains only components that are actually
131 // available in the mesh. meshInfo will contain the intersection between the
132 // components that the user wants to save and the components that are
133 // available in the mesh.
134 meshInfo = info.intersect(meshInfo);
135
136 // if the mesh has both vertex and wedge texcords, will be save just wedges
137 // because obj does not allow to save them both. In any case, also vertex
138 // texcoords will result saved as wedge texcoords in the final file.
139 if (meshInfo.hasVertexTexCoords() && meshInfo.hasFaceWedgeTexCoords()) {
140 meshInfo.setVertexTexCoords(false);
141 }
142
143 std::ofstream fp = internal::saveFileStream(filename, "obj");
144
145 std::ofstream mtlfp;
146 std::map<obj::Material, std::string> materialMap;
147
148 bool useMtl =
149 meshInfo.hasVertexColors() || meshInfo.hasFaceColors() ||
150 (meshInfo.hasTextures() &&
151 (meshInfo.hasVertexTexCoords() || meshInfo.hasFaceWedgeTexCoords()));
152 if (useMtl) {
153 mtlfp = internal::saveFileStream(filename, "mtl");
154 std::string mtlFileName =
155 FileInfo::fileNameWithExtension(filename) + ".mtl";
156 fp << "mtllib ./" << mtlFileName << std::endl;
157 }
158
159 obj::Material lastMaterial;
160
161 // vertices
162 using VertexType = MeshType::VertexType;
163 for (const VertexType& v : m.vertices()) {
164 if (useMtl) { // mtl management
165 internal::writeElementMaterial<VertexType, MeshType>(
166 v, m, meshInfo, lastMaterial, materialMap, fp, mtlfp);
167 }
168 fp << "v ";
169 internal::writeDouble(fp, v.coord().x(), false);
170 internal::writeDouble(fp, v.coord().y(), false);
171 internal::writeDouble(fp, v.coord().z(), false);
172 fp << std::endl;
173
174 if constexpr (HasPerVertexNormal<MeshType>) {
175 if (meshInfo.hasVertexNormals()) {
176 fp << "vn ";
177 internal::writeDouble(fp, v.normal().x(), false);
178 internal::writeDouble(fp, v.normal().y(), false);
179 internal::writeDouble(fp, v.normal().z(), false);
180 fp << std::endl;
181 }
182 }
183 if constexpr (HasPerVertexTexCoord<MeshType>) {
184 if (meshInfo.hasVertexTexCoords()) {
185 fp << "vt ";
186 internal::writeFloat(fp, v.texCoord().u(), false);
187 internal::writeFloat(fp, v.texCoord().v(), false);
188 fp << std::endl;
189 }
190 }
191 }
192
193 // faces
194 if constexpr (vcl::HasFaces<MeshType>) {
195 using VertexType = MeshType::VertexType;
196 using FaceType = MeshType::FaceType;
197
198 // indices of vertices that do not consider deleted vertices
199 std::vector<uint> vIndices = m.vertexCompactIndices();
200
201 uint wedgeTexCoord = 1;
202 for (const FaceType& f : m.faces()) {
203 if (useMtl) { // mtl management
204 internal::writeElementMaterial(
205 f, m, meshInfo, lastMaterial, materialMap, fp, mtlfp);
206 }
207 if constexpr(HasPerFaceWedgeTexCoords<MeshType>){
208 if (meshInfo.hasFaceWedgeTexCoords()) {
209 using WedgeTexCoordType = FaceType::WedgeTexCoordType;
210 for (const WedgeTexCoordType wt : f.wedgeTexCoords()){
211 fp << "vt ";
212 internal::writeFloat(fp, wt.u(), false);
213 internal::writeFloat(fp, wt.v(), false);
214 fp << std::endl;
215 }
216 }
217 }
218
219 fp << "f ";
220 for (const VertexType* v : f.vertices()) {
221 fp << vIndices[m.index(v)]+1;
222 if constexpr(HasPerVertexTexCoord<MeshType>){
223 // we wrote texcoords along with vertices, each texcoord has
224 // the same index of its vertex
225 if (meshInfo.hasVertexTexCoords()) {
226 fp << "/" << vIndices[m.index(v)]+1;
227 }
228 }
229 if constexpr(HasPerFaceWedgeTexCoords<MeshType>){
230 // we wrote texcoords before the face; indices are
231 // consecutive and wedge coords are the same of the number
232 // of vertices of the face
233 if (meshInfo.hasFaceWedgeTexCoords()) {
234 fp << "/" << wedgeTexCoord++;
235 }
236 }
237 fp << " ";
238 }
239 fp << std::endl;
240 }
241 }
242
243 fp.close();
244 if (useMtl) {
245 mtlfp.close();
246 }
247}
