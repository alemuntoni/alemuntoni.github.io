418 // PointersComponent interface functions
419 template<typename Element, typename ElFType>
420 void importPointersFrom(const Element& e, Face* base, const ElFType* ebase)
421 {
422 if constexpr (HasAdjacentFaces<Element>) {
423 if (isAdjacentFacesAvailableOn(e)) {
424 if constexpr (N > 0) {
425 // same static size
426 if constexpr (N == Element::ADJ_FACE_NUMBER) {
427 importPtrsFrom(e, base, ebase);
428 }
429 // from dynamic to static, but dynamic size == static size
430 else if constexpr (Element::ADJ_FACE_NUMBER < 0) {
431 if (e.adjFacesNumber() == N) {
432 importPtrsFrom(e, base, ebase);
433 }
434 }
435 else {
436 // do not import in this case: cannot import from
437 // dynamic size != static size
438 }
439 }
440 else {
441 // from static/dynamic to dynamic size: need to resize
442 // first, then import
443 resize(e.adjFacesNumber());
444 importPtrsFrom(e, base, ebase);
445 }
446 }
447 }
448 }

467private:
468 template<typename Element, typename ElFType>
469 void importPtrsFrom(const Element& e, Face* base, const ElFType* ebase)
470 {
471 if (ebase != nullptr && base != nullptr) {
472 for (uint i = 0; i < e.adjFacesNumber(); ++i) {
473 if (e.adjFace(i) != nullptr) {
474 adjFace(i) = base + (e.adjFace(i) - ebase);
475 }
476 }
477 }
