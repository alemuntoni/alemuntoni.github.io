70private:
71 void checkMeshValue(uint value) const
72 {
73 if (value >= mMeshValues.size()) {
74 throw std::runtime_error(
75 "Invalid mesh value: " + std::to_string(value) +
76 "; expected value in [0, " +
77 std::to_string(mMeshValues.size()) + ")");
78 }
79 if (!mMeshValues[value].second) {
80 throw std::runtime_error(
81 "Invalid mesh value: Mesh " + mMeshValues[value].first + " (" +
82 std::to_string(value) + ") is disabled");
83 }
84 }
