43inline std::string::const_iterator findCaseInsensitive(
44 const std::string& input,
45 const std::string& substr)
46{
47 // link: https://stackoverflow.com/a/19839371/5851101
48
49 std::string::const_iterator it = std::search(
50 input.begin(),
51 input.end(),
52 substr.begin(),
53 substr.end(),
54 [](char ch1, char ch2) {
55 return std::toupper(ch1) == std::toupper(ch2);
56 });
57 return it;
58}

68template<typename T>
69std::string toString(T val)
70{
71 // if T is a pointer
72 if constexpr (std::is_pointer_v<T>) {
73 const void* address = static_cast<const void*>(val);
74 std::stringstream ss;
75 ss << address;
76 return ss.str();
77 }
78 else {
79 return std::to_string(val);
80 }
