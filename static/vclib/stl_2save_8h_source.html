38void writeStlHeader(std::ofstream& fp, bool magicsMode, bool binary)
39{
40 std::string header = "solid STL generated by VCLib";
41 if (binary) {
42 if (magicsMode) {
43 std::string p;
44 for (uint i = 0; i < 3; i++)
45 p.push_back((char)0x7f);
46 header = "COLOR=" + p + " MATERIAL=" + p + " " + p + " " + p;
47 }
48 while (header.size() < 80)
49 header.push_back(' ');
50 }
51
52 fp << header;
53 if (!binary)
54 fp << std::endl;
55}

57template<Point3Concept PointType, Point3Concept NormalType>
58void writeTriangle(
59 std::ofstream& fp,
60 const PointType& p0,
61 const PointType& p1,
62 const PointType& p2,
63 const NormalType& n,
64 uint attributes,
65 bool binary)
66{
67 if (binary) {
68 for (uint i = 0; i < 3; ++i)
69 internal::writeFloat(fp, n[i]);
70
71 for (uint i = 0; i < 3; ++i)
72 internal::writeFloat(fp, p0[i]);
73
74 for (uint i = 0; i < 3; ++i)
75 internal::writeFloat(fp, p1[i]);
76
77 for (uint i = 0; i < 3; ++i)
78 internal::writeFloat(fp, p2[i]);
79
80 internal::writeUShort(fp, attributes);
81 }
82 else {
83 fp << " facet normal " << n.x() << " " << n.y() << " " << n.z()
84 << std::endl;
85 fp << " outer loop" << std::endl;
86
87 fp << " vertex " << p0.x() << " " << p0.y() << " " << p0.z()
88 << std::endl;
89 fp << " vertex " << p1.x() << " " << p1.y() << " " << p1.z()
90 << std::endl;
91 fp << " vertex " << p2.x() << " " << p2.y() << " " << p2.z()
92 << std::endl;
93
94 fp << " endloop" << std::endl;
95 fp << " endfacet" << std::endl;
96 }
97}

112template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
113void saveStl(
114 const MeshType& m,
115 const std::string& filename,
116 const MeshInfo& info,
117 LogType& log = nullLogger,
118 bool binary = true,
119 bool magicsMode = false)
120{
121 MeshInfo meshInfo(m);
122
123 // make sure that the given info contains only components that are actually
124 // available in the mesh. meshInfo will contain the intersection between the
125 // components that the user wants to save and the components that are
126 // available in the mesh.
127 meshInfo = info.intersect(meshInfo);
128
129 std::ofstream fp = internal::saveFileStream(filename, "stl");
130
131 if constexpr (isLoggerValid<LogType>()) {
132 log.log(0, "Saving STL file");
133 }
134
135 internal::writeStlHeader(fp, magicsMode, binary);
136
137 if constexpr (HasFaces<MeshType>) {
138 using FaceType = MeshType::FaceType;
139
140 if (binary) {
141 internal::writeInt(fp, m.faceNumber());
142 }
143
144 if constexpr (vcl::isLoggerValid<LogType>()) {
145 log.startProgress("Loading STL file", m.faces());
146 }
147
148 for (const FaceType& f : m.faces()) {
149 // For each triangle write the normal, the three coords and a short
150 auto n = vcl::faceNormal(f);
151
152 unsigned short attributes = 0;
153
154 if constexpr (HasPerFaceColor<MeshType>) {
155 if (meshInfo.hasFaceColors()) {
156 if(magicsMode)
157 attributes = 32768 | f.color().toUnsignedR5G5B5();
158 else
159 attributes = 32768 | f.color().toUnsignedB5G5R5();
160 }
161 }
162
163 if (f.vertexNumber() == 3) {
164 internal::writeTriangle(
165 fp,
166 f.vertex(0)->coord(),
167 f.vertex(1)->coord(),
168 f.vertex(2)->coord(),
169 n,
170 attributes,
171 binary);
172 }
173 else {
174 std::vector<uint> tris = vcl::earCut(f);
175 for (uint i = 0; i < tris.size(); i += 3) {
176 internal::writeTriangle(
177 fp,
178 f.vertex(tris[i])->coord(),
179 f.vertex(tris[i + 1])->coord(),
180 f.vertex(tris[i + 2])->coord(),
181 n,
182 attributes,
183 binary);
184 }
185 }
186
187 if constexpr (vcl::isLoggerValid<LogType>()) {
188 log.progress(m.index(f));
189 }
190 }
191 if constexpr (vcl::isLoggerValid<LogType>()) {
192 log.endProgress();
193 }
194 }
195
196 if (!binary) {
197 fp << "endsolid VCLib" << std::endl;
198 }
199
200 fp.close();
201}
