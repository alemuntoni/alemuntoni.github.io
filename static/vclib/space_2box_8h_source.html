325 Box<PointT> intersection(const Box<PointT>& b) const
326 {
327 Box<PointT> res = *this;
328 for (uint i = 0; i < PointT::DIM; ++i) {
329 if (minP[i] < b.minP[i]) {
330 // set the minimum point to the larger value
331 res.minP[i] = b.minP[i];
332 }
333 if (maxP[i] > b.maxP[i]) {
334 // set the maximum point to the smaller value
335 res.maxP[i] = b.maxP[i];
336 }
337 if (res.minP[i] > res.maxP[i]) {
338 // if the minimum point is larger than the maximum point, the
339 // box is null
340 res.setNull();
341 return res;
342 }
343 }
344 return res;
345 }

371 void add(const PointT& p)
372 {
373 if (isNull()) {
374 // If the box was null, set both the minimum and maximum points to
375 // the given point
376 *this = Box(p);
377 }
378 else {
379 // Expand the current box to include the added point
380 minP = vcl::min(minP, p);
381 maxP = vcl::max(maxP, p);
382 }
383 }
384
397 template<typename Scalar>
398 void add(const PointT& p, Scalar radius)
399 {
400 if (isNull()) {
401 // If the box was null, initialize it with the minimum and maximum
402 // points obtained by subtracting and adding the radius to the given
403 // point
404 *this = Box(p - radius, p + radius);
405 }
406 else {
407 // Expand the current box to include the added point and all points
408 // within the radius around it
409 minP = vcl::min(minP, p - radius);
410 maxP = vcl::max(maxP, p + radius);
411 }
412 }
413
422 void add(const Box<PointT>& b)
423 {
424 // Adding a null box should do nothing
425 if (!b.isNull()) {
426 if (isNull()) {
427 *this = b; // if the current box is null, set it equal to b
428 }
429 else {
430 // expand this box to contain the minimum and maximum point of b
431 add(b.minP);
432 add(b.maxP);
433 }
434 }
435 }
436
442 void translate(const PointT& p)
443 {
444 minP += p; // translate the minimum point by p
445 maxP += p; // translate the maximum point by p
446 }
