160template<FaceMeshConcept MeshType>
161void updatePerVertexNormals(MeshType& m, bool normalize = true)
162{
163 clearPerReferencedVertexNormals(m);
164
165 using VertexType = MeshType::VertexType;
166 using FaceType = MeshType::FaceType;
167 using NormalType = VertexType::NormalType;
168
169 for (FaceType& f : m.faces()) {
170 NormalType n =
171 faceNormal(f).template cast<typename NormalType::ScalarType>();
172 for (VertexType* v : f.vertices()) {
173 v->normal() += n;
174 }
175 }
176 if (normalize)
177 normalizePerVertexNormals(m);

241template<FaceMeshConcept MeshType>
242void updatePerVertexNormalsAngleWeighted(MeshType& m, bool normalize = true)
243{
244 clearPerReferencedVertexNormals(m);
245
246 using VertexType = MeshType::VertexType;
247 using FaceType = MeshType::FaceType;
248 using NormalType = VertexType::NormalType;
249 using NScalarType = NormalType::ScalarType;
250
251 for (FaceType& f : m.faces()) {
252 NormalType n = faceNormal(f).template cast<NScalarType>();
253
254 for (uint i = 0; i < f.vertexNumber(); ++i) {
255 NormalType vec1 =
256 (f.vertexMod(i - 1)->coord() - f.vertexMod(i)->coord())
257 .normalized()
258 .template cast<NScalarType>();
259 NormalType vec2 =
260 (f.vertexMod(i + 1)->coord() - f.vertexMod(i)->coord())
261 .normalized()
262 .template cast<NScalarType>();
263
264 f.vertex(i)->normal() += n * vec1.angle(vec2);
265 }
266 }
267 if (normalize)
268 normalizePerVertexNormals(m);
269}

300template<FaceMeshConcept MeshType>
301void updatePerVertexNormalsNelsonMaxWeighted(MeshType& m, bool normalize = true)
302{
303 clearPerReferencedVertexNormals(m);
304
305 using VertexType = MeshType::VertexType;
306 using FaceType = MeshType::FaceType;
307 using NormalType = VertexType::NormalType;
308 using NScalarType = NormalType::ScalarType;
309
310 for (FaceType& f : m.faces()) {
311 NormalType n = faceNormal(f).template cast<NScalarType>();
312
313 for (uint i = 0; i < f.vertexNumber(); ++i) {
314 NScalarType e1 =
315 (f.vertexMod(i - 1)->coord() - f.vertexMod(i)->coord())
316 .squaredNorm();
317 NScalarType e2 =
318 (f.vertexMod(i + 1)->coord() - f.vertexMod(i)->coord())
319 .squaredNorm();
320
321 f.vertex(i)->normal() += n / (e1 * e2);
322 }
323 }
324 if (normalize)
325 normalizePerVertexNormals(m);
326}

342template<FaceMeshConcept MeshType, typename MScalar>
343void multiplyPerFaceNormalsByMatrix(
344 MeshType& mesh,
345 const vcl::Matrix44<MScalar>& mat,
346 bool removeScalingFromMatrix = true)
347{
348 requirePerFaceNormal(mesh);
349
350 using FaceType = MeshType::FaceType;
351
352 Matrix33<MScalar> m33 = mat.block(0, 0, 3, 3);
353 if (removeScalingFromMatrix) {
354 MScalar scaleX = std::sqrt(
355 m33(0, 0) * m33(0, 0) + m33(0, 1) * m33(0, 1) +
356 m33(0, 2) * m33(0, 2));
357 MScalar scaleY = std::sqrt(
358 m33(1, 0) * m33(1, 0) + m33(1, 1) * m33(1, 1) +
359 m33(1, 2) * m33(1, 2));
360 MScalar scaleZ = std::sqrt(
361 m33(2, 0) * m33(2, 0) + m33(2, 1) * m33(2, 1) +
362 m33(2, 2) * m33(2, 2));
363 for (int i = 0; i < 3; ++i) {
364 m33(0, i) /= scaleX;
365 m33(1, i) /= scaleY;
366 m33(2, i) /= scaleZ;
367 }
368 }
369 for (FaceType& f : mesh.faces()) {
370 f.normal() *= m33;
371 }
372}
