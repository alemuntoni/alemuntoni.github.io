254 // Component interface function
255 template<typename Element>
256 void importFrom(const Element& e)
257 {
258 if constexpr (HasBitFlags<Element>) {
259 resetBitFlags();
260 if constexpr (
261 HasPolygonBitFlags<Element> || HasTriangleBitFlags<Element>)
262 {
263 deletedBit() = e.deleted();
264 selected() = e.selected();
265 visited() = e.visited();
266 onBorder() = e.onBorder();
267 const uint UM = std::min(USER_BITS_NUMBER, e.USER_BITS_NUMBER);
268 for (uint i = 0; i < UM; ++i)
269 userBit(i) = e.userBit(i);
270 }
271 else {
272 flags() = e.flags();
273 }
274
275 }
