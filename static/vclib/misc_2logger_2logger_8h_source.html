116 void startNewTask(double fromPerc, double toPerc, const std::string& action)
117 {
118 printLine(action, START);
119
120 assert(fromPerc >= 0);
121 assert(toPerc <= 100);
122 std::pair<double, double> newP;
123 newP.first =
124 stack.top().first +
125 (stack.top().second - stack.top().first) * (fromPerc / 100);
126 newP.second = (stack.top().second - stack.top().first) * (toPerc / 100);
127 globalPercProgress = newP.first;
128 stack.push(newP);
129 updateStep();
130 }
131
132 void endTask(const std::string& action)
133 {
134 globalPercProgress = stack.top().second;
135 if (stack.size() > 1) {
136 stack.pop();
137 updateStep();
138
139 printLine(action, END);
140 }

203 void startProgress(
204 const std::string& msg,
205 uint progressSize,
206 uint percPrintProgress = 10,
207 uint startPerc = 0,
208 uint endPerc = 100)
209 {
210 assert(percPrintProgress > 0);
211 assert((endPerc - startPerc) > 0);
212 isProgressActive = true;
213 progressMessage = msg;
214 progressPerc = startPerc;
215 progressPercStep = percPrintProgress;
216 progressStep =
217 progressSize / ((endPerc - startPerc) / percPrintProgress - 1);
218 if (progressStep == 0)
219 progressStep = progressSize;
220 lastProgress = 0;
221 }

277 void progress(uint n)
278 {
279 mutex.lock();
280 assert(isProgressActive);
281 uint progress = n / progressStep;
282 if (lastProgress < progress) {
283 progressPerc = progress * progressPercStep;
284 log(progressPerc, progressMessage);
285 lastProgress = progress;
286 }
287 mutex.unlock();
288 }

298 void printLine(const std::string& msg, uint lvl)
299 {
300 LogLevel l = PROGRESS;
301 if (lvl < DEBUG){
302 l = (LogLevel) lvl;
303 }
304 std::ostream* stream = levelStream(l);
305
306 if (stream) {
307 uint s = 0;
308 s = printPercentage(*stream);
309 s += printIndentation(*stream);
310 printMessage(*stream, msg, lvl, s);
311 printElapsedTime(*stream);
312 *stream << std::endl;
313 }

340 void printMessage(std::ostream& o, const std::string& msg, uint lvl, uint n)
341 {
342 uint maxMsgSize = lineW - n;
343 if (printTimer)
344 maxMsgSize -= TIMER_SIZE;
345 switch (lvl) {
346 case ERROR:
347 maxMsgSize -= 8;
348 o << " ERROR: ";
349 break;
350 case WARNING:
351 maxMsgSize -= 10;
352 o << " WARNING: ";
353 break;
354 case PROGRESS:
355 maxMsgSize -= 1;
356 o << " ";
357 break;
358 case DEBUG:
359 maxMsgSize -= 9;
360 o << " (debug) ";
361 break;
362 case START:
363 maxMsgSize -= 7;
364 o << " Start ";
365 break;
366 case END:
367 maxMsgSize -= 5;
368 o << " End ";
369 break;
370 }
371 o << std::setw(maxMsgSize) << std::left << msg;
372 }
