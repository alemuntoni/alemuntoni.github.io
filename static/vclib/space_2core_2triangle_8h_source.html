405 static PointT circumcenter(
406 const PointT& p0,
407 const PointT& p1,
408 const PointT& p2)
409 {
410 ScalarType a2 = p1.squaredDist(p2);
411 ScalarType b2 = p2.squaredDist(p0);
412 ScalarType c2 = p0.squaredDist(p1);
413
414 PointType c = p0 * a2 * (-a2 + b2 + c2) + p1 * b2 * (a2 - b2 + c2) +
415 p2 * c2 * (a2 + b2 - c2);
416
417 c /= 2 * (a2 * b2 + a2 * c2 + b2 * c2) - a2 * a2 - b2 * b2 - c2 * c2;
418
419 return c;
420 }

451 static ScalarType area(const PointT& p0, const PointT& p1, const PointT& p2)
452 {
453 if constexpr (DIM == 2) {
454 return ((p1[0] - p0[0]) * (p2[1] - p0[1]) -
455 (p2[0] - p0[0]) * (p1[1] - p0[1])) /
456 2;
457 }
458 if constexpr (DIM == 3) {
459 return normal(p0, p1, p2).norm() / 2;
460 }
461 else {
462 // heron's formula
463 ScalarType s = perimeter() / 2;
464 return std::sqrt(
465 s * (s - p0.dist(p1)) * (s - p1.dist(p2)) * (s - p2.dist(p0)));
466 }
467 }

486 static ScalarType quality(
487 const PointT& p0,
488 const PointT& p1,
489 const PointT& p2)
490 {
491 ScalarType a = area();
492 if (a == 0)
493 return 0; // Area zero triangles have surely quality==0;
494
495 PointType d10 = p1 - p0;
496 PointType d20 = p2 - p0;
497 PointType d12 = p1 - p2;
498 double b =
499 std::min({d10.squaredNorm(), d20.squaredNorm(), d12.squaredNorm()});
500 if (b == 0)
501 return 0; // Again: area zero triangles have surely quality==0;
502 return (2 * a) / b;
503 }
504
522 static ScalarType qualityRadii(
523 const PointT& p0,
524 const PointT& p1,
525 const PointT& p2)
526 {
527 double a = p0.dist(p1);
528 double b = p2.dist(p0);
529 double c = p1.dist(p2);
530
531 double sum = (a + b + c) * 0.5;
532 double area2 = sum * (a + b - sum) * (a + c - sum) * (b + c - sum);
533
534 if (area2 <= 0)
535 return 0;
536
537 return (8 * area2) / (a * b * c * sum);
538 }
539
560 static ScalarType qualityMeanRatio(
561 const PointT& p0,
562 const PointT& p1,
563 const PointT& p2)
564 {
565 double a = p0.dist(p1);
566 double b = p2.dist(p0);
567 double c = p1.dist(p2);
568
569 double sum = (a + b + c) * 0.5; // semiperimeter
570 double area2 = sum * (a + b - sum) * (a + c - sum) * (b + c - sum);
571 if (area2 <= 0)
572 return 0;
573 return (4.0 * std::sqrt(3.0) * std::sqrt(area2)) /
574 (a * a + b * b + c * c);
575 }
