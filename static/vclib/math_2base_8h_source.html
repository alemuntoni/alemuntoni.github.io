68inline double lnOfFactorial(int n)
69{
70 static const int FAK_LEN = 1024;
71
72 // define constants
73 static const double // coefficients in Stirling approximation
74 C0 = 0.918938533204672722, // ln(sqrt(2*pi))
75 C1 = 1. / 12., C3 = -1. / 360.;
76 // C5 = 1./1260., // use r^5 term if FAK_LEN < 50
77 // C7 = -1./1680.; // use r^7 term if FAK_LEN < 20
78 // static variables
79 static double fac_table[FAK_LEN]; // table of ln(n!):
80 // remember if fac_table has been initialized
81 static bool initialized = false;
82
83 if (n < FAK_LEN) {
84 if (n <= 1) {
85 if (n < 0)
86 assert(0); //("Parameter negative in LnFac function");
87 return 0;
88 }
89 if (!initialized) { // first time. Must initialize table
90 // make table of ln(n!)
91 double sum = fac_table[0] = 0.;
92 for (uint i = 1; i < FAK_LEN; i++) {
93 sum += std::log(double(i));
94 fac_table[i] = sum;
95 }
96 initialized = true;
97 }
98 return fac_table[n];
99 }
100 else {
101 // not found in table. use Stirling approximation
102 double n1, r;
103 n1 = n;
104 r = 1. / n1;
105 return (n1 + 0.5) * std::log(n1) - n1 + C0 + r * (C1 + r * r * C3);
106 }
107}
