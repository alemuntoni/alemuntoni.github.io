43template<PolygonMeshConcept MeshType, LoggerConcept LogType = NullLogger>
44MeshType createDodecahedron(LogType& log = nullLogger)
45{
46 using VertexType = MeshType::VertexType;
47 using CoordType = VertexType::CoordType;
48 using ScalarType = CoordType::ScalarType;
49
50 MeshType mesh;
51
52 const ScalarType fi = (1 + std::sqrt(5)) / 2;
53 const ScalarType s0 = 0;
54 const ScalarType s1 = fi;
55 const ScalarType s2 = std::pow(fi, 2);
56 const ScalarType s3 = 1;
57
58 if constexpr (isLoggerValid<LogType>()) {
59 log.log(0, "Adding vertices to PolyMesh...");
60 }
61
62 mesh.addVertices(
63 CoordType(-s1, -s1, s1),
64 CoordType(s2, s3, s0),
65 CoordType(s2, -s3, s0),
66 CoordType(-s2, s3, s0),
67 CoordType(-s2, -s3, s0),
68 CoordType(s0, s2, s3),
69 CoordType(s0, s2, -s3),
70 CoordType(s3, s0, -s2),
71 CoordType(-s3, s0, -s2),
72 CoordType(s0, -s2, -s3),
73 CoordType(s0, -s2, s3),
74 CoordType(s3, s0, s2),
75 CoordType(-s3, s0, s2),
76 CoordType(s1, s1, -s1),
77 CoordType(s1, s1, s1),
78 CoordType(-s1, s1, -s1),
79 CoordType(-s1, s1, s1),
80 CoordType(s1, -s1, -s1),
81 CoordType(s1, -s1, s1),
82 CoordType(-s1, -s1, -s1));
83
84 if constexpr (isLoggerValid<LogType>()) {
85 log.log(50, "Vertices added to PolyMesh.");
86 log.log(50, "Adding faces to PolyMesh...");
87 }
88
89 mesh.reserveFaces(12);
90 mesh.addFace(14, 11, 18, 2, 1);
91 mesh.addFace(2, 17, 7, 13, 1);
92 mesh.addFace(15, 8, 19, 4, 3);
93 mesh.addFace(4, 0, 12, 16, 3);
94 mesh.addFace(16, 5, 6, 15, 3);
95 mesh.addFace(13, 6, 5, 14, 1);
96 mesh.addFace(18, 10, 9, 17, 2);
97 mesh.addFace(19, 9, 10, 0, 4);
98 mesh.addFace(17, 9, 19, 8, 7);
99 mesh.addFace(13, 7, 8, 15, 6);
100 mesh.addFace(16, 12, 11, 14, 5);
101 mesh.addFace(18, 11, 12, 0, 10);
102
103 if constexpr (isLoggerValid<LogType>()) {
104 log.log(100, "Faces added to PolyMesh.");
105 }
106
107 return mesh;
108}

122template<TriangleMeshConcept MeshType, LoggerConcept LogType = NullLogger>
123MeshType createDodecahedron(LogType& log = nullLogger)
124{
125 if constexpr (isLoggerValid<LogType>()) {
126 log.startNewTask(0, 75, "Create Polygonal Dodecahedron.");
127 }
128
129 internal::TMPSimplePolyMesh pmesh =
130 createDodecahedron<internal::TMPSimplePolyMesh>(log);
131
132 if constexpr (isLoggerValid<LogType>()) {
133 log.endTask("Create Polygonal Dodecahedron.");
134 log.log(75, "Copying vertices into TriMesh...");
135 }
136
137 MeshType mesh;
138 using ST = MeshType::VertexType::CoordType::ScalarType;
139 mesh.reserveVertices(pmesh.vertexNumber());
140 for (const auto& v : pmesh.vertices()) {
141 mesh.addVertex(v.coord().cast<ST>());
142 }
143
144 if constexpr (isLoggerValid<LogType>()) {
145 log.log(80, "Vertices copied into TriMesh.");
146 log.log(80, "Triangularize and copy Faces into TriMesh...");
147 }
148
149 for (const auto& f : pmesh.faces()) {
150 std::vector<uint> ind = earCut(f);
151 for (uint i = 0; i < ind.size(); i += 3) {
152 mesh.addFace(
153 pmesh.index(f.vertex(ind[i])),
154 pmesh.index(f.vertex(ind[i + 1])),
155 pmesh.index(f.vertex(ind[i + 2])));
156 }
157 }
158
159 if constexpr (isLoggerValid<LogType>()) {
160 log.log(100, "Faces triangularized and copied into TriMesh.");
161 }
162
163 return mesh;
164}
