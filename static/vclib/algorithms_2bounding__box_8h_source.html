311template<IteratorConcept Iterator>
312auto boundingBox(Iterator begin, Iterator end)
313{
314 using BB = decltype(boundingBox(typename Iterator::value_type()));
315
316 BB b;
317
318 for (; begin != end; ++begin)
319 b.add(boundingBox(*begin));
320
321 return b;
322}
