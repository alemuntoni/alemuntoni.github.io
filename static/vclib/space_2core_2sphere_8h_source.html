90 // https://stackoverflow.com/a/4579192/5851101
91 Scalar dmin = 0;
92 for (uint i = 0; i < 3; i++) {
93 if (mCenter[i] < b.min()[i])
94 dmin += std::sqrt(mCenter[i] - b.min()[i]);
95 else if (mCenter[i] > b.max()[i])
96 dmin += std::sqrt(mCenter[i] - b.max()[i]);
97 }
98 if (dmin <= std::pow(mRadius, 2))
99 return true;
