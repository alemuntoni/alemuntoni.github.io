90template<typename T>
91concept MeshConcept =
92 (mesh::IsDerivedFromMesh<T>::value || mesh::IsAMesh<T>::value) &&
93 mesh::HasVertexContainer<T> &&
94 requires(
95 T o,
96 const T& co,
97 typename T::VertexType v,
98 typename T::VertexType::CoordType c)
99{
100 typename T::VertexType;
101 typename T::VertexContainer;
102
103 { o.clear() } -> std::same_as<void>;
104 { o.enableSameOptionalComponentsOf(T()) } -> std::same_as<void>;
105 { o.importFrom(T()) } -> std::same_as<void>;
106 { co.index(v) } -> std::same_as<uint>;
107 { co.index(&v) } -> std::same_as<uint>;
108 { o.addVertex() } -> std::same_as<uint>;
109 { o.addVertex(c) } -> std::same_as<uint>;
110 { o.addVertices(uint()) } -> std::same_as<uint>;
111 { o.addVertices(c, c, c, c) } -> std::same_as<uint>;
112 { o.reserveVertices(uint()) } -> std::same_as<void>;
113 { o.compactVertices() } -> std::same_as<void>;
114};

269// Concept used to enable PerFaceWedgeColors member functions in Mesh class
270// they can be enabled if:
271// - they have per face optional WedgeColors, or
272// - if the mesh is a Dcel, HalfEdges have optional Color
273template <typename M>
274concept OptionalWedgeColorsConcept =
275 FaceMeshConcept<M> &&
276 (face::HasOptionalWedgeColors<typename M::FaceType> ||
277 (HasHalfEdges<M> && hedge::HasOptionalColor<typename M::HalfEdge>));
278
279// Concept used to enable PerFaceWedgeTexCoords member functions in Mesh class
280// they can be enabled if:
281// - they have per face optional WedgeTexCoords, or
282// - if the mesh is a Dcel, HalfEdges have optional TexCoord
283template <typename M>
284concept OptionalWedgeTexCoordsConcept =
285 FaceMeshConcept<M> &&
286 (face::HasOptionalWedgeTexCoords<typename M::FaceType> ||
287 (HasHalfEdges<M> && hedge::HasOptionalTexCoord<typename M::HalfEdge>));
