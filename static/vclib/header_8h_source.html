76 PlyHeader(std::istream& file, const std::string& filename = "")
77 {
78 clear();
79
80 std::string line;
81 std::getline(file, line);
82 vcl::removeCarriageReturn(line);
83 if (line.compare(0, 3, "ply") == 0) {
84 bool error = false;
85 bool firstElement = true;
86 std::string headerLine;
87 PlyElement element;
88 do {
89 vcl::Tokenizer spaceTokenizer =
90 readAndTokenizeNextNonEmptyLine(file);
91 if (!error) {
92 vcl::Tokenizer::iterator token = spaceTokenizer.begin();
93 headerLine = *token;
94 if (headerLine == "format") {
95 token++;
96 if (*token == "ascii")
97 mFormat = ply::ASCII;
98 else if (
99 *token == "binary_little_endian" ||
100 *token == "binary")
101 mFormat = ply::BINARY_LITTLE_ENDIAN;
102 else if (*token == "binary_big_endian")
103 mFormat = ply::BINARY_BIG_ENDIAN;
104 }
105 // reading a comment, may be a texture file...
106 else if (headerLine == "comment") {
107 token++;
108 if (token != spaceTokenizer.end()) {
109 if (vcl::containsCaseInsensitive(*token, "texture"))
110 {
111 ++token;
112 if (token != spaceTokenizer.end()) {
113 std::string textName = *token;
114 auto it = vcl::findCaseInsensitive(
115 textName, "<this>");
116 if (it != textName.end()) {
117 uint pos = it - textName.begin();
118 std::string fn = vcl::FileInfo::
119 fileNameWithoutExtension(filename);
120 textName =
121 textName.substr(0, pos) + fn +
122 textName.substr(
123 pos + 6, textName.size());
124 }
125 mTextureFiles.push_back(textName);
126 }
127 }
128 }
129 }
130 // I am reading a new element
131 else if (headerLine == "element") {
132 // if it is not the first element to read, it means that
133 // the previous one needs to be saved
134 if (!firstElement) {
135 // index of each element type in elements vector
136 if (element.type == ply::VERTEX)
137 mVertElemPos = mElements.size();
138 if (element.type == ply::FACE)
139 mFaceElemPos = mElements.size();
140 if (element.type == ply::EDGE)
141 mEdgeElemPos = mElements.size();
142 if (element.type == ply::TRISTRIP)
143 mTriStripElemPos = mElements.size();
144 mElements.push_back(element);
145 element = PlyElement();
146 }
147 element = readElement(spaceTokenizer);
148 firstElement = false;
149 }
150 else if (headerLine == "property") {
151 PlyProperty p = readProperty(spaceTokenizer);
152 element.properties.push_back(p);
153 }
154 // save the last element
155 else if (headerLine == "end_header") {
156 if (element.type == ply::VERTEX)
157 mVertElemPos = mElements.size();
158 if (element.type == ply::FACE)
159 mFaceElemPos = mElements.size();
160 if (element.type == ply::EDGE)
161 mEdgeElemPos = mElements.size();
162 if (element.type == ply::TRISTRIP)
163 mTriStripElemPos = mElements.size();
164 mElements.push_back(element);
165 }
166 }
167 } while (!error && headerLine != "end_header");
168 mValid = !error && hasVertices();
169 }
170 }

189 MeshInfo getInfo() const
190 {
191 MeshInfo mod;
192 // x, y, z, nx, ny, nz, red, green, blue, alpha, vertex_indices
193
194 if (mVertElemPos != UINT_NULL) {
195 mod.setVertices();
196 for (const PlyProperty& p : mElements[mVertElemPos].properties) {
197 switch (p.name) {
198 case ply::x:
199 case ply::y:
200 case ply::z: mod.setVertexCoords(); break;
201 case ply::nx:
202 case ply::ny:
203 case ply::nz: mod.setVertexNormals(); break;
204 case ply::red:
205 case ply::green:
206 case ply::blue:
207 case ply::alpha: mod.setVertexColors(); break;
208 case ply::quality: mod.setVertexQuality(); break;
209 case ply::texture_u: mod.setVertexTexCoords(); break;
210 case ply::unknown:
211 if (p.type <= ply::PropertyType::DOUBLE) {
212 mod.addVertexCustomComponent(
213 p.unknownPropertyName, (MeshInfo::DataType) p.type);
214 }
215 default: break;
216 }
217 }
218 }
219 if (mFaceElemPos != UINT_NULL) {
220 mod.setFaces();
221 for (const PlyProperty& p : mElements[mFaceElemPos].properties) {
222 switch (p.name) {
223 case ply::vertex_indices: mod.setFaceVRefs(); break;
224 case ply::nx:
225 case ply::ny:
226 case ply::nz: mod.setFaceNormals(); break;
227 case ply::red:
228 case ply::green:
229 case ply::blue:
230 case ply::alpha: mod.setFaceColors(); break;
231 case ply::quality: mod.setFaceQuality(); break;
232 case ply::texcoord: mod.setFaceWedgeTexCoords(); break;
233 case ply::unknown:
234 if (p.type <= ply::PropertyType::DOUBLE) {
235 mod.addFaceCustomComponent(
236 p.unknownPropertyName, (MeshInfo::DataType) p.type);
237 }
238 default: break;
239 }
240 }
241 }
242 if (mTriStripElemPos != UINT_NULL) {
243 mod.setFaces();
244 for (const PlyProperty& p : mElements[mTriStripElemPos].properties)
245 {
246 switch (p.name) {
247 case ply::vertex_indices: mod.setFaceVRefs(); break;
248 case ply::nx:
249 case ply::ny:
250 case ply::nz: mod.setFaceNormals(); break;
251 case ply::red:
252 case ply::green:
253 case ply::blue:
254 case ply::alpha: mod.setFaceColors(); break;
255 case ply::quality: mod.setFaceQuality(); break;
256 case ply::texcoord: mod.setFaceWedgeTexCoords(); break;
257 default: break;
258 }
259 }
260 }
261 if (mTextureFiles.size() > 0) {
262 mod.setTextures(true);
263 }
264 return mod;
265 }

357 void setInfo(
358 const MeshInfo& info,
359 std::vector<std::string> textureFileNames = std::vector<std::string>(),
360 ply::Format format = ply::BINARY_LITTLE_ENDIAN)
361 {
362 clear();
363 mFormat = format;
364 mValid = true;
365 mTextureFiles = textureFileNames;
366 if (info.hasVertices()) {
367 mVertElemPos = mElements.size();
368 PlyElement vElem;
369 vElem.type = ply::VERTEX;
370 if (info.hasVertexCoords()) {
371 PlyProperty px, py, pz;
372 px.name = ply::x;
373 px.type = info.vertexCoordsType();
374 py.name = ply::y;
375 py.type = info.vertexCoordsType();
376 pz.name = ply::z;
377 pz.type = info.vertexCoordsType();
378 vElem.properties.push_back(px);
379 vElem.properties.push_back(py);
380 vElem.properties.push_back(pz);
381 }
382 if (info.hasVertexNormals()) {
383 PlyProperty vnx, vny, vnz;
384 vnx.name = ply::nx;
385 vnx.type = info.vertexNormalsType();
386 vny.name = ply::ny;
387 vny.type = info.vertexNormalsType();
388 vnz.name = ply::nz;
389 vnz.type = info.vertexNormalsType();
390 vElem.properties.push_back(vnx);
391 vElem.properties.push_back(vny);
392 vElem.properties.push_back(vnz);
393 }
394 if (info.hasVertexColors()) {
395 PlyProperty vcr, vcg, vcb, vca;
396 vcr.name = ply::red;
397 vcr.type = info.vertexColorsType();
398 vcg.name = ply::green;
399 vcg.type = info.vertexColorsType();
400 vcb.name = ply::blue;
401 vcb.type = info.vertexColorsType();
402 vca.name = ply::alpha;
403 vca.type = info.vertexColorsType();
404 vElem.properties.push_back(vcr);
405 vElem.properties.push_back(vcg);
406 vElem.properties.push_back(vcb);
407 vElem.properties.push_back(vca);
408 }
409 if (info.hasVertexQuality()) {
410 PlyProperty vs;
411 vs.name = ply::quality;
412 vs.type = info.vertexQualityType();
413 vElem.properties.push_back(vs);
414 }
415 if (info.hasVertexTexCoords()) {
416 PlyProperty tcu, tcv, tcn;
417 tcu.name = ply::texture_u;
418 tcu.type = info.vertexTexCoordsType();
419 tcv.name = ply::texture_v;
420 tcv.type = info.vertexTexCoordsType();
421 tcn.name = ply::texnumber;
422 tcn.type = PrimitiveType::USHORT;
423 vElem.properties.push_back(tcu);
424 vElem.properties.push_back(tcv);
425 vElem.properties.push_back(tcn);
426 }
427 if (info.hasVertexCustomComponents()) {
428 for (const auto& cc : info.vertexCustomComponents()) {
429 if (cc.type <= PrimitiveType::DOUBLE) {
430 PlyProperty pp;
431 pp.name = ply::unknown;
432 pp.unknownPropertyName = cc.name;
433 pp.type = cc.type;
434 vElem.properties.push_back(pp);
435 }
436 }
437 }
438 mElements.push_back(vElem);
439 }
440 if (info.hasFaces()) {
441 mFaceElemPos = mElements.size();
442 PlyElement fElem;
443 fElem.type = ply::FACE;
444 if (info.hasFaceVRefs()) {
445 PlyProperty vids;
446 vids.list = true;
447 vids.name = ply::vertex_indices;
448 vids.type = PrimitiveType::UINT;
449 vids.listSizeType = PrimitiveType::UCHAR;
450 fElem.properties.push_back(vids);
451 }
452 if (info.hasFaceNormals()) {
453 PlyProperty fnx, fny, fnz;
454 fnx.name = ply::nx;
455 fnx.type = info.faceNormalsType();
456 fny.name = ply::ny;
457 fny.type = info.faceNormalsType();
458 fnz.name = ply::nz;
459 fnz.type = info.faceNormalsType();
460 fElem.properties.push_back(fnx);
461 fElem.properties.push_back(fny);
462 fElem.properties.push_back(fnz);
463 }
464 if (info.hasFaceColors()) {
465 PlyProperty fcr, fcg, fcb, fca;
466 fcr.name = ply::red;
467 fcr.type = info.faceColorsType();
468 fcg.name = ply::green;
469 fcg.type = info.faceColorsType();
470 fcb.name = ply::blue;
471 fcb.type = info.faceColorsType();
472 fca.name = ply::alpha;
473 fca.type = info.faceColorsType();
474 fElem.properties.push_back(fcr);
475 fElem.properties.push_back(fcg);
476 fElem.properties.push_back(fcb);
477 fElem.properties.push_back(fca);
478 }
479 if (info.hasFaceQuality()) {
480 PlyProperty fs;
481 fs.name = ply::quality;
482 fs.type = (ply::PropertyType::Enum) info.faceQualityType();
483 fElem.properties.push_back(fs);
484 }
485 if (info.hasFaceWedgeTexCoords()) {
486 PlyProperty tc, tn;
487 tc.list = true;
488 tc.listSizeType = PrimitiveType::UCHAR;
489 tc.name = ply::texcoord;
490 tc.type =
491 (ply::PropertyType::Enum) info.faceWedgeTexCoordsType();
492 tn.name = ply::texnumber;
493 tn.type = PrimitiveType::USHORT;
494 fElem.properties.push_back(tc);
495 fElem.properties.push_back(tn);
496 }
497 if (info.hasFaceCustomComponents()) {
498 for (const auto& cc : info.faceCustomComponents()) {
499 if (cc.type <= PrimitiveType::DOUBLE) {
500 PlyProperty pp;
501 pp.name = ply::unknown;
502 pp.unknownPropertyName = cc.name;
503 pp.type = (ply::PropertyType::Enum) cc.type;
504 fElem.properties.push_back(pp);
505 }
506 }
507 }
508 mElements.push_back(fElem);
509 }
510 if (info.hasEdges()) {
511 mEdgeElemPos = mElements.size();
512 PlyElement eElem;
513 eElem.type = ply::EDGE;
514 if (info.hasEdgeVRefs()) {
515 PlyProperty v1;
516 v1.name = ply::vertex1;
517 v1.type = PrimitiveType::UINT;
518 eElem.properties.push_back(v1);
519 PlyProperty v2;
520 v2.name = ply::vertex2;
521 v2.type = PrimitiveType::UINT;
522 eElem.properties.push_back(v2);
523 }
524 mElements.push_back(eElem);
525 }

528 std::string toString() const
529 {
530 std::string s;
531
532 s += "ply\nformat ";
533
534 switch (mFormat) {
535 case ply::ASCII: s += "ascii 1.0\n"; break;
536 case ply::BINARY_BIG_ENDIAN: s += "binary_big_endian 1.0\n"; break;
537 default: s += "binary_little_endian 1.0\n"; break;
538 }
539
540 s += "comment Generated by vclib\n";
541 for (const std::string& str : mTextureFiles) {
542 s += "comment TextureFile " + str + "\n";
543 }
544 for (const PlyElement& e : mElements) {
545 s += "element ";
546 switch (e.type) {
547 case ply::VERTEX:
548 s += "vertex " + std::to_string(e.numberElements) + "\n";
549 break;
550 case ply::FACE:
551 s += "face " + std::to_string(e.numberElements) + "\n";
552 break;
553 case ply::EDGE:
554 s += "edge " + std::to_string(e.numberElements) + "\n";
555 break;
556 case ply::TRISTRIP:
557 s += "tristrips " + std::to_string(e.numberElements) + "\n";
558 break;
559 case ply::MATERIAL:
560 s += "material " + std::to_string(e.numberElements) + "\n";
561 break;
562 case ply::OTHER:
563 s += e.unknownElementType + " " +
564 std::to_string(e.numberElements) + "\n";
565 break;
566 }
567 for (const PlyProperty& p : e.properties) {
568 s += "property ";
569 if (p.list) {
570 s += "list ";
571 s += typeToString(p.listSizeType) + " ";
572 }
573 s += typeToString(p.type) + " ";
574 if (p.name == ply::unknown)
575 s += p.unknownPropertyName + "\n";
576 else
577 s += nameToString(p.name) + "\n";
578 }
579 }
580 s += "end_header\n";
581
582 return s;
583 }

592 PlyElement readElement(const vcl::Tokenizer& lineTokenizer) const
593 {
594 PlyElement e;
595 vcl::Tokenizer::iterator token = lineTokenizer.begin();
596 std::string s = *(++token);
597 if (s == "vertex") {
598 e.type = ply::VERTEX;
599 e.numberElements = std::stoi(*(++token));
600 }
601 else if (s == "face") {
602 e.type = ply::FACE;
603 e.numberElements = std::stoi(*(++token));
604 }
605 else if (s == "edge") {
606 e.type = ply::EDGE;
607 e.numberElements = std::stoi(*(++token));
608 }
609 else if (s == "tristrips") {
610 e.type = ply::TRISTRIP;
611 e.numberElements = std::stoi(*(++token));
612 }
613 else {
614 e.type = ply::OTHER;
615 e.numberElements = std::stoi(*(++token));
616 e.unknownElementType = s;
617 }
618 return e;
619 }

621 PlyProperty readProperty(const vcl::Tokenizer& lineTokenizer) const
622 {
623 PlyProperty p;
624 vcl::Tokenizer::iterator token = lineTokenizer.begin();
625 std::string type = *(++token);
626 if (type == "list") {
627 p.list = true;
628 std::string typeSize = *(++token);
629 std::string typeData = *(++token);
630 std::string name = *(++token);
631 p.listSizeType = stringToType(typeSize);
632 p.type = stringToType(typeData);
633 p.name = stringToName(name);
634 if (p.name == ply::unknown)
635 p.unknownPropertyName = name;
636 }
637 else {
638 p.list = false;
639 std::string name = *(++token);
640 p.type = stringToType(type);
641 p.name = stringToName(name);
642 if (p.name == ply::unknown)
643 p.unknownPropertyName = name;
644 }
645
646 return p;
647 }
