302 // Predicate structures
303
304 // Elements can be individuated with their ID, which is an unsigned int.
305 // This struct sets its bool `value` to true if this Mesh has a Container of Elements with the
306 // given unsigned integer El
307 // Sets also `type` with a TypeWrapper contianing the Container that satisfied the condition.
308 // the TypeWrapper will be empty if no Containers were found.
309 template<uint El>
310 struct ContainerOfTypeIndexPred
311 {
312 private:
313 template <typename Cont>
314 struct SameElPred
315 {
316 static constexpr bool value = Cont::ELEMENT_TYPE == El;
317 };
318
319 public:
320 // TypeWrapper of the found container, if any
321 using type = typename vcl::FilterTypesByCondition<SameElPred, Containers>::type;
322 static constexpr bool value = NumberOfTypes<type>::value == 1;
323 };
324
325 template<ElementConcept El>
326 struct HasContainerOfPred
327 {
328 static constexpr bool value = ContainerOfTypeIndexPred<El::ELEMENT_TYPE>::value;
329 };
330
331 template<ElementConcept El>
332 struct GetContainerOf
333 {
334 private:
335 template<typename>
336 struct TypeUnwrapper {};
337
338 template<typename C>
339 struct TypeUnwrapper<TypeWrapper<C>>
340 {
341 using type = C;
342 };
343 public:
344 using type = typename TypeUnwrapper<
345 typename ContainerOfTypeIndexPred<El::ELEMENT_TYPE>::type>::type;
346 };
347};
