262 Mesh(const Mesh& oth) :
263 Args(oth)... // call auto copy constructors for all the container
264 // elements and components
265 {
266 // Set to all element containers their parent mesh (this)

269 // for each container of oth, save its base pointer
270 // will need it to update all the pointers of this mesh
271 constexpr uint N_CONTAINERS =
272 NumberOfTypes<typename Mesh<Args...>::Containers>::value;
273 std::array<const void*, N_CONTAINERS> othBases =
274 Mesh<Args...>::getContainerBases(oth);
275
276 // update all the pointers contained on each container
277 // use the base pointer of each container of oth to update all the
278 // pointers in this mesh. Each pointer of oth that was copied in this
279 // mesh, will be updated computing its offset wrt the base of oth, and
280 // then adding that offset to the new base pointer of this mesh
281 (updateReferencesOfContainerType<Args>(*this, othBases), ...);

394 void append(const Mesh& m)
395 {
396 // for each container of this mesh, save its size
397 // will need it to update all the pointers/indices of this mesh
398 constexpr uint N_CONTAINERS =
399 NumberOfTypes<typename Mesh<Args...>::Containers>::value;
400 std::array<std::size_t, N_CONTAINERS> sizes =
401 Mesh<Args...>::getContainerSizes(*this);
402
403 // for each container of the other mesh m, save its bases
404 // will need it to update all the pointers of this mesh
405 std::array<const void*, N_CONTAINERS> bases =
406 Mesh<Args...>::getContainerBases(m);
407
408 // call the append function for each container
409 (appendContainer<Args>(m), ...);
410
411 // update all the pointers/indices contained on each container
412 (updateReferencesOfContainerTypeAfterAppend<Args>(*this, bases, sizes),
413 ...);

430 template<typename OtherMeshType>
431 void importFrom(const OtherMeshType& m)
432 {
433 // This function will first:
434 // Call, for each Container and Component of the mesh, its importFrom
435 // function. In case of containers, it first creates the same number of
436 // elements in the container, and then calls the importFrom function for
437 // each new element.
438
439 (Args::importFrom(m), ...);
440
441 // Set to all element containers their parent mesh (this)
442 updateAllParentMeshPointers();
443
444 if constexpr (mesh::HasFaceContainer<Mesh<Args...>>) {
445 // Now I need to manage imports between different types of meshes
446 // (same type of meshes are already managed from importFrom member
447 // function).
448 //
449 // Generally speaking, Polygon meshes can import from any other type
450 // of mesh. We need to take care when this mesh has static vertex
451 // references number in the face container (VERTEX_NUMBER >= 3).
452 //
453 // The follwing case don't need to be managed:
454 // - import polygon mesh from triangle mesh
455 //
456 // I can manage the following cases:
457 // - import triangle mesh from polygon mesh: need triangulation
458 //
459 // I cannot manage the follwing cases:
460 // - import static non-triangle mesh from polygon mesh or from a
461 // mesh with different VERTEX_NUMBER
462
463 // in case of import from poly to triangle mesh, I need to manage
464 // triangulation of polygons and create additional triangle faces
465 // for each of the imported polygons. This function statically
466 // asserts that the import can be done.
467 using FaceContainer = typename Mesh<Args...>::FaceContainer;
468 FaceContainer::manageImportTriFromPoly(m);
469 }
470 }

476 void swap(Mesh& m2)
477 {
478 Mesh<Args...>& m1 = *this;
479
480 constexpr uint N_CONTAINERS =
481 NumberOfTypes<typename Mesh<Args...>::Containers>::value;
482 static_assert(N_CONTAINERS != 0);
483
484 // container bases of each container for m1 and m2
485 // we save the bases of the containers before swap
486 std::array<const void*, N_CONTAINERS> m1Bases =
487 Mesh<Args...>::getContainerBases(m1);
488 std::array<const void*, N_CONTAINERS> m2Bases =
489 Mesh<Args...>::getContainerBases(m2);
490
491 // actual swap of all the containers and the components of the mesh
492 // using pack expansion: swap will be called for each of the containers
493 // (or components!) that compose the Mesh
494 using std::swap;
495 (swap((Args&) m1, (Args&) m2), ...);
496
497 // Set to all elements their parent mesh
498 m1.updateAllParentMeshPointers();
499 m2.updateAllParentMeshPointers();
500
501 // update all the refs to m1 and m2: old base of m1 is now "old
502 // base" of m2, and viceversa
503 (updateReferencesOfContainerType<Args>(m1, m2Bases), ...);
504 (updateReferencesOfContainerType<Args>(m2, m1Bases), ...);
505 }

1611 template<uint I, typename Cont, typename Array, typename... A>
1612 static void setContainerBase(const Mesh<A...>& m, Array& bases)
1613 {
1614 // since this function is called using pack expansion, it means that
1615 // Cont could be a mesh component and not a cointainer. We check if Cont
1616 // is a container
1617 if constexpr (mesh::ElementContainerConcept<Cont>) {
1618 static_assert(I >= 0 && I != UINT_NULL);
1619 bases[I] = m.Cont::mElemVec.data();
1620 }

1623 template<typename... A>
1624 static auto getContainerBases(const Mesh<A...>& m)
1625 {
1626 using Containers = Mesh<A...>::Containers;
1627
1628 // the number of containers in Mesh<A...>
1629 constexpr uint N_CONTAINERS = NumberOfTypes<Containers>::value;
1630 // each element of this array will contain the base pointer of the
1631 // vector of elements contained in each container of Mesh<A...>
1632 std::array<const void*, N_CONTAINERS> bases;
1633
1634 // for each container/component of Mesh<A...>, we set call the function
1635 // that sets the base of the container in its index
1636 (setContainerBase<IndexInTypes<A, Containers>::value, A>(m, bases),
1637 ...);
1638
1639 return bases;
1640 }
