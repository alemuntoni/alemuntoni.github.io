412 // PointersComponent interface functions
413 template<typename Element, typename ElEType>
414 void importPointersFrom(const Element& e, Edge* base, const ElEType* ebase)
415 {
416 if constexpr (HasAdjacentEdges<Element>) {
417 if (isAdjacentEdgesAvailableOn(e)) {
418 if constexpr (N > 0) {
419 // same static size
420 if constexpr (N == Element::ADJ_EDGE_NUMBER) {
421 importPtrsFrom(e, base, ebase);
422 }
423 // from dynamic to static, but dynamic size == static size
424 else if constexpr (Element::ADJ_EDGE_NUMBER < 0) {
425 if (e.adjEdgesNumber() == N) {
426 importPtrsFrom(e, base, ebase);
427 }
428 }
429 else {
430 // do not import in this case: cannot import from dynamic
431 // size != static size
432 }
433 }
434 else {
435 // from static/dynamic to dynamic size: need to resize first,
436 // then import
437 resize(e.adjEdgesNumber());
438 importPtrsFrom(e, base, ebase);
439 }
440 }
441 }
442 }

461private:
462 template<typename Element, typename ElEType>
463 void importPtrsFrom(const Element& e, Edge* base, const ElEType* ebase)
464 {
465 if (ebase != nullptr && base != nullptr) {
466 for (uint i = 0; i < e.adjEdgesNumber(); ++i) {
467 if (e.adjEdge(i) != nullptr) {
468 adjEdge(i) = base + (e.adjEdge(i) - ebase);
469 }
470 }
471 }
472 }
