167 template<MeshConcept Mesh>
168 MeshInfo(const Mesh& m)
169 {
170 setVertices();
171 setVertexCoords(
172 true, getType<typename Mesh::VertexType::CoordType::ScalarType>());
173 if constexpr (vcl::HasPerVertexNormal<Mesh>) {
174 if (vcl::isPerVertexNormalAvailable(m)) {
175 setVertexNormals(
176 true,
177 getType<
178 typename Mesh::VertexType::NormalType::ScalarType>());
179 }
180 }
181 if constexpr (vcl::HasPerVertexColor<Mesh>) {
182 if (vcl::isPerVertexColorAvailable(m)) {
183 setVertexColors(true, PrimitiveType::UCHAR);
184 }
185 }
186 if constexpr (vcl::HasPerVertexQuality<Mesh>) {
187 if (vcl::isPerVertexQualityAvailable(m)) {
188 setVertexQuality(
189 true, getType<typename Mesh::VertexType::QualityType>());
190 }
191 }
192 if constexpr (vcl::HasPerVertexTexCoord<Mesh>) {
193 if (vcl::isPerVertexTexCoordAvailable(m)) {
194 setVertexTexCoords(
195 true,
196 getType<
197 typename Mesh::VertexType::TexCoordType::ScalarType>());
198 }
199 }
200 if constexpr (vcl::HasPerVertexCustomComponents<Mesh>) {
201 auto names = m.perVertexCustomComponentNames();
202 for (auto& name : names) {
203 DataType dt = getType(m.perVertexCustomComponentType(name));
204 if (dt != PrimitiveType::NONE) {
205 addVertexCustomComponent(name, dt);
206 }
207 }
208 }
209
210 if constexpr (vcl::HasFaces<Mesh>) {
211 setFaces();
212 setFaceVRefs();
213 if (vcl::HasTriangles<Mesh>) {
214 setTriangleMesh();
215 }
216 else if (vcl::HasQuads<Mesh>) {
217 setQuadMesh();
218 }
219 else {
220 setPolygonMesh();
221 }
222 if constexpr (vcl::HasPerFaceNormal<Mesh>) {
223 if (vcl::isPerFaceNormalAvailable(m)) {
224 setFaceNormals(
225 true,
226 getType<
227 typename Mesh::FaceType::NormalType::ScalarType>());
228 }
229 }
230 if constexpr (vcl::HasPerFaceColor<Mesh>) {
231 if (vcl::isPerFaceColorAvailable(m)) {
232 setFaceColors(true, PrimitiveType::UCHAR);
233 }
234 }
235 if constexpr (vcl::HasPerFaceQuality<Mesh>) {
236 if (vcl::isPerFaceQualityAvailable(m)) {
237 setFaceQuality(
238 true, getType<typename Mesh::FaceType::QualityType>());
239 }
240 }
241 if constexpr (vcl::HasPerFaceWedgeTexCoords<Mesh>) {
242 if (vcl::isPerFaceWedgeTexCoordsAvailable(m)) {
243 setFaceWedgeTexCoords(
244 true,
245 getType<typename Mesh::FaceType::WedgeTexCoordType::
246 ScalarType>());
247 }
248 }
249 if constexpr (vcl::HasPerFaceCustomComponents<Mesh>) {
250 auto names = m.perFaceCustomComponentNames();
251 for (auto& name : names) {
252 DataType dt = getType(m.perFaceCustomComponentType(name));
253 if (dt != PrimitiveType::NONE) {
254 addFaceCustomComponent(name, dt);
255 }
256 }
257 }
258 }
259
260 if constexpr (vcl::HasEdges<Mesh>) {
261 setEdges();
262 setEdgeVRefs();
263 // if constexpr (vcl::HasPerEdgeColor<Mesh>)
264 // if (vcl::isPerEdgeColorAvailable(m))
265 // setEdgeColors(true, UCHAR);
266 }
267
268 if constexpr (vcl::HasTexturePaths<Mesh> || vcl::HasTextureImages<Mesh>)
269 {
270 if (m.textureNumber() > 0) {
271 setTextures(true);
272 }
273 }
274 }

676 [[nodiscard]] MeshInfo intersect(const MeshInfo& info) const
677 {
678 MeshInfo res;
679 for (uint i = 0; i < NUM_ELEMENTS; ++i) {
680 res.mElements[i] = mElements[i] && info.mElements[i];
681 for (uint j = 0; j < NUM_COMPONENTS; ++j) {
682 res.mPerElemComponents[i][j] =
683 mPerElemComponents[i][j] && info.mPerElemComponents[i][j];
684
685 if (res.mPerElemComponents[i][j]) {
686 res.mPerElemComponentsType(i, j) =
687 mPerElemComponentsType(i, j);
688 }
689 }
690 }
691
692 if (mType == info.mType) {
693 res.mType = mType;
694 }
695 res.mPerElemCustomComponents = mPerElemCustomComponents;
696
697 return res;
698 }

709 template<typename T>
710 static DataType getType()
711 {
712 if constexpr (std::is_same_v<T, char>)
713 return PrimitiveType::CHAR;
714 if constexpr (std::is_same_v<T, unsigned char>)
715 return PrimitiveType::UCHAR;
716 if constexpr (std::is_same_v<T, short>)
717 return PrimitiveType::SHORT;
718 if constexpr (std::is_same_v<T, unsigned short>)
719 return PrimitiveType::USHORT;
720 if constexpr (std::is_same_v<T, int>)
721 return PrimitiveType::INT;
722 if constexpr (std::is_same_v<T, uint>)
723 return PrimitiveType::UINT;
724 if constexpr (std::is_integral_v<T>)
725 return PrimitiveType::INT; // fallback to int
726 if constexpr (std::is_same_v<T, float>)
727 return PrimitiveType::FLOAT;
728 if constexpr (std::is_same_v<T, double>)
729 return PrimitiveType::DOUBLE;
730 if constexpr (std::is_floating_point_v<T>)
731 return PrimitiveType::FLOAT; // fallback to float
732 return PrimitiveType::NONE;
733 }

757template<uint ELEM_ID, MeshConcept MeshType>
758void addPerElementCustomComponent(
759 MeshType& m,
760 const MeshInfo::CustomComponent& cc)
761{
762 switch (cc.type) {
763 case PrimitiveType::CHAR:
764 m.template addPerElementCustomComponent<ELEM_ID, char>(cc.name);
765 break;
766 case PrimitiveType::UCHAR:
767 m.template addPerElementCustomComponent<ELEM_ID, unsigned char>(
768 cc.name);
769 break;
770 case PrimitiveType::SHORT:
771 m.template addPerElementCustomComponent<ELEM_ID, short>(cc.name);
772 break;
773 case PrimitiveType::USHORT:
774 m.template addPerElementCustomComponent<ELEM_ID, unsigned short>(
775 cc.name);
776 break;
777 case PrimitiveType::INT:
778 m.template addPerElementCustomComponent<ELEM_ID, int>(cc.name);
779 break;
780 case PrimitiveType::UINT:
781 m.template addPerElementCustomComponent<ELEM_ID, uint>(cc.name);
782 break;
783 case PrimitiveType::FLOAT:
784 m.template addPerElementCustomComponent<ELEM_ID, float>(cc.name);
785 break;
786 case PrimitiveType::DOUBLE:
787 m.template addPerElementCustomComponent<ELEM_ID, double>(cc.name);
788 break;
789 default: assert(0);
790 }
