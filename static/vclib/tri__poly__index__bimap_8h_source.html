112 uint triangleNumber(uint polygonIndex) const
113 {
114 assert(polygonIndex < mPolyToTri.size());
115 if (polygonIndex < mPolyToTri.size() - 1) {
116 return mPolyToTri[polygonIndex + 1] - mPolyToTri[polygonIndex];
117 }
118 else {
119 // total number of triangles minus the first triangle index of the
120 // polygon
121 return mTriToPoly.size() - mPolyToTri[polygonIndex];
122 }
123 }
124
128 void clear()
129 {
130 mTriToPoly.clear();
131 mPolyToTri.clear();
132 }
133
141 void reserve(uint nTriangles, uint nPolygons)
142 {
143 mTriToPoly.reserve(nTriangles);
144 mPolyToTri.reserve(nPolygons);
145 }
146
161 void insert(uint triangleIndex, uint polygonIndex)
162 {
163 // add the index of the polygon associated to the triangle
164 if (triangleIndex >= mTriToPoly.size()) {
165 mTriToPoly.resize(triangleIndex + 1, UINT_NULL);
166 }
167 mTriToPoly[triangleIndex] = polygonIndex;
168
169 // add the index of the triangle associated to the polygon,
170 // but only if it is the first triangle index of the polygon!
171 if (polygonIndex >= mPolyToTri.size()) {
172 mPolyToTri.resize(polygonIndex + 1, UINT_NULL);
173 }
174 if (mPolyToTri[polygonIndex] == UINT_NULL ||
175 triangleIndex < mPolyToTri[polygonIndex])
176 mPolyToTri[polygonIndex] = triangleIndex;
177 }
178
183 uint triangleNumber() const { return mTriToPoly.size(); }
184
189 uint polygonNumber() const { return mPolyToTri.size(); }
190};
