152 void insert(uint triangleIndex, uint polygonIndex)
153 {
154 // add the index of the polygon associated to the triangle
155 if (triangleIndex >= triToPoly.size()) {
156 triToPoly.resize(triangleIndex + 1, UINT_NULL);
157 }
158 triToPoly[triangleIndex] = polygonIndex;
159
160 // add the index of the triangle associated to the polygon,
161 // but only if it is the first triangle index of the polygon!
162 if (polygonIndex >= polyToTri.size()) {
163 polyToTri.resize(polygonIndex + 1, UINT_NULL);
164 }
165 if (polyToTri[polygonIndex] == UINT_NULL ||
166 triangleIndex < polyToTri[polygonIndex])
167 polyToTri[polygonIndex] = triangleIndex;
