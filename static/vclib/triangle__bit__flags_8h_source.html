96 static const uint FIRST_USER_BIT = 12;

98 // indices of the bits
99 enum {
100 DELETED = 0, // bit 0
101 SELECTED = 1, // bit 1
102 VISITED = 2, // bit 2
103 // Edge border
104 BORDER0 = 3, // bits [3, 5]
105 // Edge selection
106 EDGESEL0 = 6, // bits [6, 8]
107 // Faux edges: when representing polygonal meshes on triangle meshes,
108 // some triangle edges can be marked as "faux", meaning that they are
109 // internal on the polygon
110 FAUX0 = 9 // bits [9, 11]
111 };

309 void importFromVCGFlags(int f)
310 {
311 if (f & 0x00000010)
312 visited() = true;
313 if (f & 0x00000020)
314 selected() = true;
315 if (f & 0x00000040)
316 edgeOnBorder(0) = true;
317 if (f & 0x00000080)
318 edgeOnBorder(1) = true;
319 if (f & 0x00000100)
320 edgeOnBorder(2) = true;
321 if (f & 0x00008000)
322 edgeSelected(0) = true;
323 if (f & 0x00010000)
324 edgeSelected(1) = true;
325 if (f & 0x00020000)
326 edgeSelected(2) = true;
327 if (f & 0x00040000)
328 edgeFaux(0) = true;
329 if (f & 0x00080000)
330 edgeFaux(1) = true;
331 if (f & 0x00100000)
332 edgeFaux(2) = true;

342 int exportToVCGFlags() const
343 {
344 int f = 0;
345 if (visited())
346 f &= 0x00000010;
347 if (selected())
348 f &= 0x00000020;
349 if (edgeOnBorder(0))

352 f &= 0x00000080;

354 f &= 0x00000100;
355 if (edgeSelected(0))
356 f &= 0x00008000;
357 if (edgeSelected(1))
358 f &= 0x00010000;
359 if (edgeSelected(2))
360 f &= 0x00020000;
361 if (edgeFaux(0))
362 f &= 0x00040000;
363 if (edgeFaux(1))
364 f &= 0x00080000;
365 if (edgeFaux(2))
366 f &= 0x00100000;

376 // Component interface functions
377 template<typename Element>
378 void importFrom(const Element& e)
379 {
380 if constexpr (HasBitFlags<Element>) {
381 resetBitFlags();
382 if constexpr (HasTriangleBitFlags<Element>) {
383 flags() = e.flags();
384 }
385 else {
386 // BitFlags
387 deletedBit() = e.deleted();
388 selected() = e.selected();
389 visited() = e.visited();
390 const uint UM = std::min(USER_BITS_NUMBER, e.USER_BITS_NUMBER);
391 for (uint i = 0; i < UM; ++i)
392 userBit(i) = e.userBit(i);
393 if constexpr (HasPolygonBitFlags<Element>) {
394 // PolygonBitFlags
395 for (uint i = 0; i < 3; ++i) {
396 edgeOnBorder(i) = e.edgeOnBorder(i);
397 edgeSelected(i) = e.edgeSelected(i);
398 edgeFaux(i) = e.edgeFaux(i);
399 }
400 }
401
402 }
403 }
404 }
