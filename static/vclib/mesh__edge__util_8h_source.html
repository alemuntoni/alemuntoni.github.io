32/*
33 * Utility class that stores, for an edge of the mesh:
34 * - the index of the edge in the face
35 * - the pointer of the face
36 * - the pointers of the vertices of the edge
37 *
38 * edges can be sorted using this class
39 */
40template<FaceMeshConcept MeshType>
41class MeshEdgeUtil
42{
43public:
44 using VertexType = MeshType::VertexType;
45 using FaceType = MeshType::FaceType;
46
47 VertexType* v[2]; // Pointer to the two (ordered) vertices of the edge
48 FaceType* f; // Pointer to the face of the edge
49 int e; // Index of the edge inside the face
50
51 MeshEdgeUtil() : v {nullptr, nullptr}, f(nullptr), e(-1) {}
52 MeshEdgeUtil(FaceType& pf, uint ne)
53 {
54 v[0] = pf.vertex(ne);
55 v[1] = pf.vertexMod(ne + 1);
56 assert(v[0] != v[1]);
57
58 if (v[0] > v[1])
59 std::swap(v[0], v[1]);
60 f = &pf;
61 e = ne;
62 }
63
64 bool operator<(const MeshEdgeUtil& pe) const
65 {
66 if (v[0] < pe.v[0])
67 return true;
68 else if (v[0] > pe.v[0])
69 return false;
70 else
71 return v[1] < pe.v[1];
72 }
73
74 bool operator==(const MeshEdgeUtil& pe) const { return v[0] == pe.v[0] && v[1] == pe.v[1]; }
75 bool operator!=(const MeshEdgeUtil& pe) const { return v[0] != pe.v[0] || v[1] != pe.v[1]; }
76};
