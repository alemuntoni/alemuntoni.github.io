41template<typename T>
42concept ArrayConcept = requires(T o, const T& co)
43{
44 typename T::ValueType;
45 typename T::Scalar;
46 typename T::ConstReference;
47 typename T::Reference;
48 typename T::ConstPointer;
49 typename T::Pointer;
50 typename T::ConstIterator;

52
53 o.DIM;
54
55 { o.size(std::size_t()) } -> std::convertible_to<std::size_t>;
56 { o.empty() } -> std::convertible_to<bool>;
57
58 { o.data() } -> std::same_as<typename T::Pointer>;
