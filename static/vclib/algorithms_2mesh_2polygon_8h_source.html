56template<FaceMeshConcept MeshType, FaceConcept FaceType>
57void addTriangleFacesFromPolygon(
58 MeshType& m,
59 FaceType& f,
60 const std::vector<uint>& polygon)
61{
62 using VertexType = MeshType::VertexType;
63 using CoordType = VertexType::CoordType;
64
65 // from the ids, create a polygon of coordinates
66 std::vector<CoordType> polCoords(polygon.size());
67 for (uint i = 0; i < polygon.size(); ++i) {
68 if (polygon[i] >= m.vertexContainerSize()) {
69 throw BadVertexIndexException(
70 "Index " + std::to_string(polygon[i]) +
71 " is out of range in Vertex Container.");
72 }
73 if (m.vertex(polygon[i]).deleted()) {
74 throw BadVertexIndexException(
75 "Vertex " + std::to_string(polygon[i]) + " is deleted.");
76 }
77 polCoords[i] = m.vertex(polygon[i]).coord();
78 }
79
80 // compute earcut of the polygons
81 std::vector<uint> tris =
82 Polygon<CoordType>::earCut(polCoords.begin(), polCoords.end());
83
84 // faux edges management: create a set of unordered edges of the polygon
85 // note: we use indices from 0 to polygon.size() because that are the output
86 // indices given by the earcut algorithm
87 std::set<std::pair<uint, uint>, UnorderedPairComparator<uint>>
88 unorderedEdges;
89 for (uint i = 0; i < polygon.size(); ++i)
90 unorderedEdges.emplace(i, (i + 1) % (uint) polygon.size());
91
92 if constexpr (FaceType::VERTEX_NUMBER < 0) {
93 f.resizeVertices(3);
94 }
95
96 // set the first triangle of the loaded polygon
97 for (uint i = 0; i < f.vertexNumber(); ++i) {
98 f.setVertex(i, polygon[tris[i]]);
99 }
100
101 if constexpr (face::HasFaceBitFlags<FaceType>) {
102 if (unorderedEdges.find(std::make_pair(tris[0], tris[1])) ==
103 unorderedEdges.end())
104 f.edgeFaux(0) = true;
105 if (unorderedEdges.find(std::make_pair(tris[1], tris[2])) ==
106 unorderedEdges.end())
107 f.edgeFaux(1) = true;
108 if (unorderedEdges.find(std::make_pair(tris[2], tris[0])) ==
109 unorderedEdges.end())
110 f.edgeFaux(2) = true;
111 }
112
113 // remaining triangles, need to create more faces in the mesh
114 for (uint i = 3; i < tris.size(); i += 3) {
115 uint ff = m.addFace();
116
117 if constexpr (FaceType::VERTEX_NUMBER < 0) {
118 m.face(ff).resizeVertices(3);
119 }
120
121 for (uint j = 0; j < m.face(ff).vertexNumber(); ++j) {
122 m.face(ff).setVertex(j, polygon[tris[i + j]]);
123 }
124
125 if constexpr (face::HasFaceBitFlags<FaceType>) {
126 if (unorderedEdges.find(std::make_pair(tris[i], tris[i + 1])) ==
127 unorderedEdges.end())
128 m.face(ff).edgeFaux(0) = true;
129 if (unorderedEdges.find(std::make_pair(tris[i + 1], tris[i + 2])) ==
130 unorderedEdges.end())
131 m.face(ff).edgeFaux(1) = true;
132 if (unorderedEdges.find(std::make_pair(tris[i + 2], tris[i + 0])) ==
133 unorderedEdges.end())
134 m.face(ff).edgeFaux(2) = true;
135 }
136 }
137}
