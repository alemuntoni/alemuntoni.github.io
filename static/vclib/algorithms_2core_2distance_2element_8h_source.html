109template<Point3Concept PointType, FaceConcept FaceType, typename ScalarType>
110auto boundedDistance(
111 const PointType& p,
112 const FaceType& f,
113 ScalarType maxDist,
114 PointType& closest,
115 bool signedDist = false)
116{
117 if constexpr (TriangleFaceConcept<FaceType>) {
118 TriangleWrapper tw(
119 f.vertex(0)->coord(), f.vertex(1)->coord(), f.vertex(2)->coord());
120
121 return boundedDistance(p, tw, maxDist, closest, signedDist);
122 }
123 else {
124 ScalarType minDist = maxDist;
125
126 std::vector<uint> tris = earCut(f);
127 for (uint i = 0; i < tris.size(); i += 3) {
128 PointType w;
129 ScalarType d = boundedDistance(
130 p,
131 TriangleWrapper(
132 f.vertex(tris[i])->coord(),
133 f.vertex(tris[i + 1])->coord(),
134 f.vertex(tris[i + 2])->coord()),
135 minDist,
136 w,
137 signedDist);
138
139 if (std::abs(d) < minDist) {
140 minDist = std::abs(d);
141 closest = w;
142 }
143 }
144
145 return minDist;
146 }
147}
