33template<MeshConcept MeshType, typename VCGMeshType>
34void exportMeshToVCGMesh(const MeshType& mesh, VCGMeshType& vcgMesh)
35{
36 vcg::tri::Allocator<VCGMeshType>::AddVertices(vcgMesh, mesh.vertexNumber());
37
38 uint vi = 0;
39 for (const auto& v : mesh.vertices()) {
40 vcgMesh.vert[vi].P().X() = v.coord().x();
41 vcgMesh.vert[vi].P().Y() = v.coord().y();
42 vcgMesh.vert[vi].P().Z() = v.coord().z();
43
44 // flags
45 vcgMesh.vert[vi].Flags() = v.exportFlagsToVCGFormat();
46
47 // normal
48 if constexpr (HasPerVertexNormal<MeshType>) {
49 if (isPerVertexNormalAvailable(mesh)) {
50 if (vcg::tri::HasPerVertexNormal(vcgMesh)) {
51 vcgMesh.vert[vi].N().X() = v.normal().x();
52 vcgMesh.vert[vi].N().Y() = v.normal().y();
53 vcgMesh.vert[vi].N().Z() = v.normal().z();
54 }
55 }
56 }
57
58 // color
59 if constexpr (HasPerVertexColor<MeshType>) {
60 if (isPerVertexColorAvailable(mesh)) {
61 if (vcg::tri::HasPerVertexColor(vcgMesh)) {
62 vcgMesh.vert[vi].C()[0] = v.color().red();
63 vcgMesh.vert[vi].C()[1] = v.color().green();
64 vcgMesh.vert[vi].C()[2] = v.color().blue();
65 vcgMesh.vert[vi].C()[3] = v.color().alpha();
66 }
67 }
68 }
69
70 // quality
71 if constexpr (HasPerVertexQuality<MeshType>) {
72 if (isPerVertexQualityAvailable(mesh)) {
73 if (vcg::tri::HasPerVertexQuality(vcgMesh)) {
74 vcgMesh.vert[vi].Q() = v.quality();
75 }
76 }
77 }
78
79 // texcoord
80 if constexpr (HasPerVertexTexCoord<MeshType>) {
81 if (isPerVertexTexCoordAvailable(mesh)) {
82 if (vcg::tri::HasPerVertexTexCoord(vcgMesh)) {
83 vcgMesh.vert[vi].T().U() = v.texCoord().u();
84 vcgMesh.vert[vi].T().V() = v.texCoord().v();
85 }
86 }
87 }
88
89 // todo - custom components
90
91 vi++;
92 }
93
94 if constexpr (HasFaces<MeshType>) {
95 vcg::tri::Allocator<VCGMeshType>::AddFaces(vcgMesh, mesh.faceNumber());
96
97 uint fi = 0;
98 for (const auto& f : mesh.faces()) {
99 vcgMesh.face[fi].V(0) = &vcgMesh.vert[mesh.index(f.vertex(0))];
100 vcgMesh.face[fi].V(1) = &vcgMesh.vert[mesh.index(f.vertex(1))];
101 vcgMesh.face[fi].V(2) = &vcgMesh.vert[mesh.index(f.vertex(2))];
102
103 // flags
104 vcgMesh.face[fi].Flags() = f.exportFlagsToVCGFormat();
105
106 // normal
107 if constexpr (HasPerFaceNormal<MeshType>) {
108 if (isPerFaceNormalAvailable(mesh)) {
109 if (vcg::tri::HasPerFaceNormal(vcgMesh)) {
110 vcgMesh.face[fi].N().X() = f.normal().x();
111 vcgMesh.face[fi].N().Y() = f.normal().y();
112 vcgMesh.face[fi].N().Z() = f.normal().z();
113 }
114 }
115 }
116
117 // color
118 if constexpr (HasPerFaceColor<MeshType>) {
119 if (isPerFaceColorAvailable(mesh)) {
120 if (vcg::tri::HasPerFaceColor(vcgMesh)) {
121 vcgMesh.face[fi].C()[0] = f.color().red();
122 vcgMesh.face[fi].C()[1] = f.color().green();
123 vcgMesh.face[fi].C()[2] = f.color().blue();
124 vcgMesh.face[fi].C()[3] = f.color().alpha();
125 }
126 }
127 }
128
129 // quality
130 if constexpr (HasPerFaceQuality<MeshType>) {
131 if (isPerFaceQualityAvailable(mesh)) {
132 if (vcg::tri::HasPerFaceQuality(vcgMesh)) {
133 vcgMesh.face[fi].Q() = f.quality();
134 }
135 }
136 }
137
138 // wedge texcoords
139 if constexpr (HasPerFaceWedgeTexCoords<MeshType>) {
140 if (isPerFaceWedgeTexCoordsAvailable(mesh)) {
141 if (vcg::tri::HasPerWedgeTexCoord(vcgMesh)) {
142 vcgMesh.face[fi].WT(0).U() = f.wedgeTexCoord(0).u();
143 vcgMesh.face[fi].WT(0).V() = f.wedgeTexCoord(0).v();
144 vcgMesh.face[fi].WT(0).N() = f.textureIndex();
145 vcgMesh.face[fi].WT(1).U() = f.wedgeTexCoord(1).u();
146 vcgMesh.face[fi].WT(1).V() = f.wedgeTexCoord(1).v();
147 vcgMesh.face[fi].WT(1).N() = f.textureIndex();
148 vcgMesh.face[fi].WT(2).U() = f.wedgeTexCoord(2).u();
149 vcgMesh.face[fi].WT(2).V() = f.wedgeTexCoord(2).v();
150 vcgMesh.face[fi].WT(2).N() = f.textureIndex();
151 }
152 }
153 }
154
155 // todo - custom components
156
157 fi++;
158 }
159 }
160}
