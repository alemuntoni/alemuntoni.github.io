126 PolymorphicObjectVector(const PolymorphicObjectVector& other)
127 {
128 if constexpr (N < 0) {
129 Base::resize(other.size());
130 }
131 std::transform(
132 other.begin(), other.end(), Base::begin(), [](const auto& e) {
133 return e->clone();
134 });
135 }

162 PolymorphicObjectVector(std::size_t size, const T& value)
163 {
164 if constexpr (N >= 0) {
165 if (size != N) {
166 throw WrongSizeException(
167 "Vector must have " + std::to_string(N) + " size.");
168 }
169 fill(value);
170 }
171 else {
172 resize(size, value);
173 }

312 template<Range Rng>
313 void set(Rng&& r) requires InputRange<Rng, T>
314 {
315 uint n = std::ranges::distance(r);
316
317 if constexpr (N >= 0) {
318 n = std::min((uint) N, n);
319 }
320 else {
321 Base::resize(n);
322 }
323
324 // for each element in the range, clone it and store it in the
325 // vector
326 std::transform(
327 std::ranges::begin(r),
328 std::ranges::begin(r) + n,
329 Base::begin(),
330 [](const auto& e) {
331 return e.clone();
332 });
333 }
334
344 void fill(const T& e)
345 {
346 for (uint i = 0; i < size(); i++) {
347 Base::at(i) = e.clone();
348 }
349 }

367 void resize(uint n, const T& v) requires (N < 0)
368 {
369 if (n > Base::size()) {
370 uint oldSize = Base::size();
371 Base::resize(n);
372 for (uint i = oldSize; i < n; i++) {
373 Base::at(i) = v.clone();
374 }
375 }
376 else {
377 Base::resize(n);
378 }
