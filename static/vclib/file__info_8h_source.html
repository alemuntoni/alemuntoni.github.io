83 static bool isFileBinary(const std::string& filename)
84 {
85 const std::size_t CHECK_BUFF_SIZE = 1000;
86
87 std::ifstream file(filename, std::ios::binary | std::ios::ate);
88 std::size_t fsize = file.tellg();
89 std::size_t size = std::min(fsize, CHECK_BUFF_SIZE);
90 file.seekg(0, std::ios::beg);
91
92 std::vector<unsigned char> buffer(size);
93 if (file.read((char*)buffer.data(), size)) {
94 for (uint i = 0; i < buffer.size(); ++i) {
95 if (buffer[i] > 127)
96 return true;
97 }
98 }
99 else {
100 throw vcl::MalformedFileException("Cannot read data from file.");
101 }
102 return false;
103 }

265 static std::string addExtensionToFileName(
266 const std::string& filename,
267 const std::string& ext)
268 {
269 std::string actualfilename;
270 size_t lastindex = filename.find_last_of(".");
271 if (lastindex != std::string::npos) {
272 std::string e = filename.substr(lastindex+1, filename.size());
273 if (e == ext)
274 actualfilename = filename;
275 else
276 actualfilename = filename + "." + ext;
277 }
278 else
279 actualfilename = filename + "." + ext;
280 return actualfilename;
