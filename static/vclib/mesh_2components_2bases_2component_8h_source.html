194template<uint COMPONENT_ID, ElementOrMeshConcept T>
195bool isComponentAvailableOn(const T& obj)
196{
197 if constexpr (HasOptionalComponentOfType<T, COMPONENT_ID>) {
198 using ComponentType =
199 ComponentOfType<COMPONENT_ID, typename T::Components>;
200 const ComponentType& c = static_cast<const ComponentType&>(obj);
201 return c.isAvailable();
202 }
203 else
204 return HasComponentOfType<T, COMPONENT_ID>;
205}
