297 Scalar angle(const Point& p1) const
298 {
299 Scalar w = Base::norm() * p1.norm();
300 if (w == 0)
301 return -1;
302 Scalar t = Base::dot(p1) / w;
303 if (t > 1)
304 t = 1;
305 else if (t < -1)
306 t = -1;
307 return (Scalar) acos(t);
308 }

464 void orthoBase(Point& u, Point& v) const requires (N == 3)
465 {
466 const double locEps = double(1e-7);
467
468 Point<Scalar, 3> up(0, 1, 0);
469 u = Base::cross(up);
470
471 double len = u.norm();
472 if (len < locEps) {
473 if (std::abs(at(0)) < std::abs(at(1))) {
474 if (std::abs(at(0)) < std::abs(at(2)))
475 up = Point<Scalar, 3>(1, 0, 0); // x is the min
476 else
477 up = Point<Scalar, 3>(0, 0, 1); // z is the min
478 }
479 else {
480 if (std::abs(at(1)) < std::abs(at(2)))
481 up = Point<Scalar, 3>(0, 1, 0); // y is the min
482 else
483 up = Point<Scalar, 3>(0, 0, 1); // z is the min
484 }
485 u = Base::cross(up);
486 }
487 v = Base::cross(u);
488 }

613 Point operator*(const Eigen::Matrix<Scalar, N + 1, N + 1>& m) const
614 requires (N == 3)
615 {
616 Eigen::Matrix<Scalar, 1, N> s;
617
618 s(0) = m(0, 0) * at(0) + m(0, 1) * at(1) + m(0, 2) * at(2) + m(0, 3);
619 s(1) = m(1, 0) * at(0) + m(1, 1) * at(1) + m(1, 2) * at(2) + m(1, 3);
620 s(2) = m(2, 0) * at(0) + m(2, 1) * at(1) + m(2, 2) * at(2) + m(2, 3);
621
622 Scalar w =
623 at(0) * m(3, 0) + at(1) * m(3, 1) + at(2) * m(3, 2) + m(3, 3);
624 if (w != 0) [[likely]]
625 s = s / w;
626 return Point(s);
627 }
