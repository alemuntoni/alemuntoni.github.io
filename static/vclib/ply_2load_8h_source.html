39template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
40void loadPly(
41 MeshType& m,
42 std::istream& file,
43 const std::string& filename,
44 MeshInfo& loadedInfo,
45 LogType& log = nullLogger,
46 const LoadSettings& settings = LoadSettings())
47{
48 PlyHeader header(file, filename);
49 if (header.errorWhileLoading())
50 throw MalformedFileException("Header not valid: " + filename);
51
52 m.clear();
53
54 loadedInfo = header.getInfo();
55
56 if (settings.enableOptionalComponents)
57 enableOptionalComponentsFromInfo(loadedInfo, m);
58
59 if constexpr (HasName<MeshType>) {
60 m.name() = FileInfo::fileNameWithoutExtension(filename);
61 }
62 if constexpr (HasTexturePaths<MeshType>) {
63 m.meshBasePath() = FileInfo::pathWithoutFileName(filename);
64 }
65
66 // for logging
67 std::vector<uint> eln;
68 uint sum = 0;
69 for (const PlyElement& el : header) {
70 eln.push_back(el.numberElements);
71 sum += el.numberElements;
72 }
73
74 try {
75 uint currElems = 0;
76 uint i = 0;
77 for (const PlyElement& el : header) {
78 double beginPerc = double(currElems) / sum * 100.0;
79 double endPerc = double(currElems + eln[i]) / sum * 100.0;
80
81 switch (el.type) {
82 case ply::VERTEX:
83 log.startNewTask(beginPerc, endPerc, "Reading vertices");
84 readPlyVertices(file, header, m, log);
85 log.endTask("Reading vertices");
86 break;
87 case ply::FACE:
88 log.startNewTask(beginPerc, endPerc, "Reading faces");
89 if constexpr (HasFaces<MeshType>)
90 readPlyFaces(file, header, m, loadedInfo, log);
91 else
92 readPlyUnknownElement(file, header, el, log);
93 log.endTask("Reading faces");
94 break;
95 case ply::TRISTRIP:
96 log.startNewTask(beginPerc, endPerc, "Reading tristrips");
97 if constexpr (HasFaces<MeshType>) {
98 loadedInfo.setTriangleMesh();
99 readPlyTriStrips(file, header, m, log);
100 }
101 else
102 readPlyUnknownElement(file, header, el, log);
103 log.endTask("Reading tristrips");
104 break;
105 case ply::EDGE:
106 log.startNewTask(beginPerc, endPerc, "Reading edges");
107 if constexpr (HasEdges<MeshType>)
108 readPlyEdges(file, header, m, log);
109 else
110 readPlyUnknownElement(file, header, el, log);
111 log.endTask("Reading edges");
112 break;
113 default:
114 log.startNewTask(
115 beginPerc, endPerc, "Reading unknown elements");
116 readPlyUnknownElement(file, header, el, log);
117 log.endTask("Reading unknown elements");
118 break;
119 }
120 currElems += eln[i];
121 ++i;
122 }
123 readPlyTextures(header, m, log, settings);
124 }
125 catch (const std::runtime_error& err) {
126 m.clear();
127 throw err;
128 }
129}
