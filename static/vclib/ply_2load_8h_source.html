54template<MeshConcept MeshType, LoggerConcept LogType = NullLogger>
55void loadPly(
56 MeshType& m,
57 const std::string& filename,
58 MeshInfo& loadedInfo,
59 LogType& log = nullLogger,
60 bool enableOptionalComponents = true)
61{
62 std::ifstream file = internal::loadFileStream(filename);
63
64 ply::PlyHeader header(filename, file);
65 if (header.errorWhileLoading())
66 throw MalformedFileException("Header not valid: " + filename);
67
68 m.clear();
69
70 loadedInfo = header.getInfo();
71
72 if (enableOptionalComponents)
73 internal::enableOptionalComponents(loadedInfo, m);
74
75 if constexpr (HasName<MeshType>) {
76 m.name() = FileInfo::fileNameWithoutExtension(filename);
77 }
78 if constexpr (HasTexturePaths<MeshType>) {
79 m.meshBasePath() = FileInfo::pathWithoutFileName(filename);
80 }
81 try {
82 for (const ply::Element& el : header) {
83 switch (el.type) {
84 case ply::VERTEX: ply::loadVertices(file, header, m); break;
85 case ply::FACE: ply::loadFaces(file, header, m); break;
86 case ply::TRISTRIP: ply::loadTriStrips(file, header, m); break;
87 default: ply::readUnknownElements(file, header, el); break;
88 }
89 }
90 ply::loadTextures(header, m);
91 }
92 catch(const std::runtime_error& err) {
93 m.clear();
94 file.close();
95 throw err;
96 }
97
98 file.close();
99}
